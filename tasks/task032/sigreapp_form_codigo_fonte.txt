* PROCEDURE buscasaldos
------------------------------------------------------------
PROCEDURE buscasaldos
Private pDat

pDat = fDtoSQL(Iif(ThisForm.optSaldos.Value = 2, ThisForm.getDtInicial.Value, Date()), [23:59:59]) && 1-Informado; 2-Data Inicial; 3-Data Atual

lcQuery = [Select c.Grupos, c.Contas, d.RClis, c.Datas, c.Saldos ] + ;
			[From SigMvCcr c, SigCdCli d ] + ;
		   [Where c.cIdChaves In (] + ;
		  [Select Max(b.cIdChaves) as cIdChaves ] + ;
		    [From SigCdGcr a, SigMvCcr b ] + ;
		   [Where a.Coletors = 11 And ] + ;
		   		 [a.Codigos = b.Grupos And ] + ;
		   		 [b.Datas <= ?pDat ] + ;
		   [Group By b.Grupos, b.Contas) And ] + ;
		   [c.Contas = d.IClis And ] + ;
		   [Not c.Saldos = 0]

If (ThisForm.poDataMgr.SqlExecute(lcQuery, [LocalSaldos]) < 1)
	=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conex„o (LocalSaldos)])
	Return 0
EndIf

lnRet = 0
ThisForm.Saldos1 = [Saldo Inicial :] + Chr(13)
ThisForm.Saldos2 = [ ] + Chr(13)

Select LocalSaldos
Go Top In LocalSaldos
If Eof([LocalSaldos])
	ThisForm.Saldos1 = [Saldo Inicial :]
	ThisForm.Saldos2 = [0,00 (Nenhuma Conta Com Saldo Configurada Nos Grupos de Contas)]
Else
	Scan
		ThisForm.Saldos1 = ThisForm.Saldos1 + ;
						   Transform(LocalSaldos.Saldos, [9,999,999,999.99]) + [ = ] + Chr(13)

		ThisForm.Saldos2 = ThisForm.Saldos2 + ;
						   Alltrim(LocalSaldos.Contas) + [ / ] + Alltrim(LocalSaldos.RClis) + Chr(13)

		lnRet = lnRet + LocalSaldos.Saldos
	EndScan
EndIf
Use In LocalSaldos

Return lnRet
* Fim **************************************************************************************
ENDPROC

PROCEDURE processapagrec
Private pDataIni, pDataFin
Local lcDopes, ldDataIni, ldDataFin, lcGrupo, lcConta

lcDopes = Padr(Iif(ThisForm.ck_Pagamentos.Value = 1,[PAGAMENTO],[RECEBIMENTO]),20 )

ldDataFin = ThisForm.getDtFinal.Value

					  Hists c(100), Hist2s c(40), CondPag m(4), VlrPago m(4), nTotPago n(12,2))
Index on Emps+Str(Numes,6) Tag OrdRel					  

If lnCodOper <= 0
	lcQuery = [Select a.Emps, a.Numes, a.EmpDopNums, a.DataRcs, a.Dopes from SigCdPgr a ] + ;
				[ Inner Join SigCdPgc b on b.EmpDopNums = a.EmpDopNums ] + ;
				[ where a.Dopes = ']+lcDopes+[' and a.DataRcs between ?pDataIni and ?pDataFin ] + ;
				[ And b.Grupos = ']+lcGrupo+[' ] + ;
				Iif(not Empty(lcConta), [ And b.Contas = ']+lcConta+[' ],[]) + ;
				[ Order by a.EmpDopNums ]
Else
	lcQuery = [Select Emps, Numes, EmpDopNums, DataRcs, Dopes from SigCdPgr where Dopes = ']+lcDopes+[' and Numes = ]+Str(lnCodOper,6)
Endif				
			
If( ThisForm.poDataMgr.SQLExecute(lcQuery,[crDados1]) < 1 )
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conex„o (crDados1)')
	Return(.F.) 	
EndIf

Create Cursor crCabecalho(Empresa c(80), Titulo c(80), Periodo c(80))

=Seek(_Empr,[crSigCdEmp],[cEmps])
lcEmpresa = _Empr + [-] + crSigCdEmp.Razas
lcTitulo  = [RelatÛrio de An·lise ý Pagar/Receber - ] + Iif(lcDopes = Padr([PAGAMENTO],20),[Pagamentos],[Recebimentos])
If lnCodOper <= 0
	lcPeriodo = Iif(lcDopes = Padr([PAGAMENTO],20),[Pagamentos],[Recebimentos])+[ de : ] + Dtoc(ldDataIni) + [ ý ] + Dtoc(ldDataFin)
	lcPeriodo = lcPeriodo + [ - Grupo: ] + Alltrim(lcGrupo) + Iif(not Empty(lcConta), [ - Conta: ] + lcConta,[])
Else
	lcPeriodo = Iif(lcDopes = Padr([PAGAMENTO],20),[Pagamento],[Recebimento])+[: N∫ ] + Str(lnCodOper,6)
Endif	

Insert into crCabecalho(Empresa,Titulo,Periodo) Values (lcEmpresa,lcTitulo,lcPeriodo)

Select crDados1
Go Top in crDados1
Scan
	lcQuery = [Select a.Acertos, a.Contas, a.Datas, a.Hists, a.Moedas, a.Nopers, a.Opers, a.Tipos, a.Valos, a.Vencs, a.Empos, b.RClis from SigCdPit a ] + ;
				[Inner Join SigCdCli b on b.Iclis = a.Contas ] + ;
				[Where EmpDopNums = ']+crDados1.EmpDopNums+[']
	If( ThisForm.poDataMgr.SQLExecute(lcQuery,[crTmpPrIt]) < 1 )
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conex„o (crTmpPrIt)')
		Return(.F.)	
	Endif

	lnAcerto = 0
	lcMoeda  = []
	lcHists  = []
	lcHist2s = []
	lcCondPag= []
	lnVlrPago= 0

	lcTitulo = []

	lcEDN = crDados1.Emps + crDados1.Dopes + Str(crDados1.Numes,6)

		Return(.F.)
	EndIf
	Select crParcPag	
	
	Select crTmpPrIt
	Scan
		If lnCodOper <= 0 and not Empty(lcConta) and lcConta <> crTmpPrIt.Contas
			Loop
		Endif		
		** SigMvCcr
		lcQuery = [Select Valors, ValPags, Moedas, Hists, Hist2s, Vencs, Titulos from SigMvCcr where Emps = ']+crTmpPrIt.EmpOs+[' and Nopers = ]+Str(crTmpPrIt.Nopers,9)		
		If( ThisForm.poDataMgr.SQLExecute(lcQuery,[crTmpMccr]) < 1 )
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conex„o (crTmpMccr)')
			Return(.F.)	
		EndIf
		** SigMvPar
		lcQuery = [Select Valos, VPags, Emps, Dopes, Numes, FPags, MoeFpgs, Vencs from SigMvPar where Emps = ']+crTmpPrIt.EmpOs+[' and Nopers = ]+Str(crTmpPrIt.Nopers,9)
		If( ThisForm.poDataMgr.SQLExecute(lcQuery,[crTmpPar]) < 1 )
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conex„o (crTmpPar)')
			Return(.F.)
		EndIf
		Select crTmpPar	
		** SigMvPar
		lcQuery = [Select Valos, FPags from SigCdPbx where Nopers = ]+Str(crTmpPrIt.Nopers,9) + [ And EmpDopNums = ']+crDados1.EmpDopNums+[']				
		If( ThisForm.poDataMgr.SQLExecute(lcQuery,[crTmpPrBx]) < 1 )

			Return(.F.)	
		EndIf
		Select crTmpPrBx
		
		lcCondPag = Iif( not Eof([crTmpPrBx]), crTmpPrBx.FPags, crTmpPar.FPags )
		lnVlrPago = crTmpPrBx.Valos
		lcTitulo  = crTmpMccr.Titulos
		lcHist2s  = crTmpMccr.Hist2s
		
		Do Case
			Case crTmpPrIt.Tipos = [C]				
				Select crTmpMccr
				lnValos = crTmpMccr.Valors - crTmpMccr.ValPags + crTmpPrIt.Acertos
				lnAcerto= crTmpPrIt.Acertos
				lcMoeda = crTmpMccr.Moedas
				lcHists = crTmpMccr.Hists
				lcHist2s= crTmpMccr.Hist2s
				ldVencs = crTmpMccr.Vencs
			Case crTmpPrIt.Tipos = [P]
				If Eof([crTmpPar])					
					lnValos = crTmpMccr.Valors - crTmpMccr.ValPags + crTmpPrIt.Acertos
					lnAcerto= crTmpPrIt.Acertos
					lcMoeda = crTmpMccr.Moedas
					lcHists = crTmpMccr.Hists
					lcHist2s= crTmpMccr.Hist2s
					ldVencs = crTmpMccr.Vencs
				Else
					lnValos = crTmpPar.Valos - crTmpPar.VPags + crTmpPrIt.Acertos
					lnAcerto= crTmpPrIt.Acertos								
					lcMoeda = crTmpPar.MoeFpgs
					ldVencs = crTmpPar.Vencs
					lcHists = Alltrim( crTmpPar.FPags ) + " " + crTmpPar.Emps + ;
							  Alltrim( crTmpPar.Dopes ) + " " + Alltrim( fGerMascara( crTmpPar.Numes ) ) 
					If not Eof([crTmpMccr])
						lcHists = crTmpMccr.Hists
						lcHist2s= crTmpMccr.Hist2s
					EndIf					
				Endif				
			Case crTmpPrIt.Tipos = [M]
				lnValos = crTmpPrIt.Valos
				lnAcerto= crTmpPrIt.Acertos				
				lcMoeda = crTmpPrIt.Moedas
				ldVencs = crTmpPrIt.Vencs
				lcHists = crTmpPrIt.Hists				
		EndCase				
		lnDebito = Iif(crTmpPrIt.Opers = [D], 0, lnValos)
		lnCredito= Iif(crTmpPrIt.Opers = [C], 0, lnValos)
	
		lcHists = lcHists + Iif( not Empty(lcHist2s), Chr(13) + lcHist2s, [] )		
	
		lcCondPag = []
		lnVlrPago = []
		lnTotPago  = 0
		Select crTmpPrBx
		Scan
			lcCondPag = lcCondPag + Iif( not Empty(lcCondPag), Chr(13), []) + crTmpPrBx.FPags

ENDPROC

pdDataIni = fDtoSQL(ldDataIni)
pdDataFin = fDtoSQL(ldDataFin,'23:59:59')

=Seek(_Empr,[crSigCdEmp],[cEmps])

   lcDesJob1=''

   lcPeriodo = lcDatas + [ de : ] + Dtoc(ldDataIni) + [ ý ] + Dtoc(ldDataFin)

Insert Into crCabecalho (Empresa, Titulo, SubTitulo, Periodo, SaldoIni, SaldosCab1, SaldosCab2) ;

lcQuery = [Select a.Emps, a.DopCs, a.Nopers, a.EmpDopNums, a.Contas, a.Titulos, a.Vencs, a.Opers, a.Hists, a.EspecieNFs, ] + ;

			[From SigMvCcr a Inner Join SigCdCli b on b.iClis = a.Contas ] + ;
		   [Where a.] + lcCpoDt + [ BetWeen ?pdDataIni and ?pdDataFin And ] + ;
				  Iif(Empty(lcGrupo), [], [a.Grupos = '] + lcGrupo + [' And ]) + ;
				  Iif(Empty(lcConta), [], [a.Contas = '] + lcConta + [' And ]) + ;
				 [(a.Valors-a.ValPags) > 0 And ] + ;
				 [a.Pagos = 1 And ] + ;
				 [a.ContaPgs <> 2 And ] + ;
				 [a.Tipos <> 'C' ] + ;
				 Iif(ThisForm.opt_AnaSi.Value = 3, [And a.Tipos <> 'X'], []) + ;
		   		 Iif(lnAut = 1, [], [And ] + Iif(lnAut = 2, [Not ], []) + [a.UsuConfs = Space(10) ]) + ;
				 Iif(Empty(lcJob),[],[And a.Jobs = '] + lcJob + [' ]) +;
		   [Order By a.Emps, a.DopCs, a.Contas, a.] + lcCpoDt

If( ThisForm.poDataMgr.SQLExecute(lcQuery,[crDados1]) < 1 )
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conex„o (crDados1)')
	Return 0	
EndIf

Select * ;
  From crSigOpOpe ;
 Where RealPrevs = 2 And Imps = 1 ;
  Into Cursor crPrevs

If Not Eof([crPrevs])
	* Se Selecionou MovimentaÁ„o de Previs„o

				[From SigMvCpv a Inner Join SigCdCli b on b.iClis = a.Contas ] + ;
			   [Where a.] + lcCpoDt + [ Between ?pdDataIni and ?pdDataFin And ] + ;
					  Iif(Empty(lcGrupo), [], [a.Grupos = '] + lcGrupo + [' And ]) + ;
					  Iif(Empty(lcConta), [], [a.Contas = '] + lcConta + [' And ]) + ;
					 [(a.Valors-a.ValPags) > 0 And ] + ;
					 [a.Pagos = 1 And ] + ;
					 [a.ContaPgs <> 2 And ] + ;
					 [a.Tipos <> 'C' ] + ;
					 Iif(ThisForm.opt_AnaSi.Value = 3, [And a.Tipos <> 'X'], []) + ;
		   		     Iif(lnAut = 1, [], [And ] + Iif(lnAut = 2, [Not ], []) + [a.UsuConfs = Space(10) ]) + ;
					 Iif(Empty(lcJob),[],[And a.Jobs = '] + lcJob + [' ]) +;
			   [Order By a.Emps, a.DopCs, a.Contas, a.] + lcCpoDt

	If( ThisForm.poDataMgr.SQLExecute(lcQuery,[LocalPrev]) < 1 )
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conex„o (LocalPrev)')
		Return 0	
	EndIf
	*** Tiago - 02/12/2011 - Vianna - Coloquei a clausula 'and a.Opers = Left(c.Opers,1)' para pegar apenas as penas no SIGMVCCR que correspondem ao OPERS da operaÁ„o de tÌtulos
	Select a.* ;
	  From crDados1 a ;
	 Inner join crSigCdEmp b on b.cEmps = a.Emps ;
	 Inner join crSigOpOpe c on c.Dopes = a.DopCs ;
	 Inner Join crSigCdClc d on d.Codigos = a.Codigos ;
	 Where (b.Imps = 1) and (c.Imps = 1) and (d.Imps = 1) and a.Opers = Left(c.Opers,1) ;
	 Union All ;
	Select e.* ;
	  From LocalPrev e ;
	 Inner join crSigCdEmp f on f.cEmps = e.Emps ;
	 Inner join crSigOpOpe g on g.Dopes = e.DopCs ;
	 Inner Join crSigCdClc h on h.Codigos = e.Codigos ;
	 Where (f.Imps = 1) and (g.Imps = 1) and (h.Imps = 1) and e.Opers = Left(g.Opers,1) Into cursor crDados2
Else
	*** Tiago - 02/12/2011 - Vianna - Coloquei a clausula 'and a.Opers = Left(c.Opers,1)' para pegar apenas as penas no SIGMVCCR que correspondem ao OPERS da operaÁ„o de tÌtulos
	Select a.* ;
	  From crDados1 a ;
	 Inner join crSigCdEmp b on b.cEmps = a.Emps ;
	 Inner join crSigOpOpe c on c.Dopes = a.DopCs ;
	 Inner Join crSigCdClc d on d.Codigos = a.Codigos ;
	 Where (b.Imps = 1) and (c.Imps = 1) and (d.Imps = 1) and a.Opers = Left(c.Opers,1) Into cursor crDados2

			  ( crSigOpOpe.Opers == [DB] and crDados.Opers = [D] ) 
				nVlJuros = ThisForm.nCalcJuros(Ttod(crDados.Vencs),crDados.Valors)
			Endif

			*Convers„o para moeda selecionada		
			lnValor = crDados.Valors
			lnCtReferencia = 0
			lnCtReferencia =fBuscarCambio({}, Date(), Space(03), crDados.Moedas)
			lnValor = lnValor * lnCtReferencia / ThisForm.lnVlCotacao
			
			Append Blank in crRel
			Replace Emps 		with crDados.Emps,;
					DopCs		with crDados.DopCs,;
					Titulos 	with crDados.Titulos,;
					Conta		with crDados.Contas,;
					EmpDopNums	with Iif(Empty(Substr(crDados.EmpDopNums, 4, 1)), [], crDados.EmpDopNums),;
					NomeConta	with crDados.RClis,;
					Vencs		with Ttod(crDados.Vencs),;
					Debito		with Iif(crDados.Opers = [D], lnValor, 0 ),;
					Credito		with Iif(crDados.Opers = [C], lnValor, 0 ),;
					VlJuros		with nVlJuros,;
					VlTotal		with lnValor + nVlJuros,;
					OpDopCs		with crSigOpOpe.Opers,;
					OpLanc		with crDados.Opers in crRel

		Select crRel
		Index on Dtos(Vencs) Tag OrdRel
		
	Case (ThisForm.opt_AnaSi.Value = 2) && SintÈtico
		Create Cursor crRel(Conta c(10), NomeConta c(50), nVlrEmp1 n(14,2), nVlrEmp2 n(14,2), nVlrEmp3 n(14,2), nVlrEmp4 n(14,2), nVlrEmp5 n(14,2),;
							nVlrEmp6 n(14,2), nVlrEmp7 n(14,2), nVlrEmp8 n(14,2), nVlrEmp9 n(14,2), VlTotal n(14,2))
		Index on Conta Tag Conta	

		Create Cursor crEmps(Coluna n(2), Emps c(3))
		Index on Emps Tag Emps
		
		lnColuna = 1
		
		Select crDados
		Go Top in crDados
		Scan
			= Seek(crDados.DopCs,[crSigOpOpe],[Dopes])
			
			If not Seek(crDados.Contas,[crRel],[Conta])
				Append Blank in crRel
				Replace Conta 	  with crDados.Contas,;
						NomeConta with crDados.RClis in crRel
			EndIf

			*Convers„o para moeda selecionada		
			lnValor = crDados.Valors
			lnCtReferencia = 0
			lnCtReferencia =fBuscarCambio({}, Date(), Space(03), crDados.Moedas)
			lnValor = lnValor * lnCtReferencia / ThisForm.lnVlCotacao
			
			lnVlJuros = 0
			If( crSigOpOpe.Opers == [CR] and crDados.Opers = [C] ) or ;
			  ( crSigOpOpe.Opers == [DB] and crDados.Opers = [D] ) 		
				lnVlJuros = ThisForm.nCalcJuros(Ttod(crDados.Vencs),lnValor)
			Endif			
					
			lnVlrAtu = lnValor + lnVlJuros
			
			** Achar a coluna
			If not Seek(crDados.Emps,[crEmps],[Emps])
				Insert into crEmps(Coluna,Emps) Values (lnColuna,crDados.Emps)
				
				lcCampo = [cEmp]+Alltrim(Str(lnColuna))
				Replace &lcCampo with crDados.Emps in crCabecalho
				
				lnColuna = lnColuna + 1
			Endif
			
			cCampo   = [nVlrEmp]+Alltrim(Str(crEmps.Coluna))
* Richard em 02/01/2013 - O comportamento do Analitico È lanÁar nas colunas de dÈbito e crÈdito conforme o OPERS do
* 						  SIGMVCCR apenas e aqui est· considerando o OPERS do SIGOPOPE tambÈm causando divergÍncias
*!*				If( crSigOpOpe.Opers == [CR] )
				nVlCampo = Iif( crDados.Opers = [D], lnVlrAtu, lnVlrAtu * -1 )
*!*				Else
*!*					nVlCampo = Iif( crDados.Opers = [D], lnVlrAtu, lnVlrAtu * -1 )
*!*				Endif		
			
			Replace &cCampo with &cCampo + nVlCampo,;
					VlTotal with VlTotal + nVlCampo in crRel
		EndScan
		
		Replace nQtdEmp with lnColuna-1 in crCabecalho
		
		Select crRel
		Index on NomeConta Tag OrdRel

	Otherwise && Fluxo de Caixa
	
		Select [] as GrpSaldos, Ttod(Vencs) as Vencs, Titulos, Hists, EspecieNFs, Moedas, Valors, Opers, ;
			   9999999999.99 as SaldoDia, 9999999999.99 as SaldoTot, ;
			   9999999999.99 as Debitos,  9999999999.99 as Creditos ; 
		  From crDados ;
		 Order By Vencs, Titulos, Hists, EspecieNFs, Moedas, Valors ;
		  Into Cursor crProcessar Readwrite
		
		Update crProcessar Set SaldoDia = 0, SaldoTot = 0, Debitos = 0, Creditos = 0

		ldVct = Ctod([01/01/1900])
		lnSdD = 0
		lnSdT = lnSaldo
		Select crProcessar
		Scan
			If (ldVct <> crProcessar.Vencs)
				lnSdD = 0
			EndIf
			ldVct = crProcessar.Vencs
			
			lnVal = Abs(crProcessar.Valors)
			lnCtR = 0
			lnCtR = fBuscarCambio({}, Date(), Space(03), crProcessar.Moedas)
			lnVal = lnVal * lnCtR / ThisForm.lnVlCotacao
			
			lnSdD = lnSdD + (lnVal * Iif(crProcessar.Opers = [D], 1, -1))
			lnSdT = lnSdT + (lnVal * Iif(crProcessar.Opers = [D], 1, -1))
			
			Replace Debitos	 With Iif(crProcessar.Opers = [D], lnVal, 0), ;
					Creditos With Iif(crProcessar.Opers = [C], lnVal, 0), ;
					SaldoDia With lnSdD, ;
					SaldoTot With lnSdT In crProcessar
		EndScan

		Select crProcessar
		Go Top In crProcessar
EndCase
ENDPROC
