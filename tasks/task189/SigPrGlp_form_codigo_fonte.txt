* PROCEDURE atualizapeso
------------------------------------------------------------
PROCEDURE atualizapeso
cCompo = Alias()

TotQtd  = 0

If crSigCdPam.AutComps <> 1
	Select &cCompo.
	Scan
		Select crSigCdCom
		Go Top In crSigCdCom
		Locate For CGrus = &cCompo..CGrus And Custos = 1
		If !Eof([crSigCdCom])
			lcQuery = [Select a.cUnis, a.cUnips, b.BPesos ] + ;
					  [From SigCdPro a, SigCdGrp b ] + ;
					  [Where a.CPros = '] + &cCompo..Mats + [' And ] + ;
					  [a.CGrus = b.CGrus]

			If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crSomaGru') < 1)
				=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSomaGru - 1)')
				Return 0
			EndIf
			Go Top In crSomaGru
			If (InList(crSomaGru.BPesos,1,3))
				lcSql = [Select Fators From SigCdUni Where Cunis = ']+Iif(CrSomaGru.bPesos=1,CrSomaGru.Cunis,CrSomagru.CUnips)+[']
				=ThisForm.Podatamgr.Sqlexecute(lcsql,'LocalUni')
				lnFator = Iif(LocalUni.Fators = 0, 1, LocalUni.Fators )
				Select &cCompo.
				TotQtd = TotQtd + ( Iif(CrSomaGru.bPesos=1,&cCompo..Qtds,&cCompo..Pesos) * lnFator )
			EndIf
		EndIf		
	EndScan
	Select &cCompo.
EndIf

Return TotQtd 


ENDPROC

* PROCEDURE gravahis
------------------------------------------------------------
PROCEDURE gravahis
Local _nRegistro, _Reservado, _Inicio

Select Dopes,Estoqs,Origems,Destinos,EstOrigs,EstDests From CrSigCdOpe ;
Where 0=1 Into Cursor LocalOpe ReadWrite

Select Distinct Dopes From CrSigMvHst Into Cursor SelOperacao
Select SelOperacao
Scan
	lcSql = [Select Dopes,Estoqs,Origems,Destinos,EstOrigs,EstDests From SigCdOpe ]+;
			[Where Dopes = ']+SelOperacao.Dopes+[' ]
	ThisForm.Podatamgr.Sqlexecute(lcSql,'xTmpOpe')
	Select LocalOpe
	Append From Dbf('xTmpOpe')
EndScan
Select LocalOpe
Index on Dopes Tag dopes

Select SelOperacao
Scan
	lcSql = [Select Dopps as Dopes,1 as Estoqs,Origems,Destinos,EstOrigs,EstDests From SigCdOpd ]+;
			[Where Dopps = ']+SelOperacao.Dopes+[' ]
	ThisForm.Podatamgr.Sqlexecute(lcSql,'xTmpOpe')
	Select LocalOpe
	Append From Dbf('xTmpOpe')
EndScan

Wait Window 'Criando Chaves Primárias no Arquivo de Histórico ' Nowait
Select crSigMvHst
Go Top
_nRegistro = Reccount()

_Reservado = 0
DO While _Reservado = 0 And _nRegistro > 0
	_Reservado = fGerUniqueKey(Dtos(Datas),,,_nRegistro + 1)
EndDo
_Inicio    = _Reservado - _nRegistro

_RerSeq   = 0
DO While _RerSeq = 0 And _nRegistro > 0
	_RerSeq    = fGerUniqueKey('HISTBAR',,,_nRegistro + 1)
EndDo
_IniSeq    = _RerSeq - _nRegistro

Scan

	_Inicio = _Inicio + 1
	_IniSeq = _IniSeq + 1

	m.NewOpe = CrSigMvHst.Opers
	*** Tiago - 23/10/2015 - Definido com o Richard que vamos gravar no cidchaves a mesma letra da movimentaçăo para padronizar, assim a ordem do histórico fica correta		
*!*		=Seek(CrSigMvHst.Dopes,'LocalOpe')
*!*		If 	LocalOpe.estoqs = 1 and LocalOpe.Origems = 1 and InList(LocalOpe.EstOrigs,1,2) And ;
*!*				LocalOpe.destinos = 1 and InList(LocalOpe.EstDests,1,2)
*!*			m.NewOpe = Iif(CrSigMvHst.Opers = [E],[H],[K])
*!*		Endif

	Replace CidChaves With Dtos(Datas)+m.NewOpe+Transform(_Inicio,"@L 999999")+Thisform.Sigkey ,;
			Seqs      WITH _IniSeq

	=fRecalculaP( crSigMvHst.Emps, crSigMvHst.Grupos, crSigMvHst.Estos, crSigMvHst.CPros, crSigMvHst.Datas, crSigMvHst.CodCors, crSigMvHst.CodTams, ThisForm.poDataMgr)
	=fRecalculaC( crSigMvHst.Emps, crSigMvHst.CPros, crSigMvHst.Datas, ThisForm.poDataMgr )
	
EndScan
Wait Clear
ENDPROC

			Loop
		EndIf
		
		lnQtUtil = Utilizar

		=Seek(TmpDisp.CPros + TmpDisp.CodCors + TmpDisp.CodTams, 'TmpSaldo')

		Select TmpFinal
		Scatter To Memvar

		Select Temporario
		Append From Array Memvar
		Replace Saldo    With lnQtUtil, ;
				codTams  With TmpDisp.CodTams, ;
				Estoque  With lnQtUtil, ;
				Produzir With 0 In Temporario
		
		Replace Saldo    With TmpFinal.Saldo - lnQtUtil In TmpFinal
		Replace Produzir With TmpFinal.Produzir - lnQtUtil In TmpFinal
		Replace Disps 	 With TmpSaldo.Disps - lnQtUtil In TmpSaldo

		Select TmpSaldG
		Set Order To Cpros
		=Seek(TmpSaldo.Cpros + TmpSaldo.CodCors + TmpSaldo.CodTams)
		Replace Disps With Saldo While Cpros = TmpSaldo.Cpros And CodCors = TmpSaldo.CodCors And codTams = TmpSaldo.CodTams
		=Seek(TmpSaldo.Cpros + TmpSaldo.CodCors + TmpSaldo.CodTams)
		Scan While Cpros = TmpSaldo.Cpros And CodCors = TmpSaldo.CodCors And CodTams = TmpSaldo.CodTams And xBaixa > 0
			If TmpSaldg.Disps >= xBaixa
				Replace TmpSaldg.Disps with TmpSaldg.Disps - xBaixa
				xBaixa = 0
			Else
				xBaixa = xBaixa - TmpSaldg.Disps
				Replace TmpSaldg.Disps with 0
			Endif
		EndScan

		lcEdn = TmpFinal.Emps + TmpFinal.Dopes + Str(TmpFinal.Numes,6)
		
		lcQuery = [Select * ] + ;
				    [From SigMvIts ] + ;
				   [Where EmpDopNums = '] + lcEdn + [' And ] + ;
				   		 [CPros = '] + TmpFinal.Cpros + [']

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'TempEsti2') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TempEsti2 - 3)')
			Return 0
		EndIf

		Select TempEsti2
		Scan
			If (Citens <> TmpFinal.Citens)
				Loop
			EndIf

			If (CodCors = TmpFinal.CodCors) And (CodTams = Space(4))
				Scatter To Memvar

				pQtd = lnQtUtil
				pAqt = Qtds
				pIds = TempEsti2.cIdChaves
				
				lcQuery = [Update SigMvIts ] + ;
							 [Set CodTams = '] + TmpDisp.CodTams + [', ] + ;
							 	 [Qtds = ?pQtd, ] + ;
							 	 [AQtds = ?pAQt ] + ;
						   [Where cIdChaves = ?pIds]
				
				If (ThisForm.poDataMgr.SqlExecute(lcQuery, '') < 1)
					=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Update - 9)')
					Return 0
				EndIf

				Select crSigMvIts
				Append From Array Memvar
				Replace Qtds  With Qtds - lnQtUtil
				Replace Aqtds With Qtds
				Replace cIdChaves With fUniqueIds()
				If (Qtds = 0)
					Delete
				EndIf
				Exit
			EndIf
		EndScan
	EndScan

	Select TmpFinal
	If (TmpFinal.Saldo = 0)
		Delete
	EndIf
	Select TmpFinal
	Append From Dbf('Temporario')
	Go _RegFinal
	

	=Seek(TmpFinal.CPros + TmpFinal.CodCors + TmpFinal.CodTams, 'TmpSaldo')

If Not Seek(TmpFinal.Cpros,'TmpSaldU','Cpros')
	Insert into TmpSaldU (Cpros ) Values (TmpFinal.Cpros)
EndIf

If This.Value <> ThisForm.OldValue And TmpSaldU.KeySelm
	If MessageBox('Produto com Seleçăo Manual de estoque. '+Chr(13)+;
		'O sistema irá acionar o modo automático. Deseja Continuar?',4+16+256,'') = 7
		This.Value = ThisForm.OldValue
		Return 0
	Endif
EndIf

Do Case
	Case (This.Value = ThisForm.OldValue)
		Return

	Case (This.Value < 0)
		=Messagebox('A Quantidade a Produzir Năo Pode Ser Um Valor Negativo!!!', 32, '')
		This.Value = ThisForm.OldValue

	Case (This.Value > TmpFinal.Saldo)
		=Messagebox('A Quantidade a Produzir Năo Pode Ser Maior Que a Quantidade Da Operaçăo!!!', 32, '')
		This.Value = TmpFinal.Saldo - TmpFinal.Estoque

	Case Not Seek(TmpFinal.CPros + TmpFinal.CodCors + TmpFinal.CodTams, 'TmpSaldo') And ;
		(TmpFinal.Produzir <> TmpFinal.Saldo)
		=Messagebox('Năo Há Saldo Disponível Deste Produto No Estoque Para Reservar!!!', 32, '')
		This.Value = TmpFinal.Saldo

	Otherwise
		If (TmpSaldo.Disps + TmpFinal.Estoque >= TmpFinal.Saldo - This.Value)
			Replace TmpSaldo.Disps   With TmpSaldo.Disps + TmpFinal.Estoque - (TmpFinal.Saldo - TmpFinal.Produzir) in TmpSaldo

			Replace TmpFinal.Estoque With TmpFinal.Saldo - This.Value in TmpFinal

			Replace KeySelm With .f. in TmpSaldU

			Select TmpSaldo
			xBaixa = Saldo - Disps
			Select TmpSaldG
			Set Order To Cpros
			=Seek(TmpSaldo.Cpros + TmpSaldo.CodCors + TmpSaldo.CodTams)
			Replace Disps With Saldo While Cpros = TmpSaldo.Cpros And CodCors = TmpSaldo.CodCors And codTams = TmpSaldo.CodTams
			=Seek(TmpSaldo.Cpros + TmpSaldo.CodCors + TmpSaldo.CodTams)
			Scan While Cpros = TmpSaldo.Cpros And CodCors = TmpSaldo.CodCors And CodTams = TmpSaldo.CodTams And xBaixa > 0
				If TmpSaldg.Disps >= xBaixa
					Replace TmpSaldg.Disps with TmpSaldg.Disps - xBaixa
					xBaixa = 0
				Else
					xBaixa = xBaixa - TmpSaldg.Disps
					Replace TmpSaldg.Disps with 0
				Endif
			EndScan
		Else
			=Messagebox('Năo Há Saldo Disponível Deste Produto No Estoque Para Reservar!!!', 32, '')
			This.Value = ThisForm.OldValue
		EndIf
EndCase

ENDPROC

		llErro = .t.
	EndIf

		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (SigOpClC)')
		llErro = .t.
	EndIf

	If Not llErro And (ThisForm.poDataMgr.Commit() < 1)
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Commit)')
		llErro = .t.
	EndIf

	If llErro
		ThisForm.poDataMgr.RollBack()
	EndIf

	ThisForm.Enabled = .f.
	Do Form SigReGli With _Nump, ThisForm

	ThisForm.Cancelar.Click()
	Keyboard '{ESC}'
	Keyboard '{ESC}'
	Keyboard '{ESC}'
	Return 0
		
EndIf

ENDPROC

PROCEDURE Click
Select Cpros, Space(50) as DPros, CodCors, CodTams, Dopes, Numes, ;
	   Saldo, Estoque, Produzir, Iif(IsEmpty(TmpFinal.Obsps), [ ], [*]) as ObsPs ;
  From TmpFinal ;
 Order By Cpros, CodCors, CodTams, Dopes, Numes ;
  Into Cursor crImpressao ReadWrite
    
Go Top In crImpressao
If Eof([crImpressao])
	=MessageBox([Năo Existem Dados Para Impressăo do Relatório!!!], 0+48, [Atençăo!!!])
Else
	Select Distinct Cpros From crImpressao Into Cursor LocalProds
	Select LocalProds
	Scan
		lcQuery = [Select CPros, DPros ] + ;
				    [From SigCdPro ] + ;
				   [Where Cpros = '] + LocalProds.CPros + [']

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, [LocalBus]) < 1)
			=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (LocalBus)])
			Return .f.
		EndIf
		Select LocalBus
		Go Top In LocalBus
		If Not Eof([LocalBus])
			Update crImpressao Set DPros = LocalBus.DPros Where CPros = LocalBus.CPros
		EndIf
	EndScan

	Select crImpressao
	Go Top In crImpressao

	Report Form SigReGlp Preview NoConsole
EndIf

Return .t.
ENDPROC
