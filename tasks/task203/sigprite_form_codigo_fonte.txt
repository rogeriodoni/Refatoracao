* PROCEDURE selecionadados
------------------------------------------------------------
PROCEDURE selecionadados
Local lcQuery
Zap In crGrade
Zap In crRodape

lcQuery = [Select a.CItens, a.CPros, a.DPros, a.Obs, a.CUnis, c.Volumes, a.Opers, a.Qtds as QtdOs, ] + ;
				 [a.QtBaixas as QtdBs, a.Qtds - a.QtBaixas as QtdSs, a.Units, a.Moedas as CMoes, a.Totas ] + ;
		    [From SigMvItn a, SigCdOpe b, SigCdPro c ] + ;
		   [Where a.EmpDopNums = '] + ThisForm.EmpDopNum + [' And ] + ;
		   		 [a.Dopes = b.Dopes And ] + ;
		   		 [a.CPros = c.CPros ] + ;
		   [Order By a.CItens, a.CPros]

If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crDados') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crDados)')
	Return 0
EndIf
Go Top In crDados

Select crDados
Scan
	Scatter Memvar Memo
	m.CItens = Recno('crDados')
	m.Obs    = Nvl(m.Obs, [])
	Insert Into crGrade From Memvar
EndScan

Select CMoes, Sum(Volumes) as Volumes, Sum(QtdOs) as QtdOs, Sum(QtdBs) as QtdBs, ;
	   Sum(QtdSs) as QtdSs, Sum(Totas) as Totas ;
  From crDados ;
 Group By CMoes ;
  Into Cursor crTotal

Select crTotal
Scan
	Scatter Memvar Memo
	m.CPros = []
	Insert Into crRodape From Memvar
EndScan

Go Top In crGrade
Go Top In crRodape
* Fim **************************************************************************************
ENDPROC
