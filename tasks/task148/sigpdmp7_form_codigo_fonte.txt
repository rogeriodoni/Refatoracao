* PROCEDURE criacursor
------------------------------------------------------------
PROCEDURE criacursor
Select * From GravaNens Where 0=1 Into Cursor TmpNens ReadWrite

Select *, 000000.000 as ps2FaseAnt, Space(3) as cUnips, .f. as OsRets, .f. as Retrabs, Space(10) as MatPrincs,;
	000000.000 as PsFaseAnt, Space(60) as Descs, ;
	0 as Rateio, .f. as Visivel, 000000.000 as QtFaseAnt, 000000 as Amarracao, 00 as TrfPesa, 0 as ServAuto,;
	SPACE(20) as dopcs, 0 as retrab  ;
From GravaNensI Where 1=0 Into Cursor xNensi ReadWrite

Select xNensi
Index On Recno() For Visivel Tag Editar
Index On Cats Tag Cats
Index On Nops Tag Nops
Index On Nenvs Tag Nenvs
Index On Str(Nops, 10) + Str( Recno(), 10 ) For Rateio # 1 Tag Gravar
Index On Str(Nops, 10) + Str( Rateio, 1 ) Tag Rateio
Index On Str(Nops, 10) + CMats Tag NopCMat
Index On Str(Nops, 10) + CMats + Cats Tag NopMatCat
Index On Str(NEnvs, 10) + CMats + Cats For Visivel Tag NEnvMCat
Index On Str(NEnvs, 10) + CMats + TpOps For Visivel Tag NEnvTpops
Index On Str(NEnvs, 10) + CMats For Visivel Tag NEnvCMat
Index On Str(Nenvs,10) + Str(Nops,10) + cMats Tag nAceite
Set Order to Editar

Select * From xNensi Where 0=1 Into Cursor xNensiE ReadWrite
Index On Recno() For Visivel Tag Editar
Index On Cats Tag Cats
Index On Nops Tag Nops
Index On Nenvs Tag Nenvs
Index On Str(Nops, 10) + Str( Recno(), 10 ) For Rateio # 1 Tag Gravar
Index On Str(Nops, 10) + Str( Rateio, 1 ) Tag Rateio
Index On Str(Nops, 10) + CMats Tag NopCMat
Index On Str(Nops, 10) + CMats + Cats Tag NopMatCat
Index On Str(NEnvs, 10) + CMats + Cats For Visivel Tag NEnvMCat
Index On Str(NEnvs, 10) + CMats + TpOps For Visivel Tag NEnvTpops
Index On Str(NEnvs, 10) + CMats For Visivel Tag NEnvCMat
Set Order to Editar

Select * From xNensI Where 0=1 Into Cursor xNensiS ReadWrite
Index On Nenvs Tag Nenvs

Create Cursor xNensiR (nEnvs n(10), EmpDopNums c(31), Datas t, Destinos c(21), Retrabs c(40), Rclis c(30))
Index On Nenvs Tag Nenvs

Create Cursor xNensiN (Nops n(10), Cods c(10), Fixos n(1), Descs c(60), Qtds n(12,3), Conta2s c(10))
Index On Nops Tag Nops

Create Cursor xNensiNC (Nenvs n(10), Cods c(10), Descs c(60), Qtds n(12,3), aQtds n(12,3), Usuars c(10))
Index On Nenvs Tag Nenvs

Create Cursor xNensiP (Nenvs n(10), cMats c(14), Qtds n(12,3), Pesos n(12,3))
Index On Str(Nenvs,10) + cMats Tag Nenvs

Create Cursor CsContas (Nenvs n(10), Grupos c(10), Contas c(10) )
Index On Str(Nenvs)+Grupos+Contas Tag NenvGruCon
Index On Nenvs Tag Nenvs

Create Cursor xNensiJ (Nenvs n(10), Nops n(10), Cods c(2), Descs c(30), cRetrabs c(6))
Index on Nops Tag Nops

Select *, 0 as Autos, 00000 as BaseMins, 000000.000 as PesoAntigo, Space(30) as DRetrabs, .f. as Justs ;
From GravaMfas where 0=1 Into Cursor xMfas ReadWrite

Select xMFas
Index on Dopes + Str( Numes, 6 ) + Str( Nops ) Tag DopNumOp
Index on Nenvs Tag Nenvs
Index on Nops Tag Nops

Select *, 0 as Autos, 00000 as BaseMins, 000000.000 as PesoAntigo ;
From GravaMfas where 0=1 Into Cursor xMfas2 ReadWrite
Select xMFas2
Index on Nops Tag Nops

Create Cursor xOpi ( Nops n(10), Oper c(31), Qtds n(10,3), CPros c(14), Indiv l, Amarra n(10), ;
					Nenvs n(10), Emps c(3), Dopes c(20), Numes n(6), Conta c(10), Empds c(3), ;
					CodCors c(4), CodTams c(4), cBars n(14), Obrigs l )
Index on Nops tag Nops
Set Order to

Create Cursor xPesa ( cBars n(14), Oper c(31), CPros c(14), Qtds n(10,3), Peso n(10,3), Amarra n(10), Nenvs n(10), Emps c(3), Dopes c(20),;
					  Numes n(6), Nops n(10), Conta c(10), Obs c(10), Cors c(4), Tams c(2), MatPs c(10), Peso2 n(10,3), pesoms n(10,3), ;
					  Usulibs c(10), Pesofixs n(10,3), Especial L, Divs L)
Index On Nops Tag Nops
Index on Amarra Tag Amarra
Set Order to 

Create Cursor xMaterial ( Nenvs n(10), Cpros c(14), MatPs c(10))
Index on Nenvs Tag Nenvs

Create Cursor TmpOperacao (Codigos n(12))
Index On Codigos Tag Codigos 
Set Order To

ENDPROC

Lparameters pQuery, pForm

With ThisForm

		With .poDataMgr

			.pfSqlTabela(1).pColuna('ContaOs',  '', '', 'Ct.Origem',  80, .f.)
			.pfSqlTabela(1).pColuna('GrupoDs',  '', '', 'Gr.Destino', 80, .f.)
			.pfSqlTabela(1).pColuna('ContaDs',  '', '', 'Ct.Destino', 80, .f.)

			.AddCursor('SigCdNec',  'cIdChaves', 'GravaNens')
			.AddCursor('SigCdNei', 'cIdChaves', 'GravaNensI')
			.AddCursor('SigPdMvf',  'cIdChaves', 'GravaMFas')
		EndWith

		ThisForm.poDataMgr.CursorQuery('SigCdPam', 'crSigCdPam', .f., .f.)
		ThisForm.poDataMgr.CursorQuery('SigCdPac', 'crSigCdPac')

		ThisForm.poDataMgr.CursorQuery('SigCdOpd', 'crSigCdOpd', .f., .f.)
		Select crSigCdOpd
		Index On Dopps Tag Dopps

		=DoDefault()

		ThisForm.poDataMgr.CursorQuery('SigCdSvc', 'crSigCdSvc', .f., .f.)
		Select crSigCdSvc
		Index On Ccats + Grupos + Contas Tag CaGruCo
		Index On Grupos + Contas + Ccats Tag GruCoCa

		lcQuery = [Select Codigos, Descrs, TrfPesas, CompAgrus ] + ;
				    [From SigCdGcr ] + ;
				   [Where GerBals = 1]

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'TmpGccr') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpGccr)')
			Return 0
		EndIf
		Select TmpGccr
		Index On Codigos Tag BalCodigo
		Index On Descrs  Tag BalDescrs

		=ThisForm.Criacursor()

		_Ende = ThisForm.Pagina.Lista.Grupo_Op

		If ThisForm.plAcinserir
			ThisForm.plAcinserir   = .f.
			_Ende.Width 		   = _Ende.Width - 65
			_Ende.Consultar.Left   = _Ende.Consultar.Left - 65
			_Ende.Alterar.Left     = _Ende.Alterar.Left - 65
			_Ende.Excluir.Left     = _Ende.Excluir.Left - 65
			_Ende.Procurar.Left    = _Ende.Procurar.Left - 65
			_Ende.Listar.Left 	   = _Ende.Listar.Left - 65
		EndIf

		If ThisForm.plAcalterar
			ThisForm.plAcalterar   = .f. 
			_Ende.Alterar.Enabled  = .f.
			_Ende.Alterar.Visible  = .f.
			_Ende.Width 		   = _Ende.Width - 65
			_Ende.Excluir.Left 	   = _Ende.Excluir.Left - 65
			_Ende.Procurar.Left    = _Ende.Procurar.Left - 65
			_Ende.Listar.Left 	   = _Ende.Listar.Left - 65
		EndIf

		If ThisForm.plAcexcluir
			ThisForm.plAcexcluir   = .f. 
			_Ende.Excluir.Enabled  = .f.
			_Ende.Excluir.Visible  = .f.
			_Ende.Width 		   = _Ende.Width - 64
			_Ende.Procurar.Left    = _Ende.Procurar.Left - 64
			_Ende.Listar.Left 	   = _Ende.Listar.Left - 64
		EndIf

		Select csTemporario
		Goto Bottom

		With ThisForm.Pagina.Dados.GradeOperaçăo
			.ColumnCount  = 1
			.DeleteMark   = .f.
			.ReadOnly 	  = .t.
			.RecordSource = 'TmpOperacao'
			.Column1.FontName = 'Courier New'
			.Column1.ControlSource = 'TmpOperacao.Codigos'
			.Column1.ReadOnly = .t.
			.Refresh
		EndWith	

		ThisForm.Pagina.Lista.Grade.Column1.SetFocus
	Else
		=Messagebox('Impossível Efetuar Conexăo Com o Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith
* Fim **************************************************************************************

			_NEnv     = crSigCdNei.NEnvs

			_RecNensi = Recno()

				EndIf

				_Servico = crSigCdNei.Servicos
				If Not _Servico And Not Empty(crSigCdNei.CMats) And Empty(crSigCdNei.Cats)
					Insert Into xNensi (Emps, Dopps, Numps, Nops, NEnvs, CMats, CDescs, QtFaseAnt, Qtds, ;

						_DMat   = crSigCdPro.DPros
						_CMat   = crSigCdPro.CUnis
						_GMat   = crSigCdPro.CGrus
						_CatMat = crSigCdNei.Cats
					Else
						_BMat = Space(14)
						_DMat = ''
						_CMat = ''
						_GMat = ''
						_CatMat = '   '
					EndIf

					_TMat = crSigCdNei.TpOps

					If (crSigCdOpd.CompAgrus = 1)
						_Nenv = 0
					EndIf

					If (lnRateio <> 5)
						If Not Seek(Str(_Nenv, 10) + _BMat + _TMat, 'xNensi', 'NEnvTpops')
							Insert Into xNensi (Nops, Nenvs, CMats, CDescs, CUnis, CGrus, Visivel, Rateio, Cats, TpOps, Servicos) ;
										Values (_NEnv, _NEnv, _BMat, _DMat, _CMat, _GMat, .t., 1, _CatMat, _TMat, _Servico)
						Else
							If (xNensi.Rateio <> 1)
								Replace Rateio With 1 In xNensi
							EndIf
						EndIf

						Select xNensi
						Replace QtFaseAnt With xNensi.QtFaseAnt + crSigCdNei.Qtds, ;
								PsFaseAnt With xNensi.PsFaseAnt + crSigCdNei.Pesos, ;
								Qtds 	  With xNensi.Qtds + crSigCdNei.Qtds, ;
								Pesos 	  With xNensi.Pesos + crSigCdNei.Pesos In xNensi
					EndIf
				Else
					If Not Seek(crSigCdNei.Cats, 'xNensi', 'Cats') Or crSigCdNei.FlagIncs
						Insert Into xNensi (Emps, Dopps, Numps, CMats, CDescs, CUnis, CGrus, Visivel, Rateio, ;
											FlagIncs, Cats, Servicos, Fators, Moedas, GrupoFins) ;
									Values (crSigCdNei.Emps, crSigCdNei.Dopps, crSigCdNei.Numps, crSigCdNei.CMats, ;
											crSigCdPro.Dpros, crSigCdPro.CUnis, crSigCdPro.CGrus, .t., 0, .t., crSigCdNei.Cats, ;
											.t., crSigCdNei.Fators, crSigCdNei.Moedas, crSigCdNei.GrupoFins)
					EndIf
					Replace Qtds With xNensi.Qtds + crSigCdNei.Qtds In xNensi
				EndIf
			EndScan

			Select crSigCdNei
			Skip - 1
		EndScan
	EndIf

	If Inlist(ThisForm.pcEscolha, 'ALTERAR', 'CONSULTAR', 'EXCLUIR')
		_ChaveNens2 = TmpNens.Emps + TmpNens.Dopps + Str(TmpNens.Numps, 6)

		Scan

	EndWith
