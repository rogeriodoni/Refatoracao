PROCEDURE atualizapeso
cCompo = Alias()

TotQtd  = 0

If crSigCdPam.AutComps <> 1
	Select &cCompo.
	Scan
		Select crSigCdCom
		Go Top In crSigCdCom
		Locate For CGrus = &cCompo..CGrus And Custos = 1
		If !Eof([crSigCdCom])
			lcQuery = [Select a.cUnis, a.cUnips, b.BPesos ] + ;
					  [From SigCdPro a, SigCdGrp b ] + ;
					  [Where a.CPros = '] + &cCompo..Mats + [' And ] + ;
					  [a.CGrus = b.CGrus]

			If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crSomaGru') < 1)
				=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSomaGru - 1)')
				Return 0
			EndIf
			Go Top In crSomaGru
			If (InList(crSomaGru.BPesos,1,3))
				lcSql = [Select Fators From SigCdUni Where Cunis = ']+Iif(CrSomaGru.bPesos=1,CrSomaGru.Cunis,CrSomagru.CUnips)+[']
				=ThisForm.Podatamgr.Sqlexecute(lcsql,'LocalUni')
				lnFator = Iif(LocalUni.Fators = 0, 1, LocalUni.Fators )
				Select &cCompo.
				TotQtd = TotQtd + ( Iif(CrSomaGru.bPesos=1,&cCompo..Qtds,&cCompo..Pesos) * lnFator )
			EndIf
		EndIf		
	EndScan
	Select &cCompo.
EndIf

Return TotQtd
ENDPROC

PROCEDURE gravahis
Local _nRegistro, _Reservado, _Inicio

Select Dopes,Estoqs,Origems,Destinos,EstOrigs,EstDests From CrSigCdOpe ;
Where 0=1 Into Cursor LocalOpe ReadWrite

Select Distinct Dopes From CrSigMvHst Into Cursor SelOperacao

Scan
	lcSql = [Select Dopes,Estoqs,Origems,Destinos,EstOrigs,EstDests From SigCdOpe ]+;
			[Where Dopes = ']+SelOperacao.Dopes+[' ]
	ThisForm.Podatamgr.Sqlexecute(lcSql,'xTmpOpe')
	Select LocalOpe
	Append From Dbf('xTmpOpe')
EndScan
Select LocalOpe
Index on Dopes Tag dopes

Select SelOperacao
Scan
	lcSql = [Select Dopps as Dopes,1 as Estoqs,Origems,Destinos,EstOrigs,EstDests From SigCdOpd ]+;
			[Where Dopps = ']+SelOperacao.Dopes+[' ]
	ThisForm.Podatamgr.Sqlexecute(lcSql,'xTmpOpe')
	Select LocalOpe
	Append From Dbf('xTmpOpe')

Select crSigMvHst

_nRegistro = Reccount()

_Reservado = 0
DO While _Reservado = 0 And _nRegistro > 0
	_Reservado = fGerUniqueKey(Dtos(Datas),,,_nRegistro + 1)
EndDo
_Inicio    = _Reservado - _nRegistro

_RerSeq   = 0
DO While _RerSeq = 0 And _nRegistro > 0
	_RerSeq    = fGerUniqueKey('HISTBAR',,,_nRegistro + 1)
EndDo
_IniSeq    = _RerSeq - _nRegistro

Scan

	_Inicio = _Inicio + 1
	_IniSeq = _IniSeq + 1

	m.NewOpe = CrSigMvHst.Opers	
*** Tiago - 23/10/2015 - Definido com o Richard que vamos gravar no cidchaves a mesma letra da movimentaçăo para padronizar, assim a ordem do histórico fica correta

*!*		Endif

	Replace CidChaves With Dtos(Datas)+m.NewOpe+Transform(_Inicio,"@L 999999")+Thisform.Sigkey ,;
			Seqs      WITH _IniSeq

	=fRecalculaP( crSigMvHst.Emps, crSigMvHst.Grupos, crSigMvHst.Estos, crSigMvHst.CPros, crSigMvHst.Datas, crSigMvHst.CodCors, crSigMvHst.CodTams, ThisForm.poDataMgr)
	=fRecalculaC( crSigMvHst.Emps, crSigMvHst.CPros, crSigMvHst.Datas, ThisForm.poDataMgr )
