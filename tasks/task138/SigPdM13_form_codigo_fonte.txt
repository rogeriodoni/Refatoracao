* PROCEDURE Init
------------------------------------------------------------
PROCEDURE Init
LParameters pForm,pEnvelope

ThisForm.cntSombra.lblSombra.Caption = ThisForm.Caption
ThisForm.cntSombra.lblTitulo.Caption = ThisForm.Caption

Set DataSession To pForm.DataSessionId

With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)

	If (.poDataMgr.pnIdconn > 0)

		lcSql = [Select a.EmpDnPs, a.Cmats, a.Qtds, a.TpOps From SigCdNei a, SigCdOpd b, SigOpOpt c ]+;
				[Where a.Nenvs = ]+Str(pEnvelope,10)+[ And a.Dopps = b.Dopps And b.SepPedras = 1 ]+;
				[And b.ChkDests <> 4 And a.TpOps = c.Cods And c.chkDevs = 'S' ]+;
				[union all ]+;
				[Select a.EmpDnPs, a.Cmats, (a.Qtds*-1) as Qtds, a.TpOps From SigCdNei a, SigCdOpd b ]+;
				[Where a.Nenvs = ]+Str(pEnvelope,10)+[ And a.Dopps = b.Dopps ]+;
				[And b.ChkDests = 4 ]+;
				[Order by a.Cmats ]
		If (ThisForm.poDataMgr.SqlExecute(lcSql, 'LocalNoo') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalNoo)')
			Return 0
		EndIf
		Select LocalNoo
		Sum Qtds to _Total
		Go Top

		With ThisForm.Grade
			.RecordSource = 'LocalNoo'
			.Column1.ControlSource   = 'LocalNoo.EmpDnPs'  	
			.Column2.ControlSource   = 'LocalNoo.Cmats' 	
			.Column3.ControlSource   = 'LocalNoo.TpOPs'
			.Column4.ControlSource   = 'LocalNoo.Qtds'
			.Refresh
		EndWith
		ThisForm.Qtd_os.Value = _Total
		ThisForm.Qtd_os.Refresh
	Else
		=Messagebox('Impossível Efetuar Conexăo Com o Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith

ENDPROC
