PROCEDURE Init
Lparameters _Producao,lcNomeFrm1

ThisForm.cntSombra.lblSombra.Caption = ThisForm.Caption
ThisForm.cntSombra.lblTitulo.Caption = ThisForm.Caption

Thisform.Producao = _producao

Private pIds

pIds = Replicate(Chr(254), 20)

lcQrySend = [Select * From SigPrSnd Where cIdChaves = '] + pIds + [']

With ThisForm
	lcNomeFrm1 = Iif(Type('lcNomeFrm1')=='C' And Not Empty(lcNomeFrm1),lcNomeFrm1,.Name)
	.Name = lcNomeFrm1
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)

	If (.poDataMgr.pnIdconn > 0)
		With .poDataMgr
			.AddCursor('SigPrSnd', 'cIdChaves', 'crSigPrSnd', '', '', lcQrySend)
		EndWith

		Insert Into crSigPrSnd (Usuars, Datas, Tipos) Values (Usuar, Datetime(),Iif(ThisForm.Producao,1,0))
		
		lcSql = [Select SenhDEmps From SigCdPac ]
		ThisForm.Podatamgr.Sqlexecute(lcSql,'CrParac')

		=DoDefault()
	Else
		=Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith
* Fim **************************************************************************************

ENDPROC

PROCEDURE Valid
Local loLista

If Not Empty(This.Value)
	If ThisForm.Producao
		loLista = CreateObject('fwBuscaExt', ThisForm.poDataMgr.pnIdConn, 'SigCdOpd', ;
							   'crListaRemota', 'Dopps', This.Value, 'Seleçăo', .t.)
		If Not loLista.plAchouRegistro
			loLista.mAddColuna('Dopps', '', 'Operaçăo')
			loLista.Show()
		EndIf
		This.Value = crListaRemota.Dopps

	Else
	
		loLista = CreateObject('fwBuscaExt', ThisForm.poDataMgr.pnIdConn, 'SigCdOpe', ;
							   'crListaRemota', 'Dopes', This.Value, 'Seleçăo', .t.)
		If Not loLista.plAchouRegistro
			loLista.mAddColuna('Dopes', '', 'Operaçăo')
			loLista.Show()
		EndIf
		This.Value = crListaRemota.Dopes
	Endif
	Use In crListaRemota
Else
	This.Value = ''
EndIf
This.Refresh

ENDPROC

* PROCEDURE Click
------------------------------------------------------------
PROCEDURE Click
If Empty(crSigPrSnd.Solicits) 
	=Messagebox('Informe o Nome do Solicitante...', 32)
	ThisForm.Get_Solic.SetFocus
	Return 0
EndIf

If Empty(crSigPrSnd.Emps)
	=Messagebox('Informe a Empresa...', 32)
	ThisForm.Get_cd_empresa.SetFocus
	Return 0
EndIf

If Empty(crSigPrSnd.Operas)
	=Messagebox('Informe a Operaçăo...', 32)
	ThisForm.Get_Operaçăo.SetFocus
	Return 0
EndIf

If Empty(crSigPrSnd.Docs)
	=Messagebox('Informe o Número do Documento...', 32)
	ThisForm.Get_Doc.SetFocus
	Return 0
EndIf

If Not Thisform.Producao And Empty(crSigPrSnd.Valors) 
	=Messagebox('Informe o Valor...', 32)
	ThisForm.Get_Valor.SetFocus
	Return 0
EndIf

If Empty(crSigPrSnd.Motivos)
	=Messagebox('Informe o Motivo...', 32)
	ThisForm.Get_Motivo.SetFocus
	Return 0
EndIf
lnNum = 0
If CrParac.SenhDEmps
	lcSql = [Select nemps From SigCdEmp Where cEmps = ']+CrSigPrSnd.Emps+[']
	ThisForm.Podatamgr.Sqlexecute(lcSql,'CrEmp')
	lnNum = CrEmp.nEmps

	lcSql = [Select nfpags From SigOpFp Where Fpags = ']+CrSigPrSnd.Fpags+[']
	ThisForm.Podatamgr.Sqlexecute(lcSql,'CrfPagto')
	lnNum = lnNum + CrFPagto.NfPags + (CrSigPrSnd.Valors * 100)
Endif

lcSenha = fGerPswDiario(crSigPrSnd.Docs, crSigPrSnd.Datas, lnNum)
Replace Senhads   With lcSenha, ;
		Horas     With Time(), ;
		cIdChaves With fUniqueIds() In crSigPrSnd

ThisForm.poDataMgr.Update('crSigPrSnd')
ThisForm.poDataMgr.Commit()

=Messagebox('Favor Anotar a Senha : ' + Chr(13) + Chr(13) + lcSenha + Chr(13) + Chr(13), 32, 'Senha do Dia')

ThisForm.Release

ENDPROC
