	Scan For (Rateio = 3) And (Nenvs = _Nenv) And Not FlagIncs
		_Pesagem = _Pesagem - Pesos
	Endscan

	Scan For (Rateio = 2) And (Nenvs = _Nenv) And Not FlagIncs
		If (_RecnoRateio = 0)
			_RecnoRateio = Recno()
		Endif

		_PesoAnterior = _PesoAnterior + PsFaseAnt
	Endscan

	_DiferencaPeso = _Pesagem - _PesoAnterior
	Scan For (Rateio = 2) And (Nenvs = _Nenv) And Not FlagIncs
		Replace Pesos With Round(PsFaseAnt + (PsFaseAnt / Iif(_PesoAnterior = 0, 1, _PesoAnterior) * _DiferencaPeso), 2), ;

		_Pesagem = _Pesagem - Pesos
	Endscan

	If (_RecnoRateio <> 0) And (_Pesagem <> 0)
		Go _RecnoRateio
		Replace Pesos With Pesos + _Pesagem, ;
				Qtds  With Qtds + _Pesagem
	Endif

* PROCEDURE Click

PROCEDURE Click
Private vTipEtq

Local lnTipos, lnImpres, llImpPreco

*** mena - 10/09/2015 - pega o nome da impressora que estß no combobox *** 
lnImpres  = RECNO('crImpreV')
vNomeImp  = crImpreV.impres

With Thisform.Etiquetas
	llImpPreco = (.Opt_Preco.Value = 1)
	llImppeso = (.Opt_Peso.Value = 1)
	lnTipos    = .Opt_Tipo.Value
	vTipEtq    = Int(Val(.Opt_Tipo.Buttons(lnTipos).Tag))
*** mena 19/10/2015 - retirado para usar informacoes do cursor acima ***
*!*		lnImpres   = .Opt_Impressora.Value
*!*		vNomeImp   = .Opt_Impressora.Buttons(lnImpres).Tag
EndWith

Create Cursor dbImpressao (Cpros c(14), Dpros c(40), Qtds n(1), QtdeEtiq n(10,3), Pedido c(30), ;
						  Obs c(10), Nops n(10), Notas c(6), Aros c(50), Pesos n(12,3), Contas c(10), ;
						  EmpDopNums c(29), Reffs c(20), empos c(3), cpros1 c(14), codtams c(4))

If vTipEtq = 74 &&Etiqueta de OP
	lcSql = [Select Cpros,Nops,Qtds,ObsEtiqs as Obs,Emps,Numes,Dopes,Empos,DopeOs,NumeOs,CodTams,Pesos,CodCors,Peso2s From SigOpEtq ]+;
			[Where EmpDopNums = ']+TmpNens.Emps+TmpNens.Dopps+Str(TmpNens.Numps,6)+[' ]+;
			[Union all ]+;
			[Select Cpros,Nops,Qtds,ObsEtiqs as Obs,Emps,Numes,Dopes,Empos,DopeOs,NumeOs,CodTams,Pesos,CodCors,Peso2s From SigEtPos ]+;
			[Where EmpDopNums = ']+TmpNens.Emps+TmpNens.Dopps+Str(TmpNens.Numps,6)+[' ]+;
			[Order by Nops ]
	ThisForm.Podatamgr.Sqlexecute(lcSql,'LocalEti')

	Select LocalEti
	Scan
		ThisForm.poDataMgr.CursorQuery('SigCdPro',  'LocalPro',  'CPros',    LocalEti.CPros, 'Cpros,CGrus,SGrus,DPros,Colecoes')
		ThisForm.poDataMgr.CursorQuery('SigCdOpe',  'LocalOpe',  'Dopes',    LocalEti.DopeOs, 'Abrevs')
		ThisForm.poDataMgr.CursorQuery('SigCdPsg', 'LocalSGru', 'CGruCods', LocalPro.Cgrus + LocalPro.Sgrus, 'Descricaos')
		ThisForm.poDataMgr.CursorQuery('SigOpPic',  'LocalOpi',  'Nops',     LocalEti.Nops, 'Notas,CodTams,CodCors,Numes,dopes')

		lcAro = ''
		Select LocalOpi
		Locate For Dopes = LocalEti.Dopeos And Numes = LocalEti.NumeOs
		lcClassis = Padc(Iif(Not Empty(LocalPro.Colecoes) And LocalPro.Sgrus <> Padr('N',6),'FundiþÒo',Iif(Not Empty(LocalOpi.notas),'Os','Alianþa')),10)
		If lcClassis = Padc('Os',10)
			Select Distinct CodTams From LocalOpi Into Cursor CsSelecao Order by CodTams
			Select CsSelecao
			Scan
				lcAro = lcAro + CsSelecao.CodTams+'/'
			EndScan
		Endif

		Insert Into dbImpressao (CPros, DPros, Qtds, QtdeEtiq, Pedido, Obs, Nops, Notas, Aros, Pesos) ; 
						 Values (LocalPro.Cpros, LocalPro.DPros, 1, LocalEti.Qtds, ;
						 		 Alltrim(LocalOpe.Abrevs) + ' : ' + Alltrim(Str(LocalEti.Numeos, 6)), ;
						 		 LocalEti.Obs, LocalEti.Nops, LocalOpi.Notas, ;
						 		 Padr(lcAro,50), LocalEti.Pesos )
	EndScan

Else
	lcSql = [Select Cpros,Cbars,Qtds,ObsEtiqs as Obs,Emps,Numes,Dopes,Empos,DopeOs,NumeOs,Nops From SigOpEtq ]+;
			[Where EmpDopNums = ']+TmpNens.Emps+TmpNens.Dopps+Str(TmpNens.Numps,6)+[' Order by Cbars ]
	ThisForm.Podatamgr.Sqlexecute(lcSql,'LocalEti')

	Select LocalEti
	Scan
		ThisForm.poDataMgr.CursorQuery('SigCdPro',  'LocalPro',  'CPros', LocalEti.CPros, 'CGrus, SGrus, DPros')
		ThisForm.poDataMgr.CursorQuery('SigCdOpe',  'LocalOpe',  'Dopes', LocalEti.Dopes, 'Abrevs,globalizas')
		ThisForm.poDataMgr.CursorQuery('SigCdOpe',  'LocalOpeO',  'Dopes', LocalEti.DopeOs, 'Abrevs,globalizas')
		ThisForm.poDataMgr.CursorQuery('SigCdPsg', 'LocalSGru', 'CGruCods', LocalPro.Cgrus + LocalPro.Sgrus, 'Descricaos')

		Insert Into dbImpressao (CPros, DPros, Qtds, QtdeEtiq, Pedido, Obs) ; 
						 Values (Str(LocalEti.CBars, 14), LocalPro.DPros, 1, LocalEti.Qtds, ;
						 		 Alltrim(LocalOpe.Abrevs) + ' : ' + Alltrim(Str(LocalEti.Numeos, 6)), LocalEti.Obs )

		If InList(vTipEtq,14,39,109,113)
			ThisForm.poDataMgr.CursorQuery('SigOpPic', 'LocalOpi', 'Nops', LocalEti.Nops, 'Emps, Dopes, Numes, EmpDs')
			lcEmp = Iif(Empty(LocalOpi.EmpDs), LocalOpi.Emps, LocalOpi.EmpDs)

			lcEdn = LocalEti.Empos + LocalEti.Dopeos + Str(LocalEti.Numeos, 6)
			ThisForm.poDataMgr.CursorQuery('SigMvCab', 'LocalEest', 'EmpDopNums', lcEdn, 'ContaOs, ContaDs, Vends')
			lcconta = Iif(LocalOpeO.globalizas=1,LocalEest.ContaOs,LocalEest.ContaDs)
			If InList(vTipEtq,109,113)
				ThisForm.poDataMgr.CursorQuery('SigCdCli', 'LocalCli', 'Iclis', lcconta, 'Rclis')
				lcconta = LocalCli.Rclis
			EndIf
			If vTipEtq <> 14
				Replace Pedido With lcconta + Iif(InList(vTipEtq,39),'/'+Alltrim(Str(LocalEti.Numeos,6)),'') In dbImpressao
			Else
				Replace Contas With lcconta In dbImpressao			
			EndIf 
		EndIf
	EndScan

	If CrSigCdOpd.ChkUnis = 1 And CrSigCdOpd.Etiqs = 1
		lcSql = [Select Cpros,Cbars,Qtds,ObsEtiqs,Emps,Numes,Dopes From SigEtPos Where EmpDopNums = ']+TmpNens.Emps+TmpNens.Dopps+Str(TmpNens.Numps,6)+[']
		ThisForm.Podatamgr.Sqlexecute(lcSql,'LocalEti')

		Select LocalEti
		Scan
			_QtEtiq = LocalEti.Qtds
			Insert Into dbImpressao (CPros, Qtds) Values (LocalEti.CPros, _QtEtiq)
		EndScan

	Endif
EndIf

With Thisform.Etiquetas.Cnt_Impressora
	vnTpImpressora = .Opcao_Imp.Value
	vnAjVerts 	   = .Spn_AjVerts.Value
	vnAjHorzs 	   = .Spn_AjHorzs.Value
	vnAjDenss      = .spn_AjDenss.Value
	vnAjVelos      = .spn_AjVelos.Value
EndWith

Select dbImpressao 
Go Top

=SigOpEtq(ThisForm.poDataMgr, llImpPreco, .F., vTipEtq, vnTpImpressora, vnAjVerts, vnAjHorzs, vnAjDenss, , , , vNomeImp,.f.,.f., vnAjVelos,llImpPeso)

=MessageBox('ImpressÒo ConcluÝda!!!', 0+64, '')

ThisForm.ParentForm.Enabled = .t.
ThisForm.Release

ENDPROC
