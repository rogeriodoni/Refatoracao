* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
lcconta = Thisform.pgfprincipal.pgfiltro.get_cd_car_conta.Value
lcemp = Thisform.pgfprincipal.pgfiltro.get_Cd_Empresa.Value
ldti = Thisform.pgfprincipal.pgfiltro.get_Datai.Value
ldtf = Thisform.pgfprincipal.pgfiltro.get_Dataf.Value
llNPr = (Thisform.pgfprincipal.pgfiltro.optProcessados.Value = 1) && 1-Năo Processados; 2-Já Processados
lnPer = thisform.pgfPrincipal.pgfiltro.optPeriodo.Value

lcope = [(]

Select crSigCdOpe
Scan
	If crSigCdOpe.marca
		lcope = lcope + Iif(lcope==[(],['],[,']) + crSigCdOpe.dopes + [']
	Endif
Endscan
lcope = lcope + [) ]

Thisform.pgfprincipal.pgDados.grdope.RecordSource = []
Thisform.pgfprincipal.pgDados.grdope.Refresh()

ltru = .T.
lcquery = [select ?ltru as marca, e.titulos, a.dopes, a.numes, d.rclis, a.vencs, b.fpags, a.valos, a.datas, a.vpags, ] +;
	[d.iclis, d.endes, d.cidas, d.estas, d.nums, d.compls, d.bairs, d.ceps, d.cpfs, a.emps, a.empdopnums, a.nopers, d.Razaos, ] +;
	[d.endcobs, d.cepcobs, d.estcobs, d.baicobs, d.cidcobs, ] + ;
	[Case When d.EndCobs <> '' And Len(rtrim(d.EndCobs)) > 40 then 1 else ] + ;
	[Case When d.Endes <> '' And Len(rtrim(d.Endes) + ' ' + rtrim(d.Nums) + ' ' + rtrim(d.compls)) > 40 then 1 else 0 end end as EndErro ] +;
	[from SigMvPar a inner join SigOpFp b on a.fpags = b.fpags ] +;
	[left join SigMvCab c on a.empdopnums = c.empdopnums ] +;
	[left join SigCdCli d on c.contads = d.iclis ] +;
	[left join SigMvCcr e on a.empdopnums = e.empdopnums and a.nopers = e.nopers ] +;
	[where b.infos = 'B' and a.vpags = 0 and ] +;
	Iif(lnPer=1, [a.Vencs], [e.DtEmis]) + [ Between ?ldti and ?ldtf and ] +;
	[e.opers = 'C' and ] +;
	[c.emps = ?lcemp and a.dopes in ] + lcope + [ and a.EmpDopNums+e.titulos ] +;
	Iif(llNPr, [Not ], []) + ;
	[In (Select e.EmpDopNums+substring(e.dopeds,1,10) ] + ;
	[From SigPcOol e ] + ;
	[Where e.Tipos = 'SIGPRCNB') ] + ;
	[order by a.dopes, a.numes, a.parcs]

Thisform.podatamgr.sqlexecute(lcquery,[crFiltro])
Select crFiltro
Go top
If Reccount() = 0
	Messagebox('Nenhum dado foi encontrado',64,'Aviso')
	Return .F.
Endif

replace all Marca with .f. for EndErro = 1 && Desmarca todos os títulos que o endereço está com problemas
Go top 

Thisform.pgfprincipal.pgDados.grdope.RecordSource = [crFiltro]
*!*	Thisform.pgfprincipal.pgDados.grdope.SetAll('DynamicForeColor', 'Iif(!Empty(crFiltro.EndCobs) And Len(Alltrim(crFiltro.EndCobs)) > 40, RGB(255,0,0), ;
*!*											Iif(!Empty(crFiltro.Endes) And Len(Alltrim(crFiltro.Endes) + Alltrim(crFiltro.Nums)) > 40, RGB(255,0,0),RGB(0,0,0)))', 'Column')
Thisform.pgfprincipal.pgDados.grdope.SetAll('DynamicForeColor', 'Iif(crFiltro.EndErro = 1, RGB(255,0,0), RGB(0,0,0))', 'Column')

Thisform.pgfprincipal.pgDados.grdope.Refresh()

Thisform.pgfprincipal.pgfiltro.Enabled = .F.
Thisform.pgfprincipal.pgDados.Enabled = .T.
Thisform.pgfprincipal.pgDados.commandgroup1.btnBoleto.Enabled = !llNPr 
Thisform.pgfprincipal.ActivePage = 2

ENDPROC

* PROCEDURE geracnab
------------------------------------------------------------
PROCEDURE geracnab
Lparameters pTipo

lcconta = Thisform.pgfprincipal.pgfiltro.get_cd_car_conta.Value
lcemp = Thisform.pgfprincipal.pgfiltro.get_Cd_Empresa.Value
ldti = Thisform.pgfprincipal.pgfiltro.get_Datai.Value
ldtf = Thisform.pgfprincipal.pgfiltro.get_Dataf.Value

* Apanhando o Nome da Empresa
Thisform.poDataMgr.CursorQuery('SigCdEmp', 'crEmpresa', 'Cemps', lcemp, 'RazSocs, Cgcs, Tel1s, Ceps, Endes, Numeros, Bairs, Cidas, Estas, Compls ')
If Eof('crEmpresa') Or Empty(crEmpresa.RazSocs) Or Empty(crEmpresa.Cgcs)
	=Messagebox('Năo Foi Encontrada a Razăo Social e/ou o CNPJ da Empresa [' + lcemp + ']' + Chr(13) + ;
		'Complete os Dados no Cadastro de Empresas e Tente Nomamente!!!', 0+48, 'Atençăo!!!')
	Thisform.cntEscolha.cmbBancos.SetFocus
	Return 0
Endif

Go Top In crFiltro
Thisform.pgfprincipal.pgdados.grdope.Refresh
Go Bottom In crFiltro
Thisform.pgfprincipal.pgdados.grdope.Refresh

If pTipo = [A]
	Select crFiltro
	Count For marca To lncont
	If lncont = 0
		Messagebox('Nenhum registro foi selecionado',64,'Aviso')
		Return .F.
	Endif

	lctit = Thisform.pgfprincipal.pgfiltro.get_titban.Value
*!*		latua = .F.
	If !Empty(lctit)
*!*			latua = (Messagebox([Confirma alteraçăo do títulos para "] + lctit + [" ?],4+32,'Aviso') = 6)
		If Messagebox([Confirma geraçăo do arquivo de remessa?],4+32,'Aviso') = 7
			Return .f.
		Endif
	Else
		If Messagebox([O campo "Título Banco" năo foi preenchido. Continuar?],4+32,'Aviso') = 7
			Return .F.
		Endif
	Endif

* Apanhando os Dados do Convęnio
	Thisform.poDataMgr.CursorQuery('SigCdCli', 'crBancos', 'IClis', lcconta, 'Grupos')

	lcQuery = [Select * ] + ;
				[From SigCdCeb ] + ;
			   [Where GruContas = '] + crBancos.Grupos + lcconta + [' And ] + ;
					 [Not NAgencias = Space(6) And ] + ;
					 [Not Convenios = Space(9) ] + ;
			   [Order By NAgencias, Convenios]

	If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crConvenio') < 1)
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crConvenio)')
		Return .F.
	Endif
	Go Top In crConvenio
	If Eof('crConvenio')
		=Messagebox('Năo Foi Encontrado o Número da Agęncia e/ou o Código do Convęnio Para' + Chr(13) + ;
			'o Banco [' + lcconta + ']!!! Complete os Dados no Cadastro de Contas!!!', ;
			0+48, 'Atençăo!!!')
		Thisform.pgfprincipal.pgfiltro.get_cd_car_conta.SetFocus()
		Return .F.
	Endif
	ThisForm.pgfPrincipal.pgdados.spndias.Value = Iif(Empty(crConvenio.DiasProts), 5, crConvenio.DiasProts)
	
*** Tiago - 12/07/2011
	Do Case
		Case Alltrim(crConvenio.nbancos) = [001] && Banco do Brasil Convenios
			Thisform.cnabBrasil()
			
		Case Alltrim(crConvenio.nbancos) = [341] && Itau Convenios
			lcarq = Getfile('txt','Arquivo','OK',0,'Arquivo CNAB')
			If Empty(lcarq)
				Return .F.
			Endif
			Thisform.CnabItau()
			
		Case Alltrim(crConvenio.nbancos) = [237] && Bradesco Convenios
			Thisform.CnabBradesco()
			
		Case Inlist(Alltrim(crConvenio.nbancos),[033],[353]) && Santader Convenios
			&&thisform.CnabSantander()	
			thisform.CnabSantander240()	
				
	Endcase
Else
	* Apanhando os Dados do Convęnio
	ThisForm.poDataMgr.CursorQuery('SigCdCli', 'crBancos', 'IClis', lcconta, 'Grupos')

	lcQuery = [Select * ] + ;
			    [From SigCdCeb ] + ;
			   [Where GruContas = '] + crBancos.Grupos + lcconta + [' And ] + ;
			   		 [Not NAgencias = Space(6) And ] + ;
			   		 [Not Convenios = Space(9) ] + ;
			   [Order By NAgencias, Convenios]

	If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crConvenio') < 1)
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crConvenio)')
		Return .f.
	EndIf

	Create Cursor crcabecalho(titulo c(100), periodo c(100), empresa c(100))
	lcprot = Padl(Iif(Thisform.pgfprincipal.pgdados.spndias.Value=0, 5, Thisform.pgfprincipal.pgdados.spndias.Value), 2, [0])
	lcdata = [Período: ] + Dtoc(ldti) + [ até ] + Dtoc(ldtf) + [ - Protesto em ] + lcprot + [ dias]

	Insert Into crcabecalho(titulo,periodo,empresa) Values ('Dados para geraçăo do CNAB',lcdata,lcemp + [ - ] + crEmpresa.RazSocs)

	Select *, Space(11) as SeqNums From crFiltro Where marca Into Cursor crFiltro2

	Select crFiltro2
	Go Top

	If pTipo = [V]
		Report Form sigrecnb Preview
	Endif
Endif

ENDPROC

Local lcTit, lcBcn, lcCnv, lcAge, lcBco, lcRaz, lcCgc, lcTpC, lcRbc, lcDat, lcEnv, lnMor, ;
	  lcPri, lcPrt, lcCdC, lcDig, lcArq, lcStr, lnSeq, lcSeq, lcVct, lcVal, lcMor, ;
	  lcCpf, lcNom, lcEnd, lcCep, lcNtt, lcChr, llOks, llAtu, lcBol, lcNossoNum, lcDV

	Endif

		lcNossoNum = lcNossoNum + lcDV
