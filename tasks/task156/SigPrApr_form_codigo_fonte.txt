PROCEDURE calcideal
Local lnMoeCF,lnMoedas,lnFator,lnIdeal,lcTmpMoe,lnQtdeQs,MoeIdeal,lnFeitio,lnCustoF

MoeIdeal = CrSigCdPro.Moedas
lnFator  = CrSigCdPro.Margems
lnFeitio = CrSigCdPro.PFtios
lnCustoF = CrSigCdPro.CustoFs

ThisForm.poDataMgr.CursorQuery('SigCdMoe','CrMoe','CMoes',CrSigCdPro.MoeCusFs)
Go Top In crMoe
If !Empty(CrMoe.MoeQs)
	lcTmpMoe = CrMoe.MoeQs
	lnQtdeQs = Iif(CrMoe.QtdeQs = 0,1,CrMoe.QtdeQs)
Else
	lcTmpMoe = CrSigCdPro.MoeCusFs
	lnQtdeQs = 1
EndIf
lnMoeCF = fBuscarCotacao(lcTmpMoe,DateTime(),ThisForm.poDataMgr) * lnQtdeQs

ThisForm.poDataMgr.CursorQuery('SigCdMoe','CrMoe','CMoes',MoeIdeal)
Go Top In CrMoe
If !Empty(CrMoe.MoeQs)
	lcTmpMoe = CrMoe.MoeQs
	lnQtdeQs = Iif(CrMoe.QtdeQs = 0,1,CrMoe.QtdeQs)
Else
	lcTmpMoe = MoeIdeal
	lnQtdeQs = 1
EndIf
lnMoedas = fBuscarCotacao(lcTmpMoe,DateTime(),ThisForm.poDataMgr) * lnQtdeQs

If (CrSigCdPro.MoeCusFs <> MoeIdeal)
	lnIdeal = ((lnCustoF + lnFeitio) * lnMoeCF / lnMoedas * lnFator)
Else
	lnIdeal = ((lnCustoF + lnFeitio ) * lnFator)
EndIf

Replace pvIdeals With lnIdeal in CrSigCdPro

Return(.t.)
ENDPROC

	If !Eof()
		Scatter MemVar Memo
		Insert Into TmpPro From MemVar
		Select TmpPro
		m.CPros  = TmpPro.CPros
		m.DPros  = TmpPro.DPros
		m.ValAnt = TmpPro.PVens
		Do Case
		Case lnTipo = 1
			m.ValAtu = TmpPro.PVens + ((TmpPro.PVens * lnVariacao) / 100)
		Case lnTipo = 2
			m.ValAtu = ThisForm.CalcPreco(lnMarkUp2)
		Case lnTipo = 3	
			lnCotId  = fBuscarCotacao(TmpPro.moedas, DateTime(), ThisForm.poDataMgr)
			lnCotVd  = fBuscarCotacao(TmpPro.moevs, DateTime(), ThisForm.poDataMgr)
			lnPven   = TmpPro.Pvideals*lnCotId/lnCotVd
			lnVlVen  = (lnPven / Iif(TmpPro.Encargos<>0,TmpPro.Encargos,1))
			m.ValAtu = lnVlVen 
		EndCase
		Select CrProdutos
		Replace lMarca With 1        ,;
				CPros  With m.CPros  ,;
				DPros  With m.DPros  ,;
				ValAnt With m.ValAnt ,;
				ValAtu With m.ValAtu In CrProdutos

		Go Top
		This.Parent.Parent.Refresh
		ThisForm.Sair.Atualiza.Enabled = .T.

		m.CPros  = Space(14)
		m.DPros  = Space(40)
		m.ValAnt = 0
		m.ValAtu = 0
		If Not ThisForm.LibValAtu 
			Select CrProdutos
			Set Order To
			Go Bottom
			If !Empty(CrProdutos.CPros)
				Insert Into CrProdutos From MemVar
			EndIf
			Select CrProdutos
			This.Parent.Parent.Refresh
			Keyboard '{DNARROW}'
		EndIf
	Else
		MessageBox('Produto năo encontrado !!! Reinicie o processo.',48,'')
	EndIf
EndIf
ENDPROC
