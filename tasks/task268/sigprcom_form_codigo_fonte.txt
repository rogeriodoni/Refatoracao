PROCEDURE mnovalinha
Local lnRegs, lnRegMaxs

lnRegMaxs = 12

Insert Into CrSigCdMax (cpros,cidchaves) Values (ThisForm.Pagina.Dados.Get_Produto.Value,fUniqueIds())
Select CrSigCdMax
Count To lnRegs
Go Top

If lnRegs > lnRegMaxs
	lnI = 10
Else
	lnI = lnRegs
EndIf

With ThisForm.Pagina.Dados.GradeI
	.Refresh
	Do While lnRegs > 0
		.ActivateCell(lnI,2)
		If Empty(CrSigCdMax.emps)
			.ActivateCell(lnI,1)
			Exit
		Else
			lnRegs = lnRegs - 1
			.DoScroll(1)
		EndIf
	EndDo
EndWith
ENDPROC

PROCEDURE mrefreshget
Select CrSigCdMax
With ThisForm.Pagina.Dados
	.Get_Produto.Value  = CsCabec.cpros
	.GetDPro.Value      = CsCabec.dpros
	.GetCgru.Value      = CsCabec.cgrus
	.GetDGru.Value      = CsCabec.dgrus
	.GetIFor.Value      = CsCabec.ifors
	.GetDFor.Value      = CsCabec.rclis
	.GetRefs.Value      = CsCabec.reffs
	.Opc_Situacao.Value = Iif(CsCabec.Situas = 2,2,1)
	.Get_Produto.Refresh
	.GetDPro.Refresh
	.GetCgru.Refresh
	.GetDGru.Refresh
	.GetIFor.Refresh
	.GetDFor.Refresh
	.GetRefs.Refresh
	.Opc_Situacao.Refresh
EndWith

ThisForm.lTemCor = .F.
ThisForm.lTemTam = .F.
If (!Empty(CrSigCdMax.cpros) )
	lStrQuery = [Select b.cores,b.tams,b.tipoestos From SigCdPro a, SigCdGrp b ]+;
				[Where a.cpros = ']+CrSigCdMax.cpros+[' And b.cgrus = a.cgrus ]
	If ThisForm.poDataMgr.SqlExecute(lStrQuery,'TmpProGru') < 1
		Return(.F.)
	EndIf
	Select TmpProGru
	Go Top
	If !Eof()
		ThisForm.lTemCor    = InList(TmpProGru.tipoestos,2,4) Or TmpProGru.cores = 1
		ThisForm.lTemTam    = InList(TmpProGru.tipoestos,3,4) Or TmpProGru.tams  = 1
		ThisForm.nTipoEstos = TmpProGru.tipoestos
	EndIf
EndIf
Return(.T.)
ENDPROC

* PROCEDURE Init
------------------------------------------------------------
PROCEDURE Init
LParameters _Acesso, pcForm, pcEsc, pcDPros, pcCGrus, pcIFors, pcReffs

ThisForm.pProduto = _Acesso
If Type([_Acesso]) = [C]
	ThisForm.LockScreen = .t.
	ThisForm.WindowType = 1
Endif
 
ThisForm.pcExterno = []
ThisForm.pcCPros   = []

If Type([pcEsc]) = [C] And Type([_Acesso]) = [C]
	ThisForm.pcExterno = pcEsc
	ThisForm.pcCPros   = Padr(_Acesso, 14)
	ThisForm.pcDPros   = pcDPros
	ThisForm.pcCGrus   = pcCGrus 
	ThisForm.pcIFors   = pcIFors 
	ThisForm.pcReffs   = pcReffs 

	ThisForm.LockScreen = .f.
	ThisForm.WindowType = 1
EndIf

Private lcCPros,lcEmpO,lcEmpD

ThisForm.Busca = 1

lcQProds  = [Select a.cpros,b.dpros,b.ifors,b.reffs,b.cgrus,b.sgrus,b.situas,c.rclis,g.dgrus ]+;
			[From SigCdMax a ]+;
			[Inner Join SigCdPro b On b.cpros = a.cpros ]+;
			[Left Join SigCdCli c On b.ifors = c.iclis ]+;
			[Left Join SigCdGrp g On b.cgrus = g.cgrus ]+;
			Iif(Empty(ThisForm.pcExterno), [], [Where a.Cpros = '] + ThisForm.pcCPros + [' ]) + ;
			[Group By a.cpros,b.dpros,b.ifors,b.reffs,b.cgrus,b.sgrus,b.situas,c.rclis,g.dgrus ]

LcQprCom  = [Select a.*,b.dpros,b.ifors,b.reffs,b.cgrus,b.situas,c.rclis,g.dgrus ]+;
			[From SigCdMax a, SigCdPro b ]+;
			[Left Join SigCdCli c On b.ifors = c.iclis ]+;
			[Left Join SigCdGrp g On b.cgrus = g.cgrus ]+;
			[Where a.cpros = ?lcCPros And b.cpros = a.cpros ]+;
			[Order By a.emps,a.codtams,a.codcores ]

lcQCopia  = [Select 1 As marcas,0 As existes,a.cpros,a.emps,a.Qmaxs,a.codtams,a.codcores,a.cidchaves ]+;
			[From SigCdMax a ]+;
			[Where a.emps = ?lcEmpO And (a.cpros + a.codtams + a.codcores) Not In ]+;
 				[(Select b.cpros + b.codtams + b.codcores As cprotamcores From SigCdMax b ]+;
 				[Where b.Emps = ?lcEmpD) ]+;
			[Union All ]+;
			[Select 0 As marcas,1 As existes,a.cpros,a.emps,a.Qmaxs,a.codtams,a.codcores,a.cidchaves ]+;
			[From SigCdMax a ]+;
			[Where a.emps = ?lcEmpO And a.cpros + a.codtams + a.codcores In ]+;
 				[(Select b.cpros + b.codtams + b.codcores As cprotamcores From SigCdMax b ]+;
				[Where b.emps = ?lcEmpD) ]+;
			[Union All ]+;
			[Select 0 As marcas,2 As existes,a.cpros,a.emps,a.Qmaxs,a.codtams,a.codcores,a.cidchaves ]+;
			[From SigCdMax a ]+;
			[Where a.emps = ?lcEmpD And a.cpros + a.codtams + a.codcores Not In ]+;
 				[(Select b.cpros + b.codtams + b.codcores As cprotamcores From SigCdMax b ]+;
				[Where b.emps = ?lcEmpO)]

lcCPros = Space(14)
lcEmpO  = Space(03)
lcEmpD  = Space(03)

With ThisForm
	.poDataMgr = CreateObject('fSqlConector','SigCdMax')
	With .poDataMgr
		.AddCursor('SigCdMax', 'cpros', 'CrProdutos', '', ThisForm.Pagina.Lista.Grade, lcQProds)
		.pfSqlTabela(1).pColuna('cpros', '', '', 'Produto'   ,  108, .T.)
		.pfSqlTabela(1).pColuna('dpros', '', '', 'Descriçăo' , 285, .T.)
		.pfSqlTabela(1).pColuna('ifors', '', '', 'Fornecedor',  75, .T.)
		.pfSqlTabela(1).pColuna('reffs', '', '', 'Referęncia', 150, .T.)
		.pfSqlTabela(1).pColuna('sgrus', '', '', 'Sub Grp'   ,  45, .F.)
		.AddCursor('SigCdMax', 'cidchaves', 'CrSigCdMax', '', '', LcQprCom)
		.AddCursor('SigCdMax', 'cidchaves', 'CsCopia'   , '', '', lcQCopia)
	EndWith
	=DoDefault()
	.lTemCor  = .F.
	.lTemTam  = .F.
	.nTeste   = 0
	.AcCopiar = fChecaAcesso('SIGPRCOM','COPIAR')
	With .Pagina.Lista
		.BtnCopiar.Enabled = ThisForm.AcCopiar
	EndWith
EndWith

If ThisForm.poDataMgr.SqlExecute([Select gesind From SigCdPam ],'CrSigCdPam') < 1
	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
	Return(.F.)
EndIf

ThisForm.poDataMgr.Requery('CrProdutos')

ThisForm.Pagina.Lista.CntCopia.Visible = .F.

ThisForm.mAtivaPagina1

If Type('_Acesso') <> 'C'
	ThisForm.LockScreen = .T.
	=fCarregaGrade(ThisForm.Pagina.Lista.Grade)
	ThisForm.LockScreen = .F.
Else
	If Empty(ThisForm.pcExterno)
		ThisForm.Pagina.Lista.Grupo_op.Click(1)
	Else
		Do Case
			Case (ThisForm.pcExterno = [INSERIR])
				ThisForm.Pagina.Lista.Grupo_op.Click(1)

			Case (ThisForm.pcExterno = [ALTERAR])
				ThisForm.Pagina.Lista.Grupo_op.Click(3)

			Otherwise 
				ThisForm.Pagina.Lista.Grupo_op.Click(2)
		EndCase
	EndIf
	ThisForm.LockScreen = .F.
EndIf
ENDPROC

PROCEDURE msv_procurar
Local lcGLinhas, lcGDescrs

lcGCdPro = Alltrim(ThisForm.Pagina.Dados.Get_Produto.Value)
lcGDsPro = Alltrim(ThisForm.Pagina.Dados.GetDPro.Value)
lcGIfors = Alltrim(ThisForm.Pagina.Dados.GetIfor.Value)
lcGReffs = Alltrim(ThisForm.Pagina.Dados.GetRefs.Value)

Select CrProdutos
Set Near On

Do Case

	Case !Empty(lcGCdPro)
		Set Order To CPros
		Seek lcGCdPro
		ThisForm.Busca = 1

	Case !Empty(lcGDsPro)
		Set Order To DPros
		Seek lcGDsPro
		ThisForm.Busca = 2

	Case !Empty(lcGIfors)
		Set Order To IFors
		Seek lcGIfors
		ThisForm.Busca = 3

	Case !Empty(lcGReffs)
		Set Order To Reffs
		Seek lcGReffs
		ThisForm.Busca = 4

EndCase

Set Near Off
ENDPROC

		Return 0

	Go Top

	With This.Parent.GrdItens
		.RecordSource          = 'CsCopia'
		.Column1.ControlSource = 'CsCopia.marcas'
		.Column2.ControlSource = 'CsCopia.cpros'

		.Column4.ControlSource = 'CsCopia.codtams'

		.Refresh
	EndWith
		
	With ThisForm.Pagina.Lista.CntCopia
		.CmdProcessa.Caption = '\<Reprocessar'
		.GetEmpOs.Enabled    = .F.
		.GetEmpDs.Enabled    = .F.
		.Marcar.Visible      = (lnTotal # 0)
		.Desmarcar.Visible   = (lnTotal # 0)
		.CmdCopiar.Enabled   = (lnTotal # 0)
		.GetProduto.SetFocus

	Zap In CsCopia

If !Empty(This.Value) And ThisForm.pcEscolha = 'INSERIR'
	m.loLista = CreateObject('fwbuscaext',ThisForm.poDataMgr.pnIdConn,'SigCdPro','CrProds','cpros',This.Value,'Produtos',1000)
	If !(m.loLista.plAchouRegistro)
		m.loLista.mAddColuna('cpros','','Código')
		m.loLista.mAddColuna('dpros','','Descriçăo')
		m.loLista.Show()
	EndIf

	If LastKey() <> 27 And Used('CrProds')
		This.Value                = Evaluate('CrProds.cpros')
		This.Parent.GetDPro.Value = Evaluate('CrProds.dpros')
		If Used('TmpPro')
			Use In TmpPro
		EndIf
		lStrQuery = [Select a.cpros,a.dpros,a.cgrus,a.ifors,a.reffs,a.situas,c.rclis,g.dgrus ]+;
					[From SigCdPro a ]+;
					[Left Join SigCdGrp g On g.cgrus = a.cgrus ]+;
					[Left Join SigCdCli c On c.iclis = a.ifors ]+;
					[Where a.cpros = ']+CrProds.cpros+[' ]
		If ThisForm.poDataMgr.SqlExecute(lStrQuery,'TmpPro') < 1
			MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
			Return(.F.)
		EndIf
		
		Select TmpPro
		Go Top
		If TmpPro.situas = 2 And CrSigCdPam.gesind = 1
			MessageBox('Produto Inativo !!!',48,'')
			This.Value                = ''
			This.Parent.GetDPro.Value = ''
			This.Refresh
			This.Parent.GetDPro.Refresh
			Return(.F.)
		Else
			ThisForm.AcertaGrade(TmpPro.cpros)
			Select CrSigCdMax
			Go Top
			If !Eof()
				MessageBox('Produto já cadastrado !!!',48,'')
				Zap In CrSigCdMax
				This.Parent.GradeI.Refresh
				This.Value                 = ''
				This.Parent.GetDPro.Value = ''
				This.Refresh
				This.Parent.GetDPro.Refresh
				Return(.F.)
			EndIf
		EndIf
		
		This.Refresh
		This.Parent.GetDPro.Refresh

		Insert Into CrSigCdMax (cpros,cidchaves) Values (TmpPro.cpros,fUniqueIds())
		Select CrSigCdMax
		Go Top
				
		Insert Into CsCabec (cpros,dpros,ifors,rclis,cgrus,dgrus,reffs,situas) ;
					Values(TmpPro.cpros,TmpPro.dpros,TmpPro.ifors,TmpPro.rclis,TmpPro.cgrus,TmpPro.dgrus,;
							TmpPro.reffs,Iif(TmpPro.situas = 2,2,1))

	Else
		This.Value                = ''
		This.Parent.GetDPro.Value = ''
	EndIf
EndIf

This.Parent.GetDPro.Refresh

	If !ThisForm.MRefreshGet()

If !Empty(This.Value) And ThisForm.pcEscolha = 'INSERIR'
	m.loLista = CreateObject('fwbuscaext',ThisForm.poDataMgr.pnIdConn,'SigCdPro','CrProds','dpros',This.Value,'Produtos',1000)
	If !(m.loLista.plAchouRegistro)
		m.loLista.mAddColuna('dpros','','Descriçăo')
		m.loLista.mAddColuna('cpros','','Código')
		m.loLista.Show()
	EndIf

	If LastKey() <> 27 And Used('CrProds')
		This.Value                    = Evaluate('CrProds.dpros')
		This.Parent.Get_Produto.Value = Evaluate('CrProds.cpros')
		If Used('TmpPro')
			Use In TmpPro
		EndIf
		lStrQuery = [Select a.cpros,a.dpros,a.cgrus,a.ifors,a.reffs,a.situas,c.rclis,g.dgrus ]+;
					[From SigCdPro a ]+;
					[Left Join SigCdGrp g On g.cgrus = a.cgrus ]+;
					[Left Join SigCdCli c On c.iclis = a.ifors ]+;
					[Where a.cpros = ']+CrProds.cpros+[' ]
		If ThisForm.poDataMgr.SqlExecute(lStrQuery,'TmpPro') < 1
			MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
			Return(.F.)
		EndIf
		
		Select TmpPro
		Go Top
		If TmpPro.situas = 2 And CrSigCdPam.gesind = 1
			MessageBox('Produto Inativo !!!',48,'')
			This.Value                    = ''
			This.Parent.Get_Produto.Value = ''
			This.Refresh
			This.Parent.Get_Produto.Refresh
			Return(.F.)
		Else
			ThisForm.AcertaGrade(TmpPro.cpros)
			Select CrSigCdMax
			Go Top
			If !Eof()
				MessageBox('Produto já cadastrado !!!',48,'')
				Zap In CrSigCdMax
				This.Parent.GradeI.Refresh
				This.Value                    = ''
				This.Parent.Get_Produto.Value = ''
				This.Refresh
				This.Parent.Get_Produto.Refresh
				Return(.F.)
			EndIf
		EndIf
		
		This.Refresh
		This.Parent.Get_Produto.Refresh

		Insert Into CrSigCdMax (cpros,cidchaves) Values (TmpPro.cpros,fUniqueIds())
		Select CrSigCdMax
		Go Top
				
		Insert Into CsCabec (cpros,dpros,ifors,rclis,cgrus,dgrus,reffs,situas) ;
					Values(TmpPro.cpros,TmpPro.dpros,TmpPro.ifors,TmpPro.rclis,TmpPro.cgrus,TmpPro.dgrus,;
							TmpPro.reffs,Iif(TmpPro.situas = 2,2,1))

	Else
		This.Value                    = ''
		This.Parent.Get_Produto.Value = ''
	EndIf
EndIf
This.Refresh
This.Parent.Get_Produto.Refresh

If ThisForm.pcEscolha # 'PROCURAR'
	If !ThisForm.MRefreshGet()
		MessageBox('Favor reinicializar o processo',16,'Falha na Conexăo')
		Return(.F.)
	EndIf
EndIf

Return(.T.)
ENDPROC
