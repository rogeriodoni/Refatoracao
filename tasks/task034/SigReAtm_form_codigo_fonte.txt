------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Private pDtI, pDtF
Local lcGrupo, lcConta, lcDConta, ldDatai, ldDataf, lcGruMat, lnTotal, lcCab, lcSub, lcEmp

lcGrupo   = ThisForm.Get_Grupo.Value
lcConta   = ThisForm.Get_Conta.Value
lcDConta  = ThisForm.Get_DConta.Value
lcGruMat  = ThisForm.Get_CGru.Value
lcGgrp    = ThisForm.Get_Ggrp.Value
lcSgru    = thisform.getCSGru.Value
ldDatai   = ThisForm.Get_Datai.Value
ldDataf   = ThisForm.Get_Dataf.Value
llDestino = ThisForm.ChkDestino.Value
lnEnvelope= ThisForm.Get_Envelope.Value
lnOp      = ThisForm.Get_Nop.Value
lnOpI     = ( ThisForm.Get_Nop.Value * 10000 ) + 1
lnOpF     = ( ThisForm.Get_Nop.Value * 10000 ) + 9999
lcTGOp	  = thisform.get_tpGOp.Value
lcCodFin = thisform.getCodFinP.Value
lcClas = thisform.get_Class.Value 
lcTamA = Val(thisform.get_tamas.Value)
lcTamP = Val(thisform.get_tamps.Value)
lcTamC = Val(thisform.get_tamcs.Value)
lnAgru = thisform.opt_Agru.Value
lcMoeda = thisform.get_cd_moeda.Value

Create Cursor TmpRelat (Fase c(10),IClis c(10),RClis c(50),CMats c(14),DPros c(40) ,CPros c(14),DDPros c(40),Qtde n(12,3),TpOps c(15),;
						CUnis c(3), Operacaos c(31), Nenvs n(10), Datas D, Pesos n(12,3), Valor n(12,2), Fators n(12,3), Moecs c(3),;
						Cgrus c(3), Sgrus c(3), Cclass c(3), CodFinP c(3), Dimens c(30), DFinP c(20), DClass c(40), QtdTrabs n(12,3))
Index On Dtos(datas) + Operacaos + Str(nenvs,10) + cmats Tag Operacao
Index on TpOps + CodFinP + Str(Qtde,12,3) Tag TpCodFin
Index on Dtos(Datas) + Fase + IClis + TpOps + CUnis + CMats + Operacaos + Str(nenvs,10) + Str(Qtde,12,3) Tag FFuncMat


pDtI = fDtoSQL(ldDataI) 
pDtF = fDtoSQL(ldDataF, '23:59:59')

*** Filtra os tipos de materiais selecionados
lcTpOp = []
Select CrSigOpOpt
Go top
Count for Marca = 1 to lnTotOp
Scan
	If (CrSigOpOpt.Marca = 1 Or lnTotOp = 0)
		lcTpOp = lcTpOp + Iif(Empty(lcTpOp),[']+CrSigOpOpt.Cods+['],[,']+CrSigOpOpt.Cods+['])
	Endif
Endscan
lcTpOp = [(] + Iif(Empty(lcTpOp ),[''],lcTpOp) + [)]


lStrQuery = [Select a.Datas,a.CIdChaves,a.Emps,a.DopPs,a.NumPs,a.GrupoOs,a.ContaOs,b.CMats,b.Qtds,b.TpOps,b.Nenvs,]+;
				   [a.GrupoDs,a.ContaDs, b.Nops, b.Pesos, b.Fators, b.Servicos, b.Cats, b.Empdnps ]+;
			  [From SigCdNec a, SigCdNei b ] + ;
			[Where a.Datas BetWeen ?pDtI And ?pDtF And a.EmpDNPs = b.EmpDNPs ]+;
				Iif(Empty(lcTpOp),[],[And b.TpOps in ] + lcTpOp + [ ])+;
				Iif(Empty(lcGrupo),[],Iif(llDestino,[And a.GrupoDs = '],[And a.GrupoOs = '])+lcGrupo+[' ])+;
				Iif(Empty(lcConta),[],Iif(llDestino,[And a.ContaDs = '],[And a.ContaOs = '])+lcConta+[' ])+;
				Iif(lnEnvelope > 0,[And Nenvs = ?lnEnvelope ],[]) +;
			[Order By a.cIdChaves ]

If ThisForm.poDataMgr.SqlExecute(lStrQuery,'TmpNens') < 1
	MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpNens)')
	Return 0
EndIf

*** Quantidade total do componente trabalhados por período
lStrQuery = [Select a.Datas,a.CIdChaves,a.Emps,a.DopPs,a.NumPs,a.ContaOs,b.CMats,b.Qtds,b.TpOps,b.Nenvs,]+;
				   [a.GrupoDs,a.ContaDs, b.Nops, b.Pesos, b.Fators, b.Servicos, b.Cats, b.Empdnps, d.CodFinP, d.Cclass, d.Cunis, d.Pcuss, ]+;
					[case when e.agrupas = '' then a.GrupoOs else e.agrupas end as GrupoOs ] +;
			  [From SigCdNec a, SigCdNei b, SigOpOpt c, SigCdPro d, SigCdGcr e ] + ;
			[Where a.Datas BetWeen ?pDtI And ?pDtF And a.EmpDNPs = b.EmpDNPs And c.Cods = b.TpOps ] +;
			[And b.Cmats = d.Cpros And b.servicos = 0 And a.Retrabs = 0 And a.GrupoOs = e.Codigos ] +;
				Iif(Empty(lcGrupo),[],Iif(llDestino,[And a.GrupoDs = '],[And a.GrupoOs = '])+lcGrupo+[' ])+;
				Iif(Empty(lcConta),[],Iif(llDestino,[And a.ContaDs = '],[And a.ContaOs = '])+lcConta+[' ])+;
				Iif(lnEnvelope > 0,[And Nenvs = ?lnEnvelope ],[]) +;
				Iif(Empty(lcGruMat),[],[And d.CGrus = ']+lcGruMat+[' ]) +;
				Iif(Empty(lcGgrp),[],[And d.Mercs = ']+lcGgrp+[' ]) +;
				Iif(Empty(lcCodFin),[],[And d.CodFinP = ']+lcCodFin+[' ]) +;
				Iif(Empty(lcClas),[],[And d.Cclass = ']+lcClas+[' ]) +;
			[Order By a.cIdChaves ]
*!*	And b.flagincs = 1
If ThisForm.poDataMgr.SqlExecute(lStrQuery,'TmpProd') < 1
	MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpProd)')
	Return 0
EndIf

Select CsSigCdOpd
Go Top

*** Filtra o Tipo de operaçăo
Select * From CrSigOpOpt Where Marca > 0 Into Cursor csLocalMc
Go Top In csLocalMc
If !Eof([csLocalMc])
	Select a.* From TmpNens a, CsSigCdOpd b ;
	Where a.DopPs = b.DopPs And Iif(!Empty(lcTGOp),b.CodTgOps = lcTGOp, .t. ) ;
	Into Cursor CrSigCdNec ReadWrite
Else
	Select a.* From TmpNens a, CsSigCdOpd b ;
	Where a.DopPs = b.DopPs And Iif(!Empty(lcTGOp),b.CodTgOps = lcTGOp, .t. ) ;
	Into Cursor CrSigCdNec ReadWrite
EndIf

If (lnEnvelope > 0)
	Select Distinct Empdnps From CrSigCdNec Where Nenvs = lnEnvelope Into Cursor crEnvelope
EndIf

Select CrSigCdNec
Index On Dtos(Datas) + CIdChaves Tag Datas
Set Order To Datas
Go Top In CrSigCdNec		

lnTotal = Reccount('CrSigCdNec')
If (lnTotal <> 0)
	loBarra = CreateObject('fwprogressbar','Processando Movimentaçőes',lnTotal)
	loBarra.Titulo.FontBold = .T.
	loBarra.Show
	Scan
		loBarra.Update(.T.)

*		If (lnEnvelope > 0) And (CrSigCdNec.Nenvs <> lnEnvelope And lcEdn <> CrSigCdNec.Empdnps)
		If (lnEnvelope > 0)
			Select crEnvelope
			Go Top In crEnvelope
			Locate For Empdnps = CrSigCdNec.Empdnps
			If Eof([crEnvelope])
				Loop
			EndIf
		EndIf
		If lnOP > 0 And Not Between(CrSigCdNec.Nops,lnOpI,lnOpF)
			Loop
		EndIf
	
		lcQuery = [Select a.CPros,a.DPros,a.CGrus,a.Sgrus,a.CUnis,b.Mercs,a.Pcuss,a.Moecs,a.Cclass,a.CodFinp,a.TamPs,a.TamHs,a.TamLs From SigCdPro a, SigCdGrp b ] +;
				[Where a.CPros = ']+CrSigCdNec.CMats+[' And a.Cgrus = b.Cgrus ]
		If Thisform.PodataMgr.SqlExecute(lcQuery,'CrSigCdPro') < 1
			=MessageBox('Favor Reinicializar o Processo!!!',16,'Falha na Conexăo (CrSigCdPro)')
			Return 0
		EndIf
		Select CrSigCdPro
		If !Empty(lcGruMat) And (CrSigCdPro.CGrus <> lcGruMat)
			Loop
		EndIf
		If !Empty(lcGgrp) And (CrSigCdPro.Mercs <> lcGgrp )
			Loop
		Endif
		If !Empty(lcSgru) And (CrSigCdPro.Sgrus <> lcSgru )
			Loop
		Endif
		If !Empty(lcCodFin) And (CrSigCdPro.CodFinp <> lcCodFin )
			Loop
		Endif
		If !Empty(lcClas) And (CrSigCdPro.Cclass <> lcClas )
			Loop
		Endif
		If !Empty(lcTamP) And (CrSigCdPro.TamPs <> lcTamP )
			Loop
		Endif
		If !Empty(lcTamA) And (CrSigCdPro.TamHs <> lcTamA )
			Loop
		Endif		
		If !Empty(lcTamC) And (CrSigCdPro.TamLs <> lcTamC )
			Loop
		Endif				
		
		lcTpOps = CrSigCdNec.TpOps
		
		lcGrp = Iif(llDestino,CrSigCdNec.GrupoDs,CrSigCdNec.GrupoOs)
		ThisForm.poDataMgr.CursorQuery('SigCdGcr', 'TmpCdGcr', 'Codigos', lcGrp, 'Agrupas')
		lcGrp = Iif(Empty(TmpCdGcr.Agrupas),lcGrp,Padr(TmpCdGcr.Agrupas,10))		
		lcCta = Iif(llDestino,CrSigCdNec.Contads,CrSigCdNec.ContaOs)
		lcPed = Padr(Iif(lnAgru = 1,CrSigCdNec.Dopps+'-'+Str(CrSigCdNec.Numps,10),' '),31)
		lnEnv = Iif(lnAgru = 1,CrSigCdNec.Nenvs,0)
		ldData = Iif(lnAgru = 1,CrSigCdNec.Datas,Ctod('31/12/9999'))
		
		If lcGrp = CrSigCdPam.GrTrans
			Loop
		EndIf

		lcPro = []
		lcDpr = []
		If Not IsEmpty(CrSigCdNec.Nops) And ThisForm.chkProdutos.Value
			lcQuery = [Select top 1 a.Cpros, b.DPros ] + ;
					    [From SigOpPic a Inner join SigCdPro b on a.Cpros = b.Cpros Where Nops = ?crSigCdNec.Nops]

			If (ThisForm.poDataMgr.SqlExecute(lcQuery, [LocalPro]) < 1)
				=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (LocalPro)])
				Return 0
			EndIf
			Go Top In LocalPro
			If Not Eof([LocalPro])
				lcDpr = LocalPro.DPros
				lcPro = LocalPro.Cpros
			EndIf
		EndIf

		*** Monta dimensőes do produto
		lcDimen = ''
		ThisForm.poDataMgr.CursorQuery('SigCdGrp', 'crSigCdGrp', 'CGrus', crSigCdPro.CGrus, 'TitulosPro')
		If !Empty(Substr(crSigCdGrp.TitulosPro, 061, 060))
			lcDimen = Str(crSigCdPro.tamps,2) + Iif(Empty(Substr(crSigCdGrp.TitulosPro, 061, 020)),'p',Alltrim(Substr(crSigCdGrp.TitulosPro, 061, 020)))
			lcDimen = lcDimen + [ x ] + Str(crSigCdPro.tamhs,2) + Iif(Empty(Substr(crSigCdGrp.TitulosPro, 081, 020)),[a],Alltrim(Substr(crSigCdGrp.TitulosPro, 081, 020)))
			lcDimen = lcDimen + [ x ] + Str(crSigCdPro.tamls,2) + Iif(Empty(Substr(crSigCdGrp.TitulosPro, 101, 020)),[c],Alltrim(Substr(crSigCdGrp.TitulosPro, 101, 020)))
		Endif

		ThisForm.poDataMgr.CursorQuery('SigCdFip', 'crSigCdFip', 'Cods', crSigCdPro.CodFinP, 'Descs')
		ThisForm.poDataMgr.CursorQuery('SigCdCls', 'crSigCdCls', 'Cods', crSigCdPro.Cclass, 'Descs')
		

		Select TmpRelat
		If Not Seek(Dtos(ldData) + lcGrp + lcCta + lcTpOps + CrSigCdPro.Cunis + CrSigCdNec.CMats + lcPed + Str(lnEnv,10) )			
			ThisForm.poDataMgr.CursorQuery('SigCdCli', 'CrTmpCli', 'IClis', lcCta, 'RClis')
			Insert Into TmpRelat (Fase, Iclis, RClis, CMats, Dpros, CPros, DDPros, TpOps, Cunis, Operacaos, Nenvs, ;
								Datas, Fators, Moecs, Cclass, CodFinP, Dimens, Cgrus, Sgrus, ;
								DFinP, DClass ) ;
						  Values (lcGrp, lcCta, CrTmpCli.RClis, CrSigCdNec.CMats, CrSigCdPro.Dpros, lcPro, lcDpr, lcTpOps, CrSigCdPro.Cunis, lcPed, lnEnv, ;
						  		ldData, crSigCdNec.Fators, crSigCdPro.Moecs, crSigCdPro.Cclass, crSigCdPro.CodFinP, lcDimen, crSigCdPro.Cgrus, crSigCdPro.Sgrus, ;
						  		crSigCdFip.Descs, crSigCdCls.Descs )
		EndIf
		Replace Qtde With Qtde + CrSigCdNec.Qtds
		Replace Valor With Valor + Iif(crSigCdPro.Moecs <> lcMoeda, fCarregarCambio(crSigCdPro.Moecs, Date()) / fCarregarCambio(lcMoeda, Date()),;
		 				(CrSigCdPro.Pcuss * CrSigCdNec.Qtds))
		Replace Moecs With Iif(crSigCdPro.Moecs <> lcMoeda,lcMoeda,crSigCdPro.Moecs)

		If lnAgru = 1
			If Not Seek(Dtos(Ctod('31/12/9999')) + Replicate(Chr(255), 20) + lcTpOps + CrSigCdPro.Cunis + CrSigCdNec.CMats)
				Insert Into TmpRelat (Fase, Iclis, RClis, CMats, TpOps, Cunis, ;
							  Moecs, Datas, CodFinP, Cclass ) ;
							  Values (Replicate(Chr(255), 10), Replicate(Chr(255), 10), 'Total Geral : ', CrSigCdNec.CMats, lcTpOps, CrSigCdPro.Cunis, ;
							  crSigCdPro.Moecs, Ctod('31/12/9999'), '','')
			EndIf
			Replace Qtde with Qtde + CrSigCdNec.Qtds
			Replace Valor With Valor + Iif(crSigCdPro.Moecs <> lcMoeda, fCarregarCambio(crSigCdPro.Moecs, Date()) / fCarregarCambio(lcMoeda, Date()),;
		 				(CrSigCdPro.Pcuss * CrSigCdNec.Qtds))
			Replace Moecs With Iif(crSigCdPro.Moecs <> lcMoeda,lcMoeda,crSigCdPro.Moecs)		 				
		Endif
	EndScan
	loBarra.Complete(.T.)
EndIf

Select TmpRelat
Go top
Do Case
	Case lnAgru = 1
*!*			Set Order To tag FFuncMat
		lcCabTipo = [ - Tipo : Analítico]
		
		Select * from TmpRelat order by Datas, Fase, IClis, TpOps, CUnis, Moecs, CMats, Qtde Desc into cursor TmpRelat ReadWrite

	Case lnAgru = 2
		lcCabTipo = [ - Tipo : Modelo]
		Select CodFinp, Cunis, Sum(Qtds) as QtdTrabs, Sum(Qtds * Pcuss) as CustoTrabs from TmpProd ;
			Group By CodFinp, Cunis Order By CodFinp, QtdTrabs Desc into cursor TmpProd  ReadWrite

		Select a.TpOps, a.CodFinp as Agrupa, a.DFinP as Dagrupa, a.Cunis, a.Moecs, b.QtdTrabs, Sum(a.Qtde) as Qtde, Sum(a.Valor) as Valor, ;
			CustoTrabs, ((Sum(a.Qtde) / b.QtdTrabs) * 100) as PercMat ;
			from TmpRelat a Left Join TmpProd b On a.CodFinP = b.CodFinP And a.Cunis = b.Cunis ;
			Group By a.TpOps, a.CodFinp, a.DFinP, a.Cunis, a.Moecs, b.QtdTrabs, b.CustoTrabs Order By a.TpOps, a.CodFinp, a.Cunis into cursor TmpRelat ReadWrite
					
	Case lnAgru = 3
		lcCabTipo = [ - Tipo : Classificaçăo]
		Select Cclass, Cunis, Sum(Qtds) as QtdTrabs, Sum(Qtds * Pcuss) as CustoTrabs from TmpProd ;
			Group By Cclass, Cunis Order By Cclass, QtdTrabs Desc into cursor TmpProd  ReadWrite
			
		Select a.TpOps, a.Cclass as Agrupa, a.DClass as Dagrupa, a.Cunis, a.Moecs, b.QtdTrabs, Sum(a.Qtde) as Qtde, Sum(a.Valor) as Valor, ;
			CustoTrabs, ((Sum(a.Qtde) / b.QtdTrabs) * 100) as PercMat ;
			from TmpRelat a Left Join TmpProd b On a.Cclass = b.Cclass And a.Cunis = b.Cunis ;
			Group By a.TpOps, a.Cclass, a.DClass, a.Cunis, a.Moecs, b.QtdTrabs, b.CustoTrabs Order By a.TpOps, a.Cclass, a.Cunis into cursor TmpRelat ReadWrite

	Case lnAgru = 4
		lcCabTipo = [ - Tipo : Conta]
		Select GrupoOs, ContaOs, Cunis, Sum(Qtds) as QtdTrabs, Sum(Qtds * Pcuss) as CustoTrabs from TmpProd ;
			Group By GrupoOs, ContaOs, Cunis Order By GrupoOs, ContaOs, QtdTrabs Desc into cursor TmpProd  ReadWrite
			
		Select a.TpOps, a.Fase as Agrupa, a.Iclis + ' - ' + a.Rclis as Dagrupa, a.Cunis, a.Moecs, b.QtdTrabs, Sum(a.Qtde) as Qtde, Sum(a.Valor) as Valor, ;
			CustoTrabs, ((Sum(a.Qtde) / b.QtdTrabs) * 100) as PercMat ;
			from TmpRelat a Left Join TmpProd b On a.Fase = Padr(b.GrupoOs,10) And a.Iclis = b.ContaOs And a.Cunis = b.Cunis ;
			Group By a.TpOps, a.Fase, Dagrupa, a.Cunis, a.Moecs, b.QtdTrabs, b.CustoTrabs Order By a.TpOps, a.Fase, Dagrupa, a.Cunis into cursor TmpRelat ReadWrite

Endcase

lcCab = [Análise por Tipo de Material ] + Chr(13) + [Moeda : ] + lcMoeda + lcCabTipo
lcSub = Iif(llDestino,[Por Destino],[Por Origem]) +  [ - Período : ] + Alltrim(Dtoc(ldDatai)) + [ até ] + Alltrim(Dtoc(ldDataf))
ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'CrSigCdEmp', 'Cemps', _Empr, 'Razas')
lcEmp = _Empr + ' - ' + Iif(Eof('CrSigCdEmp'), '', CrSigCdEmp.Razas)

Create Cursor dbCabecalho (Titulo c(100), SubTitulo c(100), NomeEmpresa c(100))
Insert Into dbCabecalho (Titulo, SubTitulo, NomeEmpresa) Values (lcCab, lcSub, lcEmp)


Select TmpRelat
Go Top
ENDPROC

EndIf
