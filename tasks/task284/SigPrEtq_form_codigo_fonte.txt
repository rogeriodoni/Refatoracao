* PROCEDURE Init
------------------------------------------------------------
PROCEDURE Init
Local lnTipos, lnTop, lnHeight, lnImp, lnMaxTipos, lnMaxImpres, lnBottom1, lnBottom2, lnPrinters, llOk, lnI

ThisForm.cntSombra.lblSombra.Caption = ThisForm.Caption
ThisForm.cntSombra.lblTitulo.Caption = ThisForm.Caption
Set Step On 
With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)

	If (.poDataMgr.pnIdconn > 0)
		lcCpo = 'nMaxTpEtis, TpEtiPads, nMaxImpEti, ImpEtis, TpInstalas, AjVerts, AjHorzs, TpCBars'
		ThisForm.poDataMgr.CursorQuery('SigCdPam', 'crSigCdPam', .f., .f., lcCpo)
		Goto Top In crSigCdPam

		ThisForm.poDataMgr.CursorQuery([SigCdPac], [crSigCdPac], .f., .f., [AjDens, AjVelos, EtqSeps])
		Goto Top In crSigCdPac

		Dimension laPrinters(10,2)
		lnPrinters = APrinters(laPrinters)
		*** mena - 06/03/2015 - cria cursor e inclui nele as impressoras do windows ***
		Create Cursor crImpre (impres c(50))
		If (lnPrinters <> 0)
			For lnI = 1 To lnPrinters
				Insert Into crImpre	(impres) Values(Upper(laPrinters(lnI, 1)))
				*!*laPrinters(lnI, 1) = Upper(laPrinters(lnI, 1))
			EndFor
			*!*=ASort(laPrinters)
		EndIf

		If (Type('laPrinters') <> 'C')
			=Messagebox('Nenhuma Impressora Foi Encontrada!!! Verifique!!!', 0+48, 'Impressoras')
			Return .f.
		EndIf

		ThisForm.poDataMgr.CursorQuery('SigCdTpe', 'crSigCdTpe')
		Select crSigCdTpe
		Index On cOrdems + cEtiquetas Tag skOrdems For nSituas = 1

		Select crSigCdTpe
		Set Order to skOrdems
		Count To lnTipos For nSituas = 1

		With .Opt_Tipo
			If (lnTipos <> 0)
				lnTipoPadrao = 1
				.ButtonCount = Min(lnTipos, Max(crSigCdPam.nMaxTpEtis, 7))
				lnMaxTipos = .ButtonCount + 1
				lnI = 1
				lnTop = 10
				lnHeight = 15

				Scan For (crSigCdTpe.nSituas = 1) And (lnI < lnMaxTipos)
					If (crSigCdTpe.nTipos = crSigCdPam.TpEtiPads)
						lnTipoPadrao = lnI
					EndIf
					lcI = AllTrim(Str(lnI))
					With .Option&lcI.
						.AutoSize  = .F.
						.Width     = 197
						.Caption   = ' \<' + Chr(96 + lnI) + '. ' + AllTrim(crSigCdTpe.cEtiquetas)
						.FontSize  = 8
						.Forecolor = Rgb(90,90,90)
						.Tag       = AllTrim(Str(crSigCdTpe.nTipos))
						.Top       = lnTop
						.BackStyle = 0
					EndWith
					lnI      = lnI + 1
					lnTop    = lnTop + 20
					lnHeight = lnHeight + 20
				EndScan
			Else
				.ButtonCount = 1
				With .Option1
					.AutoSize = .F.
					.Width    = 197
					.Caption  = ' \<z. Rabicho Loja Tipo 1 (*)'
					.FontSize = 8
					.Forecolor = Rgb(90,90,90)
					.Tag      = AllTrim( Str( 1 ) )
					.Top      = lnTop
				EndWith
				lnHeight = lnHeight + 20
			EndIf
			.Enabled = (lnTipos > 1)
			.Height  = lnHeight
			.Value   = lnTipoPadrao
		EndWith

		lcSql = [Select b.Impres ] + ;
				  [From SigSyImp a, SigCdmp b ] + ;
				 [Where a.UsuAcess = '] + Usuar + [' And ] + ;
				 	   [a.CImps = b.Impres And ] + ;
				 	   [b.nTpImpres = 2 ] + ;
				 [Union All ] + ;
				[Select c.Impres ] + ;
				  [From SigCdAcG a, SigSyImp b, SigCdmp c ] + ;
				 [Where a.Usuarios = '] + Usuar + [' And ] + ;
				       [a.Grupos = b.GrAcess And ] + ;
				       [b.CImps = c.Impres And ] + ;
				       [c.nTpImpres = 2]

		If ThisForm.podatamgr.sqlexecute(lcSql,'crTmpCimp') <= 0
			Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados.',(0+48),'Erro')
			Return(.f.)
		EndIf

		Select crTmpCImp
		If (Reccount([crTmpCImp]) > 0)
			Select Distinct Impres ;
			  From crTmpCImp ;
			 Order By Impres ;
			  Into Cursor crSigCdmp ReadWrite
		Else
			lcQuery = [Select Distinct Impres ] + ;
					    [From SigCdmp ] + ;
					   [Where nTpImpres = 2 ] + ;
					   [Order By Impres]

			If (ThisForm.poDataMgr.SqlExecute(lcQuery, [crSigCdmp]) < 1)
				=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crSigCdmp)])
				Return .f.
			EndIf
		EndIf
		Select crSigCdmp
		Go Top In crSigCdmp		

*!*			lnCnt = 0
*!*			lnImp = 1
*!*			If (lnPrinters <> 0)
*!*				Select crSigCdmp
*!*				Scan
*!*					Dimension laImpOk(lnImp)
*!*					lnOk = AScan(laPrinters, Alltrim(Upper(crSigCdmp.Impres)))
*!*					If (lnOk <> 0)
*!*						lnCnt = lnCnt + 1
*!*						laImpOk(lnCnt) = laPrinters(lnOk)
*!*						lnImp = lnImp + 1
*!*					EndIf
*!*				EndScan
*!*			EndIf
*!*			lnImp = lnImp - 1

		*** mena - 05/03/2015 - a lista de impressoras sairam dos radio para o combobox *** 
		SELECT PADR(ALLTRIM(a.Impres),15)+' '+ALLTRIM(b.impres) as IDupla, b.impres, a.impres as impresS from crSigCdmp a, crImpre b where ALLTRIM(UPPER(a.impres)) like '%'+ALLTRIM(UPPER(b.impres))+'%' ;
				or ALLTRIM(UPPER(b.impres)) like '%'+ALLTRIM(UPPER(a.impres))+'%' group by b.impres, a.impres into cursor crImp READWRITE

		IF RECCOUNT('crImp') > 1
			APPEND BLANK	&& mena - 01/04/2015 se houver mais de uma impressora na lista, posiciona em impressora em branco  
		ENDIF
		
		SELECT * FROM crImp order by 1 into cursor crImpreV

					lnValDe = LocalPro.PrecoDe
				EndIf 		
			EndIf
			If Used([LocalPro]) And Not Eof([LocalPro]) And (LocalPro.PesoMs > 0)
				lnPes = LocalPro.PesoMs
			EndIf
			
			Insert Into dbImpressao (Cpros, DPros, Qtds, QtdeEtiq, Obs, PVens, empos, empdopnums, citens, Pesos, PrecoDe) ;
							 Values (LocalEestI.Cpros, LocalEestI.DPros, LocalEestI.Qtds, ;
							 		 LocalEestI.Qtds, lcEdn, lnVen, _empr, lcEdn, LocalEestI.citens, lnPes, lnValDe)
		EndScan
	EndIf
EndIf

If (ThisForm.chkLista.Value <> 1) And Not Empty(ThisForm.get_lpreco.Value)
	Select dbImpressao
	Scan
		lcQuery = [Select LPrecos, CPros, DPros, PVens, PrecoDe, VencIs, VencFs ] + ;
				    [From SigCdLpi ] + ;
				   [Where LPrecos = '] + Padr(ThisForm.get_lpreco.Value, 30) + [' And ] + ;
				   		 [CPros = '] + Padr(dbImpressao.CPros, 14) + [']

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'LocalLPreI') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalLPreI)')
			Return 0
		EndIf
		Select LocalLPreI
		Go Top In LocalLPreI
		If Not Eof([LocalLPreI])
			lnVal = LocalLPreI.PVens
			lnValDe = LocalLPreI.PrecoDe
			If Not Between(Datetime(), LocalLPreI.VencIs, LocalLPreI.VencFs)
				ThisForm.poDataMgr.CursorQuery('SigCdPro', 'LocalPro', 'CPros', Padr(LocalLPreI.CPros, 14), 'PVens, PrecoDe')
				lnVal = LocalPro.PVens
				lnValDe = LocalPro.PrecoDe
			EndIf

			Replace Obs   With LocalLPreI.LPrecos, ;
					PVens With lnVal ,;
					PrecoDe With lnValDe In dbImpressao
		EndIf
	EndScan
EndIf

Select dbImpressao
If (Reccount([dbImpressao]) = 0)
	Append Blank In dbImpressao
Endif
Go Top In dbImpressao

ThisForm.grd_Etiqueta.Refresh
* Fim **************************************************************************************
ENDPROC
