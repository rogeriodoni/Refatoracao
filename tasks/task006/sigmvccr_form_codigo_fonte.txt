PROCEDURE m_inserir
local voper,oper,grupo,conta,moed,cot,valor,soper,sgrupo,sconta,smoed,;
		scot,svalor,data,sdata,hist,shist,docu,sdocu,venc,svenc,noper,;
		snoper,emp

pData = ThisForm.PodataMgr.GetDatetime()
*** INSERIR CONTA MOVIMEN//
select craltcc
Go Top
replace craltcc.autos with .f.,;
		craltcc.usualts with usuar,;
		craltcc.datalts with pData,;
		craltcc.gruconmoes with craltcc.grupos + craltcc.contas + craltcc.moedas,;
		craltcc.cidchaves with Dtos(Datas)+Transform(fGerUniqueKey(Dtos(Datas)),"@L 999999") + ThisForm.Sigkey 
if thisform.pagina.dados.opcao_conta.value=2
	replace craltcc.valpags with craltcc.valors
	replace craltcc.pagos   with '0'
else
	replace craltcc.valpags with 0
	replace craltcc.pagos   with '1'
endif
noper   = craltcc.nopers
voper   = craltcc.vopers
oper    = craltcc.sopers
grupo   = craltcc.sgrupos
conta   = craltcc.scontas
moed    = craltcc.smoedas
cot     = craltcc.scotacaos
valor   = craltcc.svalors
soper   = craltcc.opers
sgrupo  = craltcc.grupos
sconta  = craltcc.contas
smoed   = craltcc.moedas
scot    = craltcc.cotacaos
svalor  = craltcc.valors
nlanc   = craltcc.nlancs
vlanc   = craltcc.vlancs
usualt  = craltcc.usualts
datalt  = Nvl(craltcc.datalts,Ctod(Space(08)))
contapg = craltcc.contapgs
cotusu  = craltcc.cotusus
Job		= crAltcc.Jobs
if thisform.pagina.dados.opcao_conta.value=2
	valpag = craltcc.svalors
else
	valpag = 0
endif

=fRecalculaS( CrAltcc.Grupos, Craltcc.Contas, Craltcc.Datas, Craltcc.Moedas, ThisForm.Podatamgr  )

*** INSERIR CONTRA MOVIMEN//
select craltcc
If Thisform.pcEscolha = 'INSERIR'
	append blank
Else
	Skip
Endif
replace vopers with voper ;
		nopers with noper ;
		nlancs with nlanc ;
		vlancs with vlanc ;
		opers  with oper ;
		grupos with grupo ;
		contas with conta ;
		moedas with moed ;
		cotacaos with cot ;
		valors with valor ;
		valpags with valpag ;
		emps   with _empr ;
		datas  with thisform.pagina.dados.fweditdata_data.value ;
		hists  with thisform.pagina.dados.get_hist.value ;
		hist2s with thisform.pagina.dados.get_hist2.value ;
		vencs  with thisform.pagina.dados.fweditdata_venc.value ;
		DtEmis with ThisForm.Pagina.Dados.getDtEmis.Value ;
		docus  with thisform.pagina.dados.get_docu.value ;
		Nfs    With Thisform.Pagina.Dados.Get_nota.Value ;
		autos  with .f. ;
		usualts with usualt ;
		datalts with datalt ;
		contapgs with contapg ;
		sgrupos with sgrupo ;
		scontas with sconta ;
		smoedas with smoed ;
		scotacaos with scot ;
		svalors with svalor ;
		sopers  with soper ;
		cotusus with cotusu ;
		gruconmoes with grupo + conta + moed ;
		cidchaves with Dtos(Datas)+Transform(fGerUniqueKey(Dtos(Datas)),"@L 999999") + ThisForm.Sigkey ;
		pagos with Iif(valors=valpags,'0','1') ;
		Jobs with Job

=fRecalculaS( CrAltcc.Grupos, Craltcc.Contas, Craltcc.Datas, Craltcc.Moedas, ThisForm.Podatamgr  )

Select Craltcc
Replace all Vencs  With ThisForm.Pagina.Dados.fweditdata_venc.value, ;
			DtEmis With ThisForm.Pagina.Dados.getDtEmis.Value

ENDPROC
