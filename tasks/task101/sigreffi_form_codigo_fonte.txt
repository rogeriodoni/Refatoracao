* PROCEDURE saldooper
------------------------------------------------------------
PROCEDURE saldooper
Lparameters lcEDN

Local lcQuery, lnSaldo
lcQuery = [select a.empdopnums, sum(e.valos) as valos, sum(e.vpags) as vpags from SigMvCab a ] + ;
		  [inner join SigMvItn b on b.empdopnums = a.empdopnums ] + ;
		  [inner join SigCdOpe c on c.dopes = b.dopes ] + ;		  
		  [inner join SigMvPec d on d.codigos = ] + ;
		  [cast((rtrim(cast(c.ndopes as char(4))) + replace(right(a.empdopnums, 6), ' ', '0')) as numeric(10, 0)) ] + ;
		  [inner join SigMvPar e on e.empdopnums = d.empdopnums ] + ;
		  [where a.EmpDopNums = ']+lcEDN+[' and a.chksubn = 0 and b.chksubn = 1 ] + ;
		  [group by a.empdopnums having (sum(e.valos) - sum(e.vpags)) > 0 order by 1]
		  
If ThisForm.poDataMgr.SqlExecute(lcQuery,[crSldPend]) < 1
	MessageBox('Favor Reinicializar o Processo !!!',16,'Falha na Conexăo (CrSldPend)')
	Return(.F.)
EndIf

lnSaldo = crSldPend.Valos - crSldPend.VPags
Return(lnSaldo)
ENDPROC

* PROCEDURE Init

PROCEDURE Init
Private pIds
Local lcEscape,ImpEmps

m.lcEscape = Set('Escape')
Set Escape Off
ThisForm.LockScreen = .t.

pIds = Replicate(Chr(254), 20)

lcQryGTmpD = [Select * From SigTempR Where cIdQuerys = '] + pIds + [']

With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)

	If (.poDataMgr.pnIdconn > 0)
		ThisForm.poDataMgr.CursorQuery('SigOpOpe', 'TempSigOpOpe', .f., .f., 'Dopes, Opers, RealPrevs,OpeCancs')

		Select .f. as SeleOp, Dopes, RealPrevs, Opers ;
		  From TempSigOpOpe Where OpeCancs <> 1 ;
		  Into Cursor crSigOpOpe ReadWrite

		Select crSigOpOpe
		Index On Dopes Tag Dopes
		Set Order To Dopes
		Scan
		 	If Not fAcessoTitulo(Usuar, crSigOpOpe.Dopes, .f., ThisForm.poDataMgr)
		 		Delete In crSigOpOpe
		 	EndIf
		EndScan
		Go Top

		ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'TempEmp', .f., .f., 'Cemps, Razas')

		Select .f. as ImpEmps, Cemps, Razas ;
		  From TempEmp ;
		 Order By 2 ;
		  Into Cursor cEmp ReadWrite
		  
		Select cEmp
		Index On Cemps Tag Cemps
		Set Order To Cemps

		If Seek(_Empr, 'CEmp', 'Cemps')
			m.ImpEmps = .t.
			Gather Memvar Fields ImpEmps
		EndIf

		ThisForm.poDataMgr.CursorQuery('SigCdDis', 'TmpSigCdDis', .f., .f., 'Grupos, Contas, Tipos')

		Select .t. as ImpDisps, Grupos, Contas ;
		  From TmpSigCdDis Where Tipos in (0,1) ;
		 Order By 2 ;
		  Into Cursor cDisp ReadWrite

		Select cDisp
		Index On Grupos + Contas Tag Grupos
		Index On TransForm(ImpDisps, '@!') + Grupos + Contas Tag ImpDisps
		Set Order To Grupos

		ThisForm.poDataMgr.CursorQuery('SigCdMoe', 'crSigCdMoe', .f., .f., 'CMoes, DMoes, Cotas')
		Select crSigCdMoe
		Index On CMoes Tag CMoes

		ThisForm.poDataMgr.CursorQuery('SigCdCot', 'crSigCdCot', .f., .f., 'CMoes, Datas, Valos')
		Select crSigCdCot
		Index On CMoes + Dtos(Datas) Tag CMoeData

		ThisForm.poDataMgr.CursorQuery('SigCdRec', 'crSigCdRec')
		Select crSigCdRec
		Index On Grupos Tag Grupos

		Set Escape &lcEscape.

		.get_dt_inicial.Value 	   = Date()
		.get_dt_final.Value 	   = Date()
		.get_cd_moeda.Value 	   = ''
		.get_ds_moeda.Value 	   = ''
		.opt_nr_tipo.Value 		   = 1
		.opt_situa.Value 		   = 1
		.opt_ordem_impressao.Value = 1
		
		With .Grid1
			.ColumnCount  = 3
			.RecordSource = 'cEmp'
			.Column1.ControlSource = 'cEmp.ImpEmps'
			.Column2.ControlSource = 'cEmp.Cemps'
			.Column3.ControlSource = 'cEmp.Razas'

			For I=2 To .ColumnCount
				lcCol = Alltrim(Str(I))
				.Column&lcCol..Movable   = .f.
				.Column&lcCol..Resizable = .f.
			Next
		EndWith
		
		With .Grid2
			.RecordSource = 'cDisp'
			.Column1.ControlSource = 'cDisp.ImpDisps'
			.Column2.ControlSource = 'cDisp.Grupos'
			.Column3.ControlSource = 'cDisp.Contas'

			For I=2 To .ColumnCount
				lcCol = Alltrim(Str(I))
				.Column&lcCol..Movable   = .f.
				.Column&lcCol..Resizable = .f.
				.Column&lcCol..ReadOnly  = .t.
			Next
		EndWith

		With .Grd_Oper
			.RecordSource = 'crSigOpOpe'
			.Column1.ControlSource = 'crSigOpOpe.SeleOp'
			.Column2.ControlSource = 'crSigOpOpe.Dopes'
			.Column3.ControlSource = 'crSigOpOpe.Opers'

			For I=2 To .ColumnCount
				lcCol = Alltrim(Str(I))
				.Column&lcCol..Movable   = .f.
				.Column&lcCol..Resizable = .f.
			Next
		EndWith

		=DoDefault()

		lcSql = [Select TitOpcS, TitOpcN, TitOpcA From SigCdPac]
		If ThisForm.Podatamgr.Sqlexecute(lcSql,'LocalParac') < 1
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalParac)')
			Return .f.
		EndIf
		Select LocalParac
		Go Top
		With ThisForm.OptOpcoes
			.option1.Caption = Iif(Not Empty(LocalParac.TitOpcs),LocalParac.TitOpcS,[Sim])
			.option2.Caption = Iif(Not Empty(LocalParac.TitOpcn),LocalParac.TitOpcN,[Nao])
			.option3.Caption = Iif(Not Empty(LocalParac.TitOpcA),LocalParac.TitOpcA,[Ambos])
		EndWith

		.Refresh
		.LockScreen = .f.	
	Else
		=Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith
* Fim **************************************************************************************

ENDPROC
