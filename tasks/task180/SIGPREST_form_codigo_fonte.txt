PROCEDURE Click
Local lnArq, lcArquivo, lnGeraArq, lnGeraInd, lnKey
Local lcChave, lcFiltro, lcTag, lcAlias, loBarra, lcDbc,lcCampo,lcDirAtual1

Close Tables All 
This.Enabled = .f.
ThisForm.Cancela.Enabled = .f.
ThisForm.Draw

Set Default To .\basededados\

If ThisForm.GeraArquivos.Value

	Select 0 

	Create Table ArqDBF Free ( Arquivos c(20), Dbcs c(50), Campos c(20), Tipos c(1), Tamanhos n(3) , ;
							Fracaos n(2), C_05s l, C_06s l, C_07s c(20), C_08s c(20),;
							C_09s c(20), C_10s c(20), C_11s c(20), C_12s c(20),;
							C_13s c(20), C_14s c(20), C_15s c(20), C_16s c(20) )

	Index on Arquivos + Campos Tag ArqCamp
	lnArq = ADir( laArq, '*.Dbf' )
	=ASort( laArq )

	loBarra = CreateObject( 'fwProgressBar', 'Processando Estrutura de Arquivos.', lnArq )
	loBarra.Titulo.FontBold = .t.
	loBarra.Show

	For lnGeraArq = 1 to lnArq

		lcArquivo = laArq( lnGeraArq, 1 )

		loBarra.Update( .t. )
		loBarra.SubTitulo.Caption = 'Processando Arquivo : ' + Alltrim( lcArquivo )

		If Inlist( Alltrim( lcArquivo ), 'ARQDBF.DBF', 'ARQIND.DBF', 'FOXUSER.DBF')
			Loop
		Endif

		Use &lcArquivo in 0 Alias TmpArquivo Again 

		Select TmpArquivo
		lcDbc = Alltrim( JustFName( CursorGetProp( "DataBase" ) ) ) 
		lnCampos = AFields( laCampos )

		For LnCCampos = 1 to LnCampos

			If lnCCampos = 1
				If Empty( laCampos( lnCCampos, 12 ) )
					laCampos( lnCCampos, 12 ) = Strt( lcArquivo, '.DBF', '' )
				Endif
			EndIf
			lcCampo = Padr(Upper(Alltrim(JustStem(lcArquivo))+'.'+Alltrim(laCampos( lnCCampos, 1 ))),64)

			Select ArqDBF
			Insert Into ArqDBF ( Arquivos, Dbcs, Campos, Tipos, Tamanhos, Fracaos, C_05s, C_06s, C_07s,;
								C_08s, C_09s, C_10s, C_11s, C_12s, C_13s, C_14s, C_15s, C_16s ) ;
					Values ( lcArquivo, lcDbc, ;
							laCampos( lnCCampos, 1 ), laCampos( lnCCampos, 2 ),;
							laCampos( lnCCampos, 3 ), laCampos( lnCCampos, 4 ),; 
							laCampos( lnCCampos, 5 ), laCampos( lnCCampos, 6 ),; 
							laCampos( lnCCampos, 7 ), laCampos( lnCCampos, 8 ),; 
							laCampos( lnCCampos, 9 ), laCampos( lnCCampos, 10 ),; 
							laCampos( lnCCampos, 11 ), laCampos( lnCCampos, 12 ),; 
							laCampos( lnCCampos, 13 ), laCampos( lnCCampos, 14 ),; 
							laCampos( lnCCampos, 15 ), laCampos( lnCCampos, 16 ) )
		Endfor
		Use in TmpArquivo
	Endfor
	loBarra.Complete( "Finalizando Processo de Estrutura." )
	ThisForm.Refresh

Endif

Close Tables All 

If ThisForm.GeraÍndices.Value And !File( "ArqDBF.Dbf" )
	Messagebox( 'Antes de gerar os Índices, é necessário que seja gerada a Estrutura de Arquivos...', 32, '' )
	ThisForm.Mensagem1.Caption = "Processamento Interrompido."
	ThisForm.Ok.Enabled = .t.
	ThisForm.Cancela.Enabled = .t.
	Return
Endif

If ThisForm.GeraÍndices.Value 

	Select 0 
	Use ArqDBF Order ArqCamp

	lnArq = ADir( laArq, '*.Dbf' )
	=ASort( laArq )

	If File( 'ArqInd.Dbf' )
		Delete File ArqInd.Dbf
		Delete File ArqInd.Cdx
	Endif
	
	Create Table ArqInd Free ( Arquivos c(20), Tags c(15), Indices c(240), Filtros c(240), Indexs l, c_12s c(20) )
	Index on Arquivos + Tags Tag Arquivos 
	Index on Arquivos Tag Temp Unique

	loBarra = CreateObject( 'fwProgressBar', 'Processando Índices de Arquivos.', lnArq )
	loBarra.Titulo.FontBold = .t.
	loBarra.Show

	For lnGeraInd = 1 to lnArq
	
		lcArquivo = laArq( lnGeraInd, 1 )

		loBarra.Update( .t. )
		loBarra.SubTitulo.Caption = 'Processando Arquivo : ' + Alltrim( lcArquivo )

		If Inlist( Alltrim( lcArquivo ), 'ArqDBF.DBF', 'ArqInd.DBF', 'FoxUser.DBF')
			Loop
		Endif
		Select 0
		Use &lcArquivo Alias TmpArquivo Again
		lnKey = 1
		Do While !Empty( Tag( lnKey ) )
			lcChave = Key( lnKey )
			lcFiltro = Sys( 2021, lnKey )
			lcTag = Tag( lnKey )
			Insert Into ArqInd ( Arquivos, Tags, Indices, Filtros ) ;
					Values ( lcArquivo, lcTag, lcChave, lcFiltro )

			Select TmpArquivo
			lnKey = lnKey + 1
		Enddo
		Use
	Endfor

	loBarra.Complete( "Finalizando Processo de Índice." )
	ThisForm.Refresh
Endif

Close Tables All 

ThisForm.Mensagem1.Caption = "Processamento Finalizado."
ThisForm.Ok.Enabled = .t.
ThisForm.Cancela.Enabled = .t.

If Not Used('csLogoTipo')
	Select Logos as gnLogos From SigCdPam Into Cursor csLogoTipo
	Use In SigCdPam
EndIf

ENDPROC
