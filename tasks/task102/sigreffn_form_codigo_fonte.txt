------------------------------------------------------------
* PROCEDURE godayofweek
------------------------------------------------------------
PROCEDURE godayofweek
LPARAMETERS ldData, lnDia

LOCAL ldDataRet

ldDataRet = ldData
If( DOW(ldDataRet) <> lnDia )
	IF DOW(ldData) < lnDia
		DO WHILE DOW(ldDataRet) < lnDia
			ldDataRet = ldDataRet + 1
		EndDo		
	ELSE
		DO WHILE DOW(ldDataRet) > lnDia
			ldDataRet = ldDataRet - 1
		EndDo			
	Endif
ENDIF
Return(ldDataRet)


ENDPROC

------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Private pGcm, pDat, pDtI, pDtF, lcMoeLanc
Local lcNmEmpresa, lcNmTitulo, lcNmPeriodo, lcCbEmpresa, lcCbMoeda, lcCbMoeRef, ldDtInicial, ldDtFinal, ;
	  lcCdMoeda, lcDsMoeda, lnVlCotacao, lnCtReferencia, lnVlConvertido, lnVlSaldoTotal, lnPagamentos, ;
	  lnRecebimentos, lnDoDia, lnAcumulado, lnSaldo, llImprime, lnTipoImp, lnConsAtra, lnTotAtrasos, lnValor, ;
	  lnListaAtra, lnNrPrev

ldDtInicial = ThisForm.get_dt_inicial.Value
ldDtFinal   = ThisForm.get_dt_final.Value
lcCdMoeda   = ThisForm.get_cd_moeda.Value
lcDsMoeda   = ThisForm.get_ds_moeda.Value
lnVlCotacao = ThisForm.get_vl_cotacao.Value
lnSaldos    = ThisForm.OptSaldo.Value
llImprime   = (ThisForm.Check1.Value = 1)
llConc      = (ThisForm.Conc.Value = 1)
lnOpcao     = ThisForm.Opt_Opcao.Value
lcMoeLanc   = ThisForm.Get_Cd_Moedl.Value
lnTipoImp   = ThisForm.Opt_tipo.Value 
lnConsAtra  = ThisForm.Opt_Consid_Atra.Value 
lnNrSitua   = ThisForm.opt_situa.Value
lnListaAtra = ThisForm.Opt_Lista_Atra.Value 
lnNrPrev    = ThisForm.OptPrev.Value

ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'TmpEmp', 'Cemps', _Empr, 'Razas')
lcNmEmpresa = Iif(Eof('TmpEmp'), '', TmpEmp.Razas)
lcNmTitulo  = 'Fluxo Financeiro' + Iif(llConc, ' - (Conciliado)', '')
lcNmPeriodo = 'Periodo: ' + Dtoc(ldDtInicial) + ' ŕ ' + Dtoc(ldDtFinal)
lcCbEmpresa = ''
lcCbMoeda   = 'Moeda: ' + Alltrim(lcCdMoeda) + ' - ' + Proper(Alltrim(lcDsMoeda)) + ' / ' + ;
			  'Cotaçăo: ' + Alltrim(Transform(lnVlCotacao, '999,999.99'))
lcCbMoeRef  = lcCdMoeda

If (lnNrSitua <> 3)
	lcCbEmpresa = lcCbEmpresa + '   Situaçăo: ' + Iif(lnNrSitua=1, 'Baixados', 'Pendentes')
EndIf

CREATE CURSOR crPeriodos(dFinal d(8))

Create Cursor csCabecalho (nm_empresa c(80), nm_titulo c(80), nm_periodo c(80), cb_empresa c(80), ;
						   cb_moeda c(80), cb_moe_ref c(3), ConsidAtra n(1))
Insert Into csCabecalho (nm_empresa, nm_titulo, nm_periodo, cb_empresa, cb_moeda, cb_moe_ref, ConsidAtra) ;
				 Values (lcNmEmpresa, lcNmTitulo, lcNmPeriodo, lcCbEmpresa, lcCbMoeda, lcCbMoeRef, lnConsAtra)

Create Cursor csRelatorio (Grupos c(10), Contas c(10), Saldos n(12,2), Moedas c(3), vl_Convertido n(12,2), ;
						   vl_Cotacao n(9,2), Vencs c(10), Pags n(12,2), Recs n(12,2), do_Dia n(12,2), ;
						   Acumulado n(12,2), Saldo n(12,2), Fase n(1,0), DsContas c(50), PrevPags n(12,2),;
						   PrevRecs n(12,2))

** Fluxo Semanal e Quinzenal
If( lnTipoImp = 2 Or lnTipoImp = 3 )
	ldDtAux = ldDtInicial
	If( lnTipoImp == 2 ) && Semanal
		DO WHILE ldDtAux <= ldDtFinal
			INSERT INTO crPeriodos Values(ThisForm.GoDayOfWeek(ldDtAux,7))
			ldDtAux = ldDtAux + 7
		ENDDO	
	ELSE
		ldDtAux = IIF( DAY(ldDtInicial) > 15, GOMONTH(ldDtInicial,1)-DAY(ldDtInicial), CTOD('15'+RIGHT(DTOC(ldDtInicial),8)) )
		DO WHILE ldDtAux <= ldDtFinal
			INSERT INTO crPeriodos Values(ldDtAux)
			ldDtAux = ldDtAux + IIF( DAY(ldDtAux) == 1, 16, 1 )
			ldDtAux = IIF( DAY(ldDtAux) > 15, GOMONTH(ldDtAux,1)-DAY(ldDtAux), CTOD('15'+RIGHT(DTOC(ldDtAux),8)) )
		ENDDO
	EndIf
	If( crPeriodos.dFinal < ldDtFinal )
		INSERT INTO crPeriodos Values(ldDtFinal)
	Endif
	If( crPeriodos.dFinal > ldDtFinal )
		Replace dFinal WITH ldDtFinal IN crPeriodos
	ENDIF	 
Endif	

If llImprime
	Index On Str(Fase, 1) + Contas Tag Fase
Else
	Index On Str(Fase, 1) + Grupos + Contas Tag Fase
EndIf

pKey    = fUniqueIds()
llPrevs = .F.

loBarra = CreateObject('fwprogressbar','Atualizando Arquivo Temporário...',Reccount('crSigOpOpe'))
loBarra.Show

Select crSigOpOpe
Go Top
Scan
	loBarra.Update(.T.)
	If crSigOpOpe.SeleOp
		pDop = crSigOpOpe.Dopes
		pRea = crSigOpOpe.RealPrevs
		pIds = fUniqueIds()
		If crSigOpOpe.RealPrevs = 2
			llPrevs = .T.
		EndIf
		lcQuery = [Insert Into SigTempR (Dopes, ValPres, cIdQuerys, cIdChaves) ] + ;
								[Values (?pDop, ?pRea, ?pKey, ?pIds)]

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, '') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Update - SigTempR)')
			Return 0
		EndIf
	EndIf
EndScan
loBarra.Complete(.T.)

Wait Window 'Atualizando Fisicamente o Arquivo Temporário...' NoWait

If (ThisForm.poDataMgr.Commit() < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Init - Commit)')
	Return 0
EndIf

oTotal = CreateObject('fwprogressbar', 'Processando Fluxo Financeiro...', 6)
oTotal.Top = oTotal.Top - Int(oTotal.Height / 2)
oTotal.Show

lcQuery = [Select a.Grupos, a.Contas, a.Moedas, b.cContabs ] + ;
		    [From SigMvSlc a Inner Join SigCdCli b on b.iClis = a.Contas ] + ;
			Iif(Not Empty(lcMoelanc),[ Where a.Moedas = ?lcMoeLanc ],[])+;
		    [Order By a.Grupos, a.Contas, a.Moedas]

If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crSigMvSlc') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigMvSlc)')
	Return 0
EndIf

Select a.Grupos, a.Contas, 000000000.00 as Saldos, a.Moedas, 1 as Fase, cContabs, ;
	   a.Grupos + a.Contas + a.Moedas as GruConMoes ;
  From crSigMvSlc a, cDisp b ;
 Where b.Grupos + b.Contas = a.Grupos + a.Contas And ;
 	   b.ImpDisps ;
 Order By a.Grupos, a.Contas ;
  Into Cursor csTemporario ReadWrite

Select csTemporario
Go Top
Scan

	lnSaldo = 0

	pGcm = csTemporario.GruConMoes
	pDat = fDtoSQL(ldDtInicial-Iif(lnSaldos=1,1,0), '23:59:59')

	lcQuery = [Select Top 1 Nfs, SaldoCs, Saldos, cIdChaves ] + ;
				[From SigMvCcr ] + ;
			   [Where GruConMoes = ?pGcm And ] + ;
				Iif(Not Empty(lcMoelanc),[ Moedas = ?lcMoeLanc And ],[])+;
			   		 [Datas <= ?pDat ] + ;
			   [Order By GruConMoes Asc, cIdChaves Desc]

	If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crSigMvCcr') < 1)
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigMvCcr)')
		Return 0
	EndIf

	Select crSigMvCcr
	Go Top
	If Not Eof('crSigMvCcr') and Iif(lnOpcao <> 3,Iif(lnOpcao = 1,!Empty(csTemporario.cContabs),Empty(csTemporario.cContabs)),.T.)
		lnSaldo = Iif(llConc, crSigMvCcr.Saldocs, crSigMvCcr.Saldos)
		Replace Saldos With lnSaldo In csTemporario
	EndIf
EndScan

Select csTemporario
Delete From csTemporario Where Saldos = 0
Go Top
	
lnVlSaldoTotal = 0
Select csTemporario
Go Top
oProg = CreateObject('fwprogressbar', 'Processando Saldos...', Reccount())
oProg.Titulo.FontBold = .t.
oProg.Top = oTotal.Height + oTotal.Top
oProg.Show
Scan
	oProg.Update( .t. )
	Scatter MemVar
	Insert Into csRelatorio From MemVar

	lnCtReferencia = 0
	=fBuscarCambio({}, Date(), Space(03), csRelatorio.Moedas)
	lnCtReferencia = Iif(lnCtReferencia=0, 1, lnCtReferencia)
	lnVlConvertido = (csRelatorio.Saldos * lnCtReferencia / lnVlCotacao)
	Replace vl_Convertido With lnVlConvertido, ;
		    vl_Cotacao    With lnVlCotacao In csRelatorio
	lnVlSaldoTotal = lnVlSaldoTotal + lnVlConvertido
EndScan
oProg.Complete

oTotal.Update(.t.)

Use In csTemporario

pDtI = fDtoSQL(ldDtInicial)
pDtF = fDtoSQL(ldDtFinal,'23:59:59')

lnTotAtrasos = 0
If( lnListaAtra == 1 ) 
	** Processamento dos atrasos
	lcQuery = [Select a.Grupos, a.Emps, a.DopCs, a.Opers, a.Moedas, a.Valors, a.ValPags, a.Nfs ] + ;
				[From SigMvCcr a, SigTempR b ] + ;
				[Where a.Vencs < ?pDtI ] + ;
				Iif( Not Empty(lcMoeLanc),[ And a.Moedas = ?lcMoeLanc ],[]) + [ And a.TitCancs <> 1 ] + ;
				Iif(lnNrPrev = 1,[And b.ValPres = 2], Iif(lnNrPrev = 2,[And b.ValPres <> 2],'')) + ;

	IF( lnTipoImp == 2 or lnTipoImp == 3 )
		lcCursor = [Create Cursor crRel( Tipo c(10), ]
		FOR i = 1 TO 8 
			lcData     = [dData]+ALLTRIM(STR(i))+[ c(10)]
			lcSaldoIni = [nSaldoIni]+ALLTRIM(STR(i))+[ n(12,2)]
			lcRecEst   = [nRecEst]+ALLTRIM(STR(i))+[ n(12,2)]
			lcRecConf  = [nRecConf]+ALLTRIM(STR(i))+[ n(12,2)]
			lcPagEst   = [nPagEst]+ALLTRIM(STR(i))+[ n(12,2)]
			lcPagConf  = [nPagConf]+ALLTRIM(STR(i))+[ n(12,2)]
			lcSaldoFin = [nSaldoFin]+ALLTRIM(STR(i))+[ n(12,2)]
			lcCursor = lcCursor + lcData + [,] + lcSaldoIni + [,] + lcRecEst + [,] + lcRecConf + [,] + lcPagEst + [,] + ;
								  lcPagConf  + [,] + lcSaldoFin + IIF(i <> 8,[,],[)])
		NEXT i
		&lcCursor
		lnIdx = 1
		SELECT * FROM csRelatorio WHERE Fase = 2 AND NOT EMPTY(Vencs) INTO CURSOR crAux
		SELECT crAux
		APPEND BLANK IN crRel
		Replace Tipo WITH IIF(lnTipoImp == 2,'Semana','Quinzena') IN crRel
		GO TOP IN crAux
		SCAN
			lcData     = [dData]+ALLTRIM(STR(lnIdx))
			lcSaldoIni = [nSaldoIni]+ALLTRIM(STR(lnIdx))
			lcRecEst   = [nRecEst]+ALLTRIM(STR(lnIdx))
			lcRecConf  = [nRecConf]+ALLTRIM(STR(lnIdx))
			lcPagEst   = [nPagEst]+ALLTRIM(STR(lnIdx))
			lcPagConf  = [nPagConf]+ALLTRIM(STR(lnIdx))
			lcSaldoFin = [nSaldoFin]+ALLTRIM(STR(lnIdx))
			Replace &lcData		WITH crAux.Vencs,;
					&lcSaldoIni WITH (crAux.Saldo-crAux.do_Dia),;
					&lcRecEst	WITH crAux.PrevRecs,;
					&lcRecConf  WITH crAux.Recs,;
					&lcPagEst	WITH crAux.PrevPags,;
					&lcPagConf	WITH crAux.Pags,;
					&lcSaldoFin WITH crAux.Saldo IN crRel					
			lnIdx = lnIdx + 1			
		ENDSCAN
	Endif			
Else
	Select Vencs, Max(Grupos) as Grupos, Contas, Opers, Moedas, Sum(Valors) as Valors ;
	  From csTemporariox ;
	 Group By 1, 3, 4, 5 ;
	  Into Cursor csTemporario

	Select Max(Vencs) as Vencs, Grupos, Contas, Opers, Max(Moedas) as Moedas, Sum(Valors) as Valors ;
	  From csTemporario ;
	 Group By 2, 3, 4 ;
	 Order By 3 ;
	  Into Cursor csTemporariox

	lnSaldo = lnVlSaldoTotal

	oProg = CreateObject('fwprogressbar', 'Acumulando Histórico...', RecCount('csTemporariox'))
	oProg.Titulo.FontBold = .t.
	oProg.Top = oTotal.Height + oTotal.Top
	oProg.Show
	Scan
		oProg.Update(.t.)
		
		lnPagamentos   = 0
		lnRecebimentos = 0
		lnCtReferencia = 0

		=fBuscarCambio({}, Date(), Space(03), csTemporariox.Moedas)
		lnCtReferencia = Iif(lnCtReferencia=0, 1, lnCtReferencia)
		lnVlConvertido = (csTemporariox.Valors * lnCtReferencia / lnVlCotacao)
		
		ThisForm.poDataMgr.CursorQuery('SigCdCli', 'crSigCdCli', 'IClis', csTemporariox.Contas, 'RClis')
		
		If Not Empty(csTemporariox.Grupos) And ;
		   Seek(csTemporariox.Grupos, 'crSigCdPag', 'Grupos') And ;
		   (csTemporariox.Opers = 'C') And (crSigCdPag.RelPrevs <> 2)
			lnPagamentos = lnVlConvertido
		EndIf

		If Not Empty(csTemporariox.Grupos) And ;
		   Seek(csTemporariox.Grupos, 'crSigCdRec', 'Grupos') And ;
		   (csTemporariox.Opers = 'D')
			lnRecebimentos = lnVlConvertido
		EndIf

		lnSaldo = lnSaldo + (lnRecebimentos - lnPagamentos)

		Insert Into csRelatorio(Contas, Pags, Recs, Saldo, Fase, DsContas) ;
						Values (csTemporariox.Contas, lnPagamentos, lnRecebimentos, lnSaldo, 2, crSigCdCli.RClis)
	EndScan
EndIf

Use In csTemporario
Use In csTemporariox

IF( lnTipoImp == 2 or lnTipoImp == 3 )
	SELECT crRel
Else
	Select csRelatorio
	Set Order To Fase
Endif	
Go Top
* Fim **************************************************************************************

ENDPROC

	ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'TmpSigCdEmp', .f., .f., 'Cemps, Razas')

	  From TmpSigCdEmp ;
	 Order By 2 ;

	Index On Cemps Tag Cemps

	* Apanhando as Disponibilidades Para Mostrar na Grade
	ThisForm.poDataMgr.CursorQuery('SigCdDis', 'TmpSigCdDis', .f., .f., 'Grupos, Contas, Tipos')

	Select .t. as ImpDisps, Grupos, Contas ;
	  From TmpSigCdDis Where Tipos in (0,1) ;
	 Order By 2 ;
	  Into Cursor cDisp ReadWrite

	Select cDisp
	Index On Grupos + Contas Tag Grupos
	Index On TransForm(ImpDisps, '@!') + Grupos + Contas Tag ImpDisps
	Set Order To Grupos

	ThisForm.poDataMgr.CursorQuery('SigCdPag', 'crSigCdPag', .f., .f., 'Grupos, RelPrevs')
	Select crSigCdPag
	Index On Grupos Tag Grupos

	ThisForm.poDataMgr.CursorQuery('SigCdRec', 'crSigCdRec', .f., .f., 'Grupos')

	Index On Grupos Tag Grupos

	Select crSigCdMoe
	Index On CMoes Tag CMoes

	ThisForm.poDataMgr.CursorQuery('SigCdCot', 'crSigCdCot', .f., .f., 'CMoes, Datas, Valos')

	Index On CMoes + Dtos(Datas) Tag CMoeData

	.get_dt_final.Value   = Date()
	.Get_Cd_Moeda.Value   = crSigCdPam.MoedaPs

	.Get_Ds_Moeda.Value   = TmpMoe.Dmoes
	Use In TmpMoe
	.Get_Vl_Cotacao.Value = fBuscarCotacao(crSigCdPam.MoedaPs, Date(), ThisForm.poDataMgr)

	With .Grid1
		.RecordSource = 'cEmp'
		.Column1.ControlSource = 'cEmp.ImpEmps'
		.Column2.ControlSource = 'cEmp.Cemps'
		.Column3.ControlSource = 'cEmp.Razas'

		For I=2 To .ColumnCount
			lcCol = Alltrim(Str(I))
			.Column&lcCol..Movable   = .f.
			.Column&lcCol..Resizable = .f.
			.Column&lcCol..ReadOnly  = .t.

	EndWith

	With .Grid2

		.Column1.ControlSource = 'cDisp.ImpDisps'
		.Column2.ControlSource = 'cDisp.Grupos'
		.Column3.ControlSource = 'cDisp.Contas'

		For I=2 To .ColumnCount
			lcCol = Alltrim(Str(I))
			.Column&lcCol..Movable   = .f.
			.Column&lcCol..Resizable = .f.
			.Column&lcCol..ReadOnly  = .t.
		Next
	EndWith
	
	With .Grd_Oper
		.RecordSource = 'crSigOpOpe'
		.Column1.ControlSource = 'crSigOpOpe.SeleOp'
		.Column2.ControlSource = 'crSigOpOpe.Dopes'

			lcCol = Alltrim(Str(I))
			.Column&lcCol..Movable   = .f.
			.Column&lcCol..Resizable = .f.
			.Column&lcCol..ReadOnly  = .t.

	Else

	EndIf
