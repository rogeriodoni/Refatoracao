PROCEDURE confirma
Lparameters pFoc
Private pcselbarras

Local Array laFiltro1(1,2)

Local llRetorno,lcMsg1,llChkUsuario,llChkEmpresa,lnErro1,lcMcrSql1,lnDock,lnTmpChk,;
		lnTmpApp,lnNbotao1,lcMacro1,lcOpSis1,lcOpSis2,lcTransSis1,loDatamgr1,;
		lcDirTmp1,lcArqIcone1,lcExtIcone1,lnTamIcone1, ldUltData1, ldUltData2, lnDias, llSaida1

m.llRetorno = .t.
m.lcMsg1 = ''
m.lcMacro1 = ''
m.llSupervis = .f.
m.lnErro1 = Iif(ThisForm.poDatamgr.pnIdConn>0,1,-1)
m.lcDirTmp1 = Addbs(Sys(2023))
m.llSaida1 = .f.

With Thisform
	m.llChkUsuario = .plUsuarioOk
	m.llChkEmpresa = !.get_emp.visible
	If (Usuar = [4CONTROL]) And Empty(Alltrim(crsigcdusu1.senhas))
		=MessageBox([Senha Inválida!!!], 0+16, [Violaçăo de Senha])
		Return .f.
	EndIf

	If Used('crsigcdusu1')
		Select ('crsigcdusu1')
		Set Order To ('') In ('crsigcdusu1')
		Locate For (crsigcdusu1.usuarios==.usuario.Value)
		m.llChkUsuario = (m.llChkUsuario.Or.(.Not.Empty(.usuario.Value).And..Not.Eof('crsigcdusu1').And..senha.Value==fCriptografar(crsigcdusu1.senhas)))
		m.llSupervis = (Usuar = [4CONTROL])
	EndIf

	If Used('crEmpresa1')
		Select ('crEmpresa1')
		Set Order To ('') In ('crEmpresa1')
		Locate For (crEmpresa1.cemps==.get_emp.Value)
		m.llChkEmpresa = (m.llChkEmpresa.Or.(.Not.Empty(.get_emp.Value).And..Not.Eof('crEmpresa1')))
	EndIf

	If (goSistema.NumLicencas <> 9999) And (Usuar <> [4CONTROL])
		lcQuery = [If Exists(Select 1 From SigSyCtl where LIC_Stat = 'A' and Left(LIC_Usua,10) = '] + Padr(Usuar, 10) + [' and Lic_Maqs = ']+Sys(0)+[') ] + ;
						[select 1 as LicUsos else ] + ;
						[Select Count(1) as LicUsos From SigSyCtl Where LIC_Stat = 'A' ]
		If (ThisForm.poDataMgr.SqlExecute(lcQuery, [LocClsTmp]) < 1)
			=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (LocClsTmp)])
			Return 0
		EndIf
		If Used([LocClsTmp])
			Go Top In LocClsTmp
			If Not Eof([LocClsTmp])  
				goSistema.UsoLicencas = LocClsTmp.LicUsos 
				Use In LocClsTmp
			EndIf
		EndIf
	EndIf
	
	Do Case
		Case Not m.llChkUsuario
			m.lcMsg1 = 'Usuário / Senha Inválidos.'
			.senha.Value = Space(10)
			.get_emp.Value = Space(3)
			.get_demp.Value = Space(40)
			If pFoc
				.usuario.SetFocus
			EndIf
			
		Case Not m.llChkEmpresa
			m.lcMsg1 = 'Empresa Inválida.'
			.get_emp.Value = Space(3)
			.get_demp.Value = Space(40)
			If pFoc
				.get_emp.SetFocus
			EndIf
		
		Case (goSistema.UsoLicencas >= goSistema.NumLicencas) And (Usuar <> [4CONTROL])
			 =MessageBox([A Quantidade de Licenças de Uso do Sistema Foram Excedidas.] + Chr(13) + ;
			 			 [Aguarde a Liberaçăo das Licenças em Uso e Tente Novamente.] + Chr(13) + ;
			 			 [A Quantidade Máxima de Usuários Simultâneos é ] + Alltrim(Str(goSistema.NumLicencas)), 0+64, ;
			 			 [Checagem de Licenças em Utilizaçăo])
	 		 ThisForm.Saida.Cancela.Click()
			
		OtherWise
			Public _tipoe

			.Enabled = .f.
			.Visible = .f.
			m.gcEmpr = .get_emp.Value
			m.lnDock = 0
			m.lnTmpChk = .f.
			m.lnTmpApp = .f.
			m.lnNbotao1 = 0
			_tipoe = crEmpresa1.tipos
			_empr = m.gcEmpr

			If Used('crParam1') And Used('crParam2')
				Select ('crParam1')
				Set Order To ('') In ('crParam1')
				Locate

				Select ('crParam2')
				Set Order To ('') In ('crParam2')
				Locate

				m.lnDock = Iif(BetWeen(crParam1.ndocks,0,4),crParam1.ndocks,0)
				lcArq = Sys(5)+Sys(2003)+'\SetFiscal.Ini'
				If File(lcArq)
					m.lnTmpChk = 0
					m.lnTmpApp = 0
				Else
					m.lnTmpChk = crParam1.ntmpchks
					m.lnTmpApp = crParam1.ntmpapps
				Endif
			EndIf

			m.lcMcrSql1 = [Select Distinct a.Dopes, a.Menus, a.SubMenus, a.CCentrals, a.NDopes, a.Ordes, a.EmpLancs, b.Opeinatvs, d.ConfigGers ] + ;
							[From SigCdOpe a ] + ;
					   [Left Join SigOpCdc b On a.Dopes = b.Dopes ] + ;
					   [Left Join SigOpCdd d On a.Dopes = d.Dopes ]
			
			If m.lnErro1>0
				m.lnErro1 = .poDatamgr.SQLExecute(m.lcMcrSql1, 'crTmpOpe')
			EndIf

			Select crTmpOpe
			Index On Str(nDopes, 10, 0) Tag nDopes
			Update crTmpOpe Set ConfigGers = Nvl(ConfigGers, [])

			m.lnDock = (m.lnDock-1)
			goSistema.BarraAtalho = CreateObject('fwBarraBtns')
			goSistema.BarraAtalho.Dock(m.lnDock)
			m.pcselbarras = .t.
			m.lcMcrSql1 = ([select a.programas,a.parametros,a.barraforms,b.descricaos,b.barrapict from SigCdAcB a]+;
			               [ left join sigcdprg b on b.programas+b.parametros=a.programas+a.parametros]+;
			               [ where a.usuarios=']+m.gcUsuar+[' and a.selbarras=?m.pcselbarras]+;
			               [ union all select a.programas,a.parametros,a.barraforms,b.descricaos,b.barrapict from SigCdAcB a]+;
			               [ left join sigcdprg b on b.programas+b.parametros=a.programas+a.parametros]+;
			               [ where a.selbarras=?m.pcselbarras and a.grupos in (select c.grupos from SigCdAcG c where c.usuarios=']+m.gcUsuar+[')])
			If Used('crTmpAcBarra1')
				Use In ('crTmpAcBarra1')
			EndIf
			If m.lnErro1>0
				m.lnErro1 = .poDatamgr.SQLExecute(m.lcMcrSql1,'crTmpAcBarra1')
			EndIf
			If Used('crTmpAcBarra1')
				If Used('crAcBarra1')
					Use In ('crAcBarra1')
				EndIf
				Select Distinct a.programas, a.parametros, a.barraforms, ;
					   PadR(Nvl(a.descricaos,''),100) as descricaos, ;
					   PadR(Nvl(a.barrapict,''),50) as barrapict;
				  From crTmpAcBarra1 a ;
				 Order By 1, 2 ;
				  Into Cursor crAcBarra1
			EndIf
			
			If Used('crAcBarra1')
				Select crAcBarra1
				Set Order To ('') In crAcBarra1
				Locate
				
				Scan While m.lnErro1>0
					If Not Seek(Str(Val(CrAcBarra1.Programas), 10, 0), 'crTmpOpe', 'ndopes') Or ;
						Not Nvl(crTmpOpe.Opeinatvs, 0)==1

						m.lcArqIcone1 = ''
						m.lnTamIcone1 = 0
						If Left(CrAcBarra1.BarraForms,8)='SIGMVEXP'
							If Left(CrAcBarra1.programas,2)<>'GE' And ;
								Seek(Str(Val(CrAcBarra1.Programas), 10, 0), 'crTmpOpe', 'ndopes') And ;
								Not IsEmpty(crTmpOpe.EmpLancs) And crTmpOpe.EmpLancs <> _Empr
								Loop

							EndIf
						Endif
						If Not IsEmpty(crAcBarra1.barrapict)
							m.lcExtIcone1 = JustExt(Alltrim(crAcBarra1.barrapict))
							
							If Empty(m.lcExtIcone1)
								m.lcExtIcone1 = 'ico'
							EndIf 
							
							m.lcMcrSql1 = ([select a.marqicones from SigSyIco a where a.carqicones=']+AllTrim(crAcBarra1.barrapict)+['])
							If m.lnErro1>0
								m.lnErro1 = .poDatamgr.SQLExecute(m.lcMcrSql1,'crTmpIcone1')
							EndIf
							Select ('crTmpIcone1')
							Set Order To ('') In ('crTmpIcone1')
							Locate 

							If (Not Eof('crTmpIcone1') And Not IsEmpty(crTmpIcone1.marqicones))
								m.lcArqIcone1 = (m.lcDirTmp1+goSistema.Sys2015Tmp+Sys(2015)+'.'+m.lcExtIcone1)
								m.lnTamIcone1 = StrToFile(crTmpIcone1.marqicones,m.lcArqIcone1)
								If m.lnTamIcone1==0
									m.lcArqIcone1 = ''
								EndIf 
							EndIf 
						EndIf 
						m.lnNbotao1 = (m.lnNbotao1+1)
						m.lcMacro1 = 'botao'+AllTrim(Str(m.lnNbotao1,20,0))
						goSistema.BarraAtalho.AddObject(m.lcMacro1,'botăo_da_barra')
						goSistema.BarraAtalho.&lcMacro1..ToolTiptext = AllTrim(crAcBarra1.descricaos)
						goSistema.BarraAtalho.&lcMacro1..Picture = m.lcArqIcone1
						goSistema.BarraAtalho.&lcMacro1..FormExec = AllTrim(crAcBarra1.barraforms)
						goSistema.BarraAtalho.&lcMacro1..Visible = .t.
					EndIf 
				EndScan
			EndIf
			
			goSistema.BarraAtalho.Visible = (m.lnNbotao1>0)

			If .plGetUsuario
				If .Not.Empty(.pcUsuarioAtual)
					m.lcOpSis1 = 'S'
					m.lcOpSis2 = 'USUARIO'
					m.lcTransSis1 = 'Saida do Sistema'
				Else
					.pcUsuarioAtual = m.gcUsuar
					m.lcOpSis1 = 'E'
					m.lcOpSis2 = 'USUARIO'
					m.lcTransSis1 = (Proper(AllTrim(ThisForm.Caption))+' - '+m.gcEmpr)
				EndIf
			Else
				If Empty( .pcUsuarioAtual )
					.pcUsuarioAtual = m.gcUsuar
				EndIf
				m.lcOpSis1 = 'T'
				m.lcOpSis2 = 'EMPRESA'
				m.lcTransSis1 = ('Muda Empresa - '+m.gcEmpr)
			EndIf
			If Not goSistema.Desenvolvimento
				= fTempoApp(m.lnTmpChk, m.lnTmpApp)
			EndIf
			m.lcMcrSql1 = ([select a.comiss,a.PPLinha1s,a.PPLinha2s,a.nEmps from SigCdEmp a where a.cemps=']+m.gcEmpr+['])
			If Used('SigCdEmp')
				Use In ('SigCdEmp')
			EndIf
			If m.lnErro1>0
				m.lnErro1 = .poDatamgr.SQLExecute(m.lcMcrSql1,'SigCdEmp')
			EndIf
			If Used('SigCdEmp')
				Select ('SigCdEmp')
				Locate 
			EndIf 
			m.lcMcrSql1 = [select a.desmontas,a.habembs,a.habcors,a.habtams,a.habtifs,a.SisTef,a.NumIps from SigCdPam a]
			If Used('SigCdPam')
				Use In ('SigCdPam')
			EndIf
			If m.lnErro1>0
				m.lnErro1 = .poDatamgr.SQLExecute(m.lcMcrSql1,'SigCdPam')
			EndIf
			
			Select a.* From crTmpOpe a;
				Where Not a.Menus = Space(10);
				Order By a.Menus, a.SubMenus, a.Ordes, a.EmpLancs;
				Into Cursor SigCdOpe ReadWrite

			m.lcMcrSql1 = [Select * From SigSyCcb Order By Tits]
			If Used('SigSyCcb')
				Use In ('SigSyCcb')
			EndIf
			If m.lnErro1>0
				m.lnErro1 = .poDatamgr.SQLExecute(m.lcMcrSql1,'SigSyCcb')
			EndIf
			
			m.lcMcrSql1 = [Select Distinct Dopps, Menus, Pccs, Ordes, NDopps, Emplancs ] + ;
							[From SigCdOpd ] + ;
						   [Where Menus = 1 ] + ;
						   [Order By Menus, Ordes]

			If Used('SigCdOpd')
				Use In ('SigCdOpd')
			EndIf
			If m.lnErro1>0
				m.lnErro1 = .poDatamgr.SQLExecute(m.lcMcrSql1,'SigCdOpd')
			EndIf

			m.lcMcrSql1 = [Select Distinct NGopes, DGopes, '   ' as EmpLancs ] + ;
							[From SigCdGpe ] + ;
						   [Order By NGopes, DGopes]

			If Used('SigCdGpe')
				Use In ('SigCdGpe')
			EndIf
			If m.lnErro1>0
				m.lnErro1 = .poDatamgr.SQLExecute(m.lcMcrSql1,'SigCdGpe')
			EndIf

			m.lcMcrSql1 = [Select Distinct NDopes, Dopes ] + ;
							[From SigOpOpe ] + ;
						   [Order By NDopes, Dopes]

			If Used('SigOpOpe')
				Use In ('SigOpOpe')
			EndIf
			If m.lnErro1>0
				m.lnErro1 = .poDatamgr.SQLExecute(m.lcMcrSql1,'SigOpOpe')
			EndIf
			Select ('SigCdOpd')
			Index On dopps Tag dopps
			Index On ordes Tag ordes
			Select ('SigCdOpe')
			Index On Menus + CCentrals + SubMenus + Str(ordes,3) Tag submenus
			Select ('SigCdGpe')
			Index On dgopes Tag dgopes
			Select ('SigOpOpe')
			Index On dopes Tag dopes
			Select ('SigSyCcb')
			Index On Tits Tag Tits
			
			m.ldUltData1 = Ttod(Nvl(crsigcdusu1.tDtSenhas,Ctot('01/01/1900')))
			m.ldUltData2 = Ttod(.poDatamgr.GetDateTime())
			m.lnDias = Nvl(crParam2.nDiaSenhas,0)
			m.Alertas = crParam2.Alertas			

			.Release

			goSistema.EmpresaLogada = _Empr
			goSistema.UsuarioLogado = goSistema.usuario
			goSistema.SenhaLogado   = Alltrim(crSigCdUsu1.Senhas)

			m.loLista.Show()
		EndIf
		ThisForm.get_emp.Value = Iif(LastKey()==27,Space(3),crEmpresa1.cemps)
		ThisForm.get_demp.Value = Iif(LastKey()==27,Space(40),crEmpresa1.razas)

		Try
			m.lcArqImage1 = (Addbs(Sys(2023)) + goSistema.Sys2015Tmp + Sys(2015) + '.img')
			*m.lcStrLogo1 = crEmpresa1.logos
			
			lcFoto=Strconv(Strtran(Strtran(Strtran(crEmpresa1.logos,[data:image/png;base64,],[]),[data:image/jpeg;base64,],[]),[data:image/jpg;base64,],[]),14)
			m.lcStrLogo1 = lcFoto
			
			If IsEmpty(crEmpresa1.logos)
				m.lcMcrSql1 = ('select a.logos from SigCdPam a')
				If m.lnErro1>0 
					m.lcMcrSql1 =  ThisForm.poDatamgr.SQLExecute(m.lcMcrSql1,'crTmpLogo1')
				EndIf 
				*m.lcStrLogo1 = crTmpLogo1.logos
				lcFoto=Strconv(Strtran(Strtran(Strtran(crTmpLogo1.logos,[data:image/png;base64,],[]),[data:image/jpeg;base64,],[]),[data:image/jpg;base64,],[]),14)
				m.lcStrLogo1 = lcFoto
			EndIf 
			
			If (Not IsEmpty(m.lcStrLogo1) And;
				StrToFile(m.lcStrLogo1,m.lcArqImage1)>0)
				_Screen.ImagemLocal.Picture = m.lcArqImage1
			EndIf 	
			_Screen.ImagemLocal.Visible = (Not Empty(_Screen.ImagemLocal.Picture))
		Catch
		
		EndTry 
	Else
		ThisForm.get_emp.Value = Space(3)
		ThisForm.get_demp.Value = Space(40)
	EndIf
	ThisForm.get_emp.Refresh
	ThisForm.get_demp.Refresh
	m.llRetorno = .t.
EndIf

If llRetorno And Not Empty(This.Value)
	ThisForm.Confirma(.f.)
EndIf

Return(m.llRetorno)

ENDPROC
