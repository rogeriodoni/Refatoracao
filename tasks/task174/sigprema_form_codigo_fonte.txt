PROCEDURE criapdf
Declare Integer Sleep In WIN32API Integer nMilliseconds
lPrintPDFCreator = .T.
lcDefaultPrinter = Set( "PRINTER", 2 )

Thisform.oPdf = Createobject("PDFCreator.clsPDFCreator", "PDFCreator_")

If Thisform.oPdf.cErrorDetail("Number") <> 0
	lPrintPDFCreator = .F.
	Messagebox( "Problemas para gerar o obj PDFCretor" +Chr(13) +;
		"ErrorNr: " + Str( Thisform.oPdf.cErrorDetail("Number"), 3 ) +Chr(13) +;
		"Descr: " 	+ Thisform.oPdf.cErrorDetail("Description"), 48, "Atençăo",ThisForm.ntempo )
Else
	Wait Window Chr(13) + [Aguarde... Gerando o Arquivo PDF] + Chr(13) Nowait Noclear

	With Thisform.oPdf
		.cClearCache()
		.cErrorClear()

		llPDFStart = .cStart("/NoProcessingAtStartup")
		If llPDFStart = .F.
			lPrintPDFCreator = .F.

			Messagebox( "Problemas no START do PDFCreator" +Chr(13) +;
				"ErrorNr: " + Str( Thisform.oPdf.cErrorDetail("Number"), 3 ) +Chr(13) +;
				"Descr: " 	+ Thisform.oPdf.cErrorDetail("Description"), 48, "Atençăo",ThisForm.ntempo )
		Else
			**			.cVisible = .F.
			**			.cOption("StartStandardProgram") = 0	&& é o responsável para escolher se o documento será aberto ou năo após gravaçăo automática.
			.cOption("DisableUpdateCheck")   = 1
			.cOption("UseAutosaveDirectory") = 1
			.cOption("UseAutosave") 		 = 1
			.cOption("UseStandardAuthor") 	 = 1
			.cOption("StandardAuthor") 		 = "FORTYUS"
			.cOption("StandardTitle") 		 = ""
			.cOption("AutosaveFormat") 	  	 = 0	&& 0=PDF, 1=PNG, 2=JPG, 3=BMP, 4=PCX, 5=TIFF, 6=PS, 7= EPS, 8=ASCII
			.cOption("AutosaveDirectory") 	 = Addbs( Justpath( Thisform.pcArqEmail) )
			.cOption("AutosaveFilename")  	 = Justfname( Thisform.pcArqEmail)
			.cDefaultPrinter = "PDFCreator"
			.cPrinterStop 	 = .F.
		Endif
	Endwith

	*** Mena - 08/05/2014 - seta pdf como impressora padrao para gerar arquivo ***
	lcPrn = "PDFCreator"
	Set Printer To Name (lcPrn)

	If llPDFStart = .T.
		For lnPDFCount = 1 To 200
			If Thisform.oPdf.cDefaultPrinter = "PDFCreator"
				Exit
			Endif
			Sleep(500)	&& Sleep IN WIN32API
		Next lnPDFCount

		If Thisform.oPdf.cDefaultPrinter # "PDFCreator"
			lPrintPDFCreator = .F.
			Messagebox( "Problemas para criar a Default-Printer" +Chr(13) +;
				"ErrorNr: " + Str( Thisform.oPdf.cErrorDetail("Number"), 3 ) +Chr(13) +;
				"Descr: " 	+ Thisform.oPdf.cErrorDetail("Description"), 48, "Atençăo",ThisForm.ntempo )
		Endif
	Endif
Endif

If lPrintPDFCreator = .T. And llPDFStart = .T.
	Thisform.Documento()

	Sleep( 500 )		&& Milliseconds

	lnMaxSleep  = 0
	nReadyState = 0		&& Esta var é usada dentro do -> PDF_Event()	&& C:\FrameWork\ReportPreview\PDF_Event.prg
	Do While nReadyState = 0 And lnMaxSleep <= 10
		Sleep( 1000 )		&& Milliseconds
		lnMaxSleep = lnMaxSleep +1
	Enddo
Endif

Thisform.oPdf.cDefaultPrinter = lcDefaultPrinter
Set Printer To Name (lcDefaultPrinter)

Clear Dlls "Sleep"

lnErrorNr = Thisform.oPdf.cErrorDetail("Number")
lnErrorNr = Iif( Type( "lnErrorNr" )=="N", lnErrorNr, Val( Transform( lnErrorNr ) ) )
If lnErrorNr # 0
	lPrintPDFCreator = .F.
	Messagebox( "Problemas para gerar o ARQ " + Chr(13) + Thisform.pcArqEmail + Chr(13) +;
		"ErrorNr: " + Str( lnErrorNr, 3 ) +Chr(13) +;
		"Descr: " 	+ Thisform.oPdf.cErrorDetail("Description"), 48, "Atençăo",ThisForm.ntempo )
Endif

Thisform.oPdf.cClearCache()
Thisform.oPdf.cClose()
Thisform.oPdf = Null
Wait Clear
ENDPROC

PROCEDURE memail
***************************************************
****** Envia email automatico             *********
***************************************************
LParameters tcFrom As String, tcTo As String, tcCC As String, tcAssunto As String, tcCorpo As String, tcAnexo As String, ;
			tcSmtpServer As String, tnSmtpServerPort As Integer, tcSendUserName As String, tcSendPassWord As String


Local loEmail As CDO.Message, llOk As Boolean, llEnvioOK As Boolean 

If Type( 'CreateObject("CDO.Message")' ) # "O"
	MessageBox( "Problemas para instanciar o obj CDO.Message", 48, "Atençăo" )
	Return( .F. )
EndIf

loEmail = CreateObject("CDO.Message")

With loEmail.Configuration.Fields
	.Item("http://schemas.microsoft.com/cdo/configuration/sendusing") 			  = 2
	.Item("http://schemas.microsoft.com/cdo/configuration/smtpserver") 			  = Lower( tcSmtpServer )
	.Item("http://schemas.microsoft.com/cdo/configuration/smtpconnectiontimeout") = 10
	.Item("http://schemas.microsoft.com/cdo/configuration/smtpserverport") 		  = Iif( tnSmtpServerPort = 0, 25, tnSmtpServerPort )
	.Item("http://schemas.microsoft.com/cdo/configuration/smtpauthenticate") 	  = 1
	.Item("http://schemas.microsoft.com/cdo/configuration/sendusername") 		  = Lower( tcSendUserName )
	.Item("http://schemas.microsoft.com/cdo/configuration/sendpassword") 		  = tcSendPassWord 
	.Item("http://schemas.microsoft.com/cdo/configuration/smtpusessl")	 		  = Iif( tnSmtpServerPort = 465, 1, 0 ) && mena-05/03/2014 incluido para funcionar com gmail - gmail a porta é 465
	.Update()
EndWith

llEnvioOK = .T.

With loEmail
	.To 	  = Lower( tcTo )
	.Cc 	  = Lower( Nvl( tcCC, "" ) )
	.From 	  = Lower( tcFrom )
	.Subject  = AllTrim( tcAssunto )
	.TextBody = AllTrim( tcCorpo )

	****adiciona anexos ****
	If !IsNull( tcAnexo ) And !Empty( tcAnexo )
		If File( tcAnexo )
			.AddAttachment(tcAnexo)
		Else
			llEnvioOK = .F.
			MessageBox( "Năo foi encontrado o arquivo:" +Chr(13)+ tcAnexo + " para ser anexado.", 48, "Atençăo",ThisForm.ntempo )
		EndIf
	EndIf

	Try
		If llEnvioOK = .T.
			.Send()
			llOk = .T.
		EndIf
	Catch
		Wait Window "Dados do e-mail inválidos." TimeOut 5
		llOk = .F.
	EndTry

EndWith

Release oEmail

Return llOk
ENDPROC

ThisForm.btnEmail.Enabled = ThisForm.Registry1.IsKey([PDFCreator.clsPDFCreator]) Or ThisForm.Registry1.IsKey([PDFCreatorBeta.JobQueue])

Thisform.grade.RecordSource=''

If Empty(prDopes)
	lcQuery = [Select 1 as Checks, a.EmpDopNums, a.Jobs, b.Rclis, b.Rclis, b.emails, b.grupos, b.iclis  ] + ;
			[From SigMvcab a ] + ;
			[Inner Join SigCdCli b On a.Contads = b.Iclis ] + ;
			[Where a.datatrans between ?pDti And ?pDtf ] + ;
			[And empdopnums not in(select Transacaos from sigoplog where progs = 'SIGPREMA') ] + ;
			[Order by a.EmpDopNums ]
Else
	lcQuery = [Select 1 as Checks, a.EmpDopNums, a.Jobs, b.Rclis, b.Rclis, b.emails, b.grupos, b.iclis  ] + ;
			[From SigMvcab a ] + ;
			[Inner Join SigCdCli b On a.Contads = b.Iclis ] + ;
			[Where a.EmpDopNums = '] + prDopes + [']
EndIf 
	
If thisform.podatamgr.SqlExecute(lcQuery,[TmpMvCab]) < 1
	=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (TmpMvCab)],ThisForm.ntempo)
	Return .F.
EndIf

*!*	Gravando os dados no cursor para envio dos e-mails
Select TmpMvCab
Go top	
Scan

	Insert Into crLocalTotal (Checks,grupos,Contas,Rclis,emails,prioridade,EmpDopNums) ;
					Values (TmpMvCab.Checks,TmpMvCab.grupos,TmpMvCab.iclis,TmpMvCab.Rclis,TmpMvCab.emails,[NORMAL],TmpMvCab.EmpDopNums)
					
Endscan

*!* Buscando as contas referente ao grupo parametrizado em SigCdPam (grpadats)
lcQuery	= [select 0 as Checks, c.grupos, c.iclis as Contas, c.rclis, c.emails, '' as prioridade ] + ;
	[from SigCdPam p ] + ;
	[inner  join SigCdCli c on c.grupos = p.grpadats]

If (Thisform.poDatamgr.SqlExecute(lcQuery, [crLocalPAM]) < 1)
	=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crLocal)],ThisForm.ntempo)
	Return .F.
Endif

*!*	Gravando os novos dados no cursor para envio dos e-mails
Select crLocalPAM
Scan
	lcQuery = [Select Jobs From SigClJob Where Iclis = '] + crLocalPAM.Contas + [']
	If (Thisform.poDatamgr.SqlExecute(lcQuery, [TmpClJob]) < 1)
		=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (TmpClJob)],ThisForm.ntempo)
		Return .F.
	Endif
	Select TmpClJob
	Go top
	If !Eof()
		Locate For Jobs = TmpMvCab.Jobs
		If Eof()
			Loop
		Endif 
	Endif
	
	lnChecks	= crLocalPAM.Checks
	lcgrupos	= [] 
	lcContas	= crLocalPAM.Contas
	lcRclis		= crLocalPAM.Rclis
	lcemails	= crLocalPAM.emails

	Select crLocalTotal
	Locate For Alltrim(crLocalTotal.Contas) = Alltrim(lcContas) And Alltrim(crLocalTotal.Rclis) = Alltrim(lcRclis)
	If Eof()
		Insert Into crLocalTotal (Checks,grupos,Contas,Rclis,emails,EmpDopNums,prioridade) ;
		Values (lnChecks,lcgrupos,lcContas,lcRclis,lcemails,prdopes,[NORMAL])
	Endif

	Select crLocalPAM
Endscan

Select crLocalTotal
Go Top

Thisform.grade.RecordSource='crLocalTotal'
Thisform.grade.column6.ControlSource = [crLocalTotal.Checks]
Thisform.grade.column2.ControlSource = [crLocalTotal.Contas]
Thisform.grade.column3.ControlSource = [crLocalTotal.Rclis]

Thisform.grade.column3.header1.Click() && Para ordenar por nome
Thisform.grade.Refresh()

Thisform.Refresh()

If ThisForm.Automatico
	ThisForm.btnEmail.Click()
	ThisForm.Release
	Return .f.
EndIf 
ENDPROC

------------------------------------------------------------
PROCEDURE Click
lcReceptor			= ''
lcReceptorCopia		= ''
lcAssunto			= ''
lcTxtMensagem		= ''
lcArqAnexo			= ''
lnImportancia		= 1
lcMsgRetorno		= ''
llOk = .t.

Select * From crLocaltotal Where Checks = 1 Into Cursor crLocalTotal2 ReadWrite

Select crLocaltotal2
Scan
	
	If IsEmpty(crLocaltotal2.emails)
		Loop
	EndIf 
	
	Select crLocaltotal2
	lcEmps = Substr(crLocaltotal2.EmpDopNums,1,3)
	lcDopes = Substr(crLocaltotal2.EmpDopNums,4,20)
	lcNumes = Val(Substr(crLocaltotal2.EmpDopNums,24,6))
	lcEdn = crLocaltotal2.EmpDopNums
	
	If Recno() = 1
		lcReceptor		= crLocaltotal2.emails
		lcTxtMensagem	= []
		lcTxtPrioridade	= Alltrim(crLocaltotal2.prioridade)
		lcAssunto		= []
	Else
		If !Empty(Alltrim(crLocaltotal2.emails))
			lcReceptorCopia	= lcReceptorCopia + Iif(Empty(lcReceptorCopia),[],[,]) + Alltrim(crLocaltotal2.emails)
		Endif
	Endif
	
	If File(Addbs(Sys(2023)) + [FORTYUSPDF.PDF])
		Delete File Addbs(Sys(2023)) + [FORTYUSPDF.PDF]
	EndIf 
	
	llOk = .t.
	If Not Thisform.ImpDocto(lcEdn)
		llOk = .F.
	EndIf 
			
	lcArqAnexo = ThisForm.pcarqemail
		
	** Busca dados de Email do cadastro da Empresa
	lcQuery = [Select PadEmails, PadServs, PadSenhas, pastas, PadPortas From SigCdEmp Where CEmps = '] + _Empr + [']
	If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'TmpEmpMail') < 1)
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpEmpMail)',ThisForm.ntempo)
		Return .f.
	Endif	

	lcFrom   = Lower(Alltrim(Nvl(TmpEmpMail.PadEmails,[])))	
	lcServer = Lower(Alltrim(Nvl(TmpEmpMail.PadServs,[])))
	lcSenha  = Alltrim(Nvl(TmpEmpMail.PadSenhas,[]))
	lnPorta  = Nvl(TmpEmpMail.PadPortas,0)

	Wait Window Chr(13)+ [Aguarde... gerando EMAIL] Nowait NoClear
		
	If llOk
		If !EnviaEmail(lcReceptor, lcTxtMensagem, lcAssunto, lcArqAnexo, lcFrom, lcReceptorCopia, lcServer, lcSenha, lnPorta)
			llOk = .f.
		Endif	
	EndIf 
	
	If llOk
		fGravarLog( 'T', Thisform.Name, [], lcEdn )
	EndIf 
	
Endscan

Wait Clear

If llOk
	Wait Window [Email enviado com sucesso!] Timeout 0.02
	thisform.Release()
EndIf

If Thisform.automatico 
	thisform.Release()
EndIf 
ENDPROC
