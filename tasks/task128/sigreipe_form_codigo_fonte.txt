PROCEDURE Init
Lparameters pCodigos

Private pCod

=fConfigGeral()

=DoDefault()

lcQryMala = [Select a.IClis, a.RClis, a.Endes, a.Nums, a.Compls, a.Bairs, a.Cidas, a.Ceps, a.Estas, ] + ;
				   [a.Tel1s, a.Tel2s, a.Faxs, a.Nascs, a.cIdChaves, a.Codigos, b.paises, b.Contato ] + ;
			  [From SigMlCab a, SigCdCli b ] + ;
			 [Where a.Codigos = ?pCod And ] + ;
			 	   [a.Iclis = b.Iclis ] + ;
		  	 [Order By a.IClis, a.RClis, a.Endes, a.Nums, a.Compls, a.Bairs, a.Cidas, a.Ceps, a.Estas, ] + ;
		  	 	   [a.Tel1s, a.Tel2s, a.Faxs, a.Nascs, a.cIdChaves, a.Codigos, b.Contato]

With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)
	With .poDataMgr
		.AddCursor('SigMlCab', 'cIdChaves', 'Relacao', '', '', lcQryMala)
	EndWith
EndWith

* Checando se Existe Algum Usuário Cadastrado Para Liberar o Acesso Ao Formulário
lcQuery = [Select pkChaves ] + ;
			[From SigCdAcU ] + ;
		   [Where Programas  = '] + Padr('SIGREIPE',  15) + [' And ] + ;
		   		 [Parametros = '] + Padr('LIBERAIMP',10) + [' ] + ;
		   [Order By pkChaves]

If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crSigCdAcU') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigCdAcU)')
	Return .f.
EndIf
Go Top In crSigCdAcU
If Not Eof([crSigCdAcU])
	Do Form SigSnSen With [], [SIGREIPE], [LIBERAIMP], [USUARIO] To loObj

	If (Type([loObj])=[O])
		If loObj.Acesso And Not Empty(loObj.Usuario)
			=fGravarLog([T], [MALADIRETA], Usuar, [Foi Liberada a Impressăo ] + ;
						  [de Etiquetas Pelo Usuário ] + loObj.Usuario)
		Else
			=fGravarLog([T], [MALADIRETA], Usuar, [Houve Uma Tentativa de Liberaçăo ] + ;
						  [de Impressăo Etiquetas Pelo Usuário ] + loObj.Usuario)
			Return .f.
		EndIf
		Release loObj
	EndIf
EndIf
* Fim da Checagem de Usuário Cadastrado Para Liberar o Acesso Ao Formulário

lnQueryOk = ThisForm.poDataMgr.SqlExecute('Select * From SigCdPam', 'crSigCdPam')
If (lnQueryOk < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo')
	Return 0
EndIf

If ! ThisForm.podatamgr.Cursorquery('SigCdPac','CrSigCdPac')
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo')
	Return 0
Endif

If Not Empty(pCod)
	ThisForm.Get_codigo.Value = pCod
	If ThisForm.Podatamgr.CursorQuery('SigMlItn', 'crTmp', 'Codigos', pCod)
		ThisForm.Get_desc.Value = Evaluate('crTmp.Descs')
		Use In crTmp
	EndIf
EndIf

ThisForm.opColunas.Value = 2


*** Tiago - 22/08/2012 - Muredu - Alteraçăo para imprimir etiqueta térmica de endereço
Dimension laPrinters(10,2)

lnPrinters = APrinters(laPrinters)
If (lnPrinters <> 0)
	For lnI = 1 To lnPrinters
		laPrinters(lnI, 1) = Upper(laPrinters(lnI, 1))
	EndFor
	=ASort(laPrinters)
EndIf

lcSql = [Select b.Impres ] + ;
		  [From SigSyImp a, SigCdmp b ] + ;
		 [Where a.UsuAcess = '] + Usuar + [' And ] + ;
		 	   [a.CImps = b.Impres And ] + ;
		 	   [b.nTpImpres = 2 ] + ;
		 [Union All ] + ;
		[Select c.Impres ] + ;
		  [From SigCdAcG a, SigSyImp b, SigCdmp c ] + ;
		 [Where a.Usuarios = '] + Usuar + [' And ] + ;
		       [a.Grupos = b.GrAcess And ] + ;
		       [b.CImps = c.Impres And ] + ;
		       [c.nTpImpres = 2]

If ThisForm.podatamgr.sqlexecute(lcSql,'crTmpCimp') <= 0
	Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados.',(0+48),'Erro')
	Return(.f.)
EndIf

Select crTmpCImp
If (Reccount([crTmpCImp]) > 0)
	Select Distinct Impres ;
	  From crTmpCImp ;
	 Order By Impres ;
	  Into Cursor crSigCdmp ReadWrite
Else
	lcQuery = [Select Distinct Impres ] + ;
			    [From SigCdmp ] + ;
			   [Where nTpImpres = 2 ] + ;
			   [Order By Impres]

	If (ThisForm.poDataMgr.SqlExecute(lcQuery, [crSigCdmp]) < 1)
		=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crSigCdmp)])
		Return .f.
	EndIf
EndIf
Select crSigCdmp
Go Top In crSigCdmp		

lnCnt = 0
lnImp = 1
If (lnPrinters <> 0)
	Select crSigCdmp
	Scan
		Dimension laImpOk(lnImp)
		lnOk = AScan(laPrinters, Alltrim(Upper(crSigCdmp.Impres)))
		If (lnOk <> 0)
			lnCnt = lnCnt + 1
			laImpOk(lnCnt) = laPrinters(lnOk)
			lnImp = lnImp + 1
		EndIf
	EndScan
EndIf
lnImp = lnImp - 1

With ThisForm.opt_Impressora
	lnTop = 10
	lnHeight = 15
	If (lnImp <> 0)
		.ButtonCount = Min(lnImp, Max(crSigCdPam.nMaxImpEti, 5))
		For lnI = 1 to .ButtonCount
			lcI = AllTrim(Str(lnI))
			With .Option&lcI.
				.AutoSize  = .f.
				.Caption   = ' \<' + lcI + '. ' + laImpOk(lnI)
				.FontSize  = 8
				.Forecolor = Rgb(36,84,155)
				.Tag       = Upper(laImpOk(lnI))
				.Top       = lnTop
				.Width     = 210
				.BackStyle = 0
			EndWith
			lnTop = lnTop + 20
			lnHeight = lnHeight + 20
		EndFor
	Else
		.ButtonCount = 1
		lnOk = AScan(laPrinters, 'GENÉRICO/SOMENTE TEXTO')
		With .Option1
			.AutoSize = .f.
			.Caption  = ' \<0. Genérico/Somente Texto'
			.Enabled  = (lnOk <> 0)
			.FontSize = 8
			.Forecolor = Rgb(36,84,155)
			.Tag      = 'GENÉRICO/SOMENTE TEXTO'
			.Width    = 210
			lnImp     = Iif(.Enabled, 1, 0)
		EndWith
		lnHeight = lnHeight + 20
	EndIf
	.Enabled = (lnImp > 1)
	.Height  = lnHeight
EndWith


ThisForm.Refresh

ENDPROC

------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Private pCod

pCod = Thisform.Get_Codigo.Value

Create Cursor CsCabecalho (llImpPais l, llImpCodigos l)
Append Blank
Replace llImpPais    With ( ThisForm.optpais.Value = 1 )
Replace llImpCodigos With ( ThisForm.optCodigo.Value = 1 ) 

If Empty(thisform.get_nm_operacao.Value)
	ThisForm.poDataMgr.Requery('Relacao')
Else
	Select TmpOper
	Scan
		If TmpOper.Marcas = 1 and !Empty(TmpOper.ContaEs) and TmpOper.LocalEnts <> 0
			*** Tiago - 03/06/2011 - Busca endereço de entrega se estiver preenchido
			Dimension laArray(2)
			laArray(1) = TmpOper.ContaEs
			laArray(2) = TmpOper.LocalEnts
						
			If fEndEntrega(@laArray, thisform.podatamgr)
				Select TmpOper			
				Scatter memo memvar			
				m.Ceps = laArray(04)
				m.Paises = laArray(05)
				m.Endes = laArray(06)
				m.Nums = laArray(07)
				m.Compls = laArray(08)
				m.Bairs = laArray(09)
				m.Cidas = laArray(10)
				m.Estas = laArray(11)				
				m.Contato = laArray(15)				
				Select TmpOper				
				Gather memo memvar				
			EndIf
		EndIf
	EndScan

	lcQuery = []
	For x = 1 to thisform.spn_qtdetiq.Value
		lcQuery = lcQuery  + Iif(!Empty(lcQuery),[;] + Chr(13)+Chr(10)+[Union All ;]+Chr(13)+Chr(10),[]) +;
				[Select * from TmpOper n] + Alltrim(Str(x,2,0)) + [ where marcas = 1 ] +;
				Iif(x = thisform.spn_qtdetiq.Value,[into cursor Relacao readwrite],[])
	Next
	
	=ExecScript(lcQuery)
	 
	Select Relacao
	Go top
EndIf

If (ThisForm.optContatos.Value = 1) && Usar Nome de Contato se Preenchido
	Replace All RClis With Iif(Empty(Contato), RClis, Contato) In Relacao
Endif 

If (ThisForm.optProper.Value = 1)
	Replace All RClis With Proper(RClis) In Relacao
Endif 

Select Relacao
Do Case 
	Case (ThisForm.Option.Value=1) && Alfabética
		Index On RClis + IClis Tag RClis
		If ThisForm.Opordem.value = 2
			Index On RClis + IClis DESCENDING Tag RClis
		Endif

	Case (ThisForm.Option.Value=2) && Cep
		Index On Ceps + IClis Tag Ceps
		If ThisForm.Opordem.value = 2
			Index On Ceps + IClis DESCENDING Tag Ceps
		Endif

	Case (ThisForm.Option.Value=3) && Data de Nascimento
		Index On Substr(Dtos(Nascs),5) + IClis Tag Nascs
		If ThisForm.Opordem.value = 2
			Index On Substr(Dtos(Nascs),5) + IClis DESCENDING Tag Nascs
		Endif
EndCase
Go Top In Relacao

ENDPROC
