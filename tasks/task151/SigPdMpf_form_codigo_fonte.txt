PROCEDURE Init
ThisForm.cntSombra.lblSombra.Caption = ThisForm.Caption
ThisForm.cntSombra.lblTitulo.Caption = ThisForm.Caption

With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)

	If (.poDataMgr.pnIdconn > 0)
		With .Container1
			.get_nm_operacao.Value = ''
			.get_dt_inicial.Value  = Date()
			.get_dt_final.Value    = Date()
			.get_Grupo.Value 	   = Space(10)
			.get_Conta.Value 	   = Space(10)
			.get_dConta.Value 	   = Space(10)
		EndWith

		=DoDefault()
	Else
		=Messagebox('Impossível Efetuar Conexăo Com o Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith
* Fim **************************************************************************************

ENDPROC

	laFiltro(1,2) = 1

	loLista = CreateObject('fwBuscaExt', ThisForm.poDataMgr.pnIdConn, 'SigCdGcr', ;

	If Not loLista.plAchouRegistro
		loLista.mAddColuna('Codigos', '', 'Código')
		loLista.mAddColuna('Descrs',  '', 'Descriçăo')
		loLista.Show()
	EndIf
	
	This.Value  = Iif(Lastkey()=27,[],crListaRemota.Codigos)
	thisform.container1.get_Dgrupo.Value = Iif(Lastkey()=27,[],crListaRemota.Descrs)
	
	If Not fAcessoContab(Usuar, 'C', This.Value, this, thisform.container1.get_Dgrupo, '', ThisForm.poDataMgr)
		=Messagebox('Acesso Negado!!!', 48, '')
		This.Value = ''
		thisform.container1.get_dgrupo.Value = ''
	EndIf
	Use In crListaRemota
Else
	This.Value  = ''
	thisform.container1.get_grupo.Value = ''	
EndIf

	=Messagebox('Período Inválido!!! Data Final Menor Que a Data Inicial!!!', 16, '')
	ThisForm.Container1.get_dt_inicial.SetFocus
	Return .f.
EndIf

ldDtInicial	 = ThisForm.Container1.get_dt_inicial.Value
ldDtFinal	 = ThisForm.Container1.get_dt_final.Value
lcNmOperacao = ThisForm.Container1.get_nm_operacao.Value
lcEstoqueo	 = ThisForm.Container1.get_grupo.Value
lcContao     = ThisForm.Container1.get_conta.Value
lcWhere = []
Do Case
	Case thisform.container1.optConta.Value = 1
		lcWhere = [(a.GrupoOs = '] + m.lcEstoqueo + [' And a.ContaOs = '] + m.lcContaO + [')]
	Case thisform.container1.optConta.Value = 2
		lcWhere = [(a.GrupoDs = '] + m.lcEstoqueo + [' And a.ContaDs = '] + m.lcContaO + [')]
	OtherWise
		lcWhere = [(a.GrupoOs = '] + m.lcEstoqueo + [' Or a.GrupoDs = '] + m.lcEstoqueo + [') And ] + ;
		   		 [(a.ContaOs = '] + m.lcContaO + [' Or a.ContaDs = '] + m.lcContao + [')]
Endcase
pDtI = fDtoSQL(ldDtInicial)
pDtF = fDtoSQL(ldDtFinal, '23:59:59')

lcQuery = [Select a.* ] + ;
			[From SigCdNec a, SigCdOpd o ] + ;
		   [Where a.Datas BetWeen ?pDtI And ?pDtF And ] + ;
		   		 [a.Emps = '] + _Empr + [' And ] + ;
		   		 [a.Dopps = o.Dopps And ] + ;
		   		 Iif(Empty(lcNmOperacao), [o.GrComps = 1], [a.Dopps = '] + lcNmOperacao + [']) + [ And ] + ;
		   		 lcWhere

If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'csTemporario') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (csTemporario)')
	Return 0
EndIf

ThisForm.Enabled = .f.
Select csTemporario
If (Reccount() > 0)
	Use In csTemporario

	Do Form SigPdMp7 With lcQuery, ThisForm
Else
	=Messagebox('Nenhum Registro Selecionado!!!', 0+48, '')
EndIf
ThisForm.Enabled = .t.
* Fim **************************************************************************************

ENDPROC
