PROCEDURE Init
=DoDefault()

ThisForm.poDataMgr = CreateObject('fSqlConector',thisform.Name)

lnQueryOk = ThisForm.poDataMgr.SqlExecute([Select cemps,razas From SigCdEmp where cemps = ']+_Empr+[' ],'CrSigCdEmp')
If lnQueryOk < 1
	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
	Return(.F.)
EndIf
Select CrSigCdEmp
Go Top

lnQueryOk = ThisForm.poDataMgr.SqlExecute([Select * From SigCdMoe ],'CrSigCdMoe')
If lnQueryOk < 1
	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
	Return(.F.)
EndIf
Select CrSigCdMoe
Index On cmoes Tag cmoes
Go Top

lnQueryOk = ThisForm.poDataMgr.SqlExecute([Select * From SigCdCot ],'CrSigCdCot')
If lnQueryOk < 1
	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
	Return(.F.)
EndIf
Select CrSigCdCot
Index On cmoes+DtoS(datas) Tag cmoedata
Go Top

* Inicializaçăo das variáveis do formulário
ThisForm.get_cd_grestoque.Value  = Space(10)
ThisForm.get_ds_grestoque.Value  = ''
ThisForm.get_cd_estoque.Value    = Space(10)
ThisForm.get_ds_estoque.Value    = ''
ThisForm.get_grupo.Value         = ''
ThisForm.get_grupo_desc.Value    = ''
ThisForm.get_data.Value          = Date()
ThisForm.get_moeda.Value         = ''
ThisForm.get_moeda_desc.Value    = ''
ThisForm.get_dt_cotacao.Value    = Date()
ThisForm.opt_dados_cotacao.Value = 1

ThisForm.Refresh
ENDPROC

------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Private lcData

* Variáveis de cabeçalho
Local cNomeEmpresa,cTitulo,cSubTitulo
* Variáveis de memória
Local cEstoque,cGrupo,cFornecedor,cMoeda,lcDescMoeda
Local dData,dDtCotacao,lDadosCotacao
Local cCabecalho
* Variáveis de auxílio no relatório
Local cCondicao,nValor,nCusto
Local nTotVenda,nTotCusto,nTotQtds
Local cDetalhe,cTotal
Local cto,cti
Local lcUnid, ldData, lcMoeda
Local lcEstoque, lnFator, lcFatVens
Local lnValor, lnCusto
Local cChave
ThisForm.pimprimir = .T.

* Consiste o campo data
If IsEmpty(ThisForm.get_data.Value)
	MessageBox('Data inválida !',16,'')
	ThisForm.get_data.SetFocus
	ThisForm.pimprimir = .F.
EndIf

* Consiste o campo moeda
If IsEmpty(ThisForm.get_moeda.Value)
	MessageBox('Moeda inválida !',16,'')
	ThisForm.get_moeda.SetFocus
	ThisForm.pimprimir = .F.
EndIf

* Consiste o campo data da cotaçăo
If IsEmpty(ThisForm.get_dt_cotacao.Value)
	MessageBox('Data da cotaçăo inválida !',16,'')
	ThisForm.get_dt_cotacao.SetFocus
	ThisForm.pimprimir = .F.
EndIf
If ThisForm.pimprimir

	* Inicializaçăo das variáveis
	lcGrupo        = ThisForm.get_cd_grestoque.Value
	lcEstoque      = ThisForm.get_cd_estoque.Value
	lcUnid         = ThisForm.get_grupo.Value
	ldData         = ThisForm.get_data.Value
	lcMoeda        = ThisForm.get_moeda.Value
	lcDescMoeda    = AllTrim(ThisForm.get_moeda_desc.Value)
	ldDtCotacao    = ThisForm.get_dt_cotacao.Value
	lDadosCotacao  = ThisForm.opt_dados_cotacao.Value 
	lnFator		   = ThisForm.opc_fator.Value
	llPosAtu       = (ldData = Date())

	lcNomeEmpresa = IIf(!IsEmpty(CrSigCdEmp.razas),CrSigCdEmp.razas,'')
	lcTitulo      = 'Posiçăo de Estoque p/ Unidade de Produto em ' + DtoC(ldData)
	lcSubTitulo   = IIf(IsEmpty(lcMoeda),'','Valores em '+lcDescMoeda)

	cCabecalho = PadC('Cod',3,'_')+' '
	cCabecalho = cCabecalho + PadC('Descriçăo',20,'_')+' '
	cCabecalho = cCabecalho + PadC('Fator de Venda',14,'_')+' '
	cCabecalho = cCabecalho + PadC('Qtde.',10,'_')+' '
	cCabecalho = cCabecalho + PadC('Vl.Custo',17,'_')+' '
	cCabecalho = cCabecalho + PadC('Vl.Venda',17,'_')

	* Criaçăo do arquivo cabeçalho do relatório
	Create Cursor dbCabecalho (NomeEmpresa c(80),Titulo c(80),SubTitulo c(80),Cabecalho c(200))
	Insert Into dbCabecalho (NomeEmpresa,Titulo,SubTitulo,Cabecalho) Values (lcNomeEmpresa,lcTitulo,lcSubTitulo,cCabecalho)

	* Criaçăo do cursor principal do relatório
	Create Cursor dbRelatorio (Detalhe c(200))

	* Criaçăo dos cursores necessários para o relatório
	Create Cursor TempMoe (cmoes c(3),cotacao n(14,4))
	Index On cmoes Tag cmoes

	Create Cursor dbRelat (cunis c(3),dunis c(20),qtds n(10,3),vends n(15,2),custs n(15,2),fatvens c(3))
	Index On cunis+fatvens Tag cunis
	Index On dunis+fatvens Tag descrs

	lcData = fDtoSQL(ldData,'23:59:59')
	lcGrEstIni = Iif(!IsEmpty(lcGrupo),lcGrupo,Space(10))
	lcGrEstFim = Iif(!IsEmpty(lcGrupo),lcGrupo,Replicate(Chr(254),10))
	lcEstIni   = Iif(!IsEmpty(lcEstoque),lcEstoque,Space(10))
	lcEstFim   = Iif(!IsEmpty(lcEstoque),lcEstoque,Replicate(Chr(254),10))

	lcConexao = ThisForm.poDataMgr.GetDriver()
	If lcConexao = 'foxpro'
		lcTblHis = (Addbs(JustPath(ThisForm.poDataMgr.GetDataBase()))+'SigMvHst.dbf')
		If !Used('SigMvHst')
			Use (lcTblHis) In 0
		EndIf
	EndIf

	If !IsEmpty(lcUnid)

		Wait Window 'Aguarde! Selecionando Produtos...' NoWait
		lnQueryOk = ThisForm.poDataMgr.SqlExecute([Select a.cpros,a.cunis,c.dunis,a.linhas,b.tpcustos,a.pesoms,a.pvens,]+;
			[a.moevs,a.fvendas,a.moepvs,a.custofs,a.moecusfs ]+;
		[From SigCdPro a ]+;
		[Left Join SigCdLin b On b.linhas = a.linhas ]+;
		[Left Join SigCdUni c On c.cunis = a.cunis ]+;
		[Where a.cunis = ']+lcUnid+[' ],'CrSigCdPro')
		If lnQueryOk < 1
			Return(.F.)
		EndIf
		loBarra = CreateObject('fwprogressbar','Calculando Saldos...',RecCount('CrSigCdPro'))
		loBarra.Show
		Select CrSigCdPro
		Go Top
		Scan
			loBarra.Update(.T.,'Produto: ' + CrSigCdPro.cpros)
			lcCodPro   = CrSigCdPro.cpros
			If Used('TmpSaldo')
				Use In TmpSaldo
			EndIf
			If llPosAtu
				lnQueryOk = ThisForm.poDatamgr.SqlExecute([Select sqtds From SigMvEst ]+;
				[Where cpros = ']+lcCodPro+[' And empgruests Between ']+_Empr+lcGrEstIni+lcEstIni+[' ]+;
					[And ']+_Empr+lcGrEstFim+lcEstFim+[' ],'TmpSaldo')
				If lnQueryOk < 1
					Return(.F.)
				EndIf
				Select TmpSaldo
				Go Top
				If Eof() Or TmpSaldo.sqtds = 0
					Select CrSigCdPro
					Loop
				EndIf
			Else
				If lcConexao = 'foxpro'
					Select SigMvHst
					Set Order To empcpro Descending
					Set Near On
					Seek _Empr + lcCodPro + DtoS(ldData)
					Set Near Off
					
					If !(!SigMvHst.sqtds = 0 And SigMvHst.emps = _Empr And (IsEmpty(lcGrupo) Or SigMvHst.grupos = lcGrupo) And ;
						(IsEmpty(lcEstoque) Or SigMvHst.estos = lcEstoque) And SigMvHst.cpros = lcCodPro)
						Select CrSigCdPro
						Loop
					EndIf
				Else
					lcEGEInis = _Empr + lcGrEstIni + lcEstIni
					lcEGEFims = _Empr + lcGrEstFim + lcEstFim
					lStrQuery = [Select Top 1 SQtds,Datas From SigMvHst ]+;
								[Where CPros = "]+lcCodPro+[" And EmpGruEsts Between "]+lcEGEInis+[" And "]+lcEGEFims+[" ]+;
									[And Datas <= ?lcData ]+;
								[Order By CIdChaves Desc ]
					If ThisForm.poDatamgr.SqlExecute(lStrQuery,'TmpSaldo') < 1
						MessageBox('Favor reinicializar o processo !!!',16,'Falha na Conexăo (TmpSaldo)')
						Return(.F.)
					EndIf
					Select TmpSaldo
					Go Top
					If Eof() Or TmpSaldo.SQtds = 0
						Select CrSigCdPro
						Loop
					EndIf
				EndIf
			EndIf
		
			lnValor  = CrSigCdPro.pvens * Iif(llPosAtu,TmpSaldo.sqtds,Iif(lcConexao = 'foxpro',SigMvHst.sqtds,TmpSaldo.sqtds))
			lnValor  = lnValor * fCarregarCambio(CrSigCdPro.moevs,ldData) / fCarregarCambio(lcMoeda,ldData)
			lnCusto  = CrSigCdPro.custofs
			If Nvl(CrSigCdPro.tpcustos,'') = '2'
				lnCusto = CrSigCdPro.pesoms * CrSigCdPro.custofs
			EndIf
			lnCusto  = lnCusto * Iif(llPosAtu,TmpSaldo.sqtds,Iif(lcConexao = 'foxpro',SigMvHst.sqtds,TmpSaldo.sqtds)) * fCarregarCambio(CrSigCdPro.moecusfs,ldData) / fCarregarCambio(lcMoeda,ldData)
			lnValFat = IIf(CrSigCdPro.fvendas = 0,1,CrSigCdPro.fvendas * fCarregarCambio(CrSigCdPro.moepvs,ldData))
			lnValor  = lnValor * lnValFat
			
			Select TempMoe
			Set Order To cmoes
			=Seek(CrSigCdPro.moevs,'TempMoe','cmoes')
			If Eof()
				Append Blank
				Replace TempMoe.cmoes   With CrSigCdPro.moevs
				Replace TempMoe.cotacao With fCarregarCambio(CrSigCdPro.moevs,ldData)
			EndIf
					
			Select TempMoe
			=Seek(lcMoeda,'TempMoe','cmoes')
			If Eof()
				Append Blank
				Replace TempMoe.cmoes   With lcMoeda
				Replace TempMoe.cotacao With fCarregarCambio(lcMoeda,ldData)
			EndIf

			Select TempMoe
			=Seek(CrSigCdPro.moecusfs,'TempMoe','cmoes')
			If Eof()
				Append Blank
				Replace TempMoe.cmoes   With CrSigCdPro.moecusfs
				Replace TempMoe.cotacao With fCarregarCambio(CrSigCdPro.moecusfs,ldData)
			EndIf

			Select TempMoe
			=Seek(CrSigCdPro.moepvs,'TempMoe','cmoes')
			If Eof()
				Append Blank
				Replace TempMoe.cmoes   With CrSigCdPro.moepvs
				Replace TempMoe.cotacao With fCarregarCambio(CrSigCdPro.moepvs,ldData)
			EndIf
			
			Select dbRelat
			Set Order To cunis
			
			If lnFator = 1
				lcFatVens = CrSigCdPro.moepvs
			Else
				lcFatVens = '   '
			EndIf
			
			=Seek(CrSigCdPro.cunis + lcFatVens,'dbRelat','cunis')
			
			If Eof()
				Append Blank
				Replace dbRelat.cunis   With CrSigCdPro.cunis
				Replace dbRelat.fatvens With lcFatVens
				Replace dbRelat.dunis   With Nvl(CrSigCdPro.dunis,'')
			EndIf
			
			Replace dbRelat.qtds  With dbRelat.qtds  + Iif(llPosAtu,TmpSaldo.sqtds,Iif(lcConexao = 'foxpro',SigMvHst.sqtds,TmpSaldo.sqtds))
			Replace dbRelat.vends With dbRelat.vends + lnValor
			Replace dbRelat.custs With dbRelat.custs + lnCusto
			Select CrSigCdPro
		EndScan
		loBarra.Complete(.T.)

	Else

		Wait Window 'Aguarde! Selecionando Produtos...' NoWait
		If llPosAtu
			lnQueryOk = ThisForm.poDataMgr.SqlExecute([Select cpros,sqtds,cidchaves From SigMvEst ]+;
			[Where empgruests Between ']+_Empr+lcGrEstIni+lcEstIni+[' And ']+_Empr+lcGrEstFim+lcEstFim+[' ]+;
			[Group By cidchaves,cpros,sqtds ],'TmpSaldo')
			If lnQueryOk < 1
				Return(.F.)
			EndIf
		Else	
			lnQueryOk = ThisForm.poDataMgr.SqlExecute([Select cpros From SigMvHst ]+;
			[Where empgruests Between ']+_Empr+lcGrEstIni+lcEstIni+[' And ']+_Empr+lcGrEstFim+lcEstFim+[' ]+;
			[Group By cpros ],'TmpSaldo')
			If lnQueryOk < 1
				Return(.F.)
			EndIf
		EndIf
		Select TmpSaldo
		Go Top
		If !Eof()
			loBarra = CreateObject('fwprogressbar','Calculando Saldos...',RecCount('TmpSaldo'))
			loBarra.Show
			Scan
				loBarra.Update(.T.,'Produto: ' + TmpSaldo.cpros)
				lcCodPro = TmpSaldo.cpros
				If Used('TmpPro')
					Use In TmpPro
				EndIf
				lnQueryOk = ThisForm.poDataMgr.SqlExecute([Select a.cpros,a.cunis,c.dunis,a.linhas,b.tpcustos,a.pesoms,a.pvens,]+;
					[a.moevs,a.fvendas,a.moepvs,a.custofs,a.moecusfs ]+;
				[From SigCdPro a ]+;
				[Left Join SigCdLin b On b.linhas = a.linhas ]+;
				[Left Join SigCdUni c On c.cunis = a.cunis ]+;
				[Where a.cpros = ']+lcCodPro+[' ],'TmpPro')
				If lnQueryOk < 1
					Return(.F.)
				EndIf
				If !(llPosAtu)
					If lcConexao = 'foxpro'
						Select SigMvHst
						Set Order To empcpro Descending
						Set Near On
						Seek _Empr + lcCodPro + DtoS(ldData)
						Set Near Off
						
						If !(!SigMvHst.sqtds = 0 And SigMvHst.emps = _Empr And (IsEmpty(lcGrupo) Or SigMvHst.grupos = lcGrupo) And ;
							(IsEmpty(lcEstoque) Or SigMvHst.estos = lcEstoque) And SigMvHst.cpros = lcCodPro)
							Select CrSigCdPro
							Loop
						EndIf
					Else	
						lcEGEInis = _Empr + lcGrEstIni + lcEstIni
						lcEGEFims = _Empr + lcGrEstFim + lcEstFim
						If Used('TmpHis')
							Use In TmpHis
						EndIf
						lStrQuery = [Select Top 1 SQtds From SigMvHst ]+;
									[Where CPros = "]+lcCodPro+[" And EmpGruEsts Between "]+lcEGEInis+[" And "]+lcEGEFims+[" ]+;
										[And datas <= ?lcData ]+;
									[Order By CIdChaves Desc ]
						If ThisForm.poDatamgr.SqlExecute(lStrQuery,'TmpHis') < 1
							MessageBox('Favor reinicializar o processo !!!',16,'Falha na Conexăo (TmpHis)')
							Return(.F.)
						EndIf
						Select TmpHis
						Go Top
						If Eof()
							Select TmpSaldo
							Loop
						EndIf
					EndIf
				EndIf

				lnValor  = TmpPro.pvens * Iif(llPosAtu,TmpSaldo.sqtds,Iif(lcConexao = 'foxpro',SigMvHst.sqtds,TmpHis.sqtds))
				lnValor  = lnValor * fCarregarCambio(TmpPro.moevs,ldData) / fCarregarCambio(lcMoeda,ldData)
				lnCusto  = TmpPro.custofs
				If Nvl(TmpPro.tpcustos,'') = '2'
					lnCusto = TmpPro.pesoms * TmpPro.custofs
				EndIf
				lnCusto  = lnCusto * Iif(llPosAtu,TmpSaldo.sqtds,Iif(lcConexao = 'foxpro',SigMvHst.sqtds,TmpHis.sqtds)) * fCarregarCambio(TmpPro.moecusfs,ldData) / fCarregarCambio(lcMoeda,ldData)
				lnValFat = IIf(TmpPro.fvendas = 0,1,TmpPro.fvendas * fCarregarCambio(TmpPro.moepvs,ldData))
				lnValor  = lnValor * lnValFat
				
				Select TempMoe
				Set Order To cmoes
				=Seek(TmpPro.moevs,'TempMoe','cmoes')
				If Eof()
					Append Blank
					Replace TempMoe.cmoes   With TmpPro.moevs
					Replace TempMoe.cotacao With fCarregarCambio(TmpPro.moevs,ldData)
				EndIf
						
				Select TempMoe
				=Seek(lcMoeda,'TempMoe','cmoes')
				If Eof()
					Append Blank
					Replace TempMoe.cmoes   With lcMoeda
					Replace TempMoe.cotacao With fCarregarCambio(lcMoeda,ldData)
				EndIf

				Select TempMoe
				=Seek(TmpPro.moecusfs,'TempMoe','cmoes')
				If Eof()
					Append Blank
					Replace TempMoe.cmoes   With TmpPro.moecusfs
					Replace TempMoe.cotacao With fCarregarCambio(TmpPro.moecusfs,ldData)
				EndIf

				Select TempMoe
				=Seek(TmpPro.moepvs,'TempMoe','cmoes')
				If Eof()
					Append Blank
					Replace TempMoe.cmoes   With TmpPro.moepvs
					Replace TempMoe.cotacao With fCarregarCambio(TmpPro.moepvs,ldData)
				EndIf
				
				Select dbRelat
				Set Order To cunis
				
				If lnFator = 1
					lcFatVens = TmpPro.moepvs
				Else
					lcFatVens = '   '
				EndIf
				
				=Seek(TmpPro.cunis + lcFatVens,'dbRelat','cunis')
				
				If Eof()
					Append Blank
					Replace dbRelat.cunis   With TmpPro.cunis
					Replace dbRelat.fatvens With lcFatVens
					Replace dbRelat.dunis   With Nvl(TmpPro.dunis,'')
				EndIf
				
				Replace dbRelat.qtds  With dbRelat.qtds  + Iif(llPosAtu,TmpSaldo.sqtds,Iif(lcConexao = 'foxpro',SigMvHst.sqtds,TmpHis.sqtds))
				Replace dbRelat.vends With dbRelat.vends + lnValor
				Replace dbRelat.custs With dbRelat.custs + lnCusto
				Select TmpSaldo
			EndScan
			loBarra.Complete(.T.)
		EndIf

	EndIf

	loBarraRel = CreateObject('fwprogressbar','Finalizando o processo...',RecCount('dbRelat') + 3)
	loBarraRel.Show
	
	Select dbRelat
	Go Top

	nTotQtds  = 0
	nTotCusto = 0
	nTotVenda = 0

	Scan
		loBarraRel.Update(.T.)
		cDetalhe = PadR(dbRelat.cunis,3) + ' '
		cDetalhe = cDetalhe + PadR(dbRelat.dunis,20) + ' '
		cDetalhe = cDetalhe + PadC(dbRelat.fatvens,14) + ' '
		cDetalhe = cDetalhe + PadL(Transform(dbRelat.qtds,'999999.999'),10) + ' '
		cDetalhe = cDetalhe + PadL(Transform(dbRelat.custs ,'99,999,999,999.99'),17) + ' '
		cDetalhe = cDetalhe + PadL(Transform(dbRelat.vends ,'99,999,999,999.99'),17)

		Insert Into dbRelatorio (Detalhe) Values (cDetalhe)

		nTotQtds  = nTotQtds  + dbRelat.qtds 
		nTotCusto = nTotCusto + dbRelat.custs 
		nTotVenda = nTotVenda + dbRelat.vends 
	EndScan

	loBarraRel.Update(.T.)

	cTotal = Space(40)
	cTotal = cTotal + Replicate('_',10) + ' '
	cTotal = cTotal + Replicate('_',17) + ' '
	cTotal = cTotal + Replicate('_',17) + ' '

	Insert Into dbRelatorio (Detalhe) Values (cTotal)

	cTotal = PadL('Total : ',40)
	cTotal = cTotal + PadL(Transform(nTotQtds,'999999.999'),10) + ' '
	cTotal = cTotal + PadL(Transform(nTotCusto,'99,999,999,999.99'),17) + ' '
	cTotal = cTotal + PadL(Transform(nTotVenda,'99,999,999,999.99'),17) + ' '

	loBarraRel.Update(.T.)

	Insert Into dbRelatorio (Detalhe) Values (cTotal)

	* Inclui no relatório as informaçőes referentes a cotaçăo
	If lDadosCotacao = 1 And !IsEmpty(lcMoeda)
		Select TempMoe
		Go Top
		If !Eof()
			Insert Into dbRelatorio (Detalhe) Values ('')
			Insert Into dbRelatorio (Detalhe) Values ('')
			cDetalhe = 'Cotaçőes de ' + DtoC(ldDtCotacao)
			Insert Into dbRelatorio (Detalhe) Values (cDetalhe)
			Insert Into dbRelatorio (Detalhe) Values ('')
			cDetalhe = PadC('Moeda',5)+ ' '
			cDetalhe = cDetalhe + PadC('Cotaçăo',14)
			Insert Into dbRelatorio (Detalhe) Values (cDetalhe)
			cDetalhe = Replicate('_',5) + ' '
			cDetalhe = cDetalhe + Replicate('_',14)
			Insert Into dbRelatorio (Detalhe) value (cDetalhe)

			Scan
				cDetalhe = PadR(TempMoe.cmoes,5) + ' '
				cDetalhe = cDetalhe + PadL(Transform(TempMoe.cotacao,'999,999,999.99'),14)
				Insert Into dbRelatorio (Detalhe) Values (cDetalhe)
			EndScan
		EndIf
	EndIf

	loBarraRel.Update(.T.)

	Select dbRelatorio
EndIf
ENDPROC
