PROCEDURE processamento
Private ldDtInicial,ldDtFinal
Private llCpfs,llRclis,llUltima
Private lnNrTipo,lnNrOpcao,lnOrdem
Private lcCdEmpresa,lcDsEmpresa,lcQuebra,lcGroupBy
Private lcCdCarGrupo,lcDsCarGrupo,lcCdCarConta,lcDsCarConta
Private lcCdDepGrupo,lcDsDepGrupo,lcCdDepConta,lcDsDepConta
Private lcCdEmiGrupo,lcDsEmiGrupo,lcCdEmiConta,lcDsEmiConta
Private lcVerData,lcVerEmpresa,lcVerOpcao,lcVerAberto
Private lcVerCarGrupo,lcVerCarConta,lcVerDepGrupo,lcVerDepConta,lcVerEmiGrupo,lcVerEmiConta
Private lcNomeEmpresa,lcTitulo,lcPeriodo,lcOperacao,lcCarteira,lcDeposito,lcEmissor,lcOrdenaPor

lnNrTipo     = ThisForm.opt_nr_tipo.Value
ldDtInicial  = fDtoSQL(ThisForm.get_dt_inicial.Value)
ldDtFinal    = fDtoSQL(ThisForm.get_dt_final.Value,'23:59:59')
lcCdEmpresa  = ThisForm.get_cd_empresa.Value
lcDsEmpresa  = ThisForm.get_ds_empresa.Value
lnNrOpcao    = ThisForm.opt_nr_opcao.Value
lnOrdem      = ThisForm.cmbOrdem.Value
llUltima	 = ThisForm.chkUltima.Value
lcCdCarGrupo = ThisForm.get_cd_Car_grupo.Value
lcDsCarGrupo = ThisForm.get_ds_Car_grupo.Value
lcCdCarConta = ThisForm.get_cd_Car_conta.Value
lcDsCarConta = ThisForm.get_ds_Car_conta.Value
lcCdDepGrupo = ThisForm.get_cd_dep_grupo.Value
lcDsDepGrupo = ThisForm.get_ds_dep_grupo.Value
lcCdDepConta = ThisForm.get_cd_dep_conta.Value
lcDsDepConta = ThisForm.get_ds_dep_conta.Value
lcCdEmiGrupo = ThisForm.get_cd_emi_grupo.Value
lcDsEmiGrupo = ThisForm.get_ds_emi_grupo.Value
lcCdEmiConta = ThisForm.get_cd_emi_conta.Value
lcDsEmiConta = ThisForm.get_ds_emi_conta.Value
llCPFs = .f.
llRclis = .f.

lcVerData     = Iif(lnNrTipo = 3,'a.dtents between ?ldDtInicial and ?ldDtFinal','b.vencs between ?ldDtInicial and ?ldDtFinal')
lcVerEmpresa  = Iif(IsEmpty(lcCdEmpresa),'',' and a.Emps = ?lcCdEmpresa')
lcVerOpcao    = Iif(lnNrOpcao = 1,' and a.notas <> "'+SPACE(6)+'"',Iif(lnNrOpcao=2,' and a.notas = "'+SPACE(6)+'"',''))
lcVerCarGrupo = Iif(IsEmpty(lcCdCarGrupo),'',' and a.grucarts = ?lcCdCarGrupo')
lcVerCarConta = Iif(IsEmpty(lcCdCarConta),'',' and a.concarts = ?lcCdCarConta')
lcVerDepGrupo = Iif(IsEmpty(lcCdDepGrupo),'',' and a.grupos = ?lcCdDepGrupo')
lcVerDepConta = Iif(IsEmpty(lcCdDepConta),'',' and a.contas = ?lcCdDepConta')
lcVerEmiGrupo = Iif(IsEmpty(lcCdEmiGrupo),'',' and a.grclis = ?lcCdEmiGrupo')
lcVerEmiConta = Iif(IsEmpty(lcCdEmiConta),'',' and a.iclis = ?lcCdEmiConta')
lcVerAberto   = Iif(lnNrTipo=1,' and c.tipos = "DB"','')
lcGroupBy     = Iif(llUltima, 'a.Nums', 'a.Nums, a.Dtents, a.Vencs, a.Bancos, a.Agencias, ' + ;
							  'a.Ncontas, a.Ncheques, a.Operacaos')

lcTitulo = 'Relatório de Cheques - '
Do Case
	Case lnNrTipo = 1
		lcTitulo = lcTitulo + 'Abertos'
	Case lnNrTipo = 2
		lcTitulo = lcTitulo + 'Geral'
	Case lnNrTipo = 3
		lcTitulo = lcTitulo + 'Data de Operaçăo'
	Case lnNrTipo = 4
		lcTitulo = lcTitulo + 'Data de Vencimento'
EndCase

lcTitulo = lcTitulo + ' - Ordem: '
Do Case
	Case lnOrdem = 1  && Carteira
		 lcTitulo = lcTitulo + 'Carteira'
		 lcQuebra = '"Grupo/Conta Carteira: " + a.GruCarts + "/" + a.ConCarts'
		 lcOrdenaPor = 'a.GruCarts, a.ConCarts, a.Operacaos, a.Nums'
	Case lnOrdem = 2  && Depósito
		 lcTitulo = lcTitulo + 'Depósito'
		 lcQuebra = '"Grupo/Conta Depósito: " + a.Grupos + "/" + a.Contas'
		 lcOrdenaPor = 'a.Grupos, a.Contas, a.Operacaos, a.Nums'
	Case lnOrdem = 3  && Emissor
		 lcTitulo = lcTitulo + 'Emissor'
		 lcQuebra = '"Grupo/Conta Emissor: " + a.GrClis + "/" + a.Iclis'
		 lcOrdenaPor = 'a.GrClis, a.Iclis, a.Operacaos, a.Nums'
	Case lnOrdem = 4  && Data de Entrada
		 lcTitulo = lcTitulo + 'Data de Entrada'
		 lcQuebra = '"Entrada: " + dtoc(a.DtEnts)'
		 lcOrdenaPor = 'a.DtEnts, a.Operacaos, a.Nums'
	Case lnOrdem = 5  && CPF
		 lcTitulo = lcTitulo + 'C.P.F.'
		 lcQuebra = '"C.P.F.: " + a.Cpfs'
		 lcOrdenaPor = 'a.Cpfs, a.Operacaos, a.Nums'
	Case lnOrdem = 6  && Nome do Cliente
		 lcTitulo = lcTitulo + 'Nome do Cliente'
		 lcQuebra = '"Nome: " + a.Rclis'
		 lcOrdenaPor = 'a.Rclis, a.Operacaos, a.Nums'
EndCase	

ThisForm.podatamgr.CursorQuery('SigCdEmp','crSigCdEmp', 'cEmps', _empr)
lcNomeEmpresa = CRSigCdEmp.Razas
lcPeriodo     = 'Período: ' + Dtoc(ldDtInicial) + ' ŕ ' + Dtoc(ldDtFinal) + ' - ' + ;
				'Empresa: ' + Iif(IsEmpty(lcCdEmpresa),'Geral', lcCdEmpresa + ' - ' + ;
				Alltrim(lcDsEmpresa))
lcCarteira    = 'Grupo / Conta Carteira : ' + Iif(IsEmpty(lcCdCarGrupo), 'Todos', ;
				Alltrim(lcCdCarGrupo) + ' - ' + Alltrim(lcDsCarGrupo)) + ' / ' + ;
				Iif(IsEmpty(lcCdCarConta), 'Todas', Alltrim(lcCdCarConta) + ' - '+ ;
				Alltrim(lcDsCarConta))
lcDeposito    = 'Grupo / Conta Deposito : ' + Iif(IsEmpty(lcCdDepGrupo), 'Todos', ;
				Alltrim(lcCdDepGrupo) + ' - ' + Alltrim(lcDsDepGrupo)) + ' / ' + ;
				Iif(IsEmpty(lcCdDepConta), 'Todas', Alltrim(lcCdDepConta) + ' - ' + ;
				Alltrim(lcDsDepConta))
lcEmissor     = 'Grupo / Conta Emissor  : ' + Iif(IsEmpty(lcCdEmiGrupo), 'Todos', ;
				Alltrim(lcCdEmiGrupo) + ' - ' + Alltrim(lcDsEmiGrupo)) + ' / ' + ;
				Iif(IsEmpty(lcCdEmiConta), 'Todas', Alltrim(lcCdEmiConta) + ' - ' + ;
				Alltrim(lcDsEmiConta))

lcOperacao = 'Operaçăo : ' 
Select csOperacaos
Scan
	If csOperacaos.Marcas
		lcOperacao = lcOperacao + Iif(Len(Alltrim(lcOperacao))>10, ' / ', '') + ;
					 Alltrim(csOperacaos.Operacaos)
	EndIf
EndScan

Select csSelecionados
Scan
	Do Case
		Case csSelecionados.Campos = 'C.P.F.'
			 llCpfs  = (lnOrdem # 5)
		Case csSelecionados.Campos = 'Nome'
			 llRclis = (lnOrdem # 6)
	EndCase	
	llCPFt = IIF(csSelecionados.Campos = 'C.P.F. T',.t.,.f.)

EndScan

Create Cursor CsCabecalho (NomeEmpresa c(80), Titulo c(80), Periodo c(80), Operacao c(200), ;
						   Carteira c(80), Deposito c(80), Emissor c(80), Tipo n(1), ;
						   Cpfs l(1), Rclis l(1), cpfst l(1))

Insert Into CsCabecalho (NomeEmpresa, Titulo, Periodo, Operacao, Carteira, Deposito, ;
						 Emissor, Tipo, Cpfs, Rclis, Cpfst) ;
			     Values (lcNomeEmpresa, lcTitulo, lcPeriodo, lcOperacao, lcCarteira, ;
			     		 lcDeposito, lcEmissor, lnOrdem, llCpfs, llRclis, llCpft)

Set Talk Windows
Set Talk On
ThisForm.MousePointer = 11
*** Mena 13/12/2013 - incluida descricao da conta no relatorio ordenado por Depósito ***
*** mena 03/05/2015 - incluido cpf de terceiro
vSql = [Select a.Nums, a.NumLotes, a.Dtents, a.GruCarts, a.ConCarts, a.Grupos, ]+;
	   [a.Contas+'-'+e.rclis as Contas, a.Grclis, a.Iclis, a.Dtsis, b.Vencs, b.Valors, b.Emps, ]+;
	   [b.Bancos,b.Agencias,b.Ncontas,b.Ncheques, ]+;
	   [b.Bancos + '-' + b.Agencias + '-' + b.Ncontas + '-' + b.Ncheques as Cheques, ]+;
	   [a.codigos as Operacaos , d.Cpfs, d.Rclis, a.hists, f.cpfs as cpfsT ]+;
       [From SigCqChm a, SigCdCli d, SigCdCli e, ]+;
       [SigChe b left join SigCdCli f on b.grupot = f.grupos and b.contat = f.iclis ] +; 
 	   [Where a.Nums = b.Nums and b.Iclis=d.Iclis and a.contas = e.Iclis and ]+;
	   lcVerData+lcVerEmpresa+lcVerOpcao+lcVerCarGrupo+lcVerCarConta+lcVerDepGrupo+;
	   lcVerDepConta+lcVerEmiGrupo+lcVerEmiConta
ThisForm.podatamgr.SqlExecute(vSql,"csRelatorio0")

If llUltima
	Select *,Space(Len(lcQuebra)+20) as Quebra From CsRelatorio0 Where 0 = 1 Into Cursor CsResultado ReadWrite
	
	Select a.Nums, a.NumLotes, a.Dtents, a.GruCarts, a.ConCarts, a.Grupos, ;
		   a.Contas, a.Grclis, a.Iclis, a.DtSis, ;
		   a.Vencs, a.Valors, a.Emps, Cheques,  ;
		   a.Operacaos, a.Cpfs, a.Rclis, a.cpfst, ;
		   &lcQuebra. as Quebra, a.hists ;
		   From csRelatorio0 a, csOperacaos c ;
		   Where c.Marcas and a.Operacaos = c.Operacaos &lcVerAberto ;
		   Into Cursor CsSelecao Order by a.Nums,a.DtSis Descending
	
	Select CsSelecao
	Scan
		lcNum = Nums
		Scatter Memvar memo 
		Select CsResultado
		Append Blank
		Gather Memvar memo
		Select CsSelecao
		Scan While Nums = lcNum
		EndScan
		Skip -1
	EndScan

	Select a.* From CsResultado a ;
		Order By &lcOrdenaPor. ;
	Into Cursor csRelatorio 

Else

	Select a.Nums, a.NumLotes, a.Dtents, a.GruCarts, a.ConCarts, a.Grupos, ;
		   a.Contas, a.Grclis, a.Iclis, a.Dtsis, ;
		   a.Vencs, a.Valors, a.Emps, Cheques,  ;
		   a.Operacaos, a.Cpfs, a.Rclis, &lcQuebra. as Quebra, a.hists, a.cpfst ;
		   From csRelatorio0 a, csOperacaos c ;
		   Where c.Marcas and a.Operacaos = c.Operacaos &lcVerAberto ;
		   Order By &lcOrdenaPor. ;
		   Into Cursor csRelatorio 

EndIf

ThisForm.MousePointer = 0
Set Talk Off

Select csRelatorio
Go Top
ENDPROC

PROCEDURE btnReport.Click
If This.Value <> 4
	* consiste o campo data inicial
	If IsEmpty(ThisForm.get_dt_inicial.Value)
		Messagebox('Data inicial inválida !',16,'')
		ThisForm.get_dt_inicial.SetFocus
		Return(.f.)
	Endif
	
	* consiste o campo data final
	If IsEmpty(ThisForm.get_dt_final.Value)
		Messagebox('Data final inválida !',16,'')
		ThisForm.get_dt_final.SetFocus
		Return(.f.)
	Endif
	
	* consiste os campos de datas
	If ThisForm.get_dt_final.Value < ThisForm.get_dt_inicial.Value
		Messagebox('Data inicial maior que data final !',16,'')
		ThisForm.get_dt_inicial.SetFocus
		Return(.f.)
	Endif

	* Consiste a Grade de Operaçőes
	Select * From csOperacaos Where Marcas Into Cursor csTmpOpe
	If _Tally < 1
		Messagebox('Selecione Ao Menos Uma Operaçăo!',16,'')
		ThisForm.grdOperacaos.Column1.Check1.SetFocus
		Return(.f.)
	Endif

	* Consiste a Grade de Campos de Impressăo
	Select * From csImprimir Where Marcas Into Cursor csSelecionados
	If _Tally < 1
		Messagebox('Selecione Ao Menos Um Campo Para Impressăo!',16,'')
		ThisForm.grdImprimir.Column1.Check1.SetFocus
		Return(.f.)
	Endif

	* Consiste a Ordem
	Select csImprimir
	Scan
		If !csImprimir.Marcas
			If (csImprimir.Campos = 'C.P.F.' .and. ThisForm.cmbOrdem.Value = 5) .or. ;
			   (csImprimir.Campos = 'Nome'   .and. ThisForm.cmbOrdem.Value = 6)
				Messagebox('Esta Ordem Necessita Que Selecione o Campo de Impressăo ' + ;
							Alltrim(csImprimir.Campos) + '!', 16, '')
				ThisForm.cmbOrdem.SetFocus
				Return(.f.)
			Endif
		Endif
	EndScan

Endif

DODEFAULT()

ENDPROC
