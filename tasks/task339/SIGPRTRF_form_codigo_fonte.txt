PROCEDURE Init
With ThisForm
	.poDataMgr = CreateObject('fSqlConector')
	If (.poDataMgr.pnIdconn > 0)
		
		lcSql = [Select ClockEcfs From SigCdPac ]
		ThisForm.Podatamgr.Sqlexecute(lcSql,'LocalParac')

		=DoDefault()
		
		ThisForm.Retorno = ' '

		With Thisform.Cnt_Referencia.Grade
			.RecordSource = 'CrProduto'
			.Column1.ControlSource = 'CrProduto.Cpros'
			.Column2.ControlSource = 'CrProduto.Dpros'
			.Refresh
		EndWith
		
		Select CrProduto
		Append Blank
		
		ThisForm.Cnt_Referencia.Grade.column1.SetFocus

	Else
		=Messagebox('Impossivel Efetuar ConexŃo com Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith

ENDPROC

PROCEDURE LostFocus
If Lastkey() = 13
	Select CrProduto
	If Empty(cpros)
		ThisForm.Altera.SetFocUs
	Else
		_lnRecno = Recno()
		Locate For Empty(Cpros)
		If Eof()
			Append Blank
		EndIf
		Locate For Recno() = _LnRecno
		Keyboard '{DNARROW}'
	Endif
EndIf

ENDPROC

------------------------------------------------------------
* PROCEDURE Valid
------------------------------------------------------------
PROCEDURE Valid
If This.Value = ThisForm.AntCPro
	Return
EndIf

If Not Empty(This.Value)
	m.loLista = CreateObject('fwBuscaExt',ThisForm.poDataMgr.pnIdConn,'SigCdPro','CrProds','cpros',This.Value,'Produtos',1000)
	If !(m.loLista.plAchouRegistro)
		m.loLista.mAddColuna('cpros','','C¾digo')
		m.loLista.mAddColuna('dpros','','DescrińŃo')
		m.loLista.Show()
	EndIf
	If Used('CrProds')
		This.Value = Evaluate('CrProds.Cpros')
		Replace CrProduto.Dpros With CrProds.dpros in CrProduto
		This.Parent.Parent.Column2.Text1.Value = Evaluate('CrProds.Dpros')
	EndIf
EndIf
Return(.T.)

ENDPROC

PROCEDURE Click
Select CrProduto
Count to x for Not Empty(Cpros)
If x=0
	MessageBox('Nenhum Produto Informado!!!',0+16,'')
	Return .f.
Endif

ThisForm.Cnt_Mensagem.Visible = .t.
ldDataI = Datetime()
_w = LocalParac.ClockEcfs / 3
_x = 13
_y = 1
DO While Datetime() - ldDataI < LocalParac.ClockEcfs
	_Tempo = Datetime() - ldDataI
	If _x >= 13
		_x = 1
	Else
		_x = _x + 1
	EndIf
	If _Tempo > _W
		If _Tempo < _W * 2
			For lnBot = 1 To 13
				lcMacro = 'ThisForm.Cnt_Mensagem.botao'+Alltrim(Str(lnBot,2))
				&lcMacro..BackColor = 65535
			Next
		Else
			For lnBot = 1 To 13
				lcMacro = 'ThisForm.Cnt_Mensagem.botao'+Alltrim(Str(lnBot,2))
				&lcMacro..Backcolor = 4227072
			Next
		EndIf
	Else
		For lnBot = 1 To 13
			lcMacro = 'ThisForm.Cnt_Mensagem.botao'+Alltrim(Str(lnBot,2))
			&lcMacro..Backcolor = 255
		Next
	EndIf
	
	For lnBot = 1 To 13
		lcMacro = 'ThisForm.Cnt_Mensagem.botao'+Alltrim(Str(lnBot,2))
		&lcMacro..Visible = lnBot <= _x
	Next
	
Enddo
ThisForm.Retorno = 'Ok'
ThisForm.Release
ENDPROC
