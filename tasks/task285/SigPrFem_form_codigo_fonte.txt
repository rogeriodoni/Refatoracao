------------------------------------------------------------
PROCEDURE posbalanco
Private pDat, pNot
Local _Origem, _Destino, _Grupo, _Conta, _Data, _DatauBal, _Saldoi, _Saidas, _FalhaAdimitida

ldDatai  = ThisForm.Get_datai.Value
ldDataf  = ThisForm.Get_dataf.Value
lcEmp    = _Empr
lcCodMat = crSigCdPam.Ouros
_Trabalhado = 0

ldPrimBal = Date()

lcSql = [Select Rclis, GerBals, PagFals, RecFals From SigCdCli where Iclis = ']+CrSigMvEst.Estos+[']
If Thisform.Podatamgr.Sqlexecute(lcSql,'CrSigCdCli') < 1
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crsjcli)')
	Return 0
EndIf
	
=Seek(CrSigMvEst.Grupos,'CrSigCdGcr','Codigos')

If ( CrSigCdCli.GerBals <> 1 Or CrSigCdGcr.GerBals <> 1 )
	Return 0
Endif

lcQuery = [Select Datas, codigos ] + ;
		    [From SigCdFcx ] + ;
		   [Where Emps = '] + _Empr + [' And ] + ;
		   		 [Grupos = '] + crSigMvEst.Grupos + [' And ] + ;
		   		 [Contas = '] + crSigMvEst.Estos + [' ]

If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'LocalFecha') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalFecha)')
	Return 0
EndIf

For lnCt = 1 to 2
	
	Select TmpResumo
	Set Order To
	Zap

						Replace QtdeEnts  With QtdeEnts + LocalEesti.Qtds,;
						        PesoEnts  With PesoEnts + lnQtde,;
								PesoFabre With PesoFabre + lnQtde
					Else
						Replace QtdeSais With QtdeSais + LocalEesti.Qtds,;
								PesoSais With PesoSais + lnQtde,;
								PesoFabrs with pesofabrs + lnQtde
					EndIf
				EndIf

				If _Destino
					If Not Seek(_Grupo + _Conta + LocalEestI.CPros, 'TmpResumo')
						Insert Into TmpResumo (Grupo, Conta, CMats, CUnis, Varias, Agregas, visivel ) ;
									   Values (_Grupo, _Conta, LocalEestI.CPros, TmpPro.CUnis, TmpPro.Varias, LocalGru.nAgMts, .t.)
					EndIf

					Select TmpResumo
					If (_Operacao = 'E')
						Replace QtdeEnts With QtdeEnts + LocalEesti.Qtds,;
								PesoEnts With PesoEnts + lnQtde,;
								PesoFabre With PesoFabre + lnQtde
					Else
						Replace QtdeSais With QtdeSais + LocalEesti.Qtds,;
								PesoSais With PesoSais + lnQtde,;
								PesoFabrs With PesoFabrs + lnQtde
					EndIf
				EndIf
			Endif
		EndScan
	EndScan
	loBarrap.Complete

	Select TmpResumo
	Replace Flag3 with .t. For Grupo + Conta = _Grupo + _Conta

	If (CrSigCdGcr.UnifBals = 1)
		lcMat = crSigCdPam.Ouros
		Select TmpResumo
		If Not Seek(_Grupo + _Conta + lcMat)
			Append Blank
			Replace cmats   With lcMat,;
					Grupo   With _Grupo,;
					conta   with _Conta,;
					Visivel With .t.
		Endif
			
		Select [ ] as Agrupar,Sum(PesoEnts) as pEnts, Sum(PesoSais) as pSais ;
		  From TmpResumo ;
		 Where Grupo + Conta = _Grupo + _Conta And Cmats <> lcMat And Varias <> 1 And Agregas <> 1 ;
		  Into Cursor csTotal group by 1
		Go Top In csTotal

		Select TmpResumo
		=Seek(_Grupo + _Conta)
		Scan while Grupo + Conta = _Grupo + _Conta
			If (Cmats = lcMat)
				Replace PesoEnts With PesoEnts + csTotal.pEnts
				Replace PesoSais With PesoSais + csTotal.pSais
				Replace QtdeEnts With QtdeEnts + csTotal.pEnts
				Replace QtdeSais With QtdeSais + csTotal.pSais
			Else
				If Agregas <> 1
					Replace Visivel With .f.,;
							Flag3   With .f.
				EndIf
			Endif
		EndScan
	EndIf

	**** Buscar Saldo Anterior
	lcSql = [Select * From SigOpCfe Where Codigos = ]+Str(LocalFecha.Codigos,6)+[ And Emps= ']+_Empr+[' Order by Codigos, cpros ]
	If ThisForm.Podatamgr.Sqlexecute(lcSql,'CrSaldoI') < 1
		MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo (CrSaldoI)')
		Return 0
	EndIf

	Select CrSaldoI
	Index On Cpros Tag Cpros

	Select TmpResumo
	_cMat = Space(14)
	Set Order to GrConMat
	=Seek(_Grupo + _Conta)
	Scan While Grupo + Conta = _Grupo + _Conta
		Store 0 to _Saldoi, _Saidas, _Pesagem

		=Seek(TmpResumo.CMats, 'TmpPro', 'CPros')
		If Not llTipoQ And TmpResumo.Cmats # _cmat
			Store 0 to _FalhaAdmitida
			_cmat = tmpResumo.cmats
		EndIf
		
		Select CrSaldoI
		=Seek(TmpResumo.Cmats)
		_Saldoi = CrSaldoI.Pesagems

		Select TmpResumo
		Replace Saldoi  With _Saldoi, ;
				FReal   With _Saldoi + TmpResumo.QtdeEnts - TmpResumo.QtdeSais - TmpResumo.Pesagem, ;
				FAdmin  With _FalhaAdmitida, ;
				Saldof  With _Saldoi + TmpResumo.QtdeEnts - TmpResumo.QtdeSais - TmpResumo.Pesagem - _FalhaAdmitida, ;
				PfTrabs With Iif(_Trabalhado <> 0, (Saldof / _Trabalhado * 100), 0)
		
		If (TmpResumo.Saldof <> 0) And (CrSigCdCli.PagFals = 1 Or CrSigCdCli.RecFals = 1)
			Replace Flag  With .t., ;
					Flag2 With (CrSigCdCli.PagFals = 1 Or CrSigCdCli.RecFals = 1) In TmpResumo

		EndIf
	EndScan

	Select CrSaldoI
	_cMat = Space(14)
	Scan
		If CrSaldoI.Pesagems = 0 
			Loop
		EndIf
		=Seek(CrSaldoI.Cpros, 'TmpPro', 'CPros')
		=Seek(TmpPro.Cgrus,'LocalGru','Cgrus')

		_Saldoi = CrSaldoI.Pesagems

		Select TmpResumo
		Locate For Grupo + Conta + cMats = _Grupo + _Conta + CrSaldoI.Cpros
		If Eof()
			Insert Into TmpResumo (Grupo, Conta, CMats, CUnis, Varias, Agregas, Visivel) ;
						   Values (LocalNens.GrupoOs, LocalNens.ContaOs, CrSaldoI.Cpros, ;
						   		   TmpPro.CUnis, TmpPro.Varias, LocalGru.nAgMts, .t.)

			Select TmpResumo
			Replace Saldoi  With _Saldoi, ;
					FReal   With _Saldoi + TmpResumo.QtdeEnts - TmpResumo.QtdeSais - TmpResumo.Pesagem, ;
					FAdmin  With 0 , ;
					Saldof  With _Saldoi + TmpResumo.QtdeEnts - TmpResumo.QtdeSais - TmpResumo.Pesagem - 0, ;
					PfTrabs With 0
		Endif
	EndScan

	Select TmpResumo
	=Seek(_Grupo + _Conta + lcCodMat)
	Scan while Grupo + Conta + cMats = (_Grupo + _Conta + lcCodMat)
		If lnCt = 2
			If Not Seek(_Grupo, 'Saldos')
				Insert Into Saldos (Grupos, Contas, Emps) Values (_Grupo, _Conta, _Empr)
			EndIf
			Replace Qtde With Qtde + TmpResumo.SaldoF In Saldos
		Else
			If Not Seek(_Grupo , 'SaldoAnt')
				Insert Into SaldoAnt (Grupos, Contas, Emps) Values (_Grupo, _Conta, _Empr)
			EndIf
			Replace Qtde With Qtde + TmpResumo.SaldoF In SaldoAnt
		EndIf
	EndScan
Next
Return .t.

		   		 [a.Datas BetWeen ?pDtI And ?pDtF And ] + ;

	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crProducao)')

lcQuery = [Select CPros, Nops ] + ;
		    [From SigOpPic]
If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'Tmpopi') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpOpi)')
	Return 0

Index On Nops Tag Nops

Select Distinct EmpDnps, Servicos, CMats, Pesos, ChaveB, TpOps, Qtds, Nops ;
  From crProducao ;
 Where Not Servicos ;
Into Cursor crSigCdNei ReadWrite

From crProducao ;
Order By Datas, Emps, Dopps, Numps, GrupoOs, ContaOs, GrupoDs, ContaDs, cIdChaves ;
Into Cursor crSigCdNec

Scan

	=Seek(crSigCdNec.Dopps, 'crSigCdOpd', 'Dopps')

	llEOrigem  = .f.
	llEDestino = .f.
	llSOrigem  = .f.
	llSDestino = .f.

	If (crSigCdOpd.Origems = 1)

		If (Seek(crSigCdNec.GrupoOs + crSigCdNec.ContaOs, 'TmpSaida') Or Seek(crSigCdNec.GrupoOs + Space(10), 'TmpSaida')) And (crSigCdOpd.EstOrigs = 2)

		EndIf
	EndIf

	If (crSigCdOpd.Destinos = 1)
		If (Seek(crSigCdNec.GrupoDs + crSigCdNec.ContaDs, 'TmpEntra') Or Seek(crSigCdNec.GrupoDs + Space(10), 'TmpEntra')) And (crSigCdOpd.EstDests = 1)

		EndIf
		If (Seek(crSigCdNec.GrupoDs + crSigCdNec.ContaDs, 'TmpSaida') Or Seek(crSigCdNec.GrupoDs + Space(10), 'TmpSaida')) And (crSigCdOpd.EstDests = 2)
			llSDestino = .t. && Tiago - 18/10/11 - Tava llEDestino e acho que tava errado
		EndIf
	EndIf

	If Not llEOrigem And Not llEDestino And Not llSOrigem And Not llSDestino
		Loop
	EndIf

	lcEdn = crSigCdNec.Emps + crSigCdNec.Dopps + Str(crSigCdNec.Numps, 10)

	_Material = Space(14)

	Select crSigCdNei
	=SEEK(lcEdn)
	Scan While EmpDnps = lcEdn
		lcChave1 = crSigCdNec.GrupoOs + crSigCdNec.ContaOs + crSigCdNei.TpOps
		lcChave2 = crSigCdNec.GrupoOs + Space(10) + crSigCdNei.TpOps

				_Material = Iif(Empty(TmpPro.MatPrincs), crSigCdPam.Ouros, TmpPro.MatPrincs)
			EndIf
			If CrSigCdNei.Cmats <> _Material
				Loop
			Endif
		Else
			If CrSigCdGcr.UnifBals = 3
				=Seek(CrSigCdNei.Nops,'TmpOpi','Nops')
				If CrSigCdNei.Nops = 0
					_Material = CrSigCdNei.Cmats

					=Seek(TmpOpi.Cpros,'TmpPro','Cpros')
					_Material = Iif(Empty(TmpPro.MatPrincs), crSigCdPam.Ouros, TmpPro.MatPrincs)

			If llEOrigem
				lcTemp  = 'TmpEntra'
				lcAlias = 'Entradas'

				lcTemp  = 'TmpSaida'
				lcAlias = 'Saidas'
			EndIf

lnLinha = lnLinha + 1
Insert Into TmpImprime (Linha, Cabec, Titulo, Valor) Values (lnLinha, .f., 'Saídas : ', lnTotalSaida)

lnLinha = lnLinha + 1

lnLinha = lnLinha + 1

Insert Into TmpImprime (Linha, Cabec, Titulo, Valor, Traco) Values (lnLinha, .f., 'Diferenca : ', lnSaldoIni + lnTotalEntra - lnTotalSaida + lnSaldoaFun - lnPesagem - lnSaldoFunc - lnFalhaFunc, .t.)

		Insert Into TmpImprime2 (Linha2, Cabec2, Titulo2, Valor2) Values (lnLinha2, .f., Saldos.Grupos, Saldos.Qtde)
