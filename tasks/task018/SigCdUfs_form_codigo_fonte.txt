* PROCEDURE Pagina.Dados.Grupo_Salva.Salva.Click
------------------------------------------------------------
PROCEDURE Pagina.Dados.Grupo_Salva.Salva.Click
Select CrSigCdUfs

If Inlist( Thisform.pcEscolha,'INSERIR','ALTERAR')

	If IsEmpty( Thisform.Pagina.Dados.getEstado.value )
		=messagebox('U.F. Inválido.',0+64,'Atençăo!')
		Thisform.Pagina.Dados.getEstado.setfocus
		Return
	Else
	
		If ThisForm.poDataMgr.ChkRegister('SigCdUfs','estados',Thisform.Pagina.Dados.getEstado.value) and Inlist( Thisform.pcEscolha,'INSERIR')
			Messagebox( 'U.F. Já Cadastrado.',0+64,'Atençăo!' )
			Thisform.Pagina.Dados.getEstado.SetFocus
			Return
		Endif
	
	Endif

    m.Ipis      = padr(upper(alltrim(thisform.Pagina.Dados.cmbIpi.value)),1)
	   
    select CrSigCdUfs
    Replace ipis with m.ipis
	   
ENDIF

IF inlist(thisform.pcEscolha,'EXCLUIR')
	IF ThisForm.poDataMgr.ChkRegister('SigCdCli','estas',ThisForm.Pagina.Dados.getEstado.Value) OR ;
		ThisForm.poDataMgr.ChkRegister('SigCdCee','estas',ThisForm.Pagina.Dados.getEstado.Value) OR ;
		ThisForm.poDataMgr.ChkRegister('SigCdCll','estas',ThisForm.Pagina.Dados.getEstado.Value)
		MESSAGEBOX('Estado năo pode ser excluido pois está sendo usado em Clientes!',0+64,goSistema.NomeSistema)
		RETURN .f.
	ENDIF
ENDIF


=DoDefault()

ThisForm.mAtivapagina1
ENDPROC
