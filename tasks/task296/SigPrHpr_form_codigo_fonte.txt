* PROCEDURE Init
------------------------------------------------------------
PROCEDURE Init
Lparameters pFrm

ThisForm.cntSombra.lblSombra.Caption = ThisForm.Caption
ThisForm.cntSombra.lblTitulo.Caption = ThisForm.Caption

Private pDtI, pDtF

Local lcGO, lcCO, lcGD, lcCD, lcAce

ThisForm.poDataMgr = CreateObject([fSqlConector])

lcAce = fChecaAcesso([SIGPRHPR],[AUDITORIA])

ThisForm.ParentForm = pFrm

ThisForm.Get_Data.Visible   = .f.
ThisForm.Say6.Visible       = .f.
ThisForm.GetGruOri.Value    = []
ThisForm.GetConOri.Value    = []
ThisForm.GetGruDes.Value    = []
ThisForm.GetConDes.Value    = []
ThisForm.GetDesGruOri.Value = []
ThisForm.GetDesConOri.Value = []
ThisForm.GetDesGruDes.Value = []
ThisForm.GetDesConDes.Value = []

ThisForm.grd_historico.RecordSource = []

pDtI = fDtoSQL(pdDataIni)
pDtF = fDtoSQL(pdDataFin, [23:59:59])

With ThisForm.poDataMgr
	lnQueryOk = .SqlExecute([Select a.emps,a.empos,a.grupos,a.estos,a.cpros,a.dopes,a.numes,a.datas,a.auditors,a.dtaudits,a.qtds,a.opers,]+;
				[a.sqtds,a.obs,usuars,a.cidchaves, a.pesos, a.spesos, '   ' as cunis ]+;
		   [From SigMvHst a ]+;
		  [Where a.empgruests = "]+_Empr+pcCdGrupo+pcCdConta+[" And a.cpros = "]+pcCdProduto+[" ]+;
				[And a.datas Between ?pDtI And ?pDtF ]+;
	   [Order By a.emps,a.grupos,a.estos,a.cidchaves,a.opers ],"CrSigMvHst")
	If lnQueryOk < 1
		MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
		Return(.F.)
	EndIf

	lnQueryOk = .SqlExecute([Select cpros,cunis,cunips From SigCdPro Where cpros = "]+pcCdProduto+[" ],"TmpPro")
	If lnQueryOk < 1
		MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
		Return(.F.)
	EndIf
	Select TmpPro
	Go Top

	lnQueryOk = .SqlExecute([Select Cestos From SigCdUni Where cunis = "]+TmpPro.cunis+[" ],"TmpUni")
	If lnQueryOk < 1
		MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
		Return(.F.)
	EndIf
	Select TmpUni
	Go Top

	ThisForm.lbl_Produto.Caption = "Produto : " +;
			 AllTrim(pcCdProduto) + " - " + AllTrim(pcDsProduto) + Space(10) +;
			 "Período: " + Dtoc(pdDataIni) + " ŕ " + Dtoc(pdDataFin)+Iif(TmpUni.cestos='3'," Unid.Peso:"+TmpPro.cunips,'')

	Select CrSigMvHst
	Replace all cunis with TmpPro.cunis
	Index On Pesos Tag Pesos
	Index On DTOS(datas) Tag datas
	Go Bottom
	
	ThisForm.get_Auditors.Value = CrSigMvHst.Auditors
	ThisForm.getObs.Value       = Nvl(CrSigMvHst.obs,[])
	ThisForm.Get_usuario.Value  = CrSigMvHst.Usuars

	With ThisForm.grd_historico
		.RecordSource = "CrSigMvHst"
		.ColumnCount  = Iif(TmpUni.Cestos = '3', 9, 7)
		.Column1.ControlSource = "CrSigMvHst.datas"
		.Column2.ControlSource = "CrSigMvHst.numes"
		.Column3.ControlSource = "CrSigMvHst.dopes"
		.Column4.ControlSource = "CrSigMvHst.cunis"
		.Column5.ControlSource = "CrSigMvHst.qtds"
		.Column6.ControlSource = "CrSigMvHst.opers"
		.Column7.ControlSource = "CrSigMvHst.sqtds"
		If TmpUni.Cestos = '3'
			.Column8.ControlSource = "CrSigMvHst.Pesos"
			.Column9.ControlSource = "CrSigMvHst.sPesos"
		EndIf
	EndWith

	If Used("CrSigMvCab")
		Use In CrSigMvCab
	EndIf
	lnQueryOk = .SqlExecute([Select grupoos,contaos,grupods,contads From SigMvCab ]+;
		  [Where empdopnums = "]+Iif(!IsEmpty(CrSigMvHst.empos),CrSigMvHst.empos,CrSigMvHst.emps)+;
				CrSigMvHst.dopes+Str(CrSigMvHst.numes,6)+[" ],"CrSigMvCab")
	If lnQueryOk < 1
		MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
		Return(.F.)
	EndIf
	
	ThisForm.GetGruOri.Value = CrSigMvCab.grupoos
	ThisForm.GetConOri.Value = CrSigMvCab.contaos
	ThisForm.GetGruDes.Value = CrSigMvCab.grupods
	ThisForm.GetConDes.Value = CrSigMvCab.contads

	lcGO = CrSigMvCab.grupoos
	lcCO = CrSigMvCab.contaos
	lcGD = CrSigMvCab.grupods
	lcCD = CrSigMvCab.contads

	If Used("CrSigCdGcr")
		Use In CrSigCdGcr
	EndIf
	If Used("CrSigCdCli")
		Use In CrSigCdCli
	EndIf
	
	lnQueryOk = .SqlExecute([Select codigos,descrs From SigCdGcr ]+;
		  [Where codigos = "]+lcGO+[" Or codigos = "]+lcGD+[" ],"CrSigCdGcr")
	If lnQueryOk < 1
		MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
		Return(.F.)
	EndIf
	
	lnQueryOk = .SqlExecute([Select iclis,rclis From SigCdCli ]+;
		  [Where iclis = "]+lcCO+[" Or iclis = "]+lcCD+[" ],"CrSigCdCli")
	If lnQueryOk < 1
		MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
		Return(.F.)
	EndIf
	
	Select CrSigCdGcr
	Index On codigos Tag codigos
	Go Top

	Select CrSigCdCli
	Index On iclis Tag iclis
	Go TOp
	
	ThisForm.GetDesGruOri.Value = Iif(Seek(lcGO,"CrSigCdGcr","codigos"),CrSigCdGcr.descrs,"")
	ThisForm.GetDesConOri.Value = Iif(Seek(lcCO,"CrSigCdCli","iclis"),CrSigCdCli.rclis,"")
	ThisForm.GetDesGruDes.Value = Iif(Seek(lcGD,"CrSigCdGcr","codigos"),CrSigCdGcr.descrs,"")
	ThisForm.GetDesConDes.Value = Iif(Seek(lcCD,"CrSigCdCli","iclis"),CrSigCdCli.rclis,"")
EndWith

ThisForm.GetObs.Refresh
ThisForm.GetGruOri.Refresh
ThisForm.GetConOri.Refresh
ThisForm.GetGruDes.Refresh
ThisForm.GetConDes.Refresh
ThisForm.GetDesGruOri.Refresh
ThisForm.GetDesConOri.Refresh
ThisForm.GetDesGruDes.Refresh
ThisForm.GetDesConDes.Refresh
ThisForm.Get_Auditors.Refresh
ThisForm.Get_DtAudits.Refresh

ThisForm.chkAuditado.Visible = lcAce
With ThisForm.grd_historico
	.SetAll("DynamicBackColor","Iif(!IsEmpty(CrSigMvHst.Auditors),RGB(220,255,220),RGB(255,255,255) )","Column") && Verde Claro
EndWith

ThisForm.Grd_Historico.Refresh
ThisForm.Grd_Historico.Column1.SetFocus
ENDPROC

Private lcDtHis

With ThisForm.poDataMgr

	If This.Value = 1
		Select CrSigMvHst
		Replace CrSigMvHst.auditors With Usuar
		lnQueryOk = .SqlExecute([Update SigMvHst Set auditors = "]+Usuar+[" Where cidchaves = "]+CrSigMvHst.cidchaves+[" ],"")
		If lnQueryOk < 1
			MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
			ThisForm.poDataMgr.RollBack()
			Return(.F.)
		EndIf

		Replace CrSigMvHst.dtaudits With Date()
		ldDtHis = Date()
		lcHrHis = Time()
		lcHrHis = Substr(lcHrHis,1,2) + Substr(lcHrHis,4,2) + Substr(lcHrHis,7,2)
		lcDtHis = Evaluate([{^]+TransForm((DtoS(Max(ldDtHis,CtoD("01/01/1900")))+lcHrHis),"@R 9999-99-99 99:99:99")+[}])
		lnQueryOk = .SqlExecute([Update SigMvHst Set dtaudits = ?lcDtHis Where cidchaves = "]+CrSigMvHst.cidchaves+[" ],"")
		If lnQueryOk < 1
			MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
			ThisForm.poDataMgr.RollBack()
			Return(.F.)
		EndIf
	Else
		Select CrSigMvHst
		Replace CrSigMvHst.auditors With Space(10)
		lnQueryOk = .SqlExecute([Update SigMvHst Set auditors = "]+Space(10)+[" Where cidchaves = "]+CrSigMvHst.cidchaves+[" ],"")
		If lnQueryOk < 1
			MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
			ThisForm.poDataMgr.RollBack()
			Return(.F.)
		EndIf

		Select CrSigMvHst
		Replace CrSigMvHst.dtaudits With {}
		lnQueryOk = .SqlExecute([Update SigMvHst Set dtaudits = Null Where cidchaves = "]+CrSigMvHst.cidchaves+[" ],"")
		If lnQueryOk < 1
			MessageBox("Favor reinicializar o processo.",16,"Falha na Conexăo")
			ThisForm.poDataMgr.RollBack()
			Return(.F.)
		EndIf
	EndIf
	ThisForm.poDataMgr.Commit()
EndWith
