* PROCEDURE custoproducao
------------------------------------------------------------
PROCEDURE custoproducao
Lparameters pNop, pProduto

ThisForm.poDataMgr.CursorQuery('SigCdPro', 'LocalPro', 'CPros', pProduto, 'MoeCusfs, CustoFs, pVens, Moevs ')
ThisForm.poDataMgr.CursorQuery('SigCdMoe', 'crMoe', 'CMoes', LocalPro.MoeCusFs)
Go Top In crMoe
If Not Empty(crMoe.Moeqs)
	m.Moedas = crMoe.Moeqs
	_Qtdeqs = Iif(crMoe.Qtdeqs = 0, 1, crMoe.Qtdeqs)
Else
	m.Moedas = LocalPro.MoeCusFs
	_Qtdeqs = 1
EndIf
Moev = fBuscarCotacao(m.Moedas, TmpNens.Datas, ThisForm.poDataMgr)  * _Qtdeqs
lnCp = 0

Select CrSigCdNei
Scan for Nops = pNop
	If Empty(Cmats) Or Servicos
		Loop
	EndIf

	ThisForm.poDataMgr.CursorQuery('SigCdMoe', 'crMoe', 'CMoes', CrSigCdNei.MoeCusFs)
	Go Top In crMoe
	If Not Empty(crMoe.Moeqs)
		m.Moedas = crMoe.Moeqs
		_Qtdeqs = Iif(crMoe.Qtdeqs = 0, 1, crMoe.Qtdeqs)
	Else
		m.Moedas = CrSigCdNei.MoeCusFs
		_Qtdeqs = 1
	EndIf
	Moem = fBuscarCotacao(m.Moedas, TmpNens.Datas, ThisForm.poDataMgr)  * _Qtdeqs

	lnCp = lnCp + (CrSigCdNei.Qtds * CrSigCdNei.CustoFs * Moem / Moev )
EndScan

lcSql = [Select a.cMats,c.Retrabs,a.Qtds,a.Coefs,a.Fators,a.Moedas From SigCdNei a, SigCdNec c, SigCdSvc b ]+;
		[Where a.Nops = ]+Str(pNop)+[ And a.Servicos = 1 and ]+;
		[a.cMats <> ' ' And a.cmats = b.ccats and b.Cpros <> ' ' And a.EmpDnps = c.EmpDnPs ]
Thisform.Podatamgr.Sqlexecute(lcSql,'CsxNensi')
Select csxNensI
Scan
	If csxNensI.Retrabs And csxNensI.Coefs # 0
		_Valor = Round(csxNensI.Coefs * csxNensI.Qtds, 2)
	Else
		_Valor = Round(csxNensI.Fators * csxNensI.Qtds, 2)
	EndIf
	_Moeda = csxNensi.Moedas
	If _Valor = 0
		ThisForm.poDataMgr.CursorQuery('SigCdPro', 'LocalPro', 'CPros', CsXNensI.CMats, 'MoeCusfs, CustoFs, pVens, Moevs ')
		ThisForm.poDataMgr.CursorQuery('SigCdGrp', 'LocalGru', 'CGrus', LocalPro.CGrus, 'pvCompos')

		If LocalGru.pvCompos = 2
			_Valor = Round(LocalPro.pVens * csxNensI.Qtds, 2)
			_Moeda = LocalPro.Moevs
		Else
			_Valor = Round(LocalPro.CustoFs * csxNensI.Qtds, 2)
			_Moeda = LocalPro.MoeCusFs
		Endif
	Endif

	ThisForm.poDataMgr.CursorQuery('SigCdMoe', 'crMoe', 'CMoes', _Moeda)
	Go Top In crMoe
	If Not Empty(crMoe.Moeqs)
		m.Moedas = crMoe.Moeqs
		_Qtdeqs = Iif(crMoe.Qtdeqs = 0, 1, crMoe.Qtdeqs)
	Else
		m.Moedas = _Moeda
		_Qtdeqs = 1
	EndIf
	Moem = fBuscarCotacao(m.Moedas, TmpNens.Datas, ThisForm.poDataMgr)  * _Qtdeqs

	lnCp = lnCp + (_Valor * Moem / Moev )

EndScan
Replace CustoFs With lnCp In xPesa
Return .t.

ENDPROC

* PROCEDURE ratpeso
------------------------------------------------------------
PROCEDURE ratpeso
Private pDat, pQtd, pQt2
Local _Recno1, _Pesagem, _Nop, _TotalAnterior, _Diferenca, _RecnoRateio

Select crSigCdOpd
Set Order to Dopps
Seek TmpNens.Dopps

Select CrSigCdNei
Zap

Select xNensiE
Zap

Select xNensI
Set Order To

Select xNensiE
Append From Dbf('xNensI')
Sum Pesos To _TotPeso For Visivel And Not Servicos
Set Order to Rateio
Delete All For Empty(Nops) And Empty(CMats) And Empty(Cats) And (Rateio <> 1)
Go Top

_Tpop = TpOps

If InList(crSigCdOpd.Etiqs, 1, 3)
	_Amarra = Recno('xPesa')

	Select Nenvs, MatPs, Peso as Pesos ;
	From xPesa Where Recno() = _Amarra ;
	Into Cursor xRateio

	Select xRateio
	Scan
		_Nenv 		  = Nenvs
		_MatP 		  = MatPs
		_Pesagem 	  = Pesos
		_PesoAnterior = 0
		_RecnoRateio  = 0

		Select xNensiE
		Set Order To Nenvs
		llTemOuro = .f.
		If Seek(_Nenv)
			Scan While Nenvs = _Nenv
				If CMats = _MatP
					llTemOuro = .t.
					Exit
				EndIf
			EndScan
		EndIf

		If Not llTemOuro
			Insert Into xNensiE (Nenvs, Nops, Numps, CMats, PsFaseAnt, Pesos, Qtds, Rateio) ;
						Values (_Nenv, _Nenv, TmpNens.Numps, _MatP, _Pesagem, _Pesagem, _Pesagem, 2)
		EndIf

		Scan For Rateio = 3 And NEnvs = _Nenv And Not FlagIncs
			_Pesagem = _Pesagem - Pesos
		EndScan

		Scan For Rateio = 2 And NEnvs = _Nenv And Not FlagIncs
			If (_RecnoRateio = 0)
				_RecnoRateio = Recno()
			EndIf
			_PesoAnterior = _PesoAnterior + PsFaseAnt
		EndScan

		_DiferencaPeso = _Pesagem - _PesoAnterior
		Scan For Rateio = 2 And NEnvs = _Nenv And Not FlagIncs
			Replace Pesos With Round(PsFaseAnt + (PsFaseAnt / Iif(_PesoAnterior = 0, 1, _PesoAnterior) * _DiferencaPeso), 2), ;
					Qtds  With Round(PsFaseAnt + (PsFaseAnt / Iif(_PesoAnterior = 0, 1, _PesoAnterior) * _DiferencaPeso), 2)
			_Pesagem = _Pesagem - Pesos			
		EndScan

		If (_RecnoRateio <> 0) And (_Pesagem <> 0)
			Go _RecnoRateio
			Replace Pesos With Pesos + _Pesagem, ;
					Qtds  With Qtds  + _Pesagem
		EndIf
	EndScan
EndIf

Select xNensiE
Set Order To Gravar
Sum Pesos To _TotGrava For Not Servicos

Select xNensiE
Scan For Qtds <> 0 Or Pesos <> 0 Or CrSigCdOpd.ReiNops = 1
	ThisForm.parentform.poDataMgr.CursorQuery('SigCdPro', 'LocalPro', 'CPros', xNensiE.CMats, 'CGrus, Varias, MoeCusfs, CustoFs, pVens, Moevs ')
	ThisForm.parentform.poDataMgr.CursorQuery('SigCdGrp', 'LocalGru', 'CGrus', LocalPro.CGrus, 'CEstoqs,pvCompos')
