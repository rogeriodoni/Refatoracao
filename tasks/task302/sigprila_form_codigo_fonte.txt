* PROCEDURE atuapreco
------------------------------------------------------------
PROCEDURE atuapreco
Local lnTipo, lcArq, lcNomArq
lnTipo = ThisForm.cntplanilha.OptTipo.Value 
lcArq = ThisForm.cntplanilha.GetPlanilha.Value
lcNomArq = Alltrim(JustFname(lcArq))

Select TmpPlanilha
Go Top 

loBarraLista = Createobject('fwprogressbar','Importando Lista...',Reccount('TmpPlanilha'))
loBarraLista.Top = loBarraLista.Top - Int(loBarraLista.Height/2) - 1
loBarraLista.Titulo.FontBold = .T.
loBarraLista.Show

Scan

	If IsEmpty(TmpPlanilha.Referencia)
		Select TmpPlanilha	
		Loop 
	EndIf 
	
	loBarraLista.Update(.T.,'Referencia: ' +TmpPlanilha.Referencia)
	
	
	plCodProd = TmpPlanilha.Referencia
	
	lnPreDe = Iif(Type([TmpPlanilha.PrecoVen])=[N],TmpPlanilha.PrecoVen,Val(TmpPlanilha.PrecoVen))
	lnPVens = Iif(Type([TmpPlanilha.PrecoEsp])=[N],TmpPlanilha.PrecoEsp,Val(TmpPlanilha.PrecoEsp))
	
	lcProdOff = Alltrim(TmpPlanilha.ProdOff)
	
	If lnPVens = 0
		lnPVens = lnPreDe
		lnPreDe = 0
	EndIf 
	
	=Thisform.Podatamgr.CursorQuery([SigCdPro],[crSigCdPro],Iif(lnTipo=1,[Cpros],Iif(lnTipo=2,[Dpro2s],[Reffs])),plCodProd,[Cpros,Dpros,PVens,Moevs,CUnis,PesoMs,cGrus,PrecoDe])
	
	Select crSigCdPro
	If Reccount([crSigCdPro])=0 Or IsEmpty(crSigCdPro.Cpros)
		Select TmpPlanilha	
		Loop 
	EndIf 
	
	lcCpros  = crSigCdPro.Cpros
	lcMoevs  = crSigCdPro.Moevs
	lnValAnt = crSigCdPro.pVens
	lnValDe	 = crSigCdPro.PrecoDe
	lcMoeAnt = []
	
	If lnPVens <> crSigcdPro.PVens Or lnPreDe <> crSigcdPro.PrecoDe
		Insert Into CrSigPrPre (Cpros, cIdChaves, PVens, PVensAnt, Moevs, MoevsAnt, Datas) Values ;
					(lcCpros, fUniqueIds(), lnPVens, lnValAnt, lcMoevs, lcMoeAnt, Datetime())
	EndIf 
	
	Select 	CrSigImpPr
	Append Blank
	Replace Cpros 		With lcCpros ,;
			Referencia	With TmpPlanilha.Referencia ,;
			PrecoVen	With lnPreDe, ;
			PrecoEsp	With lnPVens, ;
			DtInicial	With TmpPlanilha.DtInicial, ;
			DtFinal		With TmpPlanilha.DtFinal, ;
			UsuImp		With Usuar, ;
			DataImp		With Datetime(), ;
			Utilizado 	With 0, ;
			Arquivos	With lcNomArq, ;
			CidChaves 	With fUniqueIds() In CrSigImpPr
	
	
	
	*-- 17/07/2017 - Varrer a tabela de tamanho e corrigir o preço conforme o PVens do Produto
	=Thisform.Podatamgr.CursorQuery([SigCdGrp],[crSigCdGrp],[cGrus],crSigCdPro.cGrus,[AtuPreTam])
	If crSigCdGrp.AtuPreTam = 1
		lcQuery = [Select * From SigPrTam Where Cpros = ']+lcCpros+[' ]	
		If ThisForm.poDatamgr.SqlExecute(lcQuery,[crSigPrTam]) <1
			=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Update - crSigPrTam)])
			Return 0
		EndIf 
		
		Select crSigPrTam
		Go Top
		Scan
			lnValTam = 0
			
			If Not Empty(crSigPrTam.Percs)
				lnValTam = lnPVens + (lnPVens * (crSigPrTam.Percs/100))
			Else
				lnValTam = lnPVens
			EndIf 
			
			Replace Valor with lnValTam In crSigPrTam
			
		EndScan 
	EndIf 
	
	lcUpdate = [Update SigCdPro Set pVens = ]+Alltrim(Str(lnPVens,12,5)) + ;
						[ , PrecoDe = ]+Alltrim(Str(lnPreDe,12,5)) + ;
						Iif(Not Empty(lcProdOff),[,Prodoff = ]+Iif(lcProdOff='S','1','0'),[]) + ;
				[ Where CPros = ']+lcCpros+[' ] 
	If ThisForm.poDatamgr.SqlExecute(lcUpdate) <1
		=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Update - CrSigImpPr)])
		Return 0
	EndIf 
	
	lcDelete = [Delete from SigPrPrt Where CPros = ']+lcCpros+[' ] 
	If ThisForm.poDatamgr.SqlExecute(lcDelete) <1
		=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Delete - 1)])
		Return 0
	EndIf 
	
*!*		lcDelete = [Delete from SigPrWeb Where CPros = ']+lcCpros+[' ] 
*!*		If ThisForm.poDatamgr.SqlExecute(lcDelete) <1
*!*			=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Delete - 2)])
*!*			Return 0
*!*		EndIf 
		
	If Not Thisform.poDataMgr.Update('CrSigImpPr')
		=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Update - CrSigImpPr)])
		Return 0
	Endif
	
	If Not Thisform.poDataMgr.Update('CrSigPrPre')
		=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Update - CrSigPrPre)])
		Return 0
	Endif
	
	If Not Thisform.poDataMgr.Update('CrSigPrTam')
		=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Update - CrSigPrTam)])
		Return 0
	Endif
	
	If (Thisform.poDataMgr.Commit() < 1)
		=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Commit)])
		Return 0
	EndIf
	
	Select TmpPlanilha
Endscan
loBarraLista .Complete(.T.)

ENDPROC

------------------------------------------------------------
* PROCEDURE criaplanilha
------------------------------------------------------------
PROCEDURE criaplanilha
If Empty(Thisform.Cntplanilha.CmbTipos.Value)
	MessageBox('Tipo Inválido',16,'')
	ThisForm.Cntplanilha.CmbTipos.SetFocus
	Return .f.
EndIf 

If Empty(Thisform.Cntplanilha.getPlanilha.Value)
	MessageBox('Arquivo Inválido',16,'')
	ThisForm.Cntplanilha.cmdgetp.SetFocus
	Return .f.
EndIf 

If Used([TmpPlanilha])
	Use in TmpPlanilha
EndIf 

lcFile = Addbs(Sys(5)+Sys(2003))
If File(lcFile+'TmpPlanilha.dbf')	
	Delete File lcFile+'TmpPlanilha.dbf'
EndIf 

lcCursor = 'Create Table TmpPlanilha ('

FOR nCnt = 1 TO ThisForm.cntplanilha.List1.ListCount
   lcCursor = lcCursor + ThisForm.cntplanilha.List1.List(nCnt,2) + ','
EndFor

If Right(lcCursor,1) = ','
	lcCursor = Substr(lcCursor,1,Len(lcCursor)-1)
EndIf 

lcCursor = lcCursor +')'
lcCursor = STRTRAN(lcCursor, '|', ',')  
&lcCursor
llRecno = .f.

Try
	Select TmpPlanilha 
	lcArq = Alltrim(Thisform.Cntplanilha.getPlanilha.Value)
	Append From Alltrim(Thisform.Cntplanilha.getPlanilha.Value) Type Xl5
	llRecno = .t.
Catch
	llRecno = .F.
EndTry 

If Not llRecno 
	lcMsg = 'Năo foi possível criar o cursor temporario.' + Chr(13) 
	lcMsg = lcMsg + 'Verifique se a planilha năo está aberta ou que esteja salva com o Formato :'+Chr(13)
	lcMsg = lcMsg + 'Microsoft Excel 5.0/95 (.xls)'
	MessageBox(lcMsg,16,'Problema na Leitura da Planilha')
	Return .f.
EndIf 

Select TmpPlanilha 	
If Thisform.Cntplanilha.ChkCabecalho.Value = 1
	Go top 
	Delete In TmpPlanilha 
EndIf 





ENDPROC

		If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crTmpTam') < 1)
			=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpTam)')
			Return 0
		Endif

		Select crTmpTam
		Go Top
		If Reccount([crTmpTam])=0
			lcTam = []
		Endif
	Endif

	If Not Empty(lcCor)
		lcQuery = [Select * From SigCdCor Where Cods = ']+lcCor+[' ]
		If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crTmpCor') < 1)
			=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpCor)')
			Return 0
		Endif

		Select crTmpCor
		Go Top
		If Reccount([crTmpCor])=0
			lcCor = []
		Endif
	Endif

	lcQuery = [Select TipoEstos, Mercs, Cores, Tams, Embs, Cgrus, Dgrus, Pesos, Entregas, mtPrimas, LocalPdr ] + ;
		[From SigCdGrp ] + ;
		[Where CGrus = '] + crSigCdPro.CGrus + [' ] + ;
		[Order By TipoEstos, Mercs, Cores, Tams, Embs, Cgrus, Dgrus, Pesos, Entregas]

	If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crTmpGru') < 1)
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpGru)')
		Return 0
	Endif

	llPular = .F.
	If Empty(lcTam) And Inlist(crTmpGru.TipoEstos,3,4)
		llPular = .T.
	Endif

	If Empty(lcCor) And Inlist(crTmpGru.TipoEstos,2,4)
		llPular = .T.
	Endif

	If llPular
		Select csPrSemCT
		Append Blank
		Replace Referencia With crSigCdPro.Cpros ,;
			Qtds	   With Nvl(lnPecas,0) ,;
			Pesos	   With Nvl(lnPesos,0) ,;
			Unidade	   With Nvl(lcCunis,[]) ,;
			Valor	   With Nvl(lnPvens,0) In csPrSemCT
	Else

		If lcEmps <> lcXEmps Or lcEmpds <> lcXEmpDs
			lcXEmps  = lcEmps
			lcXEmpDs = lcEmpds
			lnVezes  = 1000
		Endif

		If lnVezes > 999
			lnNumes = fGerUniqueKey(Alltrim(lcDopes) + lcEmps)
			Insert Into CrSigMvCab (Datars,DataS,Dopes,Emps,EmpDs,Numes,MascNum,GrupoOs,ContaOs,GrupoDs,ContaDs,CIdChaves,DtAlts,EmpDopNums, usuals, Usuars, PRAZOENTS) ;
				Values (pDt,pDt,lcDopes,lcEmps,lcEmpds,lnNumes,fGerMascara(lnNumes),lcGrupoOs,lcContaOs,lcGrupoDs,lcContaDs,fUniqueIds(),Datetime(),lcEmps + lcDopes + Str(lnNumes,6),Usuar, Usuar, ldPrz)
			lnVezes = 1
		Endif


		Insert Into CrSigMvItn (Cpros,Dopes,Emps,Numes,opers,Qtds,CItens,Units,Totas,Moedas,Dpros,AQtds,CUnis,CIdChaves,DtAlts,EmpDopNums,Pesos,codbarras) ;
			Values (lcCpros,lcDopes,lcEmps,lnNumes,lcOpers,lnPecas,lnVezes,lnPvens,;
			lnPvens * lnPecas,lcMoevs,lcDpros,lnPecas,lcCunis,fUniqueIds(),Date(),lcEmps + lcDopes + Str(lnNumes,6),lnPesos,lnBarra)

		*--Lanca o Barra no SIGMVITS
		If (Not Empty(lcCor) Or Not Empty(lcTam))
			Insert Into CrSigMvIts (Emps,Dopes,Numes,Cpros,CodTams,CodCors,Qtds,AQtds,codbarras,Pesos,CIdChaves,EmpDopNums,chksubn,CItens) ;
				Values (lcEmps,lcDopes,lnNumes,lcCpros,lcTam,lcCor,lnPecas,lnPecas,lnCBars,lnPesos,fUniqueIds(),lcEmps + lcDopes + Str(lnNumes,6),.F.,lnVezes)
		Endif

		If lnEstoqs = 1

			pCidC = Dtos(pDt)+lcOpers+Transform(fGerUniqueKey(Dtos(pDt)),"@L 999999")+Thisform.Sigkey
			pSeq  = fGerUniqueKey('HISTBAR')

			Insert Into CrSigMvHst (Cpros,Datars,DataS,DtAudits,Dopes,EmpOs,Emps,opers,Numes,Qtds,Units,Totas,Grupos,;
				Estos,CIdChaves,EmpDopNums,EmpGruEsts,Oridopnums,Seqs,Pesos, Usuars, codbarras, CodTams, CodCors) ;
				Values (lcCpros,pDt,pDt,Null,lcDopes,lcEmps,lcEmps,;
				lcOpers,lnNumes,lnPecas,lnPvens,lnPvens * lnPecas,lcGrupoOs,;
				lcContaOs,pCidC,lcEmps + lcDopes + Str(lnNumes,6),;
				lcEmps + lcGrupoOs + lcContaOs,lcEmps + lcDopes + Str(lnNumes,6),pSeq,lnPesos, Usuar, ;
				lnCBars, lcTam, lcCor)

			=SqlCalcP2( CrSigMvHst.Emps, CrSigMvHst.Grupos, CrSigMvHst.Estos, CrSigMvHst.Cpros, CrSigMvHst.DataS, CrSigMvHst.CodCors, CrSigMvHst.CodTams, Thisform.poDataMgr )
			=fRecalculaC( CrSigMvHst.Emps, CrSigMvHst.Cpros, CrSigMvHst.DataS, Thisform.poDataMgr )

		Endif

		If Not Empty(lnCBars) And lcOpers = 'S'
			lcUpdate = [Update SigOpEtq Set Contas = Space(10), Grupos = Space(10) Where Cbars = ]+Str(lnCBars,14)
			If Thisform.poDataMgr.SqlExecute(lcUpdate) < 1
				Messagebox('Favor reinicializar o processo !!!',16,'Falha na Conexăo (lcUpdate - Transferencia)')
				Return .F.
			Endif
		Endif

		lnVezes = lnVezes + 1
	Endif
	Select csTransf
Endscan
loBarraLista .Complete(.T.)

lnOldSel = Select()
Select csPrNAOCad
Go Top
If Reccount([csPrNAOCad])>0
	Messagebox([Houve Produtos năo Importados]+Chr(13)+Chr(13)+[Arquivo : ]+Addbs(Sys(5)+Sys(2003))+'TRANSFERENCIA_NAO_IMPORTADA.XLS',0+64,[Atençăo],5000)
	Select csPrNAOCad
	Copy To Addbs(Sys(5)+Sys(2003))+'TRANSFERENCIA_NAO_IMPORTADA' Xl5
Endif

Select csPrSemCT
Go Top
If Reccount([csPrSemCT])>0
	Messagebox([Houve produtos năo Importados]+Chr(13)+Chr(13)+[Arquivo : ]+Addbs(Sys(5)+Sys(2003))+'PRODUTOS_SEM_CORTAMANHO.XLS',0+64,[Atençăo],5000)
	Select csPrSemCT
	Copy To Addbs(Sys(5)+Sys(2003))+'PRODUTOS_SEM_CORTAMANHO' Xl5
Endif

Select(lnOldSel)

lcMsg = 'Importaçăo da Planilha de Tranferencia: ' + Alltrim(Thisform.cntplanilha.getPlanilha.Value)
=fGravarLog('T', lcDopes, [SIGPRILA], lcMsg, Usuar )

If lnErro1>0
	lnErro1 = Iif(Thisform.poDataMgr.Update('CrSigMvCab'), 1, -1)
Endif

Endif
If lnErro1>0
	lnErro1 = Iif(Thisform.poDataMgr.Update('CrSigMvIts'), 1, -1)
Endif
If lnErro1>0
	lnErro1 = Iif(Thisform.poDataMgr.Update('CrSigMvHst'), 1, -1)
Endif

Endif
If lnErro1>0
	lnErro1 = Iif(fRecalculaC(.T.,.F.,.F.,Thisform.poDataMgr), 1, -1)
Endif
If lnErro1>0

Endif
If lnErro1 < 1

	Thisform.poDataMgr.Rollback()
	Return
Endif

ENDPROC

* PROCEDURE listapreco
------------------------------------------------------------
PROCEDURE listapreco
&& Rotina da Importaçăo da Lista de Preço

Private plPrecos,plCodLista,pPro, plEmps
plPrecos   = Replicate (Chr(254),30)	
plCodLista = Replicate (Chr(254),30)
pPro	   = Replicate (Chr(254),10)
plEmps	   = Replicate (Chr(254),3)

Select Distinct NomeLista, Emps From TmpPlanilha Into Cursor crNomeLista
Select crNomeLista && Se tiver mais de 1 lista por planilha

loBarraLista = CreateObject('fwprogressbar','Importando Lista...',Reccount('crNomeLista'))
loBarraLista.Top = loBarraLista.Top - Int(loBarraLista.Height/2) - 1
loBarraLista.Titulo.FontBold = .t.
loBarraLista.Show
Scan

 	loBarraLista.Update(.T.,'Lista: ' +crNomeLista.NomeLista)
 	plCodLista = crNomeLista.NomeLista
 	plEmps 	   = crNomeLista.Emps
 	
 	If Not Empty(plEmps)
		lcQuery = [Select lPrecos ] + ;
				    [From SigCdLpc ] + ;
				   [Where Emps = '] + plEmps + [']

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, [LocalBus]) < 1)
			=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (LocalBus 1)])
			Select crNomeLista
			Return .f.
		EndIf
		Go Top In LocalBus
		If Not Eof([LocalBus])
			=Messagebox([Já Existe uma Lista de Preço Cadastrado para a Empresa : ] + plEmps, 48, [Atençăo!!!])
			Select crNomeLista
			loop
		EndIf
	EndIf
	
 	Select crNomeLista
	ThisForm.Podatamgr.Requery('crSigCdLpc')
	Select crSigCdLpc
	If Eof('crSigCdLpc')
		Append Blank
		Replace lPrecos	 With crNomeLista.NomeLista ,;
				Codigos  With fGerUniqueKey('SigCdLpc') ,; 
				Formulas With 2 ,;
				Ncomiss  With 2 ,;
				Nvencs   With 2 ,;
				Tipos    WITH ' ' ,;
				Emps	 With Nvl(crNomeLista.Emps,'') ,;
				CidChaves With fUniqueIds() In crSigCdLpc

		*--------- Produtos
		ThisForm.Podatamgr.Requery('crSigCdLpi')
		Select * from TmpPlanilha Where NomeLista = crNomeLista.NomeLista Into Cursor crTmpPlanilha
		Select crTmpPlanilha

		Replace nQtdes with Reccount('crTmpPlanilha') In crSigCdLpc
		
		loBarraProLi = CreateObject('fwprogressbar','Importando Lista...',Reccount('crTmpPlanilha'))
		loBarraProLi.Top = loBarraProLi.Top + Int(loBarraProLi.Height/2) + 1
		loBarraProLi.Show
		Scan 
			pPro = crTmpPlanilha.cPros
			ThisForm.Podatamgr.Requery('crSigCdPro')
			loBarraProLi.Update(.T.,'Produto: ' + crSigCdPro.cPros + ' - ' + crSigCdPro.dPros)	
			Select 	crSigCdLpi
			Append Blank
			Replace cGrus 	With crSigCdPro.cGrus ,;
					cPros	With crSigCdPro.cPros ,;
					dPros	With crSigCdPro.dPros ,;
					MoeVs	With crSigCdPro.MoeVs ,;
					pCuss	With crSigCdPro.pCuss ,;	
					Ean13	With crSigCdPro.Ean13 ,;					
					Reffs	With crSigCdPro.Reffs ,;	
					lPrecos	With crTmpPlanilha.NomeLista ,;
					pVens	With Iif(Type([crTmpPlanilha.Valor])=[N],crTmpPlanilha.Valor,Val(crTmpPlanilha.Valor)), ;
					PrecoDe	With Iif(Type([crTmpPlanilha.PrecoDe])=[N],crTmpPlanilha.PrecoDe,Val(crTmpPlanilha.PrecoDe)), ;
					Vencis	With crTmpPlanilha.DataIni, ;
					Vencfs	With crTmpPlanilha.DataFim, ;
					CidChaves With fUniqueIds() In crSigCdLpi
		EndScan 
		loBarraProLi.Complete(.T.)
		
		If Not ThisForm.poDataMgr.Update('crSigCdLpc')
			=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Update - crSigCdLpc)])
			Return 0
		EndIf
		
		If Not ThisForm.poDataMgr.Update('crSigCdLpi')
			=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Update - crSigCdLpi)])
			Return 0
		EndIf
		
		If (ThisForm.poDataMgr.Commit() < 1)
			=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Commit)])
			Return 0
		EndIf	
	Else
		MessageBox('Lista ' + Alltrim(plCodLista) + ' já cadastrada'+Chr(13)+'Processo Abortado',16,'')
		Exit
		Return .f.
	EndIf 	
EndScan 
loBarraLista .Complete(.T.)
ENDPROC

* PROCEDURE pedidocons
------------------------------------------------------------
PROCEDURE pedidocons

&& Rotina da Importaçăo da Planilha de transferencias
Private pDt
pDt = fDtoSQL(Datetime())

If Used([csPrNAOCad])
	Use In csPrNAOCad
Endif

If Used([csPrSemCT])
	Use In csPrSemCT
Endif

Create Cursor csPrSemCT( Referencia c(25), Unidade c(3), Qtds N(12,2), Pesos N(12,2), Valor N(12,2))

Select crSigCdPac

=Thisform.poDataMgr.CursorQuery([SigOpCdd],[crSigOpCdd],[Dopes],lcDopes)

	Return .F.
Endif

	Messagebox([Grupo ou Conta de Destino năo definido na Movimentaçăo ]+Alltrim(crSigCdOpe.Dopes),0+64,[Atençăo])
	Return .F.
Endif

lcOpers   = Iif(crSigCdOpe.opers=1,'E','S')

lcContaDs = crSigCdOpe.ConDests

lcContaCs = crSigOpCdd.ContasCC

lnQtdIni = 0
Scan For !Deleted()
	If IsEmpty(TmpPlanilha.Empds) Or IsEmpty(TmpPlanilha.Emps)
		Loop
	Endif
	lnQtdIni = lnQtdIni + 1
Endscan

Select Distinct * From TmpPlanilha Into Cursor csTransf Readwrite

Select csTransf
Index On Empds Tag Empresa
Go Top

lnQtdFim = 0
Scan For !Deleted()
	If IsEmpty(csTransf.Empds) Or IsEmpty(TmpPlanilha.Emps)
		Loop
	Endif
	lnQtdFim = lnQtdFim + 1
Endscan

If lnQtdIni <> lnQtdFim
	Messagebox([Há Registros duplicados, Favor Verificar a Planilha.],0+48,[Atençăo])
	Return .F.
Endif

lnVezes  = 1
lcXEmps  = [X]
lcXEmpDs = [X]

loBarraLista = Createobject('fwprogressbar','Importando Transferencias...',Reccount('csTransf'))
loBarraLista.Top = loBarraLista.Top - Int(loBarraLista.Height/2) - 1
loBarraLista.Titulo.FontBold = .T.
loBarraLista.Show

Scan

	loBarraLista.Update(.T.,'Empresa: ' +csTransf.Empds)

	lcEmps  = csTransf.Emps
	lcEmps  = Padr(Alltrim(lcEmps),3)
	lcEmpDs = csTransf.Empds
	lcEmpDs = Padr(Alltrim(lcEmpDs),3)
	lnPecas = csTransf.Qtds
	lcProd  = csTransf.Cpros
	lnCBars = 0
	lcCor   = csTransf.Cors
	lcTam	= csTransf.Tams
	lnPeso	= csTransf.Pesos
	lcCta 	= csTransf.ContaOs
	ldPrz	= csTransf.Prazo

	If lnPecas = 0
		Select csTransf
		Loop
	Endif

	If Empty(lcEmps) Or Empty(lcEmpDs)
		Select csTransf
		Loop
	Endif

	If Empty(lcCta)
		Select csTransf
		Loop
	Endif

	If Empty(lcProd)
		Select csTransf
		Loop
	Endif

	If Not Empty(lcTam)
		lcQuery = [Select * From SigCdTam Where Cods = ']+lcTam+[' ]
		If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crTmpTam') < 1)
			=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpTam)')
			Return 0
		Endif

		Select crTmpTam
		Go Top
		If Reccount([crTmpTam])=0
			lcTam = []
		Endif
	Endif

	If Not Empty(lcCor)
		lcQuery = [Select * From SigCdCor Where Cods = ']+lcCor+[' ]
		If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crTmpCor') < 1)
			=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpCor)')
			Return 0
		Endif

		Select crTmpCor
		Go Top
		If Reccount([crTmpCor])=0
			lcCor = []
		Endif
	Endif

	lnBarra = 0
	If Not Empty(lnCBars) And (Not Empty(lcCor) Or Not Empty(lcTam))
		lnBarra = 0
	Else
		lnBarra = lnCBars
	Endif

	=Thisform.poDataMgr.CursorQuery([SigCdEmp],[crSigCdEmp],[Cemps],lcEmps,[Cemps])
	Select crSigCdEmp
	If Reccount([crSigCdEmp])=0
		Select csTransf
		Loop
	Endif

	=Thisform.poDataMgr.CursorQuery([SigCdEmp],[crSigCdEmp2],[Cemps],lcEmpDs,[Cemps])
	Select crSigCdEmp2
	If Reccount([crSigCdEmp2])=0
		Select csTransf
		Loop
	Endif

	=Thisform.poDataMgr.CursorQuery([SigCdCli],[crSigCdCli],[Iclis],lcCta,[Iclis])
	Select crSigCdCli
	If Reccount([crSigCdCli])=0
		Select csTransf
		Loop
	Endif

	=Thisform.poDataMgr.CursorQuery([SigCdPro],[crSigCdPro],Iif(lnTipo=1,[Cpros],Iif(lnTipo=2,[Dpro2s],[Reffs])),lcProd,[Cpros,Dpros,PVens,Moevs,CUnis,PesoMs,ForaLinha,Situas,custofs,moecusfs,cgrus])

	Select crSigCdPro
	If Reccount([crSigCdPro])=0 Or IsEmpty(crSigCdPro.Cpros)

		Select csPrNAOCad
		Append Blank
		Replace Produto 	With Nvl(lcProd,[]) ,;
			Quantidade  With Nvl(lnPecas,0) ,;
			Cor	   		With Nvl(lcCor,[]) ,;
			Tamanho	    With Nvl(lcTam,[]) ,;
			Barra	    With Nvl(lnCBars,0) ,;
			Origem 		With Nvl(lcEmps,[]) ,;
			Destino		With Nvl(lcEmpDs,[]) In csPrNAOCad

		Select csTransf
		Loop
	Endif

	If crSigCdPro.ForaLinha = 1
		If Messagebox([Produto ]+Alltrim(crSigCdPro.Cpros)+[ Descontinuado, Deseja importar mesmo descontinuado?],4+32,[Atençăo]) <> 6
			Select csTransf
			Loop
		Endif

	Endif

	If crSigCdPro.Situas = 2
		If Messagebox([Produto ]+Alltrim(crSigCdPro.Cpros)+[ Inativo, Deseja Importar mesmo Inativo?],4+32,[Atençăo]) <> 6
			Select csTransf
			Loop
		Endif
	Endif

	lcCpros = crSigCdPro.Cpros
	lcDpros = crSigCdPro.Dpros
	If lnTpPre = 1
		lnPvens = crSigCdPro.PVens
		lcMoevs = crSigCdPro.Moevs
	Else
		lnPvens = crSigCdPro.custofs
		lcMoevs = crSigCdPro.moecusfs
	Endif
	lcCunis = crSigCdPro.CUnis
	lnPesos = Iif(Empty(lnPeso),crSigCdPro.PesoMs,lnPeso)

	If Not Empty(lnBarra)

		=Thisform.poDataMgr.CursorQuery([SigOpEtq],[crSigOpEtq],[Cbars],lnBarra,[Cbars,CPros])

		If Reccount([crSigOpEtq])=0
			Select csTransf
			Loop
		Endif

		If crSigOpEtq.Cpros <> lcCpros
			Select csTransf
			Loop
		Endif
	Endif

	lcQuery = [Select TipoEstos, Mercs, Cores, Tams, Embs, Cgrus, Dgrus, Pesos, Entregas, mtPrimas, LocalPdr ] + ;
		[From SigCdGrp ] + ;
		[Where CGrus = '] + crSigCdPro.CGrus + [' ] + ;
		[Order By TipoEstos, Mercs, Cores, Tams, Embs, Cgrus, Dgrus, Pesos, Entregas]

	If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crTmpGru') < 1)
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpGru)')
		Return 0
	Endif

	llPular = .F.
	If Empty(lcTam) And Inlist(crTmpGru.TipoEstos,3,4)
		llPular = .T.
	Endif

	If Empty(lcCor) And Inlist(crTmpGru.TipoEstos,2,4)
		llPular = .T.
	Endif

	If llPular
		Select csPrSemCT
		Append Blank
		Replace Referencia With lcCpros ,;
			Qtds	   With Nvl(lnPecas,0) ,;
			Pesos	   With Nvl(lnPesos,0) ,;
			Unidade	   With Nvl(lcCunis,[]) ,;
			Valor	   With Nvl(lnPvens,0) In csPrSemCT
	Else
		If lcEmps <> lcXEmps Or lcXEmpDs <> lcEmpDs
			lcXEmps  = lcEmps
			lcXEmpDs = lcEmpDs
			lnVezes  = 1000
		Endif

		If lnVezes > 999

		Return .F.
	Endif
	Go Top In crSigCdMtI
Endif

	If Not Used([LocalMtz])
		=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (CursorQuery - LocalMtz)])

	Endif
Else
	lcQuery = [Select IClis ] + ;
		[From SigCdCli ] + ;

If CrSigCdGcr2.LimCres = 1
	lcTitulo = 'Crédito'

	_Saldo = _Saldo + _Valor - _ValorAnt

	_chequec = 0
	_chequep = 0
	Select LocalContas
	Scan
		lcQuery = [Select sum(a.valors) as Valors, a.fpags ]+;
			[From SigChe a, SigCdOpt b Where a.GrClis = ']+_Grupo+[' ]+;
			[And a.Iclis = '] + LocalContas.IClis + [' And a.umovs = b.operacaos And b.tipos='DB' Group By a.Fpags ]

		If (Thisform.poDataMgr.SqlExecute(lcQuery, 'Cheques') < 1)
			=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Cheques)')
			Return .F.
		Endif

		Select Cheques
		Scan
			If IsEmpty(Fpags)
				_chequep = _chequep + valors
			Else
				=Thisform.poDataMgr.CursorQuery('SigOpFp','crSigOpFp','Fpags',xPar.Fpags)
				*--Rafael - 30/06/2016 - Nao checa o Limite de Credito pela condiçăo de pagamento
				If crSigOpFp.ChkLimCre = 2
					Loop
				Endif
				If crSigOpFp.ValPends # 1
					_chequec = _chequec + valors
				Else
					_chequep = _chequep + valors
				Endif
			Endif
		Endscan
		_chequec = Round( _chequec / _CotLim, 2 )
		_chequep = Round( _chequep / _CotLim, 2 )

		If crSigCdPam.ChqlCreds
			_Saldo = _Saldo + _chequec
		Endif
	Endscan
Endif

If CrSigCdGcr2.LimEstoqs = 1 And !IsEmpty( crSigCdPam.MoeLimes )
	lcTitulo = 'Estoque'
	lcChave  = CrSigMvCab.Emps + CrSigMvCab.Dopes + Str(CrSigMvCab.Numes,6)

	_MoeLim = crSigCdPam.MoeLimes
		
	_CotLim = fCarregarCambio(crSigCdPam.MoeLimes, _pData)

	If CrSigCdGcr2.LimCres <> 1
		_Limite = 0
		_Saldo  = 0

		Select crSigClLcr
		Scan
			_CotLmc = fCarregarCambio(Moedas, _pData)
			_Limite = _Limite + ( LimCres * Iif(Moedas#_MoeLim,_CotLmc / _CotLim, 1 ) )
		Endscan
	Endif
	=Thisform.poDataMgr.CursorQuery('SigCdOpe','crSigCdOpe','dopes',CrSigMvCab.Dopes)

	Select LocalContas
	Scan
		If crSigCdOpe.EstDests = 1
			lcQuery = [Select a.Cpros, Sum(a.Sqtds) as Qtds From SigMvEst a, SigCdGcr b ]+;
				[Where a.Estos = '] + LocalContas.IClis + [' And a.Grupos = b.codigos And b.chkLimEsts <> 2 Group by a.Cpros ]
			If (Thisform.poDataMgr.SqlExecute(lcQuery, 'CrEst') < 1)
				=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Estoque)')
				Return .F.
			Endif

			lcQuery = [Select Cpros, Sum(Qtds*-1) as Qtds From SigMvItn Where EmpDopNums = ']+lcChave+[' Group By Cpros ]
			If (Thisform.poDataMgr.SqlExecute(lcQuery, 'CrEsti') < 1)
				=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Estoque 2)')
				Return .F.
			Endif

			Select Cpros, Sum(qtds) As qtds From crEst Group By Cpros ;
				Union All;
				Select Cpros, Sum(qtds) As qtds From crTpmMvItn Group By Cpros;
				Union All;
				Select Cpros, Sum(qtds) As qtds From crEsti Group By Cpros;
				Into Cursor TmpEst
		Else
			lcQuery = [Select a.Cpros, Sum(a.Sqtds) as Qtds From SigMvEst a, SigCdGcr b ]+;
				[Where a.Estos = '] + LocalContas.IClis + [' And a.Grupos = b.Codigos And b.ChkLimEsts <> 2 Group by a.Cpros ]
			If (Thisform.poDataMgr.SqlExecute(lcQuery, 'TmpEst') < 1)
				=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpEst)')
				Return .F.
			Endif
		Endif

		Select TmpEst
		Scan
			=Thisform.poDataMgr.CursorQuery('SigCdPro','CrTmpPro','Cpros',TmpEst.Cpros, [Cpros, Moevs, pvens, fvendas, moepvs, pesoms, custofs, moecusfs, Linhas, cunis, cproeqs, sgrus, mercs,CodFinP])
			=Thisform.poDataMgr.CursorQuery('SigCdLin','CrTmpLin','Linhas',CrTmpPro.Linhas)

			If crSigCdPam.TpPrecos = 1
				m.moeEst = CrTmpPro.Moevs
				m.vlEst  = Round( CrTmpPro.pvens * Iif( CrTmpPro.fvendas # 0 And !IsEmpty( CrTmpPro.moepvs ), CrTmpPro.fvendas, 1 ) , 2 )
				If CrTmpLin.TpVendas = '2'
					m.vlEst = CrTmpPro.Pesoms * m.vlEst
				Endif
			Else
				m.vlEst = CrTmpPro.Custofs
				If CrTmpLin.TpCustos = '2'
					m.vlEst = CrTmpPro.Pesoms * m.vlEst
				Endif
				m.moeEst = CrTmpPro.Moecusfs
			Endif
			m.vlEst = m.vlEst * TmpEst.qtds

			If _MoeLim # m.moeEst
				_Saldo = _Saldo + ( m.vlEst * fCarregarCambio(m.moeEst, _pData) / _CotLim )
			Else
				_Saldo = _Saldo + m.vlEst
			Endif
		Endscan
	Endscan
Endif

llUpdate = .T.

If CrSigCdGcr2.LimCres = 1 Or (CrSigCdGcr2.LimEstoqs = 1 And !IsEmpty( crSigCdPam.MoeLimes ) )

	If _Limite <> 0 And _Saldo = 0
		_Saldo = _Valor
	Endif

	If _Saldo > _Limite And _ValidaLim

		Messagebox( 'O Limite de '+lcTitulo+' da conta de ' + Alltrim( _Tipo ) + ' é de : ' + ;
			Transf( _Limite, '999,999,999.99' ) + ' ' + Alltrim( _MoeLim ) + '. ' + Chr( 13 ) + ;
			'O saldo desta conta somado ao valor ' + Chr( 13 ) + ;
			'da movimentaçăo excedeu o limite de '+lcTitulo+' : ' + ;
			Transf( _Saldo, '999,999,999.99' ) + ' ' + Alltrim( _MoeLim ) + '. ' + Chr( 13 ) , 48, '' )
		Do Form SigOpSen With 'LIMITE', _Conta, ' ', CrSigMvCab.Numes, _pData, CrSigMvCab.Emps To _Retorno
		llOk = ( Left( _Retorno, 1 ) = "*" )

		If !llOk
			llUpdate = .F.
			Return .F.
		Else
			llUpdate = .T.
		Endif
	EndIf
	
	ThisForm.ValorAcumulado = ThisForm.ValorAcumulado + _Valor 
	
	lcUpdate = [Update SigCdMti set ]+_CmpUp+[ = ]+Alltrim(Str((_Valor + _SaldoMov),13,2))+[ Where cIdChaves = ']+crSigCdMtI.cIdChaves+[']
	If (Thisform.poDataMgr.SqlExecute(lcUpdate) < 1)
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (SigCdMti)')
		Return .F.
	Endif

Endif

Return .T.

ENDPROC

------------------------------------------------------------
* PROCEDURE pedidofab
------------------------------------------------------------
PROCEDURE pedidofab

&& Rotina da Importaçăo da Planilha de transferencias
Private pDt
pDt = fDtoSQL(Datetime())

If Used([csPrNAOCad])
	Use In csPrNAOCad
Endif

If Used([csPrSemCT])
	use in csPrSemCT
EndIf 

Create Cursor csPrNAOCad(Origem c(10),Destino c(10),Produto c(20),Quantidade N(12,2),Grupo c(10),Barra N(14,0), Cor c(4),Tamanho c(4))

Create Cursor csPrSemCT( Referencia c(25), Unidade c(3), Qtds N(12,2), Pesos N(12,2), Valor N(12,2))

Local lnErro1
lnErro1 = Thisform.poDataMgr.pnIdConn

lnTipo = Thisform.cntplanilha.OptTipo.Value
lnTpPre = Thisform.cntplanilha.OptPreco.Value

lcEmpresa = []
If (Thisform.poDataMgr.SQLExecute([Select Cemps From SigCdEmp Where ChkEsc = 1], [crChkEsc]) < 1)
	=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo- crChkEsc)])
	Return .F.
Endif

lcEmpresa = crChkEsc.Cemps

Select TmpPlanilha
Select Distinct Contaos, Dopes, PrazoEnts From TmpPlanilha ;
	where Not Empty(Contaos) And Not Empty(Cpros) And Not Empty(Qtds) And Not Empty(Dopes) ;
	into Cursor csMovPlan Readwrite

Select csMovPlan
Go Top
Scan

	lcDopes = csMovPlan.Dopes
	lcGrupo = []

	=Thisform.poDataMgr.CursorQuery([SigCdOpe],[crSigCdOpe],[Dopes],lcDopes)
	=Thisform.poDataMgr.CursorQuery([SigOpCdd],[crSigOpCdd],[Dopes],lcDopes)

	Select crSigCdOpe
	If Empty(crSigCdOpe.GruOrigs)
		=Thisform.poDataMgr.CursorQuery([SigCdCli],[crTmpCli],[Iclis],csMovPlan.Contaos,[Grupos])
		lcGrupo = crTmpCli.Grupos
	Else
		lcGrupo = crSigCdOpe.GruOrigs
	Endif

	If Empty(crSigCdOpe.GruDests) Or Empty(crSigCdOpe.ConDests)
		Messagebox([Grupo ou Conta de Destino năo definido na Movimentaçăo ]+Alltrim(crSigCdOpe.Dopes),0+64,[Atençăo])
		Select csMovPlan
		Loop
	Endif

	lcOpers   = Iif(crSigCdOpe.opers=1,'E','S')
	lcGrupoOs = lcGrupo
	lcContaOs = csMovPlan.Contaos
	lcGrupoDs = crSigCdOpe.GruDests
	lcContaDs = crSigCdOpe.ConDests
	lnEstoqs  = crSigCdOpe.Estoqs
	lnVezes   = 1000
	lcGrupoCs = crSigOpCdd.GruposCC
	lcContaCs = crSigOpCdd.ContasCC

	Select * From TmpPlanilha Where Contaos = lcContaOs And Dopes = lcDopes Into Cursor csPedFab Readwrite
	Select csPedFab
	Go Top
	Scan

		lcEmps  = lcEmpresa
		lcEmps  = Padr(Alltrim(lcEmps),3)
		lcEmpDs = []
		lcEmpDs = Padr(Alltrim(lcEmpDs),3)
		lnPecas = csPedFab.Qtds
		lcProd  = csPedFab.Cpros
		lnCBars = 0
		lcCor   = csPedFab.Cors
		lcTam	= csPedFab.Tams
		lnPeso	= csPedFab.Pesos
		lnUnits	= csPedFab.Units
		lcCta 	= lcContaOs
		ldPrz	= csPedFab.PrazoEnts
		lnBarra = 0
		lcMoeda = csPedFab.Moedas

		If lnPecas = 0
			Select csPedFab
			Loop
		Endif

		If Empty(lcEmps)
			Select csPedFab
			Loop
		Endif

		If Empty(lcCta)
			Select csPedFab
			Loop
		Endif

		If Empty(lcProd)
			Select csPedFab
			Loop
		Endif

		If Not Empty(lcTam)
			lcQuery = [Select * From SigCdTam Where Cods = ']+lcTam+[' ]
			If (Thisform.poDataMgr.SQLExecute(lcQuery, 'crTmpTam') < 1)
				=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpTam)')
				Return 0
			Endif

			Select crTmpTam
			Go Top
			If Reccount([crTmpTam])=0
				lcTam = []
			Endif
		Endif

		If Not Empty(lcCor)
			lcQuery = [Select * From SigCdCor Where Cods = ']+lcCor+[' ]
			If (Thisform.poDataMgr.SQLExecute(lcQuery, 'crTmpCor') < 1)
				=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpCor)')
				Return 0
			Endif

			Select crTmpCor
			Go Top
			If Reccount([crTmpCor])=0
				lcCor = []
			Endif
		Endif

		=Thisform.poDataMgr.CursorQuery([SigCdCli],[crSigCdCli],[Iclis],lcCta,[Iclis])
		Select crSigCdCli
		If Reccount([crSigCdCli])=0
			Select csPedFab
			Loop
		Endif

		=Thisform.poDataMgr.CursorQuery([SigCdPro],[crSigCdPro],Iif(lnTipo=1,[Cpros],Iif(lnTipo=2,[Dpro2s],[Reffs])),lcProd,[Cpros,Dpros,PVens,Moevs,CUnis,PesoMs,ForaLinha,Situas,custofs,moecusfs,cgrus])

		Select crSigCdPro
		If Reccount([crSigCdPro])=0 Or IsEmpty(crSigCdPro.Cpros)


			Select csPrNAOCad
			Append Blank
			Replace Produto 	With Nvl(lcProd,[]) ,;
				Quantidade  With Nvl(lnPecas,0) ,;
				Cor	   		With Nvl(lcCor,[]) ,;
				Tamanho	    With Nvl(lcTam,[]) ,;
				Barra	    With Nvl(lnCBars,0) ,;
				Origem 		With Nvl(lcEmps,[]) ,;
				Destino		With Nvl(lcEmpDs,[]) In csPrNAOCad

			Select csPedFab
			Loop
		Endif

		*!*			If crSigCdPro.ForaLinha = 1
		*!*				If MessageBox([Produto ]+Alltrim(crSigCdPro.Cpros)+[ Descontinuado, Deseja importar mesmo descontinuado?],4+32,[Atençăo]) <> 6
		*!*					Select csPedFab
		*!*					loop
		*!*				EndIf
		*!*
		*!*			EndIf
		*!*
		*!*			If crSigCdPro.Situas = 2
		*!*				If MessageBox([Produto ]+Alltrim(crSigCdPro.Cpros)+[ Inativo, Deseja Importar mesmo Inativo?],4+32,[Atençăo]) <> 6
		*!*					Select csPedFab
		*!*					loop
		*!*				EndIf
		*!*			EndIf

		lcCpros = crSigCdPro.Cpros
		lcDpros = crSigCdPro.Dpros

		If lnTpPre = 1
			lnPvens = crSigCdPro.PVens
			lcMoevs = crSigCdPro.Moevs
		Else
			lnPvens = crSigCdPro.custofs
			lcMoevs = crSigCdPro.moecusfs
		Endif

		If Not Empty(lnUnits)
			lnPvens = lnUnits
		Endif

		If Not Empty(lcMoeda)
			lcMoevs = lcMoeda
		Endif

		lcCunis = crSigCdPro.CUnis
		lnPesos = Iif(Empty(lnPeso),crSigCdPro.PesoMs,lnPeso)

		If Not Empty(lnBarra)

			=Thisform.poDataMgr.CursorQuery([SigOpEtq],[crSigOpEtq],[Cbars],lnBarra,[Cbars,CPros])

			If Reccount([crSigOpEtq])=0
				Select csPedFab
				Loop
			Endif

			If crSigOpEtq.Cpros <> lcCpros
				Select csPedFab
				Loop
			Endif
		Endif

		lcQuery = [Select TipoEstos, Mercs, Cores, Tams, Embs, Cgrus, Dgrus, Pesos, Entregas, mtPrimas, LocalPdr ] + ;
			[From SigCdGrp ] + ;
			[Where CGrus = '] + crSigCdPro.CGrus + [' ] + ;
			[Order By TipoEstos, Mercs, Cores, Tams, Embs, Cgrus, Dgrus, Pesos, Entregas]

		If (Thisform.poDataMgr.SQLExecute(lcQuery, 'crTmpGru') < 1)
			=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpGru)')
			Return 0
		Endif

		llPular = .F.
		If Empty(lcTam) And Inlist(crTmpGru.TipoEstos,3,4)
			llPular = .T.
		Endif

		If Empty(lcCor) And Inlist(crTmpGru.TipoEstos,2,4)
			llPular = .T.
		Endif

		If llPular
			Select csPrSemCT
			Append Blank
			Replace Referencia With lcCpros ,;
				Qtds	   With Nvl(lnPecas,0) ,;
				Pesos	   With Nvl(lnPesos,0) ,;
				Unidade	   With Nvl(lcCunis,[]) ,;
				Valor	   With Nvl(lnPvens,0) In csPrSemCT
		Else

			If lnVezes > 999
				lnNumes = fGerUniqueKey(Alltrim(lcDopes) + lcEmps)
				Insert Into CrSigMvCab (Datars,DataS,Dopes,Emps,EmpDs,Numes,MascNum,GrupoOs,Contaos,GrupoDs,ContaDs,CIdChaves,DtAlts,EmpDopNums, usuals, Usuars, PrazoEnts, grupoccs, contaccs) ;
					Values (pDt,pDt,lcDopes,lcEmps,lcEmpDs,lnNumes,fGerMascara(lnNumes),lcGrupoOs,lcCta,lcGrupoDs,lcContaDs,fUniqueIds(),Datetime(),lcEmps + lcDopes + Str(lnNumes,6),Usuar, Usuar, ldPrz, lcGrupoCs, lcContaCs)
				lnVezes = 1
			Endif

			Insert Into CrSigMvItn (Cpros,Dopes,Emps,Numes,opers,Qtds,CItens,Units,Totas,Moedas,Dpros,AQtds,CUnis,CIdChaves,DtAlts,EmpDopNums,Pesos,codbarras) ;
				Values (lcCpros,lcDopes,lcEmps,lnNumes,lcOpers,lnPecas,lnVezes,lnPvens,;
				lnPvens * lnPecas,lcMoevs,lcDpros,lnPecas,lcCunis,fUniqueIds(),Date(),lcEmps + lcDopes + Str(lnNumes,6),lnPesos,lnBarra)

			*--Lanca o Barra no SIGMVITS
			If (Not Empty(lcCor) Or Not Empty(lcTam))
				Insert Into CrSigMvIts (Emps,Dopes,Numes,Cpros,CodTams,CodCors,Qtds,AQtds,codbarras,Pesos,CIdChaves,EmpDopNums,chksubn,CItens) ;
					Values (lcEmps,lcDopes,lnNumes,lcCpros,lcTam,lcCor,lnPecas,lnPecas,lnCBars,lnPesos,fUniqueIds(),lcEmps + lcDopes + Str(lnNumes,6),.F.,lnVezes)
			Endif

			If lnEstoqs = 1

				pCidC = Dtos(pDt)+lcOpers+Transform(fGerUniqueKey(Dtos(pDt)),"@L 999999")+Thisform.Sigkey
				pSeq  = fGerUniqueKey('HISTBAR')

				Insert Into CrSigMvHst (Cpros,Datars,DataS,DtAudits,Dopes,EmpOs,Emps,opers,Numes,Qtds,Units,Totas,Grupos,;
					Estos,CIdChaves,EmpDopNums,EmpGruEsts,Oridopnums,Seqs,Pesos, Usuars, codbarras, CodTams, CodCors) ;
					Values (lcCpros,pDt,pDt,Null,lcDopes,lcEmps,lcEmps,;
					lcOpers,lnNumes,lnPecas,lnPvens,lnPvens * lnPecas,lcGrupoOs,;
					lcContaOs,pCidC,lcEmps + lcDopes + Str(lnNumes,6),;
					lcEmps + lcGrupoOs + lcContaOs,lcEmps + lcDopes + Str(lnNumes,6),pSeq,lnPesos, Usuar, ;
					lnCBars, lcTam, lcCor)

				=SqlCalcP2( CrSigMvHst.Emps, CrSigMvHst.Grupos, CrSigMvHst.Estos, CrSigMvHst.Cpros, CrSigMvHst.DataS, CrSigMvHst.CodCors, CrSigMvHst.CodTams, Thisform.poDataMgr )
				=fRecalculaC( CrSigMvHst.Emps, CrSigMvHst.Cpros, CrSigMvHst.DataS, Thisform.poDataMgr )

			Endif

			lnVezes = lnVezes + 1
		Endif
		Select csPedFab
	Endscan

	lnVezes =  1

	Select csMovPlan
Endscan

lnOldSel = Select()
Select csPrNAOCad
Go Top
If Reccount([csPrNAOCad])>0
	Messagebox([Houve Produtos năo Importados]+Chr(13)+Chr(13)+[Arquivo : ]+Addbs(Sys(5)+Sys(2003))+'TRANSFERENCIA_NAO_IMPORTADA.XLS',0+64,[Atençăo],5000)
	Select csPrNAOCad
	Copy To Addbs(Sys(5)+Sys(2003))+'TRANSFERENCIA_NAO_IMPORTADA' Xl5
Endif

Select csPrSemCT
Go Top
If Reccount([csPrSemCT])>0
	Messagebox([Houve produtos năo Importados]+Chr(13)+Chr(13)+[Arquivo : ]+Addbs(Sys(5)+Sys(2003))+'PRODUTOS_SEM_CORTAMANHO.XLS',0+64,[Atençăo],5000)
	Select csPrSemCT
	Copy To Addbs(Sys(5)+Sys(2003))+'PRODUTOS_SEM_CORTAMANHO' Xl5
Endif

Set Step On
Select CrSigMvItn
Go Top
Select EmpDopNums, Sum(Totas) As Total From CrSigMvItn Group By EmpDopNums Order By EmpDopNums Into Cursor csTmpTot Readwrite

Select csTmpTot
Go Top
Scan
	Update CrSigMvCab Set Valos = csTmpTot.Total Where EmpDopNums = csTmpTot.EmpDopNums
	Select csTmpTot
Endscan
ThisForm.ValorAcumulado = 0
*-- Rafael - 14/11/2017 - Limite de credito pela Matriz contabil
If Inlist( crSigCdOpe.limcres, 7 )
	Select CrSigMvCab
	Go Top
	Scan
		If Not Empty(crSigOpCdd.GruposCC) And Not Empty(crSigOpCdd.ContasCC) And Not Empty(crSigOpCdd.GruposEmt) And Not Empty(crSigOpCdd.ContasEmt)
			If !Thisform.LimiteCreditoMatriz( crSigOpCdd.GruposCC, crSigOpCdd.ContasCC, crSigOpCdd.GruposEmt, crSigOpCdd.ContasEmt, 'Centro de Custo' )
				Return .F.
			Endif
		Endif
	Endscan
Endif

Select(lnOldSel)


lcMsg = 'Importaçăo da Planilha de Tranferencia: ' + Alltrim(Thisform.cntplanilha.getPlanilha.Value)
=fGravarLog('T', lcDopes, [SIGPRILA], lcMsg, Usuar )

If lnErro1>0
	lnErro1 = Iif(Thisform.poDataMgr.Update('CrSigMvCab'), 1, -1)
Endif
If lnErro1>0
	lnErro1 = Iif(Thisform.poDataMgr.Update('CrSigMvItn'), 1, -1)
Endif
If lnErro1>0
	lnErro1 = Iif(Thisform.poDataMgr.Update('CrSigMvIts'), 1, -1)
Endif
If lnErro1>0
	lnErro1 = Iif(Thisform.poDataMgr.Update('CrSigMvHst'), 1, -1)
Endif
If lnErro1>0
	lnErro1 = Iif(SqlCalcP2(.T.,Thisform.poDataMgr), 1, -1)
Endif
If lnErro1>0
	lnErro1 = Iif(fRecalculaC(.T.,.F.,.F.,Thisform.poDataMgr), 1, -1)
Endif
If lnErro1>0
	lnErro1 = Thisform.poDataMgr.Commit()
Endif
If lnErro1 < 1
	Messagebox('Falha na Gravacao dos Dados.',0+16,'')
	Thisform.poDataMgr.Rollback()
	Return
Endif

ENDPROC

PROCEDURE pedacesso

&& Rotina da Importaçăo da Planilha de transferencias

If Used([csPrNAOCad])
	use in csPrNAOCad
EndIf 

If Used([csPrSemCT])

PROCEDURE cotacao
LParameters _xMoeda, _xData
Local _xCotacao, lcAlias

lcAlias   = Alias()
_xCotacao = 1
_xData    = Iif(Type('_xData') = 'D', _xData, CrSigMvCab.Datas)

If Not IsEmpty( _xMoeda ) 
	lcSql = [Select Cotas From SigCdMoe where Cmoes = ']+_xMoeda+[']
	If ThisForm.poDataMgr.Sqlexecute(lcsql,'LocalMoe') < 1
		MessageBox([Favor reiniciar o Processo.],0+16,[Falha de conexăo (LocalMoe)])
		Return .F.
	EndIf 
	If LocalMoe.Cotas
		If !Seek( _xMoeda, "TmpCot" )
			_xCotacao = fBuscarCotacao(_xMoeda,_xData,ThisForm.poDataMgr)
			Insert Into TmpCot ( Cmoes, Valos ) Values ( _xMoeda, _xCotacao )
		Else
			If TmpCot.Valos = 0
				_xCotacao = fBuscarCotacao(_xMoeda,_xData,ThisForm.poDataMgr)
				Replace Valos With _xCotacao In TmpCot
			EndIf
			_xCotacao = TmpCot.Valos
		Endif

		*** Tiago - 08/09/2015 - Alterado para gravar a cotaçăo na tabela SIGMVMOV pois no SIGMVITN já está gravado convertido e perde a cotaçăo caso tenha sido alterado
		Select TmpTotal
		Set Order To tag Moeds		
		If !Seek(_xMoeda)
			Insert Into TmpTotal (Moeds) Values (_xMoeda)
		Endif
		Replace MoeVals With _xCotacao In TmpTotal		
	Else
		_xCotacao = 1
	EndIf
EndIf

If Not IsEmpty(lcAlias)
	Select (lcAlias)
EndIf

Return _xCotacao

ENDPROC
