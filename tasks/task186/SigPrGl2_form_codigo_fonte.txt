EndIf

------------------------------------------------------------
* PROCEDURE Click
------------------------------------------------------------
PROCEDURE Click
Private pEdI, pEdF, pItn

Select TmpCabec
Set Order To EmpDopNum

_Contador = 0
Go top
lcJob = TmpCabec.Jobs
Scan For Flag
	_Contador = _Contador + 1
	*** Tiago - 10/04/2014 - SRCOM - Alterado para checar operaçőes com JOB diferente
	If lcJob <> TmpCabec.Jobs
		Messagebox([Năo é permitido gerar OPs de operaçőes com Jobs diferentes.],48,[Aviso])
		Return .f.
	Endif	
EndScan

If (_Contador = 0)
	=Messagebox('Nenhuma Operaçăo Foi Selecionada!!!', 32, '')
	ThisForm.GradeOperaçăo.Column1.SetFocus
	Return 0
EndIf

*****Preparando Estoque Disponível
lcCidQuerys = fUniqueIds()

lcQuery = [Select * From SigCdCeg Where TpCads <> 1]
If ThisForm.PodataMgr.SqlExecute(lcquery,'TmpCeg') < 1
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpCeg)')
	Return 0
EndIf

Select CrSigTempd
Zap

Select TmpCeg
If Reccount() > 0
	Scan
		Select CrSigTempd
		Append Blank
		Replace Grupos    With TmpCeg.Grupos,;
				Contas    With TmpCeg.Contas,;
				codObs    With TmpCeg.Priors,;
				Emps      With TmpCeg.Emps  ,;
				CidChaves With fUniqueIds(),;
				CidQuerys With lccidquerys
	EndScan
Else
	Select CrSigTempd
	Append Blank
	Replace Grupos    With crSigCdPam.GrupoEsts,;
			Contas    With crSigCdPam.contaEsts,;
			CodObs    With 1,;
			Emps      With _Empr,;
			CidChaves With fUniqueIds(),;
			CidQuerys With lccidquerys
EndIf

If !ThisForm.Podatamgr2.UpDate('CrSigTempd')
	=ThisForm.Podatamgr2.RollBack()
	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo (CrSigTempD)')
	Return .f.
EndIf
If ThisForm.PoDataMgr2.commit() < 1
	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo (Commit)')
	Return .f.
EndIf

*** Tiago - 04/10/11 - Vianna - Inclui o Union para trazer o estoque das prioridades que tiver só grupo
lcQuery = [Select a.*, b.CodObs as Priors From SigMvEst a, SigTempd b ]+;
		[Where a.Grupos = b.Grupos And a.Estos = b.Contas And a.Emps = b.Emps And a.Sqtds > 0 ] +;
		[Union ] +;
		[Select a.*, b.CodObs as Priors From SigMvEst a, SigTempd b ] +;
		[Where a.Grupos = b.Grupos and b.Contas = '' And a.Emps = b.Emps And a.Sqtds > 0]
		
If ThisForm.PodataMgr.SqlExecute(lcquery,'TmpEstoque') < 1
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpEstoque)')
	Return 0
EndIf

ThisForm.PodataMgr2.fSqlApagarTmp('SigTempD', lcCidquerys )

Create Cursor TmpSaldo ( CPros c(14), CodCors c(4), CodTams c(4), Saldo n(12,3), Disps n(12,3), Fabrs n(12,3), DispFs n(12,3) )
Index on CPros + CodCors + CodTams Tag CPros

Create Cursor TmpSaldg ( Emps c(3), Grupos c(10), Estos c(10), CPros c(14), CodCors c(4), CodTams c(4), Saldo n(12,3), Disps n(12,3), Priors n(2), Reservs n(12,3) )
Index on CPros + CodCors + CodTams + Str(priors,2) + Grupos + Estos + Emps Tag CPros
Index On Emps + Grupos + Estos + CPros + CodCors + CodTams Tag GruEstPro

Select TmpEstoque
Scan
	Select TmpSaldo
	If Not Seek(TmpEstoque.Cpros + TmpEstoque.CodCors + TmpEstoque.CodTams)
		Insert Into TmpSaldo ( CPros, CodCors, CodTams, Saldo, Disps );
					Values ( TmpEstoque.CPros, TmpEstoque.CodCors, TmpEstoque.CodTams, 0, 0 )
	EndIf
	Replace Saldo With Saldo + TmpEstoque.Sqtds,;
			Disps With Disps + TmpEstoque.Sqtds In TmpSaldo

	Insert Into TmpSaldg ( Grupos, Estos, CPros, CodCors, CodTams, Saldo, Disps, Priors, Emps );
				Values ( TmpEstoque.Grupos, TmpEstoque.Estos, TmpEstoque.CPros, TmpEstoque.CodCors,;
						 TmpEstoque.CodTams, TmpEstoque.SQtds, TmpEstoque.SQtds, TmpEstoque.Priors, TmpEstoque.Emps )
Endscan

ThisForm.poDataMgr.CursorQuery('SigCdOpe', 'crSigCdOpe', 'Dopes', crSigCdPam.TransfRes)

If Not Empty(crSigCdPam.TransfRes) And Not Eof('crSigCdOpe') And (crSigCdOpe.Estoqs <> 1)
	pEdI = _Empr + crSigCdPam.TransfRes + Str(0, 6)
	pEdF = _Empr + crSigCdPam.TransfRes + Str(999999, 6)
	
	lcQuery = [Select EmpDopNums, GrupoOs, ContaOs, Emps, Dopes, Numes ] + ;
			    [From SigMvCab ] + ;
			   [Where EmpDopNums BetWeen ?pEdI And ?pEdF ] + ;
			   [Order By EmpDopNums]

	If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'TempEest') < 1)
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TempEest)')
		Return 0
	EndIf

	Select TempEest
	Scan
		lcEdn = TempEest.EmpDopNums
		ThisForm.poDataMgr.CursorQuery('SigMvItn', 'TempEestI', 'EmpDopNums', lcEdn)

		Select TempEestI
		Scan
			If (Qtds - QtBaixas > 0)
				pItn = TempEestI.CItens
				
				lcQuery = [Select * ] + ;
						    [From SigMvIts ] + ;
						   [Where EmpDopNums = '] + lcEdn + [' And ] + ;
						   		 [CItens = ?pItn]

				If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'TempEsti2') < 1)
					=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TempEsti2)')
					Return 0
				EndIf
				Select TempEsti2
				Go Top
				If Eof('TempEsti2')
					Select TmpSaldo
					If Not Seek(TempEestI.Cpros)
						Insert Into TmpSaldo (Cpros) Value (TempEestI.CPros)
					EndIf
					Replace Saldo With Saldo - (TempEestI.Qtds - TempEestI.QtBaixas), ;
							Disps With Disps - (TempEestI.Qtds - TempEestI.QtBaixas)

					Select TmpSaldg
					Set Order To GruEstPro
					If !Seek( TempEest.Emps + TempEest.GrupoOs + TempEest.ContaOs + TempEesti.Cpros )
						Insert Into TmpSaldg ( Emps, Grupos, Estos, Cpros, Priors )  ;
									Values ( TempEest.Emps, TempEest.GrupoOs, TempEest.ContaOs, TempEesti.CPros, 99 )
					Endif
					Replace Saldo With Saldo - (TempEestI.Qtds - TempEestI.QtBaixas), ;
							Disps With Disps - (TempEestI.Qtds - TempEestI.QtBaixas)
				Else
					Select TempEsti2
					Scan
						_Saldo = TempEsti2.Qtds - TempEsti2.QtBaixas
						Select TmpSaldo
						If Not Seek(TempEsti2.Cpros + TempEsti2.CodCors + TempEsti2.CodTams)
							Insert Into TmpSaldo (Cpros, CodCors, CodTams) ;
										   Value (TempEsti2.CPros, TempEsti2.CodCors, TempEsti2.CodTams)
						EndIf
						Replace Saldo With Saldo - _Saldo, ;
								Disps With Disps - _Saldo
						Select TmpSaldg
						Set Order To GruEstPro
						If !Seek( TempEest.Emps + TempEest.GrupoOs + TempEest.ContaOs + TempEsti2.Cpros + TempEsti2.CodCors + TempEsti2.CodTams )
							Insert Into TmpSaldg ( Emps, Grupos, Estos, Cpros, CodCors, CodTams, Priors ) Value ;
												( TempEest.Emps, TempEest.GrupoOs, TempEest.ContaOs, ;
												TempEsti2.CPros, TempEsti2.CodCors, TempEsti2.CodTams, 99 )
						Endif
						Replace Saldo with Saldo - ( _Saldo ),;
								Disps with Disps - ( _Saldo )
					EndScan
				EndIf
			EndIf
		EndScan
	EndScan
EndIf

Create Cursor TmpFinal (Emps c(3), Dopes c(20), Numes n(6), CPros c(14), Qtds n(10,3), Peso n(9,3), ;
						Saldo n(10,3), Estoque n(10,3), Produzir n(10,3), Obs m Null, Obsps m Null, ;
						Datas d Null, Entregas d Null, CodCors c(4), CodTams c(4), Linhas c(10), ;
						Citens n(10), Reffs c(40), Notas c(6), dpros c(40), GrupoDs c(10), ContaDs c(10),;
						KeySelM L, Fabrs n(10,3), KeyPdes L, Jobs c(10))
Index On Cpros + CodCors + CodTams Tag Cpros

Select TmpItens
Set Key To 
Set Order to CPros
Scan
	Select TmpCabec
	Seek TmpItens.Emps + TmpItens.Dopes + Str( TmpItens.Numes, 6 )
	If Not Flag
		Loop
	EndIf

	Select TmpOper
	Seek TmpItens.Dopes

	Select TmpItens
	If (Saldo > 0)
		Store 0 To _Estoque, _Produzir
		If (TmpOper.ChkObs <> 1 and Not IsEmpty(TmpItens.Obs)) Or Not Seek(TmpItens.CPros + TmpItens.CodCors + TmpItens.CodTams, 'TmpSaldo') Or ;
		   Empty(crSigCdPam.TransfRes) Or (TmpOper.Reservas = 2 And Not ThisForm.Reserva) Or TmpSaldo.Disps < 0
			_Produzir = TmpItens.Saldo
		Else
			=Seek(TmpItens.CPros + TmpItens.CodCors + TmpItens.CodTams, 'TmpSaldo')
			_Estoque = TmpSaldo.Disps
			If (TmpSaldo.Disps >= TmpItens.Saldo)
				Replace TmpSaldo.Disps With TmpSaldo.Disps - TmpItens.Saldo
			Else
				_Produzir = TmpItens.Saldo - TmpSaldo.Disps
				Replace TmpSaldo.Disps With 0
			EndIf
		EndIf

		ThisForm.poDataMgr.CursorQuery('SigCdPro', 'crSigCdPro', 'CPros', TmpItens.CPros)
		
		Insert Into TmpFinal (Emps, Dopes, Numes, CPros, Qtds, Peso, Saldo, Estoque, Produzir, Obsps, ;
							  Obs, Datas, Entregas, CodCors, CodTams, Linhas, Citens, Reffs, Notas, ;
							  Dpros, GrupoDs, ContaDs, Jobs) ;
					  Values (TmpItens.Emps, TmpItens.Dopes, TmpItens.Numes, TmpItens.CPros, ;
					  		  TmpItens.Qtds, TmpItens.Peso, TmpItens.Saldo, TmpItens.Saldo - _Produzir, ;
					  		  _Produzir, Nvl(TmpItens.Obs,''), Nvl(TmpCabec.Obs, ''), Nvl(TmpCabec.Datas, Ctod('')), ;
					  		  Nvl(TmpCabec.Entregas, Ctod('')), TmpItens.CodCors, TmpItens.CodTams, ;
					  		  TmpItens.Linhas, TmpItens.CItens, crSigCdPro.Reffs, TmpItens.Notas, ;
					  		  TmpItens.Dpros, TmpCabec.Grupods, TmpCabec.Contads, TmpCabec.Jobs )
	EndIf
EndScan

Select TmpSaldo
Scan
	If Saldo # Disps
		xBaixa = Saldo - Disps
		Select TmpSaldG
		Set Order To Cpros
		=Seek(TmpSaldo.Cpros + TmpSaldo.CodCors + TmpSaldo.CodTams)
		Scan While Cpros = TmpSaldo.Cpros And CodCors = TmpSaldo.CodCors And CodTams = TmpSaldo.CodTams And xBaixa > 0
			If TmpSaldg.Disps >= xBaixa
				Replace TmpSaldg.Disps with TmpSaldg.Disps - xBaixa
				xBaixa = 0
			Else
				xBaixa = xBaixa - TmpSaldg.Disps
				Replace TmpSaldg.Disps with 0
			Endif
		EndScan
	Endif
EndScan

If Not Empty(crSigCdPac.DopEsts)

	Create Cursor TmpFabr (priors n(2), Nops n(10), Fases c(10), Cpros c(14), CodCors c(4), CodTams c(4), Qtds n(10,3), disps n(12,3), Reservs n(12,3) )
	Index On Cpros + CodCors + CodTams + Str(Priors,2) + Str(Nops,10) Tag Cpros

	llFalse = .f.
	lcSql = [Select a.Nops, a.Cpros, a.CodCors, a.CodTams, sum(a.Qtds) as Qtds From SigOpPic a, SigCdNec b ]+;
			[Where a.Dopes = ']+CrSigCdPac.DopEsts+[' And a.EmpDopNops = b.EmpDnps And b.Chksubn = ?llFalse ]+;
			[And a.Emps = ']+_Empr+[' group by a.Nops,a.Cpros, a.Codcors, a.Codtams ]
	=ThisForm.Podatamgr.Sqlexecute(lcSql,'TmpOpi')
			
	Select TmpOpi
	Scan

		Select TmpSaldo
		If Not Seek(TmpOpi.Cpros + TmpOpi.CodCors + TmpOpi.CodTams)
			Insert Into TmpSaldo ( CPros, CodCors, CodTams );
						Values ( TmpOpi.CPros, TmpOpi.CodCors, TmpOpi.CodTams)
		EndIf
		Replace Fabrs  With Fabrs  + TmpOpi.qtds,;
				DispFs With DispFs + TmpOpi.qtds In TmpSaldo

		Insert Into TmpFabr (Nops, Cpros, CodCors, CodTams, Qtds, Priors) Values ;
			(TmpOpi.Nops, TmpOpi.Cpros, TmpOpi.CodCors, TmpOpi.CodTams, TmpOpi.Qtds, 0 )

		lcSql = [Select GrupoDs From SigPdMvf Where Nops = ]+Str(TmpOpi.Nops,10)+[ Order by CidChaves Desc ]
		=ThisForm.Podatamgr.Sqlexecute(lcSql,'TmpMfas')

		Go Top
		Replace Fases With Tmpmfas.GrupoDs in TmpFabr
		
		Store 0 to _Estoque, _Produzir

		If Seek(TmpOpi.Cpros + TmpOpi.CodCors + TmpOpi.CodTams,'TmpFinal','Cpros')
			If TmpSaldo.Fabrs >= TmpFinal.Produzir
				_Estoque = TmpFinal.Produzir
				_Produzir = 0
				Replace TmpSaldo.Dispfs With TmpSaldo.Dispfs - TmpFinal.Produzir In TmpSaldo
			Else
				_Estoque  = TmpSaldo.Fabrs
				_Produzir = TmpFinal.Produzir - TmpSaldo.Fabrs
				Replace Dispfs With 0 In TmpSaldo
			EndIf
			Replace Produzir With _Produzir,;
					Fabrs    With _Estoque In TmpFinal
		EndIf
	EndScan

	Select TmpSaldo
	Scan
		If Fabrs # Dispfs
			xBaixa = Fabrs - DispFs
			Select TmpFabr
			Set Order To Cpros
			=Seek(TmpSaldo.Cpros + TmpSaldo.CodCors + TmpSaldo.CodTams)
			Scan While Cpros = TmpSaldo.Cpros And CodCors = TmpSaldo.CodCors And CodTams = TmpSaldo.CodTams And xBaixa > 0
				If ( TmpFabr.Qtds - TmpFabr.Disps ) >= xBaixa
					Replace TmpFabr.Disps With TmpFabr.Disps + xBaixa in TmpFabr
					xBaixa = 0
				Else
					xBaixa = xBaixa - ( TmpFabr.Qtds - TmpFabr.Disps)
					Replace TmpFabr.Disps With Qtds in TmpFabr
				Endif
			EndScan
		Endif
	EndScan

	Create Cursor TmpFinalG ( Flag c(1), CPros c(14), CodCors c(4), CodTams c(4), Linhas c(10), Qtds n(10,3), ;
					Saldo n(10,3), Estoque n(10,3), Produzir n(10,3), Fabrs n(10,3), Produzir2 n(10,3), ;
					TotVenda n(10,3), QtdMins n(10,3), KeySelM L, KeySelMP L, UsuLibs c(10))
	Index on Cpros + CodCors + CodTams Tag Cpros
	
	Select Cpros, CodCors, CodTams, Linhas, Sum(Qtds) as Qtds, sum(Saldo) as Saldo, sum(Estoque) as Estoque,;
		sum(Produzir) as Produzir, sum(Fabrs) as Fabrs From TmpFinal ;
	Into Cursor Selecao Group by Cpros, CodCors, CodTams, Linhas
	
	If CrSigCdPac.nMeses > 0
		llData = Gomonth(Date(),-CrSigCdPac.nmeses)

		lcSql = [Select a.cpros, a.qtds, b.Caixas, b.copers, b.opers, a.opers ]+;
				[From SigMvItn a, SigCdOpe b, SigMvCab c ]+;
				[Where a.EmpDopNums = c.EmpDopNums And a.Emps = ']+_Empr+[' And c.datas >= ?llData ]+;
				[And a.dopes=b.dopes and b.tipoops in(4,5) ]

		ThisForm.Podatamgr.Sqlexecute(lcSql,'LocalEest')

		Select cpros, Sum(qtds*Iif(( Caixas=1 and copers=1) or (caixas<>1 and opers=1) Or (caixas<>1 and opers=3 and opers='E'),1,-1)) as Qtds ;
		From LocalEest Group by 1 Into cursor Vendas ReadWrite
		Select Vendas
		Index On cpros Tag Cpros
	EndIf

	Select TmpFinalG
	Scatter Memvar blank
	
	Select Selecao
	Scan
		Scatter Memvar
		m.flag = '+'
		lcSql = [Select QtMinFabs From SigCdPro Where Cpros = ']+Selecao.Cpros+[']
		ThisForm.PodataMgr.Sqlexecute(lcSql,'CrSigCdPro')
		m.QtdMins = 0
		If ( CrSigCdPac.GerPcps = 2 And Not ThisForm.Reserva ) Or ;
			( CrSigCdPac.GerPcps <> 2 And ThisForm.Reserva )
			m.QtdMins = crSigCdPro.QtMinFabs
		EndIf
		
		If Used('Vendas') And Seek(m.Cpros,'Vendas','Cpros')
			m.TotVenda = Vendas.Qtds
		Else
			m.TotVenda = 0
		EndIf
		m.Produzir2 = Iif(m.QtdMins>0 And m.produzir > 0 And m.Produzir < m.QtdMins, m.QtdMins - m.Produzir,0 )
		Select TmpFinalG
		Append Blank
		Gather Memvar
	EndScan
	
EndIf

Select TmpItens
Set Order To EmpdopNum
Set Key To TmpCabec.Emps + TmpCabec.Dopes + Str(TmpCabec.Numes, 6)
Go Top

ThisForm.Enabled = .f.
If ThisForm.Reserva
	ThisForm.Processar.Enabled = .f.
EndIf

If Not Empty(crSigCdPac.DopEsts)
	Do Form SigPrGlx With ThisForm, ThisForm.Datasessionid, ThisForm.Reserva, ThisForm.EmphPdr, ThisForm.Automatico, ThisForm.Numerodaop, Thisform.PorDestino
Else
	Do Form SigPrGlp With ThisForm, ThisForm.Datasessionid, ThisForm.Reserva, ThisForm.poDataMgr, ThisForm.Emphpdr, ThisForm.Automatico, ThisForm.Numerodaop
Endif
ENDPROC
