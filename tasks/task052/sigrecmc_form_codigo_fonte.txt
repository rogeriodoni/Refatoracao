PROCEDURE mchkvalid
Local llRetorno

m.llRetorno = .T.

* Consiste o campo operaçăo
lcCdEstoque = ThisForm.getCdEstoque.Value
If !IsEmpty(lcCdEstoque) And !(ThisForm.poDataMgr.ChkRegister('SigCdCli','iclis',lcCdEstoque))
   ThisForm.pcMsg = 'Conta inválida.'
   ThisForm.getCdEstoque.SetFocus
   m.llRetorno = .F.
   EndIf
   
* Consiste o campo data inicial
If IsEmpty(ThisForm.get_dt_inicial.Value)
	ThisForm.pcMsg = 'Data inicial inválida !'
	ThisForm.get_dt_inicial.SetFocus
	m.llRetorno = .F.
EndIf

* Consiste o campo data final
If IsEmpty(ThisForm.get_dt_final.Value)
	ThisForm.pcMsg = 'Data final inválida !'
	ThisForm.get_dt_final.SetFocus
	m.llRetorno = .F.
EndIf

* Consiste os campos de datas
If ThisForm.get_dt_inicial.Value > ThisForm.get_dt_final.Value
	ThisForm.pcMsg = 'Data inicial maior que data final !'
	ThisForm.get_dt_inicial.SetFocus
	m.llRetorno = .F.
EndIf

* Consiste o campo moeda
If IsEmpty(ThisForm.get_moeda.Value) And ThisForm.Opt_SelRel.Value = 1
   ThisForm.pcMsg = 'Moeda Inválida.' + Chr(13) + 'Informe o tipo de moeda'
   ThisForm.get_moeda.SetFocus
   m.llRetorno = .F.
EndIf

* Consiste número de anos para comparativo
If Year(ThisForm.get_dt_final.Value) - Year(ThisForm.get_dt_inicial.Value) > 5 And ThisForm.Comparativo.Value = 1
	ThisForm.pcMsg = 'Intervalo máximo de anos para comparativo é 6 !'
	ThisForm.get_dt_inicial.SetFocus
	m.llRetorno = .F.
EndIf

Return(m.llRetorno)
ENDPROC
