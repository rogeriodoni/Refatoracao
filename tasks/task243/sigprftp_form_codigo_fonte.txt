@nPasswords )

nRasConn = 0

nFtp = InternetConnect( nInternet, cFtpLoc, INTERNET_DEFAULT_FTP_PORT, cFtpUsu, cFtpSen, INTERNET_SERVICE_FTP, INTERNET_FLAG_PASSIVE, 0 )

	=ThisForm.Inf("Erro na conexăo com o servidor FTP. "+allt(Thisform.InErrorCase(GetLastError())),"R" )

CREATE CURSOR ftpserver (;
	nome C(240), ;
	Tipo C(10) , ;
	Tama C(12) , ;
	data C(16))
index on nome tag nome
set order to nome

cOldDate = SET("DATE")
SET DATE TO DMY
*** Ir al directorio de la KB de FoxPro
FtpSetCurrentDirectory( nFtp, cFtpLocIni )
*** Empezar el recorrido de busqueda
cFindData = REPLICATE( CHR(0), 320  )
nFind = FtpFindFirstFile( ;
	nFtp, ;
	NULL, ;
	@cFindData, ;
	INTERNET_FLAG_RAW_DATA, ;
	0 )
nSeguir = 1
DO WHILE nSeguir # 0
	*** Obtener el nombre de la matriz
	cNome = SUBSTR( cFindData, 45, AT(CHR(0), SUBSTR(cFindData,45)) -1 )
	*** Directorio o Fichero
	IF Thisform.CToWord( SUBSTR( cFindData, 1, 4 ) )= FILE_ATTRIBUTE_DIRECTORY
		cTipo = "Diretório"
        nTama = 0
	ELSE
		cTipo = "Arquivo"
		*** Tamańo del fichero
		nTama = ThisForm.CToWord( SUBSTR( cFindData, 33, 4 ) )
	ENDIF
	*** Cambiar el formato de Fecha y Hora
	cSystemTime = REPLICATE( CHR(0), 16 )
	FileTimeToSystemTime( ;
		SUBSTR( cFindData, 21, 8 ), ;
		@cSystemTime )
	*** Obtener cada uno de los elementos
	cDia = PADL( ThisForm.CToWord( SUBSTR( ;
		cSystemTime, 7, 2 ) ), 2, "0" )
	cMes = PADL( ThisForm.CToWord( SUBSTR( ;
		cSystemTime, 3, 2 ) ), 2, "0" )
	cAno = STR( ThisForm.CToWord( SUBSTR( ;
		cSystemTime, 1, 2 ) ), 4 )
	cHora = PADL( ThisForm.CToWord( SUBSTR( ;
		cSystemTime, 9, 2 ) ), 2, "0" )
	cMinuto = PADL( ThisForm.CToWord( SUBSTR( ;
		cSystemTime, 4, 2 ) ), 2, "0" )
	tFechaHora = cDia+"/"+cMes+"/"+cAno+" "+cHora+":"+cMinuto
	*** Insertar el dato en el cursor
	INSERT INTO ftpserver ;
		VALUES ( cNome, ;
			cTipo, ;
			transform(nTama,"9999,999,999"),;
			tFechaHora )
	*** Continuar con la búsqueda
	cFindData = REPLICATE( CHR(0), 320 )
	nSeguir = InternetFindNextFile( ;
		nFind, ;
		@cFindData )
ENDDO

InternetCloseHandle( nFtp )
InternetCloseHandle( nInternet )


SET DATE TO (cOldDate)
sele FtpServer
GO TOP
RETURN .t.
ENDPROC

PROCEDURE renameftpfile
LPARAMETERS lcOldFile, lcNewFile 

#DEFINE FILE_ATTRIBUTE_NORMAL          128
#DEFINE FILE_ATTRIBUTE_DIRECTORY        16
#DEFINE INTERNET_FLAG_RAW_DATA  1073741824

#DEFINE INTERNET_OPEN_TYPE_PRECONFIG     0
#define INTERNET_OPEN_TYPE_DIRECT        1
#define INTERNET_OPEN_TYPE_PROXY         3
#DEFINE INTERNET_DEFAULT_FTP_PORT       21
#DEFINE INTERNET_SERVICE_FTP             1
#DEFINE INTERNET_FLAG_PASSIVE     14217728
#DEFINE FILE_ATTRIBUTE_NORMAL          128

#define cNULL	   CHR(0)

DECLARE LONG InternetOpen ;
	IN "wininet.dll" ;
	STRING   lpszAgent, ;
	LONG     dwAccessType, ;
	STRING   lpszProxyName, ;
	STRING   lpszProxyBypass, ;
	LONG     dwFlags
DECLARE LONG InternetConnect ;
	IN "wininet.dll" ;
	LONG     hInternetSession, ;
	STRING   lpszServerName, ;
	LONG     nServerPort, ;
	STRING   lpszUsername, ;
	STRING   lpszPassword, ;
	LONG     dwService, ;
	LONG     dwFlags, ;
	LONG     dwContext
DECLARE INTEGER InternetCloseHandle ;
	IN "wininet.dll" ;
	LONG     hInet
DECLARE LONG GetLastError ;
	IN WIN32API
DECLARE Integer FtpRenameFile IN WinInet ;
		Integer nConnect_Handle, ;
		String @lpcRemoteFile, ;
		String @lpcNewFile
	      

LOCAL fResult
nInternet = InternetOpen( "vfp", INTERNET_OPEN_TYPE_DIRECT, "", "", 0 )
IF nInternet = 0
	=ThisForm.Inf("Erro na conexăo com o servidor. "+allt(Thisform.InErrorCase(GetLastError())),"R" )
	InternetCloseHandle( nFtp )
	RETURN .F.
ENDIF

*** Conexión con el servicio FTP
nFtp = InternetConnect( nInternet, cFtpLoc, INTERNET_DEFAULT_FTP_PORT, cFtpUsu, cFtpSen, INTERNET_SERVICE_FTP, INTERNET_FLAG_PASSIVE, 0 )
IF nFtp = 0
	=ThisForm.Inf("Erro na conexăo com o servidor FTP. "+allt(Thisform.InErrorCase(GetLastError())),"R" )
	RETURN .F.
ENDIF

lcOldFile = lower(allt(lcOldFile)) + cNULL
lcNewFile = lower(allt(lcNewFile)) + cNULL

fResult = FtpRenameFile(nftp, @lcOldFile, @lcNewFile)

InternetCloseHandle( nFtp )
InternetCloseHandle( nInternet )

IF fResult = 1
   RETURN .T.
ELSE
   RETURN .F.
ENDIF
ENDPROC

		lcArq1
