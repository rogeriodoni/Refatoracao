PROCEDURE calcula_saldo
If crDistribui.QtdDistr <= 0
	Replace crDistribui.QtdDistr With 0
EndIf

Select crDistribui
lnReg = Recno()
Sum QtdDistr for Cpros = crDisponivel.Cpros to lnQtdDistr
Go lnReg

If crDisponivel.Sqtds - lnQtdDistr < 0
	Messagebox('Saldo insuficiente para movimentar.',48,'Aviso')
	Replace QtdDistr With 0 in crDistribui
	Select crDistribui
	lnReg = Recno()
	Sum QtdDistr for Cpros = crDisponivel.Cpros to lnQtdDistr
	Go lnReg
Endif 

Replace QtdSaida With lnQtdDistr in crDisponivel
Replace QtdSaldo With Sqtds - lnQtdDistr in crDisponivel

This.Pagina.dados.pageframe1.page2.grdDisponivel.Refresh()
This.Pagina.dados.pageframe1.page2.grdDistribui.Refresh()



ENDPROC

*!*						Select b.cEmps, Case When d.SQtds is null Then 0000000 Else d.Sqtds End As QtdSaldo, 
*!*						Case When e.Qtds is null Then 0000000 Else e.Qtds End As QtdDistr,

lcEgc = []
lcEmps = []

Scan

	Select a.CPros, b.Dpros, a.CodCors, a.CodTams, Sum(a.SQtds) as Sqtds, 0000000 As QtdSaida, Sum(a.SQtds) As QtdSaldo, b.Cgrus, b.Sgrus
	From SigMvEst a, SigCdpro b
	Where a.EmpGruEsts in <<lcEgc>>
	  And a.Cpros = b.Cpros 
	  And a.SQtds > 0 
	Group By a.CPros, b.Dpros, a.CodCors, a.CodTams, b.Cgrus, b.Sgrus	
EndText

If thisform.podatamgr.SqlExecute(lcSQL, [crDisponivel] ) < 1
	MessageBox( [Problemas no Select dos Produtos com Saldo], 48, "Atençăo" )
	Return(0)
EndIf

Text tO lcSQL NoShow TextMerge
	Select b.cEmps, Case When d.SQtds is null Then 0000000 Else d.Sqtds End As QtdSaldo, 0 As QtdDistr, Space(10) as Locals, a.CPros, a.CodCors, a.CodTams, c.Dopes
	from (Select Distinct a.CPros, a.CodCors, a.CodTams From SigMvEst a
	Where a.EmpGruEsts in <<lcEgc>> And a.SQtds > 0) a 
	Inner Join SigCdEmp b On b.cEmps not in <<lcEmps>>
	Inner Join SigCdOpe c On b.DirColOuts = c.Dopes
	Left Join SigMvEst d On d.EmpGruEsts = b.cEmps + c.GruDests + c.ConDests And d.CPros + d.CodCors + d.CodTams = a.CPros + a.CodCors + a.CodTams
	Order By a.CPros, a.CodCors, a.CodTams, b.cEmps
EndText

If thisform.podatamgr.SqlExecute(lcSQL, [crDistribui] ) < 1
	MessageBox( [Problemas no Select dos Produtos Distribuidos], 48, "Atençăo" )
	Return(0)
EndIf

Select crDistribui
Index ON cPros+CodCors+CodTams Tag Tag1
Set Order To Tag1

logrdDistribui.RecordSourceType = 1 
logrdDistribui.RecordSource = [crDistribui]

Select crDisponivel
Index on Cgrus Tag Cgrus
Set Relation TO cPros+CodCors+CodTams INTO crDistribui

logrdDisponivel.RecordSourceType = 1 
logrdDisponivel.RecordSource = [crDisponivel]

PROCEDURE btnProcessar.Click
Select crDistribui

	Messagebox([Nenhum produto foi informado para distribuir],48,[Aviso])
	Return .f.

Count for QtdSaldo < 0 to lnQtdNeg
If lnQtdNeg > 0
	Messagebox([Existe produtos distribuídos que estăo ficando com saldo negativo.] + Chr(13) +[Favor checar antes de processar.],48,[Aviso])
	Return .f.
Endif

*** Atualiza tabela temporária com os estoques que serăo distribuídos
lccidquerys = fUniqueIds()
Select crSigCdCeg
If Reccount() > 0
	Scan
		Select CrSigTempd
		Append Blank
		Replace Grupos    With crSigCdCeg.Grupos,;
				Contas    With crSigCdCeg.Contas,;
				codObs    With crSigCdCeg.Priors,;
				Emps      With crSigCdCeg.Emps  ,;
				CidChaves With fUniqueIds(),;
				CidQuerys With lccidquerys
	EndScan
Else
	Select CrSigTempd
	Append Blank
	Replace Grupos    With crSigCdPam.GrupoEsts,;
			Contas    With crSigCdPam.contaEsts,;
			CodObs    With 1,;
			Emps      With _Empr,;
			CidChaves With fUniqueIds(),;
			CidQuerys With lccidquerys
EndIf

If !ThisForm.Podatamgr2.UpDate('CrSigTempd')
	=ThisForm.Podatamgr2.RollBack()
	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo (CrSigTempD)')
	Return .f.
EndIf

	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo (Commit)')
	Return .f.
Endif

*** Busca os produtos que estăo nos estoques que serăo distribuídos
lcQuery = [Select a.*, b.CodObs as Priors From SigMvEst a, SigTempd b ]+;
		[Where a.Grupos = b.Grupos And a.Estos = b.Contas And a.Emps = b.Emps And a.Sqtds > 0 ] +;
		[Union ] +;
		[Select a.*, b.CodObs as Priors From SigMvEst a, SigTempd b ] +;
		[Where a.Grupos = b.Grupos and b.Contas = '' And a.Emps = b.Emps And a.Sqtds > 0]
		
If ThisForm.PodataMgr.SqlExecute(lcquery,'TmpEstoque') < 1
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpEstoque)')
	Return 0
EndIf

ThisForm.PodataMgr2.fSqlApagarTmp('SigTempD', lcCidquerys )

Index on CPros + CodCors + CodTams + Str(priors,2) + Grupos + Estos + Emps Tag CPros

Zap

*!*			Insert Into TmpSaldo ( CPros, CodCors, CodTams, Saldo, Disps );
*!*						Values ( TmpEstoque.CPros, TmpEstoque.CodCors, TmpEstoque.CodTams, 0, 0 )
*!*		EndIf
*!*		Replace Saldo With Saldo + TmpEstoque.Sqtds,;
*!*				Disps With Disps + TmpEstoque.Sqtds In TmpSaldo

	Insert Into TmpSaldg ( Grupos, Estos, CPros, CodCors, CodTams, Saldo, Disps, Priors, Emps );
				Values ( TmpEstoque.Grupos, TmpEstoque.Estos, TmpEstoque.CPros, TmpEstoque.CodCors,;
						 TmpEstoque.CodTams, TmpEstoque.SQtds, TmpEstoque.SQtds, TmpEstoque.Priors, TmpEstoque.Emps )

Index On EmpDs + Grupos + Estos + Emps + Dopes + Str(Numes,6) Tag EmpDopNum

	If crDistribui.QtdDistr # 0

			If TmpSaldg.Disps >= xBaixa && Reservs

				Insert Into TmpEstoque (Cpros, CodCors, CodTams, Emps, dopes, Numes, Grupos, Estos, Estoque, EmpDs, Locals ) Values ;
