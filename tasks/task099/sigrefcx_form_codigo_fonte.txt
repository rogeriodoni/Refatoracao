* PROCEDURE cabecalho
------------------------------------------------------------
PROCEDURE cabecalho
Parameters lcDetalhe, lcSemLinha, pUsuarios, pQtde, ldDtCab

lnTipoOpc = ThisForm.Opc_Tipo.Value

=This.Linha('',2,pUsuarios,ldDtCab)
=This.Linha( 1,2,pUsuarios,ldDtCab)
If pQtde
	=This.Linha(PadC(' ' + AllTrim(lcDetalhe ) + ' ',48,'*'),2,pUsuarios,ldDtCab)
Else
	=This.Linha(PadC(' ' + AllTrim(lcDetalhe ) + ' ',40,'*'),2,pUsuarios,ldDtCab)
EndIf
If !lcSemLinha
	=This.Linha('',2,pUsuarios,ldDtCab)
EndIf
ENDPROC

------------------------------------------------------------
* PROCEDURE linha
------------------------------------------------------------
PROCEDURE linha
LParameters _Linha, _Tipo, _Usuario, _Datas

lcArq = Iif(ThisForm.Segunda, [dbTmpRel2], [dbTmpRel])

If !Type('_Usuario') = 'C'
	_Usuario = ''
EndIf

Do Case
	Case Type('_Linha') = 'U' Or Type('_Linha') = 'L'
		Insert Into &lcArq. (detalhe,tipo,usuars,datas) Values (Space(40),_Tipo,_Usuario,_Datas)
	Case Type('_Linha') = 'C'
		If Empty(_Linha)
			Insert Into &lcArq. (detalhe,tipo,usuars,datas) Values (Space(40),_Tipo,_Usuario,_Datas)
		Else
			Insert Into &lcArq. (detalhe,tipo,usuars,datas) Values (_Linha,_Tipo,_Usuario,_Datas)
		EndIf
	Case Type('_Linha') = 'N'
		If _Linha = 1
			Insert Into &lcArq. (detalhe,tipo,usuars,datas) Values (Replicate('=',ThisForm.MaxCols),_Tipo,_Usuario,_Datas)
		EndIf
EndCase
ENDPROC

	EndIf

	Select TmpUsuarios
	If Inlist(1,m.ResLanca,m.ResCheque,m.ResCartao,m.ResConta,m.ResVende,m.ResForma,m.ResLanFor,m.ResQtdes,m.ResvDets)

EndScan
loBarraRes.Complete(.T.)

If lnOptCol = 1 And m.Resvdets = 1
	Select dbRelIni
	Go Top
	If !Eof()
		=ThisForm.Rel_Ini2(ldDtImp)
	EndIf

Set Order To Usuars
Go Top

Return(.T.)

* PROCEDURE impressao
------------------------------------------------------------
PROCEDURE impressao
Local lnImpress, lcLastUsuar, llEncontrou
lnImpress = 0
llEncontrou = .F.

If ThisForm.OptCol.Value = 1

	Ok = .F.
	lnEtqImpres = 0
	If (APrinters(gaPrinters) = 0)
		=MessageBox('O Painel de Controle Năo Possui Impressoras Configuradas!!!', 0+48, '')
		Return .f.
	Endif

	* Apanhando Dados de Configuraçăo de Impressoras (SigCdmp)
	pTip = 1
	lcSQLCmd = [Select * From SigCdmp ] + ;
				[Where nTpImpres = ?pTip]
				
	lnSQLErr = ThisForm.Podatamgr.SQLExecute(lcSQLCmd, 'Tmp1')
	If (lnSQLErr < 1)				&& Caso Falhe ao Executar a Instruçăo SQL Acima
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (SigCdmp)')
		Return .F.
	EndIf

	llFlags = .f.
	Select *, llFlags as lMarcas, llFlags as lSituas ;
	  From Tmp1 ;
	  Into Cursor csSigCdmp ReadWrite

	Select csSigCdmp
	Index On Impres Tag Imprime For ( lSituas = .T.)
	Index On Impres Tag Impres
	Goto Top In csSigCdmp

	lcSql = [Select b.* From SigSyImp a, SigCdmp b Where a.UsuAcess = ']+Usuar+[' And a.cImps = b.Impres And b.nTpImpres = ?pTip ]+;
			[Union all ]+;
			[Select c.* From SigCdAcG a, SigSyImp b, SigCdmp c Where a.Usuarios=']+Usuar+[' ]+;
			[And a.Grupos = b.GrAcess And b.Cimps = c.Impres And c.nTpImpres = ?pTip]

	If ThisForm.Podatamgr.sqlexecute(lcSql,'CrTmpCimp') <= 0
		Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados.',(0+48),'Erro')
		Return .f.
	EndIf

	Select CrTmpCimp
	If Reccount() > 0
		Select Impres From CrTmpCimp Into Cursor LocalcImp Group by Impres ReadWrite
		llFlags = .f.
		Select *, llFlags as lMarcas, llFlags as lSituas ;
		  From LocalcImp ;
		  Into Cursor csSigCdmp ReadWrite

		Select csSigCdmp
		Index On Impres Tag Imprime For ( lSituas = .T.)
		Index On Impres Tag Impres
		Goto Top In csSigCdmp
	Endif
 	For I=1 to APrinters(gaPrinters)
 		lcImpres = Upper(gaPrinters(I, 1))
 		If Seek(lcImpres, 'csSigCdmp', 'Impres')
	 		lnEtqImpres = lnEtqImpres + 1
	 		Select csSigCdmp
	 		Replace lMarcas With .T., ;
	 				lSituas With .T.
	 	Endif
	Next
	
	If (lnEtqImpres = 0)
		Messagebox( 'Nenhuma Impressora de Boleto Está Configurada no' + Chr(13) + ;
					'Painel de Controle.  Contate o Suporte Técnico.', 0+48, ;
					'Sem Impressora Configurada no Painel de Controle' )
		Return ''
	Else
		If (lnEtqImpres # 1)	&& Se Tiver Mais Que Uma Impressora, Habilita Formulário
			Do Form SigOpIpB	&& Seleciona Impressora Para Etiqueta / Boletos	
		EndIf
	EndIf
	
	Select csSigCdmp
	Locate for lSituas

	Set Printer To Name (AllTrim(CsSigCdmp.impres))
	Set Device To Printer
	???  Chr(13) + Chr(10)

	lcLastUsuar = Space(10)
	Select dbRelatorio
	Scan
		If dbRelatorio.usuars <> lcLastUsuar And !Empty(lcLastUsuar) And lnImpress <> 1
			??? Chr(12)
		EndIf
		??? dbRelatorio.detalhe + Chr(13) + Chr(10)
		lcLastUsuar = dbRelatorio.Usuars
	EndScan
	???  Chr(13) + Chr(10)
	???  Chr(13) + Chr(10)
	???  Chr(13) + Chr(10)
	???  Chr(13) + Chr(10)
	Set Device To Screen
	Set Printer To Default

Else
	Report Form SigReFc2 To Printer Prompt NoConsole
EndIf	

Use In dbRelatorio
ENDPROC

------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Local lnOpcTipo, ldDtIni, ldDtFim

lnOpcTipo = ThisForm.Opc_Tipo.Value
lnOpcCol  = ThisForm.Optcol.Value
ldDtIni   = ThisForm.Get_Dt_Inicial.Value
ldDtFim   = ThisForm.Get_Dt_Final.Value


ThisForm.MaxCols = Iif(ThisForm.OptCol.Value = 1,40,254)

Create Cursor dbTmpRel(detalhe c(ThisForm.MaxCols) Null,tipo n(1),usuars c(10),datas d)
Index On usuars Tag Usuars

Create Cursor dbTmpRel2(detalhe c(ThisForm.MaxCols) Null,tipo n(1),usuars c(10),datas d)
Index On usuars Tag Usuars

If lnOpcCol = 1
	For lnVez = 1 To 2
		ThisForm.Segunda = (lnVez = 2)
		If !(ThisForm.Processa(ldDtIni,ldDtFim))
			Return(.F.)
		EndIf
	Next
	
	Select * From dbTmpRel ;
	Union All ;
	Select * From dbTmpRel2 ;
	Into Cursor dbRelatorio ReadWrite
Else
	If lnOpcTipo = 1
		For lnVez = 1 To 2
			ThisForm.Segunda = (lnVez = 2)
			If !(ThisForm.Processa(ldDtIni,ldDtFim))
				Return(.F.)
			EndIf
		Next
		Select * From dbTmpRel ;
		Union All ;
		Select * From dbTmpRel2 ;
		Into Cursor dbRelatorio ReadWrite
	Else
	
		llFirst = .T.
		lnDias  = (ldDtFim - ldDtIni + 1)
		loBarraDias    = CreateObject('fwprogressbar','Verificando as datas...',lnDias)
		loBarraDias.Top = loBarraDias.Top - loBarraDias.Height
		loBarraDias.Show
		ldDtProc = ldDtIni
		Do While ldDtProc <= ldDtFim
			loBarraDias.Update(.T.,'Dia : ' + Dtoc(ldDtProc))
			For lnVez = 1 To 2
				ThisForm.Segunda = (lnVez = 2)
				If !(ThisForm.Processa(ldDtProc,ldDtProc))
					Return(.F.)
				EndIf
			Next
			If llFirst
				Select * From dbTmpRel ;
				Union All ;
				Select * From dbTmpRel2 ;
				Into Cursor dbRelatorio ReadWrite
				llFirst = .F.
			Else
				Select dbTmpRel
				Go Top
				Scan
					Scatter Memo Memvar
					Insert Into dbRelatorio From MemVar
					Select dbTmpRel
				EndScan
				Select dbTmpRel2
				Go Top
				Scan
					Scatter Memo Memvar
					Insert Into dbRelatorio From MemVar
					Select dbTmpRel2
				EndScan
			EndIf
			ldDtProc = ldDtProc + 1
		EndDo
		loBarraDias.Complete(.T.)
	EndIf
EndIf

Select dbRelatorio
Go Top
ENDPROC

=DoDefault()
* Fim **************************************************************************************
ENDPROC
