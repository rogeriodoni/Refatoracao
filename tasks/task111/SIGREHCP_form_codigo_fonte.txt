------------------------------------------------------------
* PROCEDURE Init
------------------------------------------------------------
PROCEDURE Init
With ThisForm
	.loNegocios = CreateObject([cnSigRehcp], .Name)

	If (.loNegocios.poDataMgr.pnIdconn > 0)
		.getDtInicial.Value = Date()
		.getDtFinal.Value   = Date()
		.getCPros.Value 	= Space(14)
		.getDPros.Value 	= Space(40)
		.getCEmps.Value 	= Space(03)
		.getDEmps.Value 	= Space(40)

		.loObj = CreateObject([Relation])

		.loObj.AddProperty([DataInicial], .getDtInicial.Value)
		.loObj.AddProperty([DataFinal],   .getDtFinal.Value)
		.loObj.AddProperty([Produto],     [])
		.loObj.AddProperty([Empresa],     [])
		.loObj.AddProperty([CalcCustos],  00)
		.loObj.AddProperty([TrataErro],   [])
		.loObj.AddProperty([TipoImps],    00)

		* Validando Campos de Empresa
		.loNegocios.poDataMgr.CursorQuery([SigCdPaC], [rCustParaC], .f., .f., [CalcCustos])
		If Used([rCustParaC]) And Not Eof([rCustParaC])
			.loObj.CalcCustos = rCustParaC.CalcCustos
			If (.loObj.CalcCustos = 1) && 1-Produto; 2-Empresa + Produto
				.getCEmps.Visible 	= .f.
				.getDEmps.Visible 	= .f.
				.getCEmps.Enabled   = .f.
				.getDEmps.Enabled 	= .f.
				.lblEmpresa.Visible = .f.
			EndIf
		EndIf
				
		=DoDefault()
	Else
		=Messagebox([Impossível Efetuar Conexăo Com o Servidor de Banco de Dados...], 0+48, [CreateObject])
		Return .F.
	EndIf
EndWith
* Fim **************************************************************************************
ENDPROC

PROCEDURE processamento
With ThisForm
	.loObj.DataInicial	= .getDtInicial.Value
	.loObj.DataFinal	= .getDtFinal.Value
	.loObj.Produto		= .getCPros.Value
	.loObj.Empresa		= .getCEmps.Value
	.loObj.TipoImps     = .optTipImps.Value

	llRet = .loNegocios.Processamento(.loObj)

	If Not llRet
		Return .f.
	EndIf
EndWith
* Fim **************************************************************************************
ENDPROC

------------------------------------------------------------
* PROCEDURE btnReport.Click
------------------------------------------------------------
PROCEDURE btnReport.Click
If (This.Value <> 4) && Se Năo Clicou em Saída
	With ThisForm
		.loObj.DataInicial	= .getDtInicial.Value
		.loObj.DataFinal	= .getDtFinal.Value
		.loObj.Produto		= .getCPros.Value
		.loObj.Empresa		= .getCEmps.Value

		.loNegocios.ValidaBotoes(ThisForm.loObj)	
		
		llRet = .loObj.TrataErro
	EndWith

	Do Case
		Case llRet = [DataInicial]
			=Messagebox([Data Inicial Inválida!!!], 0+48, [Atençăo!!!])
			ThisForm.getDtInicial.SetFocus
			Return .f.

		Case llRet = [DataFinal]
			=Messagebox([Data Final Inválida!!!], 0+48, [Atençăo!!!])
			ThisForm.getDtInicial.SetFocus
			Return .f.

		Case llRet = [Periodo]
			=Messagebox([Data Final Menor Que a Inicial!!!], 0+48, [Atençăo!!!])
			ThisForm.getDtInicial.SetFocus
			Return .f.
	EndCase
EndIf

=DoDefault()
* Fim **************************************************************************************
ENDPROC

PROCEDURE Valid
Local loLista

If Not Empty(This.Value)
	loLista = CreateObject('fwBuscaExt', ThisForm.loNegocios.poDataMgr.pnIdConn, 'SigCdPro', ;
						   'crListaRemota', 'DPros', Padr(This.Value, 40), 'Seleçăo', 500)

	If Not loLista.plAchouRegistro
		loLista.mAddColuna('DPros', '', 'Descriçăo')
		loLista.mAddColuna('CPros', '', 'Código')
		loLista.Show()
	EndIf
	
	ThisForm.getCPros.Value = crListaRemota.CPros
	ThisForm.getDPros.Value = crListaRemota.DPros
	Use In crListaRemota
Else
	ThisForm.getCPros.Value = ''
	ThisForm.getDPros.Value = ''
EndIf
ThisForm.getCPros.Refresh
ThisForm.getDPros.Refresh
* Fim *********************************************************************************
ENDPROC
