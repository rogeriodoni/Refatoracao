PROCEDURE anafpags
Local lcVal, lnTot

Insert Into crImpressao (Detalhe) Values ([======= ANALITICO POR PAGAMENTOS =======])

Select Distinct FPags ;
  From crDetalhe ;
 Order By FPags ;
  Into Cursor TmpFPags

lnTot = 0

Select TmpFPags
Scan
	Select Iif(RelFechas=4, Padr([VALE], 12), FPags) as FPags, RPags, RChqs, ;
		   RelFechas, Vends, Notas, Parcs, ValPagos, Contas, Numeros, Vencs ;
	  From crDetalhe ;
	 Where FPags = TmpFPags.FPags ;
	 Order By FPags, Vends, Notas, Parcs ;
	  Into Cursor TmpOpePags

	lnSub = 0

	Select TmpOpePags
	Scan

		*** Mena - 03/04/2014 - inclui Vale na soma dos totais ***
*!*		lnTot = lnTot + Iif(RelFechas=4, 0, TmpOpePags.ValPagos)
*!*		lnSub = lnSub + Iif(RelFechas=4, 0, TmpOpePags.ValPagos)

		lnTot = lnTot + TmpOpePags.ValPagos
		lnSub = lnSub + TmpOpePags.ValPagos
		
		lcVal = Transform(TmpOpePags.ValPagos, [999,999.99])

		If (TmpOpePags.Parcs = 1) && Se For a Primeira Parcela
			*DINHEIRO   V10      999999/01 999,999.99
			Insert Into crImpressao (Detalhe) Values (Padr(TmpOpePags.FPags, 10) + [ ] + Padr(TmpOpePags.Vends, 9) + ;
					TmpOpePags.Notas + [/] + Padl(TmpOpePags.Parcs, 2, [0]) + [ ] + lcVal)
		Else
			*                    999999/01 999,999.99
			Insert Into crImpressao (Detalhe) Values ([                    ] + TmpOpePags.Notas + [/] + ;
					Padl(TmpOpePags.Parcs, 2, [0]) + [ ] + lcVal)
		EndIf

		If Not Empty(TmpOpePags.Numeros) 
			* Insere Dados do Cheque
			*         9999999999 / 999999  99/99/9999
			Insert Into crImpressao (Detalhe) Values ([         ] + ;
					TmpOpePags.Contas + [ / ] + TmpOpePags.Numeros + [  ] + Dtoc(TmpOpePags.Vencs))
		EndIf

		If Not Empty(TmpOpePags.RPags) 
			* Insere Dados da Forma de Pagamento do Subnível
			*                    999999/01 999,999.99
			Insert Into crImpressao (Detalhe) Values ([        ] + TmpOpePags.RPags)

			If Not Empty(TmpOpePags.RChqs) 
				* Insere Dados do Cheque de Pagamento do Subnível
				*         9999999999 / 999999  99/99/9999
				Insert Into crImpressao (Detalhe) Values ([         ] + TmpOpePags.RChqs)
			EndIf
		EndIf
	EndScan

	lcVal = Transform(lnSub, [999,999,999.99])
	Insert Into crImpressao (Detalhe) Values ([----------------------------------------])
	Insert Into crImpressao (Detalhe) Values ([=              SUBTOTAL : ] + lcVal)
	Insert Into crImpressao (Detalhe) Values ([])
EndScan

lcVal = Transform(lnTot, [999,999,999.99])
Insert Into crImpressao (Detalhe) Values ([----------------------------------------])
Insert Into crImpressao (Detalhe) Values ([=           TOTAL GERAL : ] + lcVal)
Insert Into crImpressao (Detalhe) Values ([----------------------------------------])
Insert Into crImpressao (Detalhe) Values ([])
* Fim **************************************************************************************
ENDPROC

PROCEDURE detalhado
Local lcVal, lnTot, lcVen

lcVen = ThisForm.getVends.Value

If Empty(lcVen)
	If ThisForm.chkVends.Value 
		Select Vends, Dopes, Sum(Iif(RelFechas=4, 0.00, ValPagos)) as Valos ;
		  From crDetalhe ;
		 Group By Vends, Dopes ;
		 Order By Dopes, Vends ;
		  Into Cursor TmpOperacoes
	Else
		Select Space(10) as Vends, Dopes, Sum(Iif(RelFechas=4, 0.00, ValPagos)) as Valos ;
		  From crDetalhe ;
		 Group By Dopes ;
		 Order By Dopes ;
		  Into Cursor TmpOperacoes
	EndIf
Else
	Select Vends, Dopes, Sum(Iif(RelFechas=4, 0.00, ValPagos)) as Valos ;
	  From crDetalhe ;
	 Where Vends = lcVen ;
	 Group By Vends, Dopes ;
	 Order By Dopes, Vends ;
	  Into Cursor TmpOperacoes
EndIf
  
Insert Into crImpressao (Detalhe) Values ([======== DETALHADO POR OPERACAO ========])

lnTot = 0
Select TmpOperacoes
Scan && Operaçőes
	lnTot = lnTot + TmpOperacoes.Valos
	
	* Imprimindo cabeçalho da Operaçăo      
	* VENDA NF AV         :   999,999,999.99
	lcVal = Transform(TmpOperacoes.Valos, [999,999,999.99])
	Insert Into crImpressao (Detalhe) Values ([= ] + TmpOperacoes.Dopes + [:   ] + lcVal)

	* Selecionando as Operaçőes Para Impressăo
	lcWhe = Iif(Empty(lcVen), [], [ And Vends = '] + Padr(lcVen, 10) + [' ])

	If Empty(lcVen)
		If ThisForm.chkVends.Value 
			Select Distinct Vends, Dopes, Numes ;
			  From crDetalhe ;
			 Where Dopes = TmpOperacoes.Dopes And Vends = TmpOperacoes.Vends ; 
			 Order By Dopes, Vends, Numes ;
			  Into Cursor TmpOpeIndivs
		Else
			Select Distinct Dopes, Numes ;
			  From crDetalhe ;
			 Where Dopes = TmpOperacoes.Dopes ; 
			 Order By Dopes, Numes ;
			  Into Cursor TmpOpeIndivs
		EndIf
	Else
		Select Distinct Dopes, Numes ;
		  From crDetalhe ;
		 Where Dopes = TmpOperacoes.Dopes And Vends = lcVen ; 
		 Order By Dopes, Numes ;
		  Into Cursor TmpOpeIndivs
	EndIf
	
	Select TmpOpeIndivs
	Scan
		Select IClis, RClis, Notas, Vends, NVends, Parcs, FPags, ValPagos, Vencs, Bancos, Agencias, Contas, Numeros, RelFechas ;	
		  From crDetalhe ;
		 Where Dopes = TmpOpeIndivs.Dopes And ; 
		 	   Numes = TmpOpeIndivs.Numes ;
		 Order By Parcs ;
		  Into Cursor TmpOpePags
		
		* Inserindo as Parcelas
		Select TmpOpePags
		Scan
			lcVal = Transform(TmpOpePags.ValPagos, [999,999,999.99])

			If (TmpOpePags.Parcs = 1) && Se For a Primeira Parcela
				*- 999999 (99) DINHEIRO    999,999,999.99
				Insert Into crImpressao (Detalhe) Values ([> ] + Padl(Alltrim(TmpOpeIndivs.Numes), 6, [0]) + ;
						[ (] + Padl(TmpOpePags.Parcs, 2, [0]) + [) ] + TmpOpePags.FPags + lcVal)
			Else
				*         (99) DINHEIRO    999,999,999.99
				Insert Into crImpressao (Detalhe) Values ([        ] + ;
						[ (] + Padl(TmpOpePags.Parcs, 2, [0]) + [) ] + TmpOpePags.FPags + lcVal)
			EndIf
			If Not Empty(TmpOpePags.Numeros) 
				* Insere Dados do Cheque
				*         9999999999 / 999999  99/99/9999
				Insert Into crImpressao (Detalhe) Values ([         ] + ;
						TmpOpePags.Contas + [ / ] + TmpOpePags.Numeros + [  ] + Dtoc(TmpOpePags.Vencs))
			EndIf
		EndScan
		
		* Inserindo o Cliente
		Select TmpOpePags
		Go Top In TmpOpePags
		If Not Eof([TmpOpePags])
			*  XXXXXXXXXX - NONONONONONONONONO 999999
			lcCli = Padr(Alltrim(TmpOpePags.IClis) + [ - ] + Alltrim(TmpOpePags.RClis), 31)
			Insert Into crImpressao (Detalhe) Values ([  ] + lcCli + [ ] + TmpOpePags.Notas)
		EndIf
		
		* Inserindo o Vendedor
		If ThisForm.chkVends.Value 
			Select TmpOpePags
			Go Top In TmpOpePags
			If Not Eof([TmpOpePags])
				*  XXXXXXXXXX - NONONONONONONONONONONONON
				If Empty(TmpOpePags.Vends)
					lcVdd = Alltrim(TmpOpePags.IClis) + [ - ] + Alltrim(TmpOpePags.RClis)
				Else
					lcVdd = Alltrim(TmpOpePags.Vends) + [ - ] + Alltrim(TmpOpePags.NVends)
				EndIf
				Insert Into crImpressao (Detalhe) Values ([  ] + lcVdd)
			EndIf
		EndIf
	EndScan
	
	Insert Into crImpressao (Detalhe) Values ([----------------------------------------])
EndScan

lcVal = Transform(lnTot, [999,999,999.99])
Insert Into crImpressao (Detalhe) Values ([=           TOTAL GERAL : ] + lcVal)
Insert Into crImpressao (Detalhe) Values ([----------------------------------------])
Insert Into crImpressao (Detalhe) Values ([])
* Fim **************************************************************************************
ENDPROC

PROCEDURE pagtoxfpags
Local lcVal, lnTot

Select Dopes, Sum(ValPagos) as Valos ;
  From crDetalhe ;
 Group By Dopes ;
 Order By Dopes ;
  Into Cursor TmpOperacoes
  
Insert Into crImpressao (Detalhe) Values ([==== RESUMO: OPERACOES X PAGAMENTOS ====])
Insert Into crImpressao (Detalhe) Values ([])

lnGer = 0
Select TmpOperacoes
Scan && Operaçőes
	Insert Into crImpressao (Detalhe) Values ([= ] + TmpOperacoes.Dopes)

	Select DetFecCxs, RelFechas, FPags, Sum(ValPagos) as Valos ;
	  From crDetalhe ;
	 Where Dopes = TmpOperacoes.Dopes ;
	 Group By DetFecCxs, RelFechas, FPags ;
	 Order By FPags ;
	  Into Cursor TmpFPags

	lnTot = 0
	Select TmpFPags
	Scan
		lnGer = lnGer + Iif(RelFechas=4, 0, TmpFPags.Valos)
		lnTot = lnTot + Iif(RelFechas=4, 0, TmpFPags.Valos)
		
		* = XXXXXXXXXXXX : 999,999,999.99
		lcVal = Transform(TmpFPags.Valos, [999,999,999.99])
		Insert Into crImpressao (Detalhe) Values ([  ] + TmpFPags.FPags + [ : ] + lcVal)
		
		If (TmpFPags.DetFecCxs = 1) And (TmpFPags.RelFechas = 3)
			Select Notas, FPags, Sum(ValPagos) as ValPagos, Vencs, Bancos, Agencias, Contas, Numeros, Outros ;	
			  From crDetalhe ;
			 Where Dopes = TmpOperacoes.Dopes And ; 
			 	   FPags = TmpFPags.FPags ;
			 Order By Notas ;
			 Group By Notas, FPags, Vencs, Bancos, Agencias, Contas, Numeros, Outros ;	
			  Into Cursor TmpOpeChqs
			
			Select TmpOpeChqs
			Scan
				* Insere Dados dos Cheques
				If Not Empty(TmpOpeChqs.Outros) && Observaçăo do Cartăo
					*  999999 xxxxxxxxxxxxxxxxxxxx 999,999.99
					lcVal = Transform(TmpOpeChqs.ValPagos, [999,999.99])					
					Insert Into crImpressao (Detalhe) Values ([  ] + TmpOpeChqs.Notas + [ ] + TmpOpeChqs.Outros + ;
							[ ] + lcVal)
				EndIf
				
				If Not Empty(TmpOpeChqs.Numeros) && Dados do Cheque
					*  999999 99/99/9999 999999 99,999,999.99

* PROCEDURE processamento

PROCEDURE processamento
Private pDtI, pDtF

Local lcEmp, lcDEp, ldDtI, ldDtF, lcUsu, lcVen, lcEdn, lcFPg

lcEmp = ThisForm.getEmps.Value
lcDEp = ThisForm.getDEmps.Value 
ldDtI = ThisForm.getDtIni.Value 
ldDtF = ThisForm.getDtFim.Value
lcUsu = ThisForm.getUsuars.Value
lcVen = ThisForm.getVends.Value
lcFPg = ThisForm.getFPags.Value

pDtI = fDtoSQL(ldDtI)
pDtF = fDtoSQL(ldDtF, [23:59:59])

* Carrega Todas As Operaçőes Que Devem Aparecer No Relatório de Fechamento Do Caixa
lcQuery = [Select o.RelFechas, o.CalcFecs, e.EmpDopNums, e.Datas, e.ContaDs, c.RClis, e.Notas, e.Usuars, ] + ;
				 [e.Valos, e.Vends, o.COpers, (o.NDopes * 1000000) + e.Numes as Codigos, e.ChkSubn ] + ;
		  	[From SigCdOpe o, SigMvCab e, SigCdCli c ] + ;
		   [Where (o.RelFechas = 1 Or o.RelFechas = 4 Or o.RelFechas = 3 And e.chkPagos = 1) And  ] + ;
			     [o.CalcFecs = 2 And  ] + ;
			     [o.Dopes = e.Dopes And  ] + ;
			     [e.Datas BetWeen ?pDtI And ?pDtF And ] + ;
			     [e.ContaDs = c.IClis And ] + ;
			     [e.lcancelas = 0 And ] + ;
			      Iif(Empty(lcUsu), [], [e.Usuars = '] + Padr(lcUsu, 10) + [' And ]) + ;
			     [Not e.EmpDopNums In ( ] + ;
				   [Select a1.EmpDopNums From SigMvPec a1 ] + ;
				    [Where a1.Codigos In ( ] + ;
			   		 	  [Select (o1.NDopes * 1000000 ) + e1.Numes as Codigos ] + ;
			   			    [From SigCdOpe o1, SigMvCab e1 ] + ;
			   			   [Where o1.RelFechas = 3 And  ] + ;
			   					 [o1.CalcFecs = 2 And  ] + ;
			   					 [o1.Dopes = e1.Dopes And  ] + ;
			   					 [e1.Datas BetWeen ?pDtI And ?pDtF)) ] + ;
		   [Order By e.EmpDopNums]

If (ThisForm.poDataMgr.SqlExecute(lcQuery, [crDados]) < 1)
	=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crDados)])
	Return .f.
EndIf

Delete From crDados Where RelFechas = 1 And ChkSubn = .t. && Apagando as VENDA NF PZ canceladas

Update crDados Set Vends = ContaDs Where Empty(Vends)

Create Cursor crPagamentos (EmpDopNums c(29), Parcs n(3), FPags c(12), Bancos c(3), Agencias c(4), ;
							Contas c(10), Numeros c(6), Valos n(12,2), MoeFpgs c(3), Vencs d(8), ;
							Outros c(20), DetFecCxs n(1))
							
* Apanhando Formas de Pagamento
Select crDados
Scan
	lcEdn = crDados.EmpDopNums 
	
	lcQuery = [Select a.EmpDopNums, a.Parcs, a.FPags, b.Formas, b.Infos, a.Bancos, a.Agencias, ]+ ;
					 [a.Contas, a.Numeros, Case When b.trocos=1 then -a.valos else a.valos end as valos, a.MoeFpgs, a.Vencs, a.Outros, b.DetFecCxs ] + ;
			    [From SigMvPar a, SigOpFp b ] + ;
			   [Where a.EmpDopNums = '] + lcEdn + [' And ] + ;
			   		 [a.FPags = b.FPags ] + ;
			   [Order By a.Parcs]
			 
	If (ThisForm.poDataMgr.SqlExecute(lcQuery, [crPags1]) < 1)
		=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crPags1)])
		Return .f.
	EndIf

	Select crPags1
	Go Top In crPags1
	If Not Eof([crPags1])
		Select * ;
		  From crPags1 ;
		 Where Iif(Empty(lcFpg), .t., FPags = Padr(lcFPg, 12)) ;
		  Into Cursor crPags ReadWrite
	Else
		*** mena - 22/09/2014 - corrigido erro de programa. estava v.valos ***	
		lcQuery = [Select b.EmpDopNums, b.Parcs, b.FPags, c.Formas, c.Infos, b.Bancos, b.Agencias, b.Contas, b.Numeros, ]+ ;
					     [Case When c.trocos=1 then -b.valos else b.valos end as valos, b.MoeFpgs, b.Vencs, b.Outros, c.DetFecCxs ] + ;
			        [From SigMvPec a, SigMvPar b, SigOpFp c ] + ;
			       [Where a.Codigos = ] + Str(crDados.Codigos, 10) + [ And ] + ;
			     	     [a.EmpDopNums = b.EmpDopNums And ] + ;
			   		     [b.FPags = c.FPags ] + ;
				   		 Iif(Empty(lcFpg), [], [And b.FPags = '] + Padr(lcFPg, 12) + [' ]) + ;
	  			   [Order By b.Parcs]
			 
		If (ThisForm.poDataMgr.SqlExecute(lcQuery, [crPags]) < 1)
			=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crPags)])
			Return .f.
		EndIf
	EndIf
	
	Select crPags
	Go Top In crPags
	If Eof([crPags]) And Not Empty(lcFpg)
		Delete In crDados
	Else
		Scan
			Scatter Memvar Memo
			m.EmpDopNums = lcEdn
			m.FPags = Iif(m.Infos = [B] And Not Empty(m.Formas), m.Formas, m.FPags)
			
			Insert Into crPagamentos From Memvar
		EndScan
	EndIf
EndScan

Select Left(a.EmpDopNums, 3) as Emps, Substr(a.EmpDopNums, 4, 20) as Dopes, Right(a.EmpDopNums, 6) as Numes, ;
	   a.Datas, a.ContaDs as IClis, a.RClis, a.Notas, a.Valos * Iif(a.COpers = 2, -1, 1) as Valos, a.Vends, ;
	   Space(30) as NVends, a.Codigos, a.Usuars, b.Parcs, b.FPags, b.Bancos, b.Agencias, b.Contas, b.Numeros, ;
	   b.Valos * Iif(a.COpers = 2, -1, 1) as ValPagos, b.MoeFpgs, b.Vencs, b.DetFecCxs, b.Outros, a.RelFechas, ;
	   a.CalcFecs, Space(32) as RPags, Space(32) as RChqs ;
  From crDados a Right Join crPagamentos b ;
 	On a.EmpDopNums = b.EmpDopNums ;
 Order By a.Datas, a.EmpDopNums, b.Parcs ;
  Into Cursor crDetalhe ReadWrite

Select crDetalhe
Scan
	If (crDetalhe.RelFechas = 3 And crDetalhe.CalcFecs = 2)
		lcEdn = crDetalhe.Emps + crDetalhe.Dopes + crDetalhe.Numes
		
		lcQuery = [Select e.Numes + (o.NDopes * 1000000) as Codigos, p.EmpDopNums ] + ;
					[From SigMvCab e, SigCdOpe o, SigMvPec p ] + ;
				   [Where e.EmpDopNums = '] + lcEdn + [' And ] + ;
				   		 [o.Dopes = e.Dopes And ] + ;
				   		 [p.Codigos = e.Numes + (o.NDopes * 1000000) ] + ;
				   [Order By p.EmpDopNums]

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, [crVista1]) < 1)
			=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crVista1)])
			Return .f.
		EndIf

		Select crVista1
		Go Top In crVista1
		If Not Eof([crVista1])
			lcQuery = [Select r.Parcs, r.FPags, r.Valos, r.Vencs, r.Contas, r.Numeros ] + ;
						[From SigMvPar r ] + ;
					   [Where r.EmpDopNums = '] + crVista1.EmpDopNums + [' And ] + ;
					   		 [r.Parcs = ] + Str(crDetalhe.Parcs, 2) + [ ] + ;
					   [Order By r.Parcs]

			If (ThisForm.poDataMgr.SqlExecute(lcQuery, [crVista]) < 1)
				=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crVista)])
				Return .f.
			EndIf

			lnPri = 1
			Select crVista
			Scan
				If Not IsNull(crVista.Fpags)
					If (lnPri > 1)
						lcQuery = [Select NComps ] + ;
								    [From SigCdUsu ] + ;
								   [Where Usuarios = '] + crDetalhe.Vends + [']
								 
						If (ThisForm.poDataMgr.SqlExecute(lcQuery, [crNVends]) < 1)
							=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crNVends)])
							Return .f.
						EndIf
						Select crNVends
						Go Top In crNVends
						If Not Eof([crNVends])
							Replace NVends With crNVends.NComps In crDetalhe
						EndIf

						Skip In crDetalhe
					EndIf
					
					Replace RPags With Str(lnPri, 2) + [ ] + crVista.Fpags + [ ] + Space(6) + ;
									   Transform(crVista.Valos, [999,999.99]) In crDetalhe

					If Not Empty(crVista.Numeros)
						Replace RChqs With crVista.Contas + [ / ] + crVista.Numeros + [  ] + ;
								Dtoc(crVista.Vencs) In crDetalhe
					EndIf

					lnPri = lnPri + 1
				EndIf
			EndScan
		EndIf
	EndIf

	lcQuery = [Select NComps ] + ;
			    [From SigCdUsu ] + ;
			   [Where Usuarios = '] + crDetalhe.Vends + [']
			 
	If (ThisForm.poDataMgr.SqlExecute(lcQuery, [crNVends]) < 1)
		=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (crNVends)])
		Return .f.
	EndIf
	Select crNVends
	Go Top In crNVends
	If Not Eof([crNVends])
		Replace NVends With crNVends.NComps In crDetalhe
	EndIf
EndScan

* Montando a Impressăo Linha a Linha
Create Cursor crImpressao (Detalhe c(40))

Set Point To [,]
Set Separator To [.]

ThisForm.Cabecalho()

If ThisForm.chkDetOpe.Value 
	ThisForm.Detalhado()
EndIf

If ThisForm.chkVendRes.Value
	ThisForm.ResVends()
EndIf

If ThisForm.chkResFpg.Value
	ThisForm.ResumoPagto()
EndIf

If ThisForm.chkOpePag.Value 
	ThisForm.PagtoXfpags()
EndIf

If ThisForm.chkAnaPag.Value 
	ThisForm.AnaFPags()
EndIf

Set Point To [.]
Set Separator To [,]

Select crImpressao
Go Top In crImpressao
* Fim **************************************************************************************
ENDPROC
