PROCEDURE Init
lcQryEest = [Select * From SigMvCab]
lcQryEesti = [Select * From SigMvItn]
lcQrySigCdLin = [Select descs,linhas,pedidos From SigCdLin Order By descs,linhas,pedidos ]

ThisForm.cntSombra.lblSombra.Caption = ThisForm.Caption
ThisForm.cntSombra.lblTitulo.Caption = ThisForm.Caption

With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)


	If (.poDataMgr.pnIdconn > 0)
		With .poDataMgr
			.AddCursor('SigMvCab', 'cIdChaves', 'crSigMvCab', '', , lcQryEest)
			.AddCursor('SigMvItn', 'cIdChaves', 'crSigMvItn', '', , lcQryEesti)
			.AddCursor('SigCdLin', 'Linhas', 'crSigCdLin', '', , lcQrySigCdLin)

			.ReQuery('crSigCdLin')
		EndWith
		
		ThisForm.Get_Datai.Value = Date() - 7

		=DoDefault()
		
		Select crSigCdLin
		Index On DESCS tag DESCS
		Index On linhas tag linhas
	Else
		=Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith
* Fim **************************************************************************************
ENDPROC

* PROCEDURE Click

PROCEDURE Click
Private lcGrupo, lcEmpresa, lcConta, lcLinha, lcOperacao, lcFornece, Chave
Local lnItens, lnNumero

lcEmpresa = ThisForm.Get_cd_Empresa.Value
lcGrupo   = ThisForm.Get_cd_GrEstoque.Value
lcConta   = ThisForm.Get_cd_Estoque.Value
lcLinha   = ThisForm.Get_Linha.Value
lcNega    = ThisForm.Get_negativo.Value
ldData    = fDtoSQL(ThisForm.Get_Datai.Value)

If IsEmpty( lcEmpresa )
	Messagebox( 'É obrigatório informar a Empresa...', 32, '' )
	ThisForm.Get_cd_Empresa.SetFocus
	Return
Endif

If IsEmpty( lcGrupo )
	Messagebox( 'É obrigatório informar o Grupo...', 32, '' )
	ThisForm.Get_cd_GrEstoque.SetFocus
	Return
Endif

If IsEmpty( lcConta )
	Messagebox( 'É obrigatório informar a Conta...', 32, '' )
	ThisForm.Get_cd_Estoque.SetFocus
	Return
Endif

If IsEmpty( lcLinha )
	Messagebox( 'É obrigatório informar a Linha...', 32, '' )
	ThisForm.Get_Linha.SetFocus
	Return
Endif

Select crSigCdLin
Set Order to Linhas
If !Seek( lcLinha )
	Messagebox( 'Esta Linha de Produçăo năo está cadastrada...',32, '' )
	ThisForm.Get_Linha.SetFocus
	Return
Endif

If IsEmpty( crSigCdLin.Pedidos )	
	Messagebox( 'Esta Linha de Produçăo năo possui uma Operaçăo cadastrada...', 32, '' )
	ThisForm.Get_Linha.SetFocus
	Return
Endif
lcOperacao = crSigCdLin.Pedidos

Set Talk On
Set Talk Window

If lcNega # "S"

	lcQuery = [	Select E.Emps, E.Grupos, E.Estos, E.CPros, E.SQtds,]+;
			  [ P.CPros, P.Linhas, P.Situas, P.QMins, P.IFors, P.PVens, P.Moevs, P.Dpros ]+;
			  [ From SigMvEst E, SigCdPro P ]+;
			  [ Where E.Emps = ?lcEmpresa And E.Grupos = ?lcGrupo And ]+;
			  [		E.Estos = ?lcConta And E.CPros = P.CPros And ]+;
			  [		E.SQtds < P.QMins And P.Linhas = ?lcLinha And ]+;
			  [		P.Situas = 1 And P.QMins > 0 ] + ;
			  [ Order By E.Emps, E.Grupos, E.Estos, E.CPros, E.SQtds,]+;
			  [ P.CPros, P.Linhas, P.Situas, P.QMins, P.IFors, P.PVens, P.Moevs, P.Dpros ]
	If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crTemp1') < 1)
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTemp)')
		Return 0
	EndIf
	Chave = lcEmpresa+lcGrupo+lcConta

	lcQuery = [ Select CPros, QMins, IFors, PVens, Moevs, Dpros ]+;
		[	From SigCdPro ]+;
		[	Where Linhas = ?lcLinha and QMins > 0 And Situas=1 And ?Chave+CPros ]+;
		[		Not In (Select empgruests+Cpros as Chave From SigMvEst ) ]
	If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crTemp2') < 1)
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTemp2)')
		Return 0
	EndIf

	Select Emps, Grupos, Estos, CPros, SQtds, QMins, QMins - SQtds As DifProds, IFors, PVens, Moevs, Dpros ;
		From crTemp1 ;
		Where Emps = lcEmpresa And Grupos = lcGrupo And ;
				Estos = lcConta And ;
				SQtds < QMins And Linhas = lcLinha And ;
				Situas = 1 And QMins > 0 ;
	Union all;
	Select lcEmpresa as Emps, lcGrupo as Grupos, lcConta as Estos, CPros, 000000000.00 as SQtds, QMins,;
			QMins As DifProds, IFors, PVens, Moevs, Dpros ;
		From CrTemp2 ;
	Into Cursor TmpMinimo

Else
	lcQuery = [ Select E.Emps, E.Grupos, E.Estos, E.CPros, E.SQtds, ]+;
			  [  P.IFors, P.CPros, P.Situas, P.Linhas, P.PVens, P.Moevs, P.dpros ]+;
			  [ From SigMvEst E, SigCdPro P ]+;
			  [ Where E.Emps = ?lcEmpresa And E.Grupos = ?lcGrupo And ]+;
			  [	E.Estos = ?lcConta And E.CPros = P.CPros And ]+;
			  [	E.SQtds < 0 And P.Linhas = ?lcLinha And ]+;
			  [	P.Situas = 1 ] + ;
			  [ Order By E.Emps, E.Grupos, E.Estos, E.CPros, E.SQtds, ]+;
			  [  P.IFors, P.CPros, P.Situas, P.Linhas, P.PVens, P.Moevs, P.dpros ]
			  			  
	If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crTemp3') < 1)
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTemp3)')
		Return 0
	EndIf

	Select Emps, Grupos, Estos, CPros, SQtds, 0 as QMins,;
			Abs(SQtds) As DifProds, IFors, PVens, Moevs, dpros ;
		From crTemp3 ;
	Into Cursor TmpMinimo
Endif
Set Talk Off

Select TmpMinimo
Go Top
If Eof() Or Lastkey() = 27
	Messagebox( 'Nenhum produto selecionado...', 32, '' )
	Return
Endif

Set Talk On
Set Talk Window

lcQuery = [Select I.CPros, I.Qtds, I.QtBxProds, I.empdopnums, E.empdopnums, E.Grupods, E.Contads, E.Dopes, E.Emps, ] + ;
		  [ O.Dopes, O.Globalizas ] + ;
		  [	From SigMvCab E, SigMvItn I, SigCdOpe O ] + ;
		  [	Where E.Dopes = O.Dopes And E.Emps = ?lcEmpresa And ] + ;
		  [	(O.Globalizas=1 or O.Globalizas=2) And ] + ;
		  [	E.Grupods = ?lcGrupo And ] + ;
		  [	E.Contads = ?lcConta And E.EmpDopNums = I.EmpDopNums ] + ;
		  [ Order By I.CPros, I.Qtds, I.QtBxProds, I.empdopnums, E.empdopnums, E.Grupods, E.Contads, E.Dopes, E.Emps, ] + ;
		  [ O.Dopes, O.Globalizas ]
If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crTemp4') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTemp4)')
	Return 0
EndIf

Select CPros, Sum(Qtds - QtBxProds) As Produzindo ;
  From crTemp4 ;
 Group By CPros ;
  Into Cursor TmpPedidos

Set Talk Off

Select TmpPedidos
Index on Cpros Tag CPros

Select TmpMinimo
Index on IFors + CPros Tag ForProd

Select M.CPros, M.IFors, M.PVens, M.Moevs, M.Dpros,;
	M.DifProds - Iif( IsNull( P.Produzindo ), 0, P.Produzindo ) as Qtds;
	From TmpMinimo M Left Join TmpPedidos P On M.CPros = P.CPros ;
	Where M.DifProds > Iif( IsNull( P.Produzindo ), 0, P.Produzindo ) ;
	Into Cursor TmpProd

Select TmpProd
Go Top
If Eof()
	Messagebox( 'Nenhum produto selecionado...', 32, '' )
	Return
Endif

Select TmpProd
Index on IFors + CPros Tag ForProd
Count to lnTotalReg

loBarra = CreateObject( 'fwprogressbar', 'Processando Pedidos...', lnTotalReg )
loBarra.Show

lcFornece = Repl( Chr( 255 ), 10 ) 

Select TmpProd
Scan 
	loBarra.Update( .t. )
	If lcFornece # TmpProd.IFors
		
		lnNumero  = fGerUniqueKey( Alltrim( lcOperacao ) + lcEmpresa )
		lcFornece = TmpProd.IFors

		lcQuery = [Select Dopes, GruOrigs, Opers ] + ;
		    [ From SigCdOpe ] + ;
		    [ Where Dopes=?lcOperacao ] + ;
		    [ Order By Dopes, GruOrigs, Opers ]
		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crSigCdOpe') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigCdOpe)')
			Return 0
		EndIf
		Go Top in ('crSigCdOpe')
		Select TmpProd
		
		Insert Into crSigMvCab ( Emps, Dopes, Numes, Datas, Datars, MascNum,;
								Grupoos, Contaos, Grupods, Contads, Usuars, EmpDopNums, CidChaves, DtAlts ) ;
				Value ( lcEmpresa, lcOperacao, lnNumero, ldData, ldData,;
						Alltrim( fGerMascara( lnNumero ) ), Iif(IsEmpty(crSigCdOpe.GruOrigs),'ESTOQUE',crSigCdOpe.GruOrigs), lcFornece, lcGrupo,;
						lcConta, Usuar, lcEmpresa+lcOperacao+Str(lnNumero,6), fUniqueIds(), Date() )
		Select crSigMvCab
		Replace Datars     With Iif(IsEmpty(Datars),Null,Datars),;
				Datas      With Iif(IsEmpty(Datas),Null,Datas),;
				DataTrans  With Iif(IsEmpty(DataTrans),Null,DataTrans),;
				PrazoEnts  With Iif(IsEmpty(PrazoEnts ),Null,PrazoEnts ),;
				Dtemis     With Iif(IsEmpty(Dtemis),Null,Dtemis),;
				Obses      With Iif(IsEmpty(obses),Null,Obses),;
				DtFechas   With Iif(IsEmpty(DtFechas),Null,DtFechas),;
				DtAudits   With Iif(IsEmpty(DtAudits),Null,DtAudits)

		lnItens = 0
	Endif
	lnItens = lnItens + 1
	Insert Into crSigMvItn ( Emps, Dopes, Numes, CItens, CPros, Qtds, Units, Moedas, opers, totas, dpros, EmpDopNums, CidChaves, DtAlts ) ;
			Value ( lcEmpresa, lcOperacao, lnNumero, lnItens, TmpProd.CPros,;
					TmpProd.Qtds, TmpProd.PVens, TmpProd.Moevs, Iif(crSigCdOpe.opers = 1,'E','S'),(TmpProd.Qtds * TmpProd.PVens),TmpProd.dpros,;
					lcEmpresa + lcOperacao + Str(lnNumero,6), fUniqueIds(), Date() )
	Select crSigMvCab
	Replace crSigMvCab.ValInis with crSigMvCab.ValInis + ( TmpProd.PVens * TmpProd.Qtds ),;
			crSigMvCab.Valos with crSigMvCab.Valos + ( TmpProd.PVens * TmpProd.Qtds ),;
			crSigMvCab.DtAlts With Date()
Endscan
loBarra.Complete( .t. )

Select Min(Datas) as Datas From CrSigMvCab Into Cursor TmpGdm

**********************************************************************************************
If Not ThisForm.poDataMgr.Update('crSigMvCab')
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Update - crSigMvCab)')
	Return 0
EndIf
If Not ThisForm.poDataMgr.Update('crSigMvItn')
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Update - crSigMvItn)')
	Return 0
EndIf
If (ThisForm.poDataMgr.Commit() < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Commit)')
	Return 0
EndIf
**********************************************************************************************

Messagebox( 'Processamento Finalizado...', 32, '' )

Return

ENDPROC
