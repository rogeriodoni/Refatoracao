PROCEDURE visualizacao
Local lnTipoRel

lnTipoRel = ThisForm.optTipoRel.Value && 1-Sintético, 2-Analítico, 3-Resumo

	Report Form SigReAd4 Preview NoConsole

		Report Form SigReAd2 Preview NoConsole
	Else
		Report Form SigReAd1 Preview NoConsole
	Endif

------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Private lcNmEmpresa, lcNmTitulo, lcNmPeriodo, lcCdEmpresa, lcNmOperacao, ldDtInicial, ldDtFinal, lnNrOpcao, lnObs
Private lnNrCusto, lnNrVenda, lnNrSaldo, lnDecimals, lcFixed, lnTipoVars, lcMacroAcres, lcMacroDesc, lcMacroTotal
Private llAnalitico, lnMargem, lnTipoRel

With ThisForm
	lcCdEmpresa  = .get_Empresa.Value
	lcNmOperacao = .get_nm_operacao.Value
	lcMoeda      = .getcmoeda.Value
	lcDMoeda     = .getDmoeda.Value
	ldDtInicial	 = fDtoSQL(.get_dt_inicial.Value)
	ldDtFinal	 = fDtoSQL(.get_dt_final.Value,"23:59:59")
	lnTipoVars   = .opt_TipoVars.Value
	lcCdVendedor = .get_vended.Value
	lnObs        = .opt_obs.Value
	llAnalitico  = (.optTipoRel.Value # 1) && 1-Sintético, 2-Analítico, 3-Resumo
	lnTipoRel    = .optTipoRel.Value 
	lnMargem     = .getMargem.Value
EndWith

lcMoedaOper  = lcMoeda
lcPeriodo    = 'PERIODO : ' + Dtoc(ldDtInicial) + ' A ' + Dtoc(ldDtFinal)
if ThisForm.Titulos
	lcTitulo     = 'RELAÇĂO DE APURAÇĂO DE DESCONTOS/ACRÉSCIMOS'
	lcSubTit     = 'MOEDA : ' + Alltrim(lcMoeda) + ' - ' + Alltrim(lcDMoeda) + ;
				   Iif(!IsEmpty(lnMargem), ' - Margem : ' + Alltrim(Str(lnMargem,5,2)) + '%', '')
	llNomeVend   = fChecaAcesso('SigReAds', 'NOMEVEND')
else
	lcTitulo     = 'RAPDE'
	lcSubTit     = Alltrim(lcMoeda) + Iif(!IsEmpty(lnMargem), ' - (' + Alltrim(Str(lnMargem,5,2)) + '%)', '')
	llNomeVend   = .f.
endif

ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'crSigCdEmp', 'cEmps', lcCdEmpresa)

* Criaçăo do Arquivo Cabeçalho do Relatório
Create Cursor csCabecalho (NomeEmp C(80), Titulo C(80), Titulo2 C(200), Titulo3 C(200), ;
						   tObs N(1), Analitico L(1), NVend L(1), Titulos L(1) )
Insert Into csCabecalho (NomeEmp, Titulo, Titulo2, Titulo3, tObs, Analitico, NVend, Titulos) ;
 				 Values (crSigCdEmp.Razas, lcTitulo, lcSubTit, lcPeriodo, lnObs, llAnalitico, llNomeVend, ThisForm.Titulos )
 				 
If Used('csRelatorio')
	Use In csRelatorio
Endif

If Used('csProdutos')
	Use In csProdutos
Endif

m.lnDecimals = Set('Decimals')
m.lcFixed = Set('Fixed')
Set Decimals To 2
Set Fixed On

Set Talk Windows
Set Talk On
******************************************************************************************************************
lcNumOp = ''
If IsEmpty(lcNmOperacao)
	Select cs_SigCdTom
	Scan
		If Marca>0
			lcNumOp = lcNumOp + Iif(!Empty(lcNumOp),[ Or ],[(]) + [o.TipoOps = ]+Alltrim(Str(cs_SigCdTom.Codigos,2))
		Endif
	EndScan
	If Len(lcNumOp) > 1
		lcNumOp = lcNumOp + [)]
	Endif
Endif

IF InList(lnTipoRel,4,5)

	lcGrupo = ''
	Select cs_SigCdGrp
	Scan
		If Marca>0
			lcGrupo = lcGrupo + Iif(!Empty(lcGrupo),[ Or ],[(]) + [p.cgrus = ']+Alltrim(cs_SigCdGrp.cgrus)+[' ]
		Endif
	EndScan
	If !EMPTY(lcGrupo)
		lcGrupo= lcGrupo+ [) ]
	Endif

	lcWhere = IIF(EMPTY(lcCdEmpresa), [], [ and b.emps = ?lcCdEmpresa ])
	lcWhere = lcWhere + IIF(EMPTY(lcCdVendedor), [], [ and a.vends = ?lcCdVendedor ])
	If !Empty(lcNmOperacao)
		lcWhere = lcWhere + [ and a.dopes = ?lcNmOperacao ]
	ELSE
		lcWhere = lcWhere + IIF(EMPTY(lcNumOp),[], [ and ] + lcNumOp + [ ])
	ENDIF

	lcWhere = lcWhere + IIF(EMPTY(lcGrupo),[], [ and ] + lcGrupo + [ ])

	lcSql = [select 1 as Tipo, b.emps, a.vends, b.dopes, b.numes, b.cpros, b.codbarras, b.opers, b.qtds, a.empdopnums, p.cgrus, ] + ;
			[(b.totas+b.valdescs) * case when t.tpgdmis = 2 then -1 else 1 end AS totas, ] + ;
			[b.valrats * case when t.tpgdmis = 2 then -1 else 1 end as valrats, ] +;
			[((b.valdescs*b.qtds)-b.valrats) * case when t.tpgdmis = 2 then -1 else 1 end as desctotal, ] +; 
			[(b.totas+b.valrats) * case when t.tpgdmis = 2 then -1 else 1 end as valpags  ] + ;
			[from sigmvcab a, sigmvitn b, sigcdope o, sigcdpro p, sigcdtom t ] + ;
			[where a.datas between ?ldDtInicial And ?ldDtFinal and a.empdopnums = b.empdopnums ] +;
			[and a.dopes = o.dopes ] + ;
			[and o.TipoOps = t.codigos ] + ;
			[and b.cpros = p.cpros ] + lcWhere + ;
			[order by 1,3,4,5 ]

	ThisForm.poDatamgr.SqlExecute(lcSql ,"csRelatorio")
	SELECT cgrus, SUM(qtds) as qtds, SUM(valpags) as valpags, SUM(totas) as totas, SUM(desctotal) as desctotal from csRelatorio group by cgrus order by 1 into cursor crCgrus

	SELECT crCgrus
	GO top
	SCAN
		SELECT csRelatorio
		APPEND BLANK
		replace Tipo With 2
		replace dopes with 'RESUMO GRUPO'
		replace cpros with crCgrus.cgrus
		replace qtds with crCgrus.qtds
		replace totas with crCgrus.totas
		replace desctotal with crCgrus.desctotal
		replace valpags with crCgrus.valpags
		SELECT crCgrus
	EndScan
	
	Select csRelatorio 
	If lnTipoRel = 5
		Delete From csRelatorio Where Tipo = 1
	EndIf 

	
else
	lcWhere = Iif(!IsEmpty(lcCdEmpresa), [a.Emps = ?lcCdEmpresa And ], [])+;
			  [a.Emps = b.Emps And a.Dopes = b.Dopes And a.Numes =  b.Numes And ]+;
			  [a.Datas Between ?ldDtInicial And ?ldDtFinal And ]+;
			  [a.Dopes = o.Dopes And ]+;
			  Iif(!IsEmpty(lcNmOperacao), [a.Dopes = ?lcNmOperacao And ], [])+;
			  Iif(!IsEmpty(lcCdVendedor), [a.Vends = ?lcCdVendedor And ], [])+;
			  [a.Vends = c.Iclis And ] + Iif(IsEmpty(lcNmOperacao),lcNumOp + [ And ],[]) + ;
			  [(o.ValPres = 1 Or o.ValPres = 3)]
	 	   
	SqlEest = [Select a.Emps, a.Dopes, a.Numes, a.MascNum, a.Vends, c.Rclis, b.nSinals, b.ValItens, b.Moeds, a.Datas, ]+;
	          [b.ValTots, a.CodObs, a.Obses, a.Usuars ]+; 
	  		  [From SigMvCab a, SigMvMov b, SigCdCli c, SigCdOpe O ]+;
	 		  [Where ]+lcWhere 
	 		  
	ThisForm.poDatamgr.SqlExecute(SqlEest,"csSqlEest")

	Select csSqlEest

	Select Emps, Dopes, Numes, MascNum, Vends, Rclis, Usuars, ;
		   Sum(Iif(nSinals < 0, -1, 1) * ValItens * fBuscarCotacao(Moeds, Datas, ThisForm.poDataMgr) / fBuscarCotacao(lcMoedaOper, Datas, ThisForm.poDataMgr)) as ValInis, ;
	       (ValTots * fBuscarCotacao(Moeds, Datas, ThisForm.poDataMgr) / fBuscarCotacao(lcMoedaOper, Datas, ThisForm.poDataMgr)) as Valos, ;
	       CodObs, 'CAB' as Tp, 1 as Soma ;
	  	   From csSqlEest ;
	       Into Cursor TmpCsRelat ;
	       Group By Emps, Dopes, Numes, MascNum, Vends, Rclis, Usuars, 9, CodObs

	lcWhere = Iif(!IsEmpty(lcCdEmpresa), [a.Emps = ?lcCdEmpresa And ], [])+;
			  [a.Datas Between ?ldDtInicial And ?ldDtFinal And ]+;
			  [Not a.Valvars = 0 And ]+;
			  [a.Dopes = o.Dopes And ]+;
			  Iif(!IsEmpty(lcNmOperacao), [a.Dopes = ?lcNmOperacao And ], [])+;
			  Iif(!IsEmpty(lcCdVendedor), [a.Vends = ?lcCdVendedor And ], []) + ;
			  [a.Vends=c.Iclis And ] + Iif(IsEmpty(lcNmOperacao),lcNumOp + [ And ],[]) + ;
			  [Not (o.ValPres = 1 Or o.ValPres = 3)]
	 	   
	SqlEest = [Select a.Emps, a.Dopes, a.Numes, a.MascNum, a.Vends, c.Rclis, a.ValInis, o.cMoes, a.Datas, ]+;
	          [a.Valos, a.CodObs, a.Obses, a.Usuars ]+;
	  		  [From SigMvCab a, SigCdCli c, SigCdOpe O ]+;
	 		  [Where ]+lcWhere

	ThisForm.poDatamgr.SqlExecute(SqlEest,"csSqlEest2")

	Select csSqlEest2

	Select Emps, Dopes, Numes, MascNum, Vends, Rclis, Usuars, ;
	       Sum(ValInis * Iif(IsEmpty(cMoes), 1, fBuscarCotacao(Cmoes, Datas, ThisForm.poDataMgr)) / fBuscarCotacao(lcMoedaOper, Datas, ThisForm.poDataMgr)) as ValInis, ;
	       (Valos * Iif(IsEmpty(cMoes), 1, fBuscarCotacao(Cmoes, Datas, ThisForm.poDataMgr)) / fBuscarCotacao(lcMoedaOper, Datas, ThisForm.poDataMgr)) as Valos, ;
	       CodObs, 'CAB' as Tp, 1 as Soma ;
	  	   From csSqlEest2 ;
	  	   Into Cursor TmpCsRela2 ;
		   Group By Emps, Dopes, Numes, MascNum, Vends, Rclis, Usuars, 9, CodObs

	* Une as Duas Selects Acima Em Uma Apenas

	lcWhere1 = Iif(lnTipoVars=1, [ValInis > Valos Or (ValInis = 0 And Not Valos = 0)], ;
			   Iif(lnTipoVars=2, [ValInis < Valos Or ValInis = 0], ;
			   Iif(lnTipoVars=3, [Not ValInis = Valos], [0=0])))

	lcWhere2 = Iif(lnTipoVars=1, [ValInis > Valos], ;
			   Iif(lnTipoVars=2, [ValInis < Valos], ;
			   Iif(lnTipoVars=3, [Not ValInis = Valos], [0=0])))

	Select Emps, Dopes, Numes, MascNum, Vends, Rclis, ValInis, Valos, CodObs, Tp, Soma, ;
	       Space(14) as CPros, 00000000000000 as CBars, Space(65) as DPros, Space(1) as Opers, ;
	       000000.00 as Qtds, 00000.000000 as Units, 000000000.00 as Totas, 000000000.00 as DifVal, Usuars ;
	       From TmpCsRelat ;
	       Where &lcWhere1. ;
	       Union all ;															
	Select Emps, Dopes, Numes, MascNum, Vends, Rclis, ValInis, Valos, CodObs,  Tp, Soma, ;
	       Space(14) as CPros, 00000000000000 as CBars, Space(65) as DPros, Space(1) as Opers, ;
	       000000.00 as Qtds, 00000.000000 as Units, 000000000.00 as Totas, 000000000.00 as DifVal, Usuars ;
	       From TmpCsRela2 ;
	       Where &lcWhere2. ;
	       Into Cursor csRelatorio1 ;
	       Order By 1, 5, 2, 3, 11

	Select Emps, Dopes, Numes, Obses From  csSqlEest ;
	Union All ;
	Select Emps, Dopes, Numes, Obses From  csSqlEest2 ;
	Into Cursor TmpObses ;
	Order By 1, 2, 3 

	Select a.*,b.Obses from csRelatorio1 a Left Join TmpObses b ;
	                   on a.Emps = b.Emps And a.Dopes = b.Dopes And a.Numes = b.Numes ;
	                   into Cursor CsRelatorio READWRITE 

	*Use DBF('csRelatorio1') Again In 0 Alias csRelatorio
	Select csRelatorio
	Index On Emps + Dopes + Str(Numes, 6) Tag EmpDopNum
	Index On Emps + Vends + Dopes + Str(Numes, 6) + Tp Tag VdDpNmTp

	**********************************************************************************************
	* Filtra os Registros Pela Margem de Tolerancia
	If !IsEmpty(lnMargem)
		Select csRelatorio
		Scan
			If Abs(100 - ((Valos*100) / Iif(ValInis#0, ValInis, 1))) < lnMargem
				Delete
			Endif
		EndScan
	Endif
	**********************************************************************************************

	If llAnalitico

		lcIdQuery=SYS(2015)
		Local InsTmpD
		
		Zap In CrSigTempR
		
		Select csRelatorio 
		Go Top
		Scan
			m.Emps       = csRelatorio.Emps
			m.Dopes      = csRelatorio.Dopes
			m.Numes      = csRelatorio.Numes
			m.MascNum    = csRelatorio.MascNum
			m.CPros      = csRelatorio.Vends
			m.Razas      = csRelatorio.RClis
			m.Valors     = csRelatorio.Valos
			m.Qtds       = csRelatorio.ValInis
			m.CodObs     = csRelatorio.CodObs
			m.Obss       = csRelatorio.Obses
			m.EmpDopNums = Padr(csRelatorio.Emps,3) + Padr(csRelatorio.Dopes,20) + Str(csRelatorio.Numes,6)
			m.CIdQuerys  = lcIdQuery
			m.CIdChaves  = fUniqueIds()
			m.Contas   	 = csRelatorio.Usuars
			Insert Into CrSigTempR From MemVar
			Select CsRelatorio
		EndScan
		
		If ThisForm.poDataMgr.Update('CrSigTempR')
			ThisForm.poDataMgr.Commit()
		EndIf
		**a.Obss as Obses,
		SqlEesti = [Select a.Emps, a.Dopes, a.Numes, a.MascNum, a.cPros As Vends, a.Razas As RClis, a.Qtds As ValInis, ]+;
		           [a.Valors As Valos, a.CodObs, a.Obss as Obses, 'ITN' as Tp, 0 as Soma, b.CPros, b.CodBarras as CBars, b.DPros, b.Opers, b.Qtds, ]+;
			   	   [b.Totas As Units, b.Totas, 99999999.99 as DifVal, a.Contas as Usuars ]+;
		       	   [From SigTempR a, SigMvItn b ]+;
		       	   [Where a.EmpDopNums = b.EmpDopNums And a.cIdQuerys = ']+lcIdQuery+[']
		ThisForm.poDatamgr.SqlExecute(SqlEesti,"csProdutos")
		  
		Select csProdutos
		Index On Emps + Dopes + Str(Numes, 6) Tag EmpDopNum
		Index On Emps + Vends + Str(Numes, 6) Tag EmpVenNum
		Set Order To EmpVenNum

		* Carregando o Valor Realmente Pago em "DifVal"
		Select csProdutos
		Set Order To EmpDopNum
		Replace all DifVal with 0
		Scan
			lnDif = 0
			ThisForm.poDataMgr.CursorQuery('SigMvMov', 'crSigMvMov', 'EmpDopNums', csProdutos.Emps + csProdutos.Dopes + Str(csProdutos.Numes, 6))
			Select crSigMvMov
			Scan 
				lnDif = lnDif + crSigMvMov.Valtots - ( crSigMvMov.Nsinals *  crSigMvMov.Valitens )
			EndScan		
			Replace DifVal With lnDif In csProdutos
		EndScan

		Select csProdutos
		Set Order To EmpVenNum
		Go Top
		oProg = CreateObject('fwprogressbar', 'Processando Produtos...', Recc() * 2)
		oProg.Top = oProg.Top - Int(oProg.Height/2)
		oProg.Show
		Do While !Eof()
		
			oProg.Update(.t.)
			lcEmp = csProdutos.Emps
			lcVen = csProdutos.Vends
			lnNum = csProdutos.Numes

			lnTotI1 = 0
			lnRec = Recno()
			lnDif = 0
			lnItn = 0
			lnGer = 0
		
			Scan While csProdutos.Emps=lcEmp And csProdutos.Vends=lcVen And csProdutos.Numes=lnNum
			* InList(csProdutos.Dopes, Padr('TR', 20), Padr('V', 20), Padr('VF', 20)) .And.
				If ! Inlist(csProdutos.Cpros,Padr("CP",14), Padr("CR",14), Padr("C",14), Padr("E",14)) And CsProdutos.Opers = "S"
					lnItn = lnItn + csProdutos.Totas
					lnTotI1 = lnTotI1 + 1
				Endif
				lnGer = lnGer + csProdutos.Totas
				lnDif = Iif(IsEmpty(lnDif), csProdutos.DifVal, lnDif)
			EndScan
			lnResto = lnDif
			If lnGer < 0
				lnResto = Abs(lnResto)
			Endif

			lnTotI2 = 0
			Go lnRec
			Scan While csProdutos.Emps=lcEmp And csProdutos.Vends=lcVen And csProdutos.Numes=lnNum

			* InList(csProdutos.Dopes, Padr('TR', 20), Padr('V', 20), Padr('VF', 20)) .And.

				If 	! Inlist(csProdutos.Cpros,Padr("CP",14), Padr("CR",14), Padr("C",14), Padr("E",14)) .And. CsProdutos.Opers = 'S'

					lnTotI2 = lnTotI2 + 1

					Rateio_TR  = Round(lnDif * (csProdutos.Totas / Iif(lnItn <> 0, lnItn, 1)), 0)
					If lnTotI1 = lnTotI2
			            lnVlVend = csProdutos.Totas + lnResto
					Else
			            lnVlVend  = csProdutos.Totas + Rateio_TR
			            lnResto = lnResto - Rateio_TR
					Endif
					Replace Totas With lnVlVend In csProdutos
				Endif
			EndScan
		Enddo

		Select csProdutos
		Scan
			oProg.Update( .t. )																			
			Insert Into csRelatorio (Emps, Dopes, Numes, MascNum, Vends, Rclis, ValInis, Valos, CodObs, Obses, Tp, ;
									 Soma, CPros, CBars, DPros, Opers, Qtds, Units, Totas, DifVal, Usuars) ;
						     Values (csProdutos.Emps, csProdutos.Dopes, csProdutos.Numes, csProdutos.MascNum, ;
						     		 csProdutos.Vends, csProdutos.RClis, 0, 0, csProdutos.CodObs, csProdutos.Obses, ;
						     		 csProdutos.Tp, csProdutos.Soma, csProdutos.CPros, csProdutos.CBars, ;
						     		 csProdutos.DPros, csProdutos.Opers, csProdutos.Qtds, csProdutos.Units, ;
						     		 csProdutos.Totas, csProdutos.DifVal, csProdutos.Usuars)
		EndScan
		oProg.Complete
		ThisForm.poDataMgr.fSqlApagarTmp('SigTempR', lcIdQuery)
	Endif

	If lnObs = 1 && Se Imprime a Observaçăo
		Select csRelatorio
		Set Order To VdDpNmTp Descending
		lcEmp = ''
		lcVen = ''
		lnNum = 0
		Scan
			If lcEmp # csRelatorio.Emps .Or. lcVen # csRelatorio.Vends .Or. lnNum # csRelatorio.Numes
				Replace Tp With 'OB' + Subs(csRelatorio.Tp,1,1) In csRelatorio
				lcEmp = csRelatorio.Emps
				lcVen = csRelatorio.Vends
				lnNum = csRelatorio.Numes
			Endif
		EndScan
		Set Order To VdDpNmTp Ascending
	Endif
	******************************************************************************************************************
	Set Talk Off

	Set Decimals To m.lnDecimals
	Set Fixed &lcFixed.

	Select csRelatorio
	Set Order To VdDpNmTp

	If (lnTipoRel = 3) && Resumo
		Select Emps, Count(1) as TotSoma ;
		       From TmpCsRelat ;
		       Group By Emps ;
		       Into Cursor csTot1 
		Index On Emps Tag Emps

		Select Emps, Count(1) as TotSoma ;
		       From TmpCsRela2 ;
		       Group By Emps ;
		       Into Cursor csTot2
	    Index On Emps Tag Emps

		Select Emps, Sum(ValInis) as ValInis, Sum(Valos) as Valos,  ;
			   Sum(Iif(ValInis > Valos, ValInis - Valos, 000000000.0000)) as Desconto, ;
			   Sum(Iif(Valos > ValInis, Valos - ValInis, 000000000.0000)) as Acrescimo, ;
			   Sum(Soma) as Soma, 000000 as TotSoma ;
	      From csRelatorio ;
	     Group By Emps ;
	      Into Cursor csRelatorio ReadWrite

		Select csRelatorio
		Scan
			If Seek(csRelatorio.Emps, 'csTot1', 'Emps')
				Replace TotSoma With TotSoma + csTot1.TotSoma In csRelatorio
			EndIf 
			If Seek(csRelatorio.Emps, 'csTot2', 'Emps')
				Replace TotSoma With TotSoma + csTot1.TotSoma In csRelatorio
			EndIf 
		EndScan
	EndIf
ENDIF

Select csRelatorio
Goto Top
******************************************************************************************************************
ENDPROC

------------------------------------------------------------
* PROCEDURE impressao
------------------------------------------------------------
PROCEDURE impressao
Local lnTipoRel

lnTipoRel = ThisForm.optTipoRel.Value && 1-Sintético, 2-Analítico, 3-Resumo

If InList(lnTipoRel,4,5)
	Report Form SigReAd4 to Printer Prompt NoConsole
Else
	If lnTipoRel = 3
		Report Form SigReAd2 to Printer Prompt NoConsole
	Else
		Report Form SigReAd1 to Printer Prompt NoConsole
	ENDIF
ENDIF

ENDPROC

------------------------------------------------------------
* PROCEDURE Init
------------------------------------------------------------
PROCEDURE Init
With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)
	With .poDataMgr
		If (.pnIdconn > 0)
			.AddCursor('SigTempR','CIdChaves','CrSigTempR','','')
			=DoDefault()
		Else
			=Messagebox('Impossível Efetuar Conexăo Com o Servidor de Banco de Dados...', 0+48, 'CreateObject')
			Return .f.
		EndIf
		.SqlExecute([Select MoePadVens, GrPadVens From SigCdPam],"crSigCdPam")
	EndWith
EndWith
* Fim **************************************************************************************

&& Seleciona Apenas As Empresas Năo Oficiais

SqlEmp = [Select cEmps, Razas, TipoNFs From SigCdEmp Where not SigCdEmp.tiponfs = 2]
ThisForm.poDatamgr.SqlExecute(SqlEmp,"JanEmp")
Select JanEmp
index on cemps tag cemps
index on razas tag razas

*-[ Carrega a lista do SigCdTom ]-*
lcSQLCmd = [Select Codigos, Descrs From SigCdTom]
If (ThisForm.poDataMgr.SQLExecute(lcSQLCmd, 'crSigCdTom') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo')
	Return 0
EndIf

Select crSigCdTom
Scan
	Insert Into Cs_SigCdTom (Marca,Codigos,Descri) ;
	Values (Iif(InList(crSigCdTom.codigos,4,5),1,0),crSigCdTom.codigos,Str(crSigCdTom.codigos,2) + '-' + crSigCdTom.descrs)
EndScan

Select cs_SigCdTom

	* consiste o campo data inicial

		messagebox('Data inicial inválida !',16,'')
		thisform.get_dt_inicial.setfocus
		return(.f.)
	endif

	* consiste o campo data final

		messagebox('Data final inválida !',16,'')
		thisform.get_dt_final.setfocus
		return(.f.)
	endif
	
	* consiste os campos de datas
	if thisform.get_dt_inicial.value > thisform.get_dt_final.value
		messagebox('Data inicial maior que data final !',16,'')
		thisform.get_dt_inicial.setfocus
		return(.f.)
	endif
endif

DODefault()
