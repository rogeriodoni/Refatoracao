PROCEDURE atualizatmp

Lparameters lnTodos

Wait Window 'Atualizando Arquivos...' NoWait

Local lnPos 
lnPos = 0

Select crOrigem
Locate For .f.

Select crDestino
Locate For .f.

If lnTodos==1
	Zap In TmpBarra

Else
	Select crDestino
	Locate For .f.
	
	Delete From TmpBarra Where Programas + Parametros Not In (Select b.Programas + b.Parametros From crDestino b Where Not Empty(b.Grupos))
	
EndIf 

Select crDestino

Scan

	lnPos = lnPos + 1 

	Scatter MemVar	
	m.BarraOrdem = lnPos
	m.SelBarras  = .f.

	If Not Empty(m.BarraForms) And Not Seek(m.Programas + m.Parametros, 'TmpBarra', 'ProPar')
		Insert Into TmpBarra From Memvar

	EndIf
	
EndScan

Select TmpBarra
Set Order to BarraOrdem
Locate For .f.
Go Top

Wait Clear

* Fim **************************************************************************************

ENDPROC

PROCEDURE montaacesso
parameters latualiza

*!*	SELECT *, 0 as inseriu from crorigem into cursor crmenuace readwrite

*!*	WITH thisform.pagina.dados.pagina.aceSSO.oleTree
*!*		cleft = .Left
*!*		ctop = .top
*!*		.left = 2000
*!*		.top = 2000

*!*		thisform.LockScreen = .T. 

*!*		SELECT crmenuace
*!*		SCAN
*!*			IF !latualiza
*!*				nreg = RECNO()
*!*				cmenu = ALLTRIM(SUBSTR(crmenuace.descricaos,1,RAT('-',crmenuace.descricaos,1)-1))
*!*				LOCATE for ALLTRIM(descricaos) == cmenu
*!*				IF !EOF()
*!*					nreg2 = RECNO()
*!*					GO nreg
*!*					IF crmenuace.inseriu = 0
*!*						cmenu = ALLTRIM(SUBSTR(crmenuace.descricaos,RAT('-',crmenuace.descricaos,1)+1))
*!*						onode = .Nodes.Add(ALLTRIM(STR(nreg2) + "_"),4,ALLTRIM(STR(nreg) + "_"),cmenu,0)
*!*						onode.tag = crmenuace.descricaos
*!*						replace inseriu with 1
*!*					ENDIF
*!*				ELSE
*!*					GO nreg
*!*					IF crmenuace.inseriu = 0
*!*						cmenu = ALLTRIM(SUBSTR(crmenuace.descricaos,AT('-',crmenuace.descricaos,1)+1)) && crmenuace.descricaos
*!*						onode = .nodes.Add(,1,ALLTRIM(STR(RECNO()) + "_"),cmenu,0)
*!*						onode.tag = crmenuace.descricaos
*!*						replace inseriu with 1
*!*					ENDIF
*!*				ENDIF
*!*			ELSE
*!*				nindex = crmenuace.npos && RECNO() &&.Nodes.Count
*!*				IF ! Empty(crmenuace.grupos) or crmenuace.expandi = 1
*!*					.Nodes.Item(nindex).Checked = IIF(!Empty(crmenuace.grupos),.t.,.f.)
*!*					.Nodes.Item(nindex).Expanded = IIF(thisform.pcescolha = 'CONSULTAR' or crmenuace.expandi=1,.t.,.f.)
*!*					.Nodes.Item(nindex).BackColor = IIF(crmenuace.expandi = 1,RGB(0,185,255),RGB(255,255,255))

*!*					FOR i = crmenuace.npos to 1 step -1 &&RECNO() .Nodes.Count
*!*						IF ISNULL(.Nodes.Item(i).Parent)
*!*							nindex = 0 && Root
*!*						ELSE
*!*							nindex = .Nodes.Item(i).Parent.index
*!*							.Nodes.Item(nindex).Expanded = IIF(thisform.pcescolha = 'CONSULTAR' or crmenuace.expandi=1,.t.,.f.)
*!*						ENDIF
*!*						i = nindex + IIF(nindex=1,0,1)
*!*					ENDFOR
*!*				Else
*!*					If (Type([.Nodes.Item(nindex)]) = [O])
*!*						.Nodes.Item(nindex).Checked = .f.
*!*						.Nodes.Item(nindex).Expanded = IIF(thisform.pcescolha = 'CONSULTAR',.t.,.f.)
*!*						.Nodes.Item(nindex).BackColor = RGB(255,255,255)
*!*					EndIf
*!*				ENDIF
*!*			ENDIF
*!*		ENDSCAN
*!*		.left = cleft
*!*		.top = ctop
*!*		.refresh()
*!*		thisform.LockScreen = .F.
*!*	Endwith

Select crAcCamp
Go Top
	
With thisform.pagina.dados.pagina.page1.grade
	.RecordSource = []
	
	Select a.cCodigos, a.cDescTelas, a.cCampos, a.cObjetos, a.cDescCamps, Nvl(b.nStatus,1) as nStatus, Nvl(b.Grupos,'') as Grupos, Nvl(b.Usuarios,'') as Usuarios, ;
		Nvl(b.PkChaves,'') as PkChaves ;
		from crAcCamp a Left Join crTmpAcTel b On a.cCodigos = b.cCodigos And a.cCampos = b.cCampos ;
		into cursor crSigAcTel ReadWrite
		
	.RecordSource = [crSigAcTel]
	.column2.ControlSource = [crSigAcTel.cDescTelas]
	.column3.ControlSource = [crSigAcTel.cDescCamps]
	.column4.ControlSource = [crSigAcTel.nStatus]
	.column4.cmbStatus.RowSource = [Padrăo,Invisível,Năo Editável,Obrigatório]
	.column4.cmbStatus.RowSourceType = 1
Endwith

PROCEDURE DragOver
Lparameters oSource, nXCoord, nYCoord, nState

Do Case
	Case (nState = 0) && Enter
		oSource.DragIcon = ThisForm.CanDropIcon
		
	Case (nState = 1) && Leave
		oSource.DragIcon = ThisForm.NoDropIcon
EndCase
* Fim **************************************************************************************

ENDPROC

------------------------------------------------------------
* PROCEDURE DragDrop
------------------------------------------------------------
PROCEDURE DragDrop
Lparameters oSource, nXCoord, nYCoord

If (oSource.Name <> This.Name) And Inlist(ThisForm.pcEscolha, 'INSERIR', 'ALTERAR')
	ThisForm.Pagina.Dados.Pagina.Cadastro.cmdAdd.Click
EndIf
* Fim **************************************************************************************

ENDPROC

------------------------------------------------------------
* PROCEDURE MouseMove
------------------------------------------------------------
PROCEDURE MouseMove
Lparameters nButton, nShift, nXCoord, nYCoord

If (nButton = 1) && Left Mouse
	If (Abs(nXCoord - ThisForm.MouseX) > ThisForm.DragThreshold Or ;
		Abs(nYCoord - ThisForm.MouseY) > ThisForm.DragThreshold ) And ;
		Inlist(ThisForm.pcEscolha, 'INSERIR', 'ALTERAR')
		THIS.Drag
	EndIf
EndIf
* Fim **************************************************************************************

ENDPROC

------------------------------------------------------------
* PROCEDURE MouseDown
------------------------------------------------------------
PROCEDURE MouseDown
Lparameters nButton, nShift, nXCoord, nYCoord

ThisForm.MouseX = nXCoord
ThisForm.MouseY = nYCoord
* Fim **************************************************************************************
ENDPROC

------------------------------------------------------------
* PROCEDURE KeyPress
------------------------------------------------------------
PROCEDURE KeyPress
Lparameters nKeyCode, nShiftAltCtrl

If (nKeyCode = 63) And (nShiftAltCtrl = 1)
	ThisForm.SelectAll(This)
EndIf
* Fim **************************************************************************************

ENDPROC
