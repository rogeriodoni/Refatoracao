PROCEDURE Init
Private llNot

llNot = .f.

With ThisForm
	.loNegocios = CreateObject('cnSigReAto', .Name)

	If (.loNegocios.poDataMgr.pnIdconn > 0)
		.getDtInicial.Value = Date()
		.getDtFinal.Value   = Date()

		.loObj = CreateObject('Relation')

		.loObj.AddProperty('DataInicial', .getDtInicial.Value)
		.loObj.AddProperty('DataFinal',   .getDtFinal.Value)
		.loObj.AddProperty('Grupo',       [])
		.loObj.AddProperty('Produto',     [])
		.loObj.AddProperty('Linha',       [])
		.loObj.AddProperty('Fornecedor',  [])
		.loObj.AddProperty('Imagem',      .f.)
		.loObj.AddProperty('Diferenca',   1)
		.loObj.AddProperty('Tipos',       1)
		.loObj.AddProperty('TrataErro',   [])

		With .GrdOperacoesE
			.ColumnCount  = 3
			.RecordSource = 'csOperacoesE'

			.Column1.ControlSource   = 'csOperacoesE.SelImp'
			.Column1.Movable 	     = .f.
			.Column1.Resizable       = .f.
			.Column1.ReadOnly        = .f.
			.Column1.Sparse          = .f.
			.Column1.Check1.AutoSize = .t.

			.Column2.ControlSource = 'csOperacoesE.Codigos'
			.Column3.ControlSource = 'csOperacoesE.Descrs'

			For I=2 To .ColumnCount
				lcCol = Alltrim(Str(I))
				.Column&lcCol..FontName = 'Courier New'
				.Column&lcCol..Movable   = .f.
				.Column&lcCol..Resizable = .f.
				.Column&lcCol..ReadOnly  = .t.
			Next
		EndWith 

		With .GrdOperacoesS
			.ColumnCount  = 3
			.RecordSource = 'csOperacoesS'

			.Column1.ControlSource   = 'csOperacoesS.SelImp'
			.Column1.Movable 	     = .f.
			.Column1.Resizable       = .f.
			.Column1.ReadOnly        = .f.
			.Column1.Sparse          = .f.
			.Column1.Check1.AutoSize = .t.

			.Column2.ControlSource = 'csOperacoesS.Codigos'
			.Column3.ControlSource = 'csOperacoesS.Descrs'

			For I=2 To .ColumnCount
				lcCol = Alltrim(Str(I))
				.Column&lcCol..FontName = 'Courier New'
				.Column&lcCol..Movable   = .f.
				.Column&lcCol..Resizable = .f.
				.Column&lcCol..ReadOnly  = .t.
			Next
		EndWith 

		If Not .loNegocios.CursorQuery('SigCdOpe',  'crSigCdOpe',  .f., .f., 'Dopes, TipoOps')
			=Messagebox('Impossível Criar o Cursor de Operaçőes...', 0+48, 'CreateObject')
			Return .f.
		EndIf

		If Not .loNegocios.CursorQuery('SigCdTom', 'crSigCdTom', .f., .f., 'Codigos, Descrs')
			=Messagebox('Impossível Criar o Cursor de Tipos de Operaçőes...', 0+48, 'CreateObject')
			Return .f.
		EndIf

		Select crSigCdTom
		Scan
			Insert Into csOperacoesE (SelImp, Codigos, Descrs) ;
							  Values (crSigCdTom.Codigos = 3, crSigCdTom.Codigos, crSigCdTom.Descrs)

			Insert Into csOperacoesS (SelImp, Codigos, Descrs) ;
							  Values (crSigCdTom.Codigos = 6, crSigCdTom.Codigos, crSigCdTom.Descrs)
		EndScan
		Go Top In csOperacoesE
		.GrdOperacoesE.Refresh

		Go Top In csOperacoesS
		.GrdOperacoesS.Refresh

		If Not .loNegocios.CursorQuery('SigCdCeg', 'crSigCdCeg', .f., .f., 'Grupos, Contas')
			=Messagebox('Impossível Criar o Cursor de Estoques...', 0+48, 'CreateObject')
			Return .f.
		EndIf

		Select crSigCdCeg
		Scan
			Scatter Memvar
			Insert Into csSigCdCeg From Memvar
		EndScan

		Select csSigCdCeg
		Set Order To GruContas
		Go Top In csSigCdCeg

		With .grdSigCdCeg
			.RecordSource    	   = 'csSigCdCeg'
			.RecordSourceType	   = 1
			.Column1.ControlSource = 'csSigCdCeg.lMarca'
			.Column2.ControlSource = 'csSigCdCeg.Grupos'
			.Column3.ControlSource = 'csSigCdCeg.Contas'

			For I=2 To .ColumnCount
				lcCol = Alltrim(Str(I))
				.Column&lcCol..FontName = 'Courier New'
				.Column&lcCol..Movable   = .f.
				.Column&lcCol..Resizable = .f.
				.Column&lcCol..ReadOnly  = .t.
			Next
		EndWith

		=DoDefault()
	Else
		=Messagebox('Impossível Efetuar Conexăo Com o Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .F.
	EndIf
EndWith
* Fim **************************************************************************************
ENDPROC

	Select Iif(Ordem = [0000000001], CPros,  Space(14)) as Código, ;
		   Iif(Ordem = [0000000001], DPros,  Space(40)) as Descriçăo, ;
		   Iif(Ordem = [0000000001], Reffs,  Space(15)) as Referęncia, ;
		   Iif(Ordem = [0000000001], CGrus,  Space(03)) as Grp, ;
		   Iif(Ordem = [0000000001], Linhas, Space(10)) as Linha, ;

		   Iif(Ordem = [0000000001], Entradas - Saidas, 000000000.00) as Saldo, ;

		   Iif(Estos = [Total :   ], Diferencas, 000000000.00) as Diferenças ;
	  From crImpressao ;
	  Into Cursor csExcell
Else 							 && Sintético
	Select CPros as Código, DPros as Descriçăo, Reffs as Referęncia, CGrus as Grp, Linhas as Linha, ;

		   Estoques as Qtde, Diferencas as Diferenças ;
	  From crImpressao ;
	  Into Cursor csExcell
EndIf

Select csExcell
Go Top

lcArqExcel1 = PutFile([Nome da Planilha], [], [XLS;TXT])

If fVerificaPasta(lcArqExcel1, 2)
	If (Lower(Alltrim(JustExt(lcArqExcel1)))=='xls')
		Copy To (lcArqExcel1) Type Xl5
	Else 
		Copy To (lcArqExcel1) Delimited With "" With Character ';'
	EndIf 

	If File(lcArqExcel1)
		=MessageBox('Planilha Gerada Com Sucesso!!!', 0+48+32, '')
	EndIf
Else
	=MessageBox('Erro na Geraçăo da Planilha,' + Chr(10) + 'Verificar Diretório ou Planilha Já Aberta...', 0+48+32, '')
EndIf

PROCEDURE btnReport.Click
If (This.Value <> 4) && Se Năo Clicou em Saída
	Select Codigos From csOperacoesE Where SelImp Into Cursor csTmpE
	
	If (Reccount('csTmpE') < 1)
		=Messagebox('Nenhum Tipo de Operaçăo de Entrada Foi Selecionado!!!', 0+48, '')
		ThisForm.grdOperacoesE.Column1.SetFocus
		Return .f.
	EndIf
	
	Select Codigos From csOperacoesS Where SelImp Into Cursor csTmpS
	
	If (Reccount('csTmpS') < 1)
		=Messagebox('Nenhum Tipo de Operaçăo de Saida Foi Selecionado!!!', 0+48, '')
		ThisForm.grdOperacoesS.Column1.SetFocus
		Return .f.
	EndIf

	Select Codigos From csTmpE Where Codigos In (Select Codigos From csTmpS) Into Cursor csTmpDE
	Select Codigos From csTmpS Where Codigos In (Select Codigos From csTmpE) Into Cursor csTmpDS
	
	If (Reccount('csTmpDE') > 0) Or (Reccount('csTmpDS') > 0)
		=Messagebox('Foram Selecionados Tipos de Operaçőes de Entrada e Saida Iguais!!!', 0+48, '')
		ThisForm.grdOperacoesE.Column1.SetFocus
		Return .f.
	EndIf	

	With ThisForm
		.loObj.DataInicial = .getDtInicial.Value
		.loObj.DataFinal   = .getDtFinal.Value
		.loObj.Grupo       = .getGrupo.Value
		.loObj.Produto     = .getCPros.Value
		.loObj.Linha 	   = .getLin.Value
		.loObj.Fornecedor  = .getFornecedor.Value
		.loObj.Imagem      = (.optImagem.Value = 1)
		.loObj.Tipos	   = .optTipos.Value
		.loObj.Diferenca   = .optDiferencas.Value

		.loNegocios.ValidaBotoes(ThisForm.loObj)	
		
		llRet = .loObj.TrataErro
	EndWith

	Do Case
		Case llRet = [DataInicial]
			=Messagebox('Data Inicial Inválida!!!', 0+48, '')
			ThisForm.getDtInicial.SetFocus
			Return .f.

		Case llRet = [DataFinal]
			=Messagebox('Data Final Inválida!!!', 0+48, '')
			ThisForm.getDtFinal.SetFocus
			Return .f.

		Case llRet = [Final < Inicial]
			=Messagebox('A Data Final Năo Pode Ser Menor Que a Inicial!!!', 0+48, '')
			ThisForm.getDtFinal.SetFocus
			Return .f.
	EndCase
EndIf

=DoDefault()
* Fim **************************************************************************************
ENDPROC
