PROCEDURE atualizatmp

Lparameters lnTodos

Wait Window 'Atualizando Arquivos...' NoWait

Local lnPos 
lnPos = 0

Select crOrigem
Locate For .f.

Select crDestino
Locate For .f.

If lnTodos==1
	Zap In TmpBarra

Else
	Select crDestino
	Locate For .f.
	
	Delete From TmpBarra Where Programas + Parametros Not In (Select b.Programas + b.Parametros From crDestino b Where Not Empty(b.Usuarios))
	
EndIf 

Select crDestino

Scan

	lnPos = lnPos + 1 

	Scatter MemVar	
	m.BarraOrdem = lnPos
	m.SelBarras  = .f.

	If Not Empty(m.BarraForms) And Not Seek(m.Programas + m.Parametros, 'TmpBarra', 'ProPar')
		Insert Into TmpBarra From Memvar

	EndIf
	
EndScan

Select TmpBarra
Set Order to BarraOrdem
Locate For .f.
Go Top

Wait Clear
* Fim **************************************************************************************

ENDPROC

PROCEDURE NodeCheck
*** ActiveX Control Event ***
LPARAMETERS node

IF Inlist(ThisForm.pcEscolha, 'INSERIR', 'ALTERAR')
	SELECT crGruAce
	GO node.index

	IF this.nodes.Item(node.Index).Checked
		Replace marca With 1

		lcIdChaves = 'GRUPO' + crGruAce.grupos
		Select crLogData
		Seek lcIdChaves
		If ! Found()
			Insert Into crLogData (cIdChaves, cOcors, cTipos) ;
				Values(lcIdChaves, 'Usuario : ' + This.Parent.Usuario.Value + ' Grupo : ' + crGruAce.grupos, 'I')
		Else
			If cTipos = 'E'
				Delete 
			EndIf
		Endif		
	ELSE
		Replace marca With 0

		lcIdChaves = 'GRUPO' + crGruAce.grupos
		Select crLogData
		Seek lcIdChaves
		If ! Found()
			Insert Into crLogData (cIdChaves, cOcors, cTipos) ;
				Values(lcIdChaves, 'Usuario : ' + This.Parent.Usuario.Value + ' Grupo : ' + crGruAce.grupos, 'E')
		Else
			If cTipos = 'I'
				Delete 
			EndIf
		Endif		
	ENDIF
ENDIF
ENDPROC
