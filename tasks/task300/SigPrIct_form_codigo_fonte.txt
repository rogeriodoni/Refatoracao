PROCEDURE processamento

Private pDtI, pDtF, pLot, pNop, lcOrdMccr, lnRegMccr

Local loProgresso, tot, Cta, Cred, Deb, Hisc, Ano, Dtct, Empc, Totreg, NumAux, NomeAux, ValFecha, lcConexao, lcCpoMccr, ;
	ValOcoTit

IniPer = Thisform.Get_Datai.Value
FinPer = Thisform.Get_Dataf.Value

lcCpoMccr = [Emps, Nopers, Opers, Tipos, Grupos, Contas, SContas, Datas, Hists, Hist2s, Valors, Cotacaos, SValors, Valocurs, Nfs, EmpdopNums, Titulos, EspecieNfs, SGRUPOS, CONTEMS,empdopncs ]

* Processo 1 :
Create Cursor Grupos (Codigos c(10), ContConts N(1))

Create Cursor SemConta (Contas c(09), DataS d Null, Hists c(70), Valors N(12,2), Ocors c(40))
Index On Contas + Dtos(DataS) Tag Conta

Create Cursor LoteProc ( Lotes N (06) )
Index On Lotes Tag Lotes

Thisform.poDataMgr.CursorQuery('SigCdEmp', 'crSigCdEmp', 'Cemps', _Empr, 'Razas')
Create Cursor Cabecalho (Empresa c(80), Titulo c(80), SubTit c(80), Periodo c(80))
Insert Into Cabecalho (Empresa, Titulo, Periodo) ;
	Values (_Empr + ' - ' + crSigCdEmp.Razas, ;
	'Relatório de Inconsistęncias de Integraçăo Contábil', ;
	'Período: ' + Dtoc(IniPer) + ' ŕ ' +  + Dtoc(FinPer))

Ocor1 = "Conta Contábil năo Cadastrada "
Ocor2 = "Empresa Contábil năo Cadastrada "
Ocor3 = "Item do Pagamento sem Centro de Custo "
Ocor4 = "PagtoXTit, Emp ou Controle năo encontrado"

Zap In MovAux

* Selecionando o Arquivo de Grupos de Conta Corrente Contabilizáveis
lcQuery = [Select Codigos, ContConts ] + ;
	[From SigCdGcr ] + ;
	[Where IntConts = 1]

If (Thisform.poDataMgr.SqlExecute(lcQuery, 'Grupos') < 1)
	=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Grupos)')
	Return 0
Endif
Select Grupos
Index On Codigos Tag Codigos

* Selecionando o Arquivo de Grupos de Conta Corrente
lcQuery = [Select Codigos, IntConts, ContConts ] + ;
	[From SigCdGcr]

If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crSigCdGcr') < 1)
	=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigCdGcr)')
	Return 0
Endif
Select crSigCdGcr
Index On Codigos Tag Codigos

* Selecionando o Arquivo de Empresas
lcQuery = [Select Cemps ] + ;
	[From SigCdEmp]

If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crSigCdEmp') < 1)
	=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigCdEmp)')
	Return 0
Endif
Select crSigCdEmp
Index On Cemps Tag Cemps

lcConexao = Thisform.poDataMgr.GetDriver()

* Selecionando o Arquivo de Conta Corrente
pDtI = fDtoSQL(IniPer)
pDtF = fDtoSQL(FinPer, '23:59:59')

If lcConexao = 'foxpro'

	lcTblMccr = (Addbs(Justpath(Thisform.poDataMgr.GetDatabase()))+'SigMvCcr.dbf')
	If !Used('SigMvCcr')
		Use (lcTblMccr) Order DataS In 0 Alias crSigMvCcr
		Select crSigMvCcr
		Set Near On
		Seek pDtI
		Set Near Off
		lnRegs = 0
		Wait Window 'Aguarde! Selecionando Registros...' Nowait
		Scan While crSigMvCcr.DataS <= pDtF
			lnRegs = lnRegs + 1
		Endscan
		Select crSigMvCcr
		Go Top
		Set Near On
		Seek pDtI
		Set Near Off
	Endif
Else

	lcQuery = [Select Emps, Dopes, Numes, Grupos, Contas, SGrupos, SContas, Datas, ] + ;
		[Hists, Hist2s, Valors, Opers, Autos, Nopers, Nfs, Titulos, Tipos, EmpDopNums,EspecieNfs, SGRUPOS,empdopncs ] + ;
		[From SigMvCcr ] + ;
		[Where Datas BetWeen ?pDtI And ?pDtF ] + ;
		[Order By Datas, Nopers]

	If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crSigMvCcr') < 1)
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigMvCcr)')
		Return 0
	Endif
	Select crSigMvCcr
	Index On DataS Tag DataS
	lnRegs = Reccount('crSigMvCcr')
	Go Top

Endif

DataProc = Ctod('  /  /  ')
Ncont = 0
lnOldNopers = 0

Select crSigMvCcr
oProg = Createobject('fwprogressbar', 'Processando Dados do Cursor...', lnRegs )
oProg.Show
Scan While DataS <= pDtF
	oProg.Update(.T.)

	Wait Window "Processando dia : " + Dtoc(DataS) + " - " + Str(Nopers) Nowait

	Scatter Memvar

	lcOrdMccr = Order('crSigMvCcr')
	lnRegMccr = Recno('crSigMvCcr')
	
	** era aqui
	** Ncont = Ncont + 1

	* Verifica se o Grupo da Conta é contabilizável
	* Se năo for, verifica se a conta é contabilizável

	lcQuery = [Select IClis, Razaos, RClis, Cpfs ] + ;
		[From SigCdCli ] + ;
		[Where IClis = '] + m.SContas + [' ] + ;
		[Order By CContabs]

	If (Thisform.poDataMgr.SqlExecute(lcQuery, 'LocalCli') < 1)
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalCli - 1)')
		Return 0
	Endif

	lcQuery = [Select IClis, RClis, Razaos, Cpfs, IntConts, CContabs,TpHists,Hists ] + ;
		[From SigCdCli ] + ;
		[Where IClis = '] + m.Contas + [' ] + ;
		[Order By IntConts, CContabs]

	If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crSigCdCli') < 1)
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigCdCli)')
		Return 0
	Endif

	Select Grupos
	Set Order To Codigos
	Seek M.Grupos
	If Eof() Or crSigMvCcr.Valors = 00
		If (crSigCdCli.IntConts <> 1)
			Loop
		Endif
	Endif

	*** Verifica se a contra partida permite contabilizaçăo
	*** É determinado pelo grupo
	*** Exemplo : Transferencia entre bancos / só é contabilizado o pagamento o titulo năo.

	Select crSigCdGcr
	Set Order To Codigos
	Seek crSigMvCcr.Sgrupos
	If (crSigCdGcr.IntConts = 3) &&  Se o grupo năo permite que a contra partida seja contabilizada
		Loop
	Endif

	*** Definiçăo do Código da Empresa Contábil
	Select crSigCdEmp
	Set Order To Cemps
	Seek crSigMvCcr.Emps

***************************** BRUNO ********************************
*!*		m.EmpCont = crSigCdEmp.

*!*		If Empty(M.EmpCont)
*!*			Select SemConta
*!*			Append Blank
*!*			Replace DataS  With crSigMvCcr.DataS, ;
*!*				Contas With crSigMvCcr.Contas, ;
*!*				Hists  With crSigMvCcr.Hists + crSigMvCcr.Hist2s, ;
*!*				Valors With crSigMvCcr.Valors, ;
*!*				Ocors  With crSigMvCcr.Emps + " - " + Ocor2
*!*		Endif
****************************** FIM *********************************

	Select crSigMvCcr

	*** Definiçăo da Conta Contábil

	Do Case
		Case Not Empty(Grupos.ContConts)
			m.Contabs = Grupos.ContConts

		Case Not Empty(crSigCdCli.CContabs)
			m.Contabs = crSigCdCli.CContabs

		Otherwise
			m.Contabs = M.Contas

			Select SemConta
			Append Blank
			Replace DataS  With crSigMvCcr.DataS, ;
				Contas With crSigMvCcr.Contas, ;
				Hists  With crSigMvCcr.Hists + crSigMvCcr.Hist2s, ;
				Valors With crSigMvCcr.Valors, ;
				Ocors  With Ocor1

			Select crSigMvCcr
	Endcase

	If (crSigMvCcr.Opers = "D")
		m.Debs  = crSigMvCcr.Valors
		m.Creds = 00
	Else
		m.Debs  = 00
		m.Creds = crSigMvCcr.Valors
	Endif

	ValLan1   = crSigMvCcr.Valors

	m.Anofis  = Str(Year(crSigMvCcr.DataS),4)
	m.Datas   = Strtran(Dtoc(crSigMvCcr.DataS),"/","")
	m.Contas  = M.Contabs
	m.Debs    = Trans(M.Debs  * 100,"@L 999999999999")
	m.Creds   = Trans(M.Creds * 100,"@L 999999999999")
	m.Hists   = Substr(Hists+Hist2s,1,70)
	m.Hists   = Strtran(M.Hists,Chr(1),'')
	m.Hists   = Strtran(M.Hists,Chr(13),'')
	m.Hists   = Strtran(M.Hists,Chr(10),'')
	m.Emps    = crSigMvCcr.Emps
	m.Nfs     = crSigMvCcr.Nfs
	m.Cecus   = crSigMvCcr.Emps

	HistPrinc = Hist2s

	If Autos And "LOTE" $ Hists
		m.Docto   = Val(Right(Alltrim(Hists),6))

		Numlote = Val(Substr(Hists,At("LOTE",Hists)+5,6))
		If Numlote = 0   &&conforme solicitado
			Numlote = Val(Right(Alltrim(Hists),6))
		Endif
		Select LoteProc
		Seek Numlote

		If Not Eof()
			Loop
		Endif
		Select crSigMvCcr
	Else
		m.Docto   = M.Nfs
	Endif
	
	IF crSigMvCcr.nopers<>lnOldNopers 
	Ncont = Ncont + 1
	lnOldNopers =crSigMvCcr.nopers
	ENDIF 
	m.Transacaos = Trans(Ncont,'@L 999999')
	m.IClis  = LocalCli.IClis
	m.Razaos = Iif(Empty(LocalCli.Razaos), LocalCli.RClis, LocalCli.Razaos)
	m.Cpfs   = LocalCli.Cpfs

	* Lançamento 1

	Select MovAux
	Append Blank
	Gather Memvar

	If (Nopers = 1)
		Loop
	Endif

	Select crSigMvCcr

	* Lançamento 2

	RecAtual    = Recno('crSigMvCcr')
	OpeAtual    = crSigMvCcr.Opers
	Controle    = crSigMvCcr.Nopers
	ControleAtu = crSigMvCcr.Nopers
	EmpLanc     = crSigMvCcr.Emps
	Manual      = .F.

	* Verificaçăo de Lançamento com Tipo = "M" --> Manual pelo Pagamento

	Thisform.poDataMgr.CursorQuery('SigMvCcr','TmpMccr','Nopers',Controle, lcCpoMccr)
	Select TmpMccr
	Scan
		If TIPOS = 'M'
			Manual = .T.
			Exit
		Endif
	Endscan

	Do Case
		Case crSigMvCcr.Autos And "LOTE" $ Hists

			* Lançamentos por lote de cheques
			* Checa se este lote já foi processado

			Select LoteProc
			Seek Numlote

			If Not Eof()
				Loop
			Else
				Append Blank
				Replace Lotes With Numlote
			Endif

			pLot = Numlote

			lcQuery = [Select Emps, NumOs ] + ;
				[From SigCqChm ] + ;
				[Where NumLotes = ?pLot]

			If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crSigCqChm') < 1)
				=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigCqChm)')
				If Used('crSigMvCcr')
					Use In crSigMvCcr
				Endif
				Return 0
			Endif

			If lcConexao = 'foxpro'

				Select crSigCqChm
				Scan
					Controle = crSigCqChm.Emps + Alltrim(Str(crSigCqChm.Numos,7))
					Select crSigMvCcr
					Set Order To VOpers
					Go Top
					If !Seek(Controle,'crSigMvCcr','VOpers')
						Controle = EmpLanc + Alltrim(Str(crSigCqChm.Numos,7))
						=Seek(Controle,'crSigMvCcr','VOpers')
					Endif

					Select crSigMvCcr
					If !Eof('crSigMvCcr')
						Scan While crSigMvCcr.VOpers = Controle
							If (Opers  <> OpeAtual) And ;
									(Grupos <> crSigCdPam.GrupoPags) And ;
									(Grupos <> crSigCdPam.GrupoRecs)
								* Lança operaçăo contrária

								*** Definiçăo da Conta Contábil

								Select crSigCdGcr
								Set Order To Codigos
								Seek crSigMvCcr.Grupos

								lcQuery = [Select IClis, RClis, Razaos, Cpfs ] + ;
									[From SigCdCli ] + ;
									[Where IClis = '] + crSigMvCcr.SContas + [' ] + ;
									[Order By CContabs]

								If (Thisform.poDataMgr.SqlExecute(lcQuery, 'LocalCli') < 1)
									=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalCli - 2)')
									Return 0
								Endif

								lcQuery = [Select CContabs, IClis, RClis, Razaos, Cpfs,TpHists,Hists ] + ;

		ThisForm.Get_DataF.SetFocus
		Return 0
	Endif

	If ThisForm.Get_DataF.value < ThisForm.Get_DataI.value
		Messagebox('A Data Final Năo Pode Ser Menor Que a Inicial!!!', 0+48, '')
		ThisForm.Get_DataF.SetFocus
		Return 0
	Endif

	If Messagebox('Confirma o Processamento ?', 4+32+256, '') = 6
		ThisForm.Processamento	
	Else
		Return 0
	EndIf
Else
	ThisForm.Release
EndIf
* Fim **************************************************************************************

ENDPROC
