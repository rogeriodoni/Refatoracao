------------------------------------------------------------
PROCEDURE reverterxml
Parameters pcCursor, pcNivAnt, pcCurDest

Local lnOldSel, lcXML

lnOldSel = Select()
lcXML = []

If Used([CrAux])
	Use in CrAux
EndIf 

Select &pcCursor.
Select cidchaves, Cods, Descs, 0 as Marcas, Space(4) as Codigo From &pcCursor. Where 1=0 Into Cursor CrAux ReadWrite

Select &pcCursor.	
Go top

Scan 			
	lcXML = Nvl(&pcCursor..Nivel,[])
	Zap in CrAux
	If Not Empty(lcXML)
		= fXmlToCursor(lcXML, [CrAux], .f.)
		Select CrAux
		Scan For CrAux.Marcas <> 0
			Select &pcCurDest.
			Append Blank 
			Replace Cods With CrAux.Cods ,;
					Descs With &pcCursor..Descs ,;
					CidChaves With CrAux.CidChaves ,;
					Codigo With &pcCursor..Cods in &pcCurDest.
			Select CrAux	
		EndScan 
	EndIf 
EndScan 

Select &pcNivAnt.
Go Top 

If Used([CrAux])
	Use in CrAux
EndIf 

Select(lnOldSel)
ENDPROC

	Iif(Not Empty(lcCon),[Conta: ] 		+ Alltrim(lcCon) + [-] + Alltrim(lcDCt) + lcChr,[]) + ;

lcSub2=	[Grande Grupo: ]  + Iif(Empty(lcSGrupro), [Global], lcSGrupro + [-] + lcSDgrupro) + lcChr +;
	[Grupo Produto: ] + Iif(Empty(lcGruPro),  [Global], lcGruPro  + [-] + lcDgruPro) + lcChr +;
	[Subgrupo: ]      + Iif(Empty(lcCSubG),   [Global], lcCSubG   + [-] + lcDSubG) + lcChr +;

			Endif
		Endif
		Select CsImpressao
	Endscan
Endif

Select CsImpressao
Index on localizas Tag localizas
Index on Substr(localizas,3,2) Tag Nivel2s
Index on Substr(localizas,5,2) Tag Nivel3s
Index on Substr(localizas,7,4) Tag Nivel4s
Go Top

Do case
Case lnOrdem = 1
	Set Order To localizas

EndCase
Go Top In CsImpressao
ENDPROC
