ENDPROC

------------------------------------------------------------
* PROCEDURE Init
------------------------------------------------------------
PROCEDURE Init
LParameters lcAcesso

WITH Thisform
	.poDatamgr = CreateObject('fSqlConector',ThisForm.Name)

	If Not .poDatamgr.pnIdconn>0
		Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados.',(0+48),'Erro')
		Return(.f.)
	Endif

	With .poDatamgr
		.AddCursor('SigOpIns','CidChaves','crSigOpIns','Cods',.f.)
		.Requery('crSigOpIns')
	EndWith

	Create Cursor csSigOpIns (Cods c(10), Descrs c(76), Tipo c(10))
	INDEX ON CODS TAG CODS
	SET ORDER TO Cods

	SELECT crSigOpIns

	SCAN  
		SCATTER MEMVAR
		SELECT csSigOpIns
		APPEND BLANK
		Replace Cods WITH m.Cods
		Replace Descrs WITH m.Descrs
		Replace Tipo WITH IIF(m.Tipos=1,"CHEQUE",Iif(m.Tipos=2,"TITULOS","FOLLOW UP"))
		SELECT crSigOpIns
	ENDSCAN

	WITH ThisForm.pagina.Lista.grade
		.RecordSource = "csSigOpIns"
		.Column1.ControlSource = "csSigOpIns.Cods"
		.Column2.ControlSource = "csSigOpIns.Descrs"
		.Column3.ControlSource = "csSigOpIns.Tipo"
	EndWith

Endwith	

=DoDefault()

ENDPROC

------------------------------------------------------------
* PROCEDURE Pagina.Lista.Grupo_op.Click
------------------------------------------------------------
PROCEDURE Pagina.Lista.Grupo_op.Click
LParameters Opcao

SELECT crSigOpIns
LOCATE FOR Cods = csSigOpIns.Cods

If !DoDefault(Opcao)
	ThisForm.mAtivapagina1
	Return
Endif

Do Case
	Case (ThisForm.pcEscolha = 'INSERIR')
		ThisForm.Pagina.Dados.optTipo.Value = 1
		ThisForm.Pagina.Dados.getCods.SetFocus

	Case (ThisForm.pcEscolha = 'PROCURAR')
		ThisForm.Pagina.Dados.getCods.SetFocus

	Case Inlist(ThisForm.pcEscolha, 'EXCLUIR', 'CONSULTAR')
		ThisForm.Pagina.Dados.Grupo_Salva.Salva.SetFocus

	Case (ThisForm.pcEscolha = 'ALTERAR')
		ThisForm.Pagina.Dados.getDescrs.SetFocus
EndCase
* Fim ***********************************************************************************
ENDPROC

------------------------------------------------------------
* PROCEDURE Pagina.Dados.Grupo_Salva.Salva.Click
------------------------------------------------------------
PROCEDURE Pagina.Dados.Grupo_Salva.Salva.Click
If Inlist(ThisForm.pcEscolha, 'INSERIR', 'ALTERAR')

	If IsEmpty(crSigOpIns.Cods)
		=MessageBox('Codigo Inválido!!!', 48, '')
		ThisForm.Pagina.Dados.getCods.SetFocus
		Return 0
	Endif

	If IsEmpty(crSigOpIns.Descrs)
		=MessageBox('Descriçăo Inválida!!!', 48, '')
		ThisForm.Pagina.Dados.getDescrs.SetFocus
		Return 0
	Endif

	If IsEmpty(crSigOpIns.Mensas)
		=MessageBox('Mensagem Inválida!!! Atençăo Esta Mensagem Será Utilizada na Impressăo ' + Chr(13) + ;
					'da Solicitaçăo!!! Portanto é Necessário Que a Mesma Seja Preenchida!!!', 48, '')
		ThisForm.Pagina.Dados.getDescrs.SetFocus
		Return 0
	Endif

	If Inlist(ThisForm.pcEscolha, 'INSERIR')
		Replace CidChaves WITH fUniqueIds()
	Endif

	SCATTER MEMVAR
	SELECT csSigOpIns
	Locate For Cods = m.cods
	If Inlist(ThisForm.pcEscolha, 'INSERIR', 'ALTERAR')
		IF EOF()
			APPEND BLANK
		ENDIF
		Replace Cods WITH m.Cods
		Replace Descrs WITH m.Descrs
		Replace Tipo WITH IIF(m.Tipos=1,"CHEQUE",Iif(m.Tipos=2,"TITULOS","FOLLOW UP"))
		SELECT crSigOpIns
	Endif

	ThisForm.Pagina.Lista.grade.Refresh

	Local m.lnErro1

	m.lnErro1 = ThisForm.poDatamgr.pnIdConn

	With ThisForm
		If m.lnErro1>0 
			m.lnErro1 = Iif(.poDataMgr.UpDate('crSigOpIns'),1,-1)
		EndIf 
		If m.lnErro1>0 
			m.lnErro1 = .poDataMgr.Commit()
		EndIf 
		If Not m.lnErro1>0 
			.poDataMgr.RollBack()
		EndIf 
	EndWith 
	If Not m.lnErro1>0 
		= MessageBox('Erro na Gravaçăo dos Dados do Cadastro de Operaçőes.',48)
	EndIf 

Endif

If Inlist(ThisForm.pcEscolha, 'EXCLUIR')

	SELECT csSigOpIns
	DELETE
	ThisForm.Pagina.Lista.grade.Refresh

Endif

=DoDefault()

ThisForm.mAtivapagina1
* Fim ***********************************************************************************
ENDPROC
