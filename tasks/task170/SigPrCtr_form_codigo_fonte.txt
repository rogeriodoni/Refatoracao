------------------------------------------------------------
* PROCEDURE carregaarquivos
------------------------------------------------------------
PROCEDURE carregaarquivos
parameters pTipo

If Empty(ThisForm.Pagina.Dados.Pageframe1.Page1.Get_Conta.Value)
	MessageBox([Favor Informar uma Conta.],0+16,[Aviso])
	Return .F.
EndIf 

Local lcDir, lcFor, lcCgc, lcChv, lcStr, lcArq, lcCCF
Local lnQtdXml, lnReg
Local Array laArqs[1]

Store 0 to lnQtdXml, lnReg
Store [] to lcDir, lcFor, lcCgc, lcChv, lcStr, lcArq, lcCCF

lcAqv = ThisForm.Pagina.Dados.Pageframe1.Page1.Get_Arquivo.Value
lcCgc = Alltrim(Strtran(Strtran(Strtran(ThisForm.Pagina.Dados.Pageframe1.Page1.Get_Cpf.Value,[.],[]),[/],[]),[-],[]))

If Alltrim(Upper(Right(Justfname(lcAqv),3))) <> 'XML'
	MessageBox([Arquivo esta em formato diferente de XML.],0+48,[Aviso])
	Return .F.
EndIf 

If Not Empty(lcAqv) And Not Empty(lcCgc)
	lcArq = Alltrim(lcAqv)
	lcStr = Alltrim(Upper(FileToStr(lcArq)))
	
	If !empty(lcStr)
		lcChv = Alltrim(StrExtract(lcStr, [<CHNFE>], [</CHNFE>]))
		lcCCF = Substr(lcChv,7,14)
		llOK = .T.
		If lcCCF <> lcCgc
			llOK = .F.
			lcMsg = [Fornecedor com CPF/CNPJ Diferente do XML, ] + Chr(13) + ;
					[Arquivo XML: ] + lcCCF + Chr(13) + ;
					[Fornecedor: ] + lcCgc + Chr(13) + ;
					[Deseja Continuar?]
			If MessageBox(lcMsg,4+32+256,[Atençăo]) = 6
				llOK = .T.
			EndIf 
		EndIf 
		If llOK
			lnReg = lnReg + 1
			If pTipo 
				ThisForm.Lerxml(lcArq)
			EndIf 		
		EndIf 				
	EndIf 
EndIf 
ENDPROC
