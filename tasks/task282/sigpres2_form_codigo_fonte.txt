PROCEDURE Init
Lparameters _Chave, pDataSes, poForm

Set DataSession to pDataSes

With ThisForm
	.Caption    = _Chave
	.cDopes     = _Chave
	.ParentForm = poForm

	.poDataMgr = CreateObject('fSqlConector', .Name)

	If (.poDataMgr.pnIdconn > 0)
		Goto Top In csTemporario

		.poDataMgr.CursorQuery('SigCdPam', 'crSigCdPam', .f., .f., 'MascNums')

		pEdn = csTemporario.Emps + csTemporario.Dopes + Str(csTemporario.Numes, 6)

		.poDataMgr.CursorQuery('SigMvPec', 'crSigMvPec',  'EmpDopNums', pEdn)
		.poDataMgr.CursorQuery('SigMvPec', 'TmpOperacao', 'EmpDopNums', pEdn)

		Select TmpOperacao
		Index On Codigos Tag Codigos

		=DoDefault()

		.AcertaBotoes
		
		lcQuery = [select isnull(b.codtams,'') as codtams, isnull(b.codcors,'') as codcors, a.* ] + ;
			[from sigmvitn a ] + ;
			[left join sigmvits b on a.empdopnums = b.empdopnums ] + ; 
			[and a.cpros = b.cpros ] + ; 
			[and a.citens = b.citens ] + ; 
			[Where a.empdopnums = ']+pEdn+[']
		
		.poDataMgr.SqlExecute(lcQuery,[crSigMvItn])
		.poDataMgr.SqlExecute(lcQuery,[xEestI])
		
		*.poDataMgr.CursorQuery('SigMvItn', 'crSigMvItn', 'EmpDopNums', pEdn)
		*.poDataMgr.CursorQuery('SigMvItn', 'xEestI', 'EmpDopNums', pEdn)

		Select csTemporario
		
		.MontaGrades
		
		.Pagina.Lista.Grade.SetAll('DynamicForeColor','Iif(CsTemporario.Nops>0,Iif(CsTemporario.ChkSubn,RGB(0,0,255),RGB(255,0,0)),Iif(CsTemporario.ChkSubn,RGB(0,0,255),Iif(CsTemporario.ChkBxParcs,RGB(255,0,255),RGB(0,0,0))))','Column')
		.Pagina.Lista.Grade.Column1.SetFocus
	Else
		=Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith
* Fim **************************************************************************************
ENDPROC

PROCEDURE Pagina.Lista.Grupo_op.Click
Parameters Opcao

If (Type('Opcao') = 'N')
	This.Value = Opcao
EndIf

ThisForm.LockScreen = .t.

If Inlist(This.Value, 1, 3, 4, 5)
	ThisForm.LockScreen = .f.
	ThisForm.mAtivaPagina1
	Return 0
EndIf

Select csTemporario
pEdn = csTemporario.Emps + csTemporario.Dopes + Str(csTemporario.Numes, 6)

NoDefault

With ThisForm
	loGBotaosalva = .Pagina.Dados.Grupo_Salva
	Do Case
		Case (This.Value = 2)
			.pcEscolha = 'CONSULTAR'
*!*				loGBotaosalva.Salva.Caption = '\<Ok'

		Case (This.Value = 5)
			.pcEscolha = 'PROCURAR'
			loGBotaosalva.Salva.Caption = '\<Procurar'
			Select csTemporario
			If Not Eof('csTemporario')
				Append Blank In csTemporario

	EndCase
	.mObjEnabled(.Pagina.Dados, .t.)
	If Not InList(This.Value, 0, 6)
		.KeyPreview = .f.
*!*			loGBotaosalva.Cancelar.Visible = (Not .pcEscolha = 'CONSULTAR')
			loGBotaosalva.Salva.Enabled = (Not .pcEscolha = [CONSULTAR])

		.Pagina.ActivePage = 2
	Else
		.KeyPreview = .t.
		ThisForm.LockScreen = .f.
		ThisForm.mAtivapagina1
		Return 0
	EndIf
EndWith

Zap In TmpOperacao

ThisForm.poDataMgr.CursorQuery('SigMvPec', 'crSigMvPec', 'EmpDopNums', pEdn)
Select crSigMvPec
Index On Codigos Tag Codigos
Scan
	Scatter MemVar Memo
	Insert Into TmpOperacao From MemVar
EndScan
Go Top In TmpOperacao

Zap In xEestI

lcQuery = [select isnull(b.codtams,'') as codtams, isnull(b.codcors,'') as codcors, b.qtds as QtdsS, a.* ] + ;
			[from sigmvitn a ] + ;
			[left join sigmvits b on a.empdopnums = b.empdopnums ] + ; 
			[and a.cpros = b.cpros ] + ; 
			[and a.citens = b.citens ] + ; 
			[Where a.empdopnums = ']+pEdn+[']
		
ThisForm.poDataMgr.SqlExecute(lcQuery,[crSigMvItn])
*ThisForm.poDataMgr.CursorQuery('SigMvItn', 'crSigMvItn', 'EmpDopNums', pEdn)

*** Mena - 27/11/2017 - agrupa quantidades e pesos por cpros+codcors+codtams
SELECT empdopnums, cpros, codcors, codtams, SUM(qtds) as qtds, SUM(qtdss) as qtdss, Max(qtbaixas) as qtbaixas, Max(qtprods) as qtprods, ;
Max(qtbxprods) as qtbxprods, 0000 as citens, SUM(tpesos) as tpesos, SUM(descvals) as descvals ;
from crSigMvItn group by empdopnums, cpros, codcors, codtams into cursor crSigMvItn

Select crSigMvItn
Index On EmpDopNums Tag EmpDopNums
Scan
	Scatter Memvar Memo
	
	If Not Empty(m.codtams) Or Not Empty(m.codcors)
		m.qtds = m.QtdsS
	EndIf 
	
	ThisForm.poDataMgr.CursorQuery('SigCdPro', 'crSigCdPro', 'CPros', crSigMvItn.CPros)
	m.Tpesos   = crSigCdPro.PesoMs * crSigMvItn.Qtds
	m.DescVals = (crSigMvItn.QtBaixas / Iif(crSigMvItn.Qtds <> 0, crSigMvItn.Qtds, 1) * 100)
	If (Type("gcTpInstalas")="C" And m.gcTpInstalas="V")
		m.DescVals = (crSigMvItn.QtBxProds / Iif(crSigMvItn.QtProds <> 0, crSigMvItn.QtProds, 1) * 100)
	Endif
	Insert Into xEestI From Memvar
EndScan

Select xEestI
Sum Qtds, QtBaixas, Tpesos, QtProds, QtBxProds To Tqtd, Tqtdb, Tpeso, TQtPrd, tQtBxprd
Append Blank
Replace Cpros    With 'TOTAL ->', ;
		Qtds     With Tqtd, ;
		Qtbaixas With Tqtdb, ;
		Tpesos   With Tpeso, ;
		QtProds  With tQtPrd,;
		QtBxProds with tqtbxprd,;
		DescVals With (QtBaixas / Iif(Qtds <> 0, Qtds, 1) * 100)
If (Type("gcTpInstalas")="C" And m.gcTpInstalas="V")
	Replace DescVals With (QtBxProds / Iif(QtProds <> 0, QtProds, 1) * 100)
Endif
Go Top

With Thisform.Pagina.Dados
	.fwgrade1.Refresh
	.GradeOperaçăo.Refresh
EndWith

ThisForm.poDataMgr.CursorQuery('SigCdCli', 'crSigCdCli', 'IClis', csTemporario.ContaOs)
If Not Eof('crSigCdCli')
	ThisForm.Pagina.Dados.Origem.Get_dconta.Value = crSigCdCli.RClis

If Not Eof('crSigCdCli')
	ThisForm.Pagina.Dados.Origem.Get_dcontad.Value = crSigCdCli.RClis
EndIf

ThisForm.poDataMgr.CursorQuery('SigCdCli', 'crSigCdCli', 'IClis', csTemporario.Vends)
If Not Eof('crSigCdCli')
	ThisForm.Pagina.Dados.Origem.Get_Dresps.Value = crSigCdCli.RClis
EndIf

ThisForm.poDataMgr.CursorQuery('SigOpCdc', 'crSigOpCdc', 'Dopes', csTemporario.Dopes)
ThisForm.poDataMgr.CursorQuery('SigCdOpe', 'crSigCdOpe', 'Dopes', csTemporario.Dopes)

With Thisform.Pagina.Dados
	.fwgrade1.Visible = .t.
	.Get_obs.Visible   = .t.
	.say_obs.Visible   = .t.
	.Get_Descr.Visible = .t.
	.say2.Visible 	   = .t.
EndWith	

ThisForm.Pagina.Dados.CmdEntrega.Visible = InList(crSigCdOpe.DtEntrs, 1, 4, 6)
