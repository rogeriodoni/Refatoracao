PROCEDURE Init
With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)

	If (.poDataMgr.pnIdconn > 0)
		=DoDefault()

		lcQuery = [Select TpInstalas ] + ;
				    [From SigCdPam]

		If (.poDataMgr.SqlExecute(lcQuery, 'crSigCdPam') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crSigCdPam)')
			Return 0
		EndIf
		Select CrSigCdPam
		Go Top
		
		*-[ Carrega a lista do SigCdTom ]-*
		lcSQLCmd = [Select Codigos, Descrs From SigCdTom]
		If (ThisForm.poDataMgr.SQLExecute(lcSQLCmd, 'crSigCdTom') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo')
			Return 0
		EndIf

		Select crSigCdTom
		Scan
			Insert Into Cs_SigCdTom (Marca,Codigos,Descri) ;
			Values (0,crSigCdTom.codigos,Str(crSigCdTom.codigos,2) + '-' + crSigCdTom.descrs)
		EndScan

		Select cs_SigCdTom
		go top
		
	Else
		=Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith

With ThisForm
	.Get_Dt_Inicial.Value = Date()
	.Get_Dt_Final.Value   = Date()
	.Refresh
EndWith

* Fim **************************************************************************************
ENDPROC

------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Private pDtI, pDtF
Local lcOperacao, ldDatai, ldDataf, lcEmpresa, lcGrupop, lcGrupo, lcConta, lcPeriodo
Local lcDescConta, lcTitulo, lcChave, lnContador, lcGrupoRel, lcContaRel

lcOperacao 	= Padr(Alltrim(ThisForm.get_operacao.Value), 20)
lnNume		= thisform.getnumes.Value
ldDatai 	= ThisForm.get_dt_inicial.Value
ldDataf 	= ThisForm.get_dt_final.Value
lcEmpresa 	= ThisForm.get_empresa.Value
lcGrupop 	= Padr(Alltrim(ThisForm.get_grupo.Value), 3)
lcGrdGru	= thisform.get_cd_ggrupo.Value
lcSubGru	= thisform.get_cd_sgrupo.Value
lcGrupo 	= Padr(Alltrim(ThisForm.get_cd_grEstoque.Value), 10)
lcConta 	= Padr(Alltrim(ThisForm.get_cd_Estoque.Value), 10)
lnOpcao     = ThisForm.OptSituas.Value
llChkOp		= (thisform.chkImpOp.Value = 1)
lnNopi      = ThisForm.getopi.Value
lnNopf      = Iif(Empty(ThisForm.getopf.Value), 999999, ThisForm.getopf.Value)
llServico	= (thisform.chkServico.Value = 1)
llPen 		= (thisform.chkPendente.Value = 1)
lnObs 		= ThisForm.GetCodObs.Value
llOrd		= (ThisForm.chkOrdem.Value = 1)

llTipoSel = .t.
lcFilTipo = []
Select cs_SigCdTom
Locate For Marca = 1
If Eof()
	llTipoSel = .f.
EndIf

ThisForm.poDataMgr.CursorQuery('SigCdCli', 'SigCdCli', 'IClis', lcConta)

ThisForm.poDataMgr.CursorQuery('SigCdOpe', 'crSigCdOpe', 'Dopes', lcOperacao)

ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'CrSigCdEmp', 'Cemps', lcEmpresa)
lcEmpresaDesc = Iif(Eof('CrSigCdEmp'), '', CrSigCdEmp.Razas)

ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'CrSigCdEmp', 'Cemps', _Empr)
lcNomeEmpresa = Iif(Eof('CrSigCdEmp'), '', CrSigCdEmp.Razas)

lcTitulo    = [Listagem de Composiçăo por Movimentaçăo ( ] + Alltrim(lcOperacao) + ;
				 Iif(!Empty(thisform.getNumes.Value),[:] + Str(thisform.getNumes.Value,6),[]) + [ )]
lcPeriodo   = 'Período: ' + Dtoc(ldDatai) + ' ŕ ' + Dtoc(ldDataf) + ' - ' + ;
			  Iif(Empty(lcEmpresa), 'Geral', 'Empresa: ' + lcEmpresa + ' - ' + Alltrim(lcEmpresaDesc))
  			  
lcSubTit	= Iif(lnNopi <> 0 ,[OP: ] + Str(lnNopi,6) + [ ŕ ] + Str(lnNopf,6),[])

Create Cursor dbCabecalho (NomeEmpresa c(80), Titulo c(80), Periodo c(150), SubTitulo c(80), lCompo L, Total m(4))
Insert Into dbCabecalho (NomeEmpresa, Titulo, Periodo, SubTitulo) ;
				 Values (lcNomeEmpresa, lcTitulo, lcPeriodo, lcSubTit)

Create Cursor TmpRelat (Emps c(3), Razas c(40), CGrus c(3), DGrus c(20), Grupo c(10), Conta c(10), DConta c(40), ;
						Tps c(1), CPros c(14), DPros c(40), cUnis c(3), Qtds n(12,3), ;
						QtPds n(10), Pesoms n(8,4), Pesos n(12,3), Datas d(8), NDias n(8), Imps n(1), ;
						reffs c(40), Mats c(14), DMats c(40), unicompos c(3), cunips c(3), Nops c(10), ObsOfs c(50), RefMats c(20))
Index on Grupo + Conta + Emps + CPros tag EmpCPro
Index on Grupo + Conta + Emps + Tps + CGrus + CPros tag EmpCGruPro
Index on Grupo + Conta + Emps + Tps + CGrus + CPros tag EGP
Index on Mats + Reffs + Cpros tag MatRef
Index on Cpros + Nops + Mats + Reffs tag CproMatRef

lcQuery = [Select Cemps, Razas ] + ;
		    [From SigCdEmp ] + Iif(Empty(lcEmpresa), [], [ Where Cemps = '] + lcEmpresa + ['])

If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'CrSigCdEmp') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (CrSigCdEmp)')
	Return 0
EndIf

pDtI = fDtoSQL(ldDataI)
pDtF = fDtoSQL(ldDataF, '23:59:59')

Wait Window 'Aguarde! Selecionando Movimentaçőes...' NoWait
lcWhere = []
If llChkOp
	lcWhere = [ a.Nops Between ?lnNopi And ?lnNopf And b.produc = 1 ] +;
				Iif(Empty(lcEmpresa),[],[And a.Emps = ']+lcEmpresa+[' ])
Else
	lcWhere = Iif(!Empty(lcOperacao),[a.Dopes = ']+lcOperacao+[' ],[])
	lcWhere = lcWhere + Iif(Empty(lcOperacao) And Not llTipoSel,[b.produc = 1 ],[]) && Se năo escolheu nenhuma operaçăo para filtrar, ai pega todas as operaçőes que geram produçăo
	lcWhere = lcWhere + Iif(lnNume <> 0,Iif(Empty(lcWhere),[],[ And ]) + [a.Numes = ] + Str(lnNume,6) + [ ],[])
	If Thisform.optData.Value = 1
		lcWhere = lcWhere +	Iif(lnNume <> 0,[],Iif(Empty(lcWhere),[],[ And ]) + [a.Datas BetWeen ?pDtI And ?pDtF ])
	Else
		lcWhere = lcWhere +	Iif(lnNume <> 0,[],Iif(Empty(lcWhere),[],[ And ]) + [a.PrazoEnts BetWeen ?pDtI And ?pDtF ])
	Endif
	lcWhere = lcWhere + Iif(Empty(lcEmpresa),[],Iif(Empty(lcWhere),[],[ And ]) + [a.Emps = ']+lcEmpresa+[' ])
	lcWhere = lcWhere + Iif(lnOpcao = 1,Iif(Empty(lcWhere),[],[ And ]) + [a.Nops = 0 ],Iif(lnOpcao = 2,Iif(Empty(lcWhere),[],[ And ]) + [a.Nops <> 0 ],[]))				
EndIf

Select cs_SigCdTom
Go Top
Scan 
	If Empty(lcFilTipo)
		lcFilTipo = Alltrim(Str(cs_SigCdTom.Codigos))
	Else
		lcFilTipo = lcFilTipo + [,] + Alltrim(Str(cs_SigCdTom.Codigos))
	EndIf 
EndScan 	
	
If not Empty(lcFilTipo)
	lcWhere = lcWhere + Iif(Empty(lcWhere),[],[ And ]) + [ b.TipoOps In (]+lcFilTipo+[) ]
EndIf 

lStrQuery = [Select a.Emps, a.Dopes, a.Numes, a.Datas, a.GrupoOs, a.ContaOs, a.GrupoDs, a.ContaDs, a.EmpDopNums, a.Nops, b.Globalizas, b.Servicos ]+;
			[From SigMvCab a Inner Join SigCdOpe b On a.Dopes = b.Dopes Where ] +;
			lcWhere + ;
			Iif(lnObs = 0,[],[ And a.CodObs = ]+ Str(lnObs) + [  ]) + ;			
			[Order By a.Emps, a.Dopes, a.Numes, a.Datas, a.GrupoOs, a.ContaOs, a.GrupoDs, a.ContaDs, a.EmpDopNums ]

If (ThisForm.poDataMgr.SqlExecute(lStrQuery, 'CrSigMvCab') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (CrSigMvCab)')
	Return 0
EndIf

Select CrSigMvCab
Index On EmpDopNums Tag EmpDopNums
Go Top

Wait Clear

Select CrSigCdEmp
oProg = CreateObject('fwprogressbar', 'Processando Empresas...', Reccount('CrSigCdEmp'))
oProg.Top = oProg.Top - (oProg.Height / 2)
oProg.Show
Scan
	oProg.Update(.T.,'Empresa: ' + CrSigCdEmp.CEmps)

	lcEmpPro = CrSigCdEmp.Cemps
	
	Select CrSigMvCab
	Count For Emps = lcEmpPro To lnRegs
	Go Top

	If Seek(lcEmpPro,'CrSigMvCab','EmpDopNums')
		
		oProg2 = CreateObject('fwprogressbar','Verificando Movimentaçőes...',lnRegs)
		oProg2.Top = oProg2.Top + (oProg2.Height / 2)
		oProg2.Show
	
		Scan While CrSigMvCab.Emps = lcEmpPro

			oProg2.Update(.T.)

			Store '' to lcGrupoRel, lcContaRel
			
			*** Tiago - 06/02/2012 - Vianna - Se for OP de servico e estiver marcado para năo filtrar
			If !llServico And Empty(lcOperacao) And Inlist(crSigMvCab.Servicos,1,2)
				Loop
			Endif
			***
			
			If !Empty(lcGrupo) or !Empty(lcConta)
				If crSigMvCab.Globalizas = 1 or crSigMvCab.Servicos = 1
					If Not Empty(lcGrupo) And (lcGrupo <> CrSigMvCab.GrupoOs)
						Loop
					Endif
					If Not Empty(lcConta) And (lcConta <> CrSigMvCab.ContaOs)
						Loop
					Endif
					lcGrupoRel = CrSigMvCab.GrupoOs
					lcContaRel = CrSigMvCab.ContaOs
				EndIf

				If crSigMvCab.Globalizas = 2 or crSigMvCab.Servicos = 2
					If Not Empty(lcGrupo) And (lcGrupo <> CrSigMvCab.GrupoDs)					
						Loop
					EndIf
					If Not Empty(lcConta) And (lcConta <> CrSigMvCab.ContaDs)
						Loop
					Endif
					lcGrupoRel = CrSigMvCab.GrupoDs
					lcContaRel = CrSigMvCab.ContaDs
				EndIf
			Else
				lcGrupoRel = Space(10) 
				lcContaRel = Space(10) 
			EndIf		

			lcChave = CrSigMvCab.EmpDopNums			

			lStrQuery = [Select a.mats, a.CPros, a.Pesos, a.Qtds, a.QtBaixas,b.Dpros as Dmats,b.CGrus,b.CUnis,b.PesoMs,b.CustoFs, b.Reffs as RefMats, ] +;
						[c.DGrus,d.FatDivs,b.Reffs, e.Dpros, a.cunis as unicompos, a.cunips, ] +;
						[Case When g.Qtds is null Then f.Qtds Else g.Qtds End as QtdPed, d.Fators, f.Citens, Space(50) as ObsOFs, a.EmpDopNums, ] +;
						[Case When g.CodCors is null Then space(4) Else g.CodCors End as CodCors, ] +;
						[Case When g.CodTams is null Then space(4) Else g.CodTams End as CodTams ] +;
						[From SigSubMv a Inner Join SigCdPro e On e.Cpros = a.Cpros ] +;
						[Inner Join SigMvItn f On f.empdopnums = a.empdopnums and f.citens = a.citem2 ] +;
						[Inner Join SigCdPro b On b.CPros = a.mats ] +;
						[Left Join SigCdGrp c On c.CGrus = b.CGrus ] +;
						[Left Join SigCdUni d On d.CUnis = b.CUnis ] +;
						[Left Join SigMvIts g On g.EmpDopNums = f.EmpDopNums And g.Citens = f.Citens ] +;
						[Where a.empdopnums = ']+lcChave+[' ] +;
						Iif(Empty(lcGrdGru),  '',[And b.Mercs = ']+lcGrdGru+[' ])+;							
						Iif(Empty(lcGrupoP),  '',[And b.CGrus = ']+lcGrupoP+[' ])+;
						Iif(Empty(lcSubGru),  '',[And b.SGrus = ']+lcSubGru+[' ])+;
						[Union All ] +;
						[Select a.mats, a.CPros, ] +;
						[Case When h.Pesos is null Then a.Pesos Else h.Pesos End as Pesos, ] +;						
						[Case When h.Qtds is null Then a.Qtds Else h.Qtds End as Qtds, ] +;
						[0 as QtBaixas, b.Dpros as DMats,b.CGrus,b.CUnis,b.PesoMs,b.CustoFs, b.Reffs as RefMats, ] +;
						[c.DGrus,d.FatDivs, f.reffs, e.Dpros, a.unicompos, a.cunips, ] +;
						[Case When g.Qtds is null Then e.Qtds Else g.Qtds End as QtdPed, ] +;
						[d.Fators, e.Citens, Cast(a.ObsOFs as char(50)) as ObsOFs, e.EmpDopNums, ] +;
						[Case When g.CodCors is null Then space(4) Else g.CodCors End as CodCors, ] +;
						[Case When g.CodTams is null Then space(4) Else g.CodTams End as CodTams ] +;						
						[From SigMvItn e Left Join SigMvIts g On e.EmpDopNums = g.EmpDopNums And e.Citens = g.Citens ] +;
						[Inner Join SigPrCpo a On e.cpros = a.cpros ] +;
						[Left Join SigSubCp h On a.Cpros = h.Cpros and a.Mats = h.Mats And h.CodTams = g.CodTams ] +;
						[Inner Join SigCdPro f On e.Cpros = f.Cpros ] +;
						[Inner Join SigCdPro b On b.CPros = a.mats ] +;
						[Left Join SigCdGrp c On c.CGrus = b.CGrus ] +;
						[Left Join SigCdUni d On d.CUnis = b.CUnis ] +;
						[Where e.empdopnums = ']+lcChave+[' ] +;
						Iif(Empty(lcGrdGru),  '',[And b.Mercs = ']+lcGrdGru+[' ])+;							
						Iif(Empty(lcGrupoP),  '',[And b.CGrus = ']+lcGrupoP+[' ])+;
						Iif(Empty(lcSubGru),  '',[And b.SGrus = ']+lcSubGru+[' ])+;		
						Iif(llOrd,[ And a.OrdTs = 0 ],[]) + ;			
						[and e.citens not in (Select e.citens ] +;
						[From SigMvItn e, SigSubMv a ] +;
						[Where e.empdopnums = ']+lcChave+[' and e.empdopnums = a.empdopnums and e.citens = a.citem2 ] +;
						[group by e.citens) ]

PROCEDURE Valid
Local loLista

If Not Empty(This.Value)
	loLista = CreateObject('fwbuscaext', ThisForm.poDataMgr.pnIdConn, 'SigCdObs', ;
						   'crListaRemota', 'Codigos', This.Value, 'Seleçăo', .t.)

	If Not loLista.plAchouRegistro
		loLista.mAddColuna('Codigos', 'XXX', 'Código')
		loLista.mAddColuna('Descrs', '', 'Descriçăo')
		loLista.Show()
	EndIf
	
	Thisform.GetCodObs.Value = crListaRemota.Codigos
	Thisform.GetDesObs.Value = crListaRemota.Descrs
	Use In crListaRemota
Else
	Thisform.GetCodObs.Value = 0
	Thisform.GetDesObs.Value = ''
EndIf

ENDPROC
