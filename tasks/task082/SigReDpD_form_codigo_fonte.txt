PROCEDURE Init
With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)

	If (.poDataMgr.pnIdconn > 0)
		lcQuery = [Select Codigos, Descrs ] + ;
				    [From SigCdGcr ] + ;
				   [Where GerBals = 1]

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'CrSigCdGcr') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexão (CrSigCdGcr)')
			Return 0
		EndIf
		Select CrSigCdGcr
		Index On Codigos Tag BalCodigo
		Index On Descrs  Tag BalDescrs

		lcQuery = [Select Cods, Descs ] + ;
				    [From SigOpOpt ] + ;
				   [Where Situas < 2 And (Emps = ' ' Or Emps = ']+_Empr+[') ]

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'CrSigOpOpt') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexão (CrSigOpOpt)')
			Return 0
		EndIf
		Select CrSigOpOpt
		Index On Cods  Tag Cods
		Index On Descs Tag Descs

		lcQuery = [Select IClis, RClis, Inativas ] + ;
				    [From SigCdCli ] + ;
				   [Where GerBals = 1]

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'CrSigCdCli') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexão (CrSigCdCli)')
			Return 0
		EndIf
		Select CrSigCdCli
		Index On IClis Tag BalCodigo
		Index On RClis Tag BalNome
		
		=Thisform.Podatamgr.Cursorquery('SigCdPam','CrSigCdPam',.f.,.f.,[GrTrans])

		*** Tiago - 17/04/2012 - Vianna - Checa os acessos que o usuário possui de tipo de produção
		lcSql = [Select 0 as Acesso, * From SigInTgo ]
		If (ThisForm.poDataMgr.SqlExecute(lcSql, 'TmpTpGop') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexão (TmpTpGop)')
			Return 0
		EndIf
		Select TmpTpGop
		Scan
			If fChecaAcesso('SIGPRGLOT',Alltrim(TmpTpGop.Codigos))
				Replace Acesso With 1 In TmpTpGop
			EndIf
		EndScan
		
		Select * From TmpTpGop Where Acesso = 1 Into Cursor CrTmpTpGop ReadWrite
		Select CrTmpTpGop
		Index on Codigos Tag Codigos
		Go Top
		
		If Reccount('CrTmpTpGop') = 1
			ThisForm.Get_TpGOp.Value = CrTmpTpGop.Codigos
		Endif
		***

		=DoDefault()
	Else
		Messagebox('Impossivel Efetuar Conexão com Servidor de Banco de Dados...',48,'Create Object')
		Return(.F.)
	EndIf
EndWith

ThisForm.CntOpe.GrdTipo.RecordSource = ''
*-[ Gera o Grid com as Operações de Produção ]-*
If ThisForm.poDataMgr.SqlExecute([Select 0 As Marca, DopPs, ndopps, CodTgOps From SigCdOpd ],'CsSigCdOpd') < 1
	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexão (CsSigCdOpd')
	Return(.F.)
Endif
*** Tiago - 17/04/2012 - Vianna - Checa operações de produção que o usuário tem acesso

	If ! fChecaAcesso('P'+padl(csSigCdOpd.ndopps,8,'0'))
		Delete
	Endif
Endscan
Select CsSigCdOpd
Go top

------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Private pDtI, pDtF
Local lcGrupo, lcConta, lcDConta, ldDatai, ldDataf, lcGruMat, lnTotal, lcCab, lcSub, lcEmp

lcGrupo   = ThisForm.Get_Grupo.Value
lcConta   = ThisForm.Get_Conta.Value
lcDConta  = ThisForm.Get_DConta.Value
lcGruMat  = ThisForm.Get_CGru.Value
lcGgrp    = ThisForm.Get_Ggrp.Value
ldDatai   = ThisForm.Get_Datai.Value
ldDataf   = ThisForm.Get_Dataf.Value
lcDopp    = Thisform.Get_Dopes.Value
lnNum     = Thisform.getNumes.Value
llImprime = Thisform.ChkTpoOp.Value
llDestino = ThisForm.ChkDestino.Value
lcTpOpt   = ThisForm.Get_tpOps.Value
llDope    = ThisForm.Chk_Dopes.Value
llDopeDt  = ThisForm.Chk_detalhe.Value
lnEnvelope= ThisForm.Get_Envelope.Value
lnOp      = ThisForm.Get_Nop.Value
lnOpI     = ( ThisForm.Get_Nop.Value * 10000 ) + 1
lnOpF     = ( ThisForm.Get_Nop.Value * 10000 ) + 9999
lcCodCli  = thisform.get_codcli.Value
lcTGOp	  = thisform.get_tpGOp.Value

Create Cursor TmpRelat (Fase c(10),IClis c(10),RClis c(50),CMats c(14),DPros c(40),CPros c(14),DDPros c(40),Qtde n(12,3),TpOps c(15),;
						CUnis c(3), Operacaos c(31), Nenvs n(10), Datas D, Pesos n(12,3), Valor n(12,2), Fators n(12,3), Peso2s n(12,3), CUniPs c(3))
Index On Dtos(datas) + Operacaos + Str(nenvs,10) + cmats Tag Operacao
Index on Fase + IClis + TpOps + CUnis + CUniPs + CMats + Operacaos + Str(nenvs,10) Tag FFuncMat

pDtI = fDtoSQL(ldDataI) 
pDtF = fDtoSQL(ldDataF, '23:59:59')

lStrQuery = [Select a.Datas,a.CIdChaves,a.Emps,a.DopPs,a.NumPs,a.GrupoOs,a.ContaOs,b.CMats,b.Qtds,b.TpOps,b.Nenvs,]+;
				   [a.GrupoDs,a.ContaDs, b.Nops, b.Pesos, b.peso2s, b.Fators, b.Servicos, b.Cats, b.Empdnps ]+;
			  [From SigCdNec a, SigCdNei b ] + ;
			 [Where a.Datas BetWeen ?pDtI And ?pDtF And ]+;
					Iif(Empty(lcGrupo), [], Iif(llDestino, [a.GrupoDs = '], [a.GrupoOs = ']) + lcGrupo + [' And ]) + ;
					Iif(Empty(lcConta), [], Iif(llDestino, [a.ContaDs = '], [a.ContaOs = ']) + lcConta + [' And ]) + ;
					Iif(Empty(lcDopp),  [], [a.Dopps = '] + lcDopp+[' And ]) + ;
					Iif(Empty(lnNum),   [], [a.NumPs = ] + Alltrim(Iif(Type([lnNum])=[C],lnNum,Str(lnNum))) + [ And ]) + ;
				   [a.EmpDNPs = b.EmpDNPs ] + ;
			[Order By a.cIdChaves ]

If ThisForm.poDataMgr.SqlExecute(lStrQuery,'TmpNens') < 1
	MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexão (TmpNens)')
	Return 0
EndIf

Select CsSigCdOpd
Go Bottom
Go Top

Select * From CsSigCdOpd Where Marca > 0 And Iif(!Empty(lcTGOp),CodTgOps = lcTGOp, .t. ) Into Cursor csLocalMc
Go Top In csLocalMc

If Empty(ThisForm.Get_Dopes.Value) And Not Eof([csLocalMc])
	Select a.* From TmpNens a, CsSigCdOpd b ;
	Where a.DopPs = b.DopPs And b.Marca > 0 ;
	Into Cursor CrSigCdNec ReadWrite
Else
	Select a.* From TmpNens a, CsSigCdOpd b ;
	Where a.DopPs = b.DopPs And Iif(!Empty(lcTGOp),b.CodTgOps = lcTGOp, .t. ) ;
	Into Cursor CrSigCdNec ReadWrite
EndIf

If (lnEnvelope > 0)
	Select Distinct Empdnps From CrSigCdNec Where Nenvs = lnEnvelope Into Cursor crEnvelope
EndIf

Select CrSigCdNec
Index On Dtos(Datas) + CIdChaves Tag Datas
Set Order To Datas
Go Top In CrSigCdNec		

lnTotal = Reccount('CrSigCdNec')
If (lnTotal <> 0)
	loBarra = CreateObject('fwprogressbar','Processando Movimentações',lnTotal)
	loBarra.Titulo.FontBold = .T.
	loBarra.Show
	Scan
		loBarra.Update(.T.)

		If Not Empty(lcTpOpt) And CrSigCdNec.TpOps <> lcTpOpt
			Loop
		EndIf

*		If (lnEnvelope > 0) And (CrSigCdNec.Nenvs <> lnEnvelope And lcEdn <> CrSigCdNec.Empdnps)
		If (lnEnvelope > 0)
			Select crEnvelope
			Go Top In crEnvelope
			Locate For Empdnps = CrSigCdNec.Empdnps
			If Eof([crEnvelope])
				Loop
			EndIf
		EndIf
		If lnOP > 0 And Not Between(CrSigCdNec.Nops,lnOpI,lnOpF)
			Loop
		EndIf
	
		*** Tiago - 25/10/11 - Vianna - Incluido filtro para trazer OPs de um determinado cliente, para analisar o que foi lançado de servicos para cada funcionário
		If !Empty(lcCodCli)
			If !Empty(crSigCdNec.Nops)
				lcQuery = [Select Case When c.globalizas = 1 Then d.Iclis Else e.Iclis End as Iclis, ] +;
							[Case When c.globalizas = 1 Then d.Rclis Else e.Rclis End as Rclis ]+;
							[From sigoppic a inner join SigMvCab b on a.EmpDopNums = b.EmpDopNums ] +;
							[inner join SigCdOpe c on b.Dopes = c.Dopes ] +;
							[inner join SigCdCli d on b.ContaOs = d.Iclis ] +;
							[inner join SigCdCli e on b.ContaDs = e.Iclis ] +;
							[where a.Nops = ?CrSigCdNec.Nops And (b.ContaOs = '] + lcCodCli + [' Or b.ContaDs = '] + lcCodCli + [' )]
			Else
				lcQuery = [Select Case When c.globalizas = 1 Then d.Iclis Else e.Iclis End as Iclis, ] +;
							[Case When c.globalizas = 1 Then d.Rclis Else e.Rclis End as Rclis ] +;
							[From SigPdMvf f Inner Join sigoppic a  On f.Nops = a.Nops inner join SigMvCab b on a.EmpDopNums = b.EmpDopNums ] +;
							[inner join SigCdOpe c on b.Dopes = c.Dopes ] +;
							[inner join SigCdCli d on b.ContaOs = d.Iclis ] +;
							[inner join SigCdCli e on b.ContaDs = e.Iclis ] +;
							[where f.EmpDnPs = '] + crSigCdNec.EmpDnPs + [' And (b.ContaOs = '] + lcCodCli + [' Or b.ContaDs = '] + lcCodCli + [' )]
			Endif
			If Thisform.PodataMgr.SqlExecute(lcQuery,'LocalCli') < 1
				=MessageBox('Favor Reinicializar o Processo!!!',16,'Falha na Conexão (LocalCli)')
				Return 0
			Endif
			If Eof('LocalCli')
				Loop
			EndIf
		EndIf
		
		If CrSigCdNec.Servicos  and !Empty(crSigCdNec.Nenvs) And Empty(CrSigCdNec.Cats) && Se for servico e o campo Cats estiver vazio, é pq é um serviço que está agregado há OP, ai só pode sari uma única vez nas movimentações
			Loop
		EndIf
		***

		lcQuery = [Select a.CPros,a.DPros,a.CGrus,a.CUnis,a.CUniPs,b.Mercs From SigCdPro a, SigCdGrp b Where a.CPros = ']+CrSigCdNec.CMats+[' And a.Cgrus = b.Cgrus ]
		If Thisform.PodataMgr.SqlExecute(lcQuery,'CrSigCdPro') < 1
			=MessageBox('Favor Reinicializar o Processo!!!',16,'Falha na Conexão (CrSigCdPro)')
			Return 0
		EndIf
		Select CrSigCdPro
		If (CrSigCdPro.CGrus <> lcGruMat) And !Empty(lcGruMat)
			Loop
		EndIf
		If (CrSigCdPro.Mercs <> lcGgrp ) And !Empty(lcGgrp)
			Loop
		EndIf
		
		lcTpOps = Iif(llImprime,CrSigCdNec.TpOps,Space(15))
		
		lcGrp = Iif(llDestino,CrSigCdNec.GrupoDs,CrSigCdNec.GrupoOs)
		lcCta = Iif(llDestino,CrSigCdNec.Contads,CrSigCdNec.ContaOs)
		lcPed = Padr(Iif(llDope,CrSigCdNec.Dopps+'-'+Str(CrSigCdNec.Numps,10),' '),31)
		lnEnv = Iif(llDope,CrSigCdNec.Nenvs,0)
		
		If lcGrp = CrSigCdPam.GrTrans
			Loop
		EndIf

		lcPro = []
		lcDpr = []
		If Not IsEmpty(CrSigCdNec.Nops) And ThisForm.chkProdutos.Value
			lcQuery = [Select top 1 a.Cpros, b.DPros ] + ;
					    [From SigOpPic a Inner join SigCdPro b on a.Cpros = b.Cpros Where Nops = ?crSigCdNec.Nops]

			If (ThisForm.poDataMgr.SqlExecute(lcQuery, [LocalPro]) < 1)
				=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexão (LocalPro)])
				Return 0
			EndIf
			Go Top In LocalPro
			If Not Eof([LocalPro])
				lcDpr = LocalPro.DPros
				lcPro = LocalPro.Cpros
			EndIf
		EndIf
		
		If llDopeDt && Impressão detalhada
			lcPed = Alltrim(CrSigCdNec.Dopps)+'-'+Alltrim(Str(CrSigCdNec.Numps,10))
			lnEnv = CrSigCdNec.Nops
		
			Select TmpRelat
			Set Order To Operacao
			If Not Seek(Dtos(CrSigCdNec.Datas) + lcPed + Str(lnEnv,10) + CrSigCdNec.cMats )
				Insert Into TmpRelat (Datas, Operacaos, Nenvs, CMats, Dpros, Cunis, CPros, DDPros, Fators ) ;
							  Values (CrSigCdNec.Datas, lcPed, lnEnv, CrSigCdNec.Cmats, CrSigCdPro.Dpros, CrSigCdPro.Cunis, lcPro, lcDpr, crSigCdNec.Fators )
			EndIf
			Replace Qtde  With Qtde  + CrSigCdNec.Qtds,;
					Pesos With Pesos + CrSigCdNec.Pesos
			Replace Valor With Valor + (CrSigCdNec.Fators * CrSigCdNec.Qtds)
		Else

			Select TmpRelat
			If Not Seek(lcGrp + lcCta + lcTpOps + CrSigCdPro.Cunis + CrSigCdPro.CuniPs + CrSigCdNec.CMats + lcPed + Str(lnEnv,10) )
				ThisForm.poDataMgr.CursorQuery('SigCdCli', 'CrTmpCli', 'IClis', lcCta, 'RClis')
				Insert Into TmpRelat (Fase, Iclis, RClis, CMats, Dpros, CPros, DDPros, TpOps, Cunis, Operacaos, Nenvs, Datas, Fators, CuniPs) ;
							  Values (lcGrp, lcCta, CrTmpCli.RClis, ;
							  		  CrSigCdNec.CMats, CrSigCdPro.Dpros, lcPro, lcDpr, lcTpOps, CrSigCdPro.Cunis,;
							  		  lcPed, lnEnv, CrSigCdNec.Datas, crSigCdNec.Fators,CrSigCdPro.CuniPs)
			EndIf
			Replace Qtde With Qtde + CrSigCdNec.Qtds,;
					Peso2s With peso2s + crSigCdNec.Peso2s
			Replace Valor With Valor + (CrSigCdNec.Fators * CrSigCdNec.Qtds)

			*** Total Geral
			If Not Seek(Replicate(Chr(255), 20) + lcTpOps + CrSigCdPro.Cunis + CrSigCdPro.CuniPs + CrSigCdNec.CMats)
				Insert Into TmpRelat (Fase, Iclis, RClis, CMats, TpOps, Cunis, CuniPs ) ;
							  Values (Replicate(Chr(255), 10), Replicate(Chr(255), 10), ;
							  		  'Total Geral : ', CrSigCdNec.CMats, lcTpOps, CrSigCdPro.Cunis, CrSigCdPro.CuniPs)
			EndIf
			Replace Qtde with Qtde + CrSigCdNec.Qtds,;
					Peso2s With peso2s + crSigCdNec.Peso2s
			Replace Valor With Valor + (CrSigCdNec.Fators * CrSigCdNec.Qtds)			
		EndIf
	EndScan
	loBarra.Complete(.T.)
EndIf

lcCab = [Análise de Produção]
lcSub = Iif(ThisForm.chkRelatorio.Value, [], Iif(llDestino,[Por Destino],[Por Origem]) +  [ - ]) + [Período : ] + Alltrim(Dtoc(ldDatai)) + [ até ] + Alltrim(Dtoc(ldDataf))
ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'CrSigCdEmp', 'Cemps', _Empr, 'Razas')
lcEmp = _Empr + ' - ' + Iif(Eof('CrSigCdEmp'), '', CrSigCdEmp.Razas)

If ThisForm.chkFuncs.Value And Not Empty(ThisForm.Get_Conta.Value)
	lStrQuery = [Select a.RClis, a.Cpfs ]+;
				  [From SigCdCli a ] + ;
				[Where a.IClis = '] + Padr(ThisForm.Get_Conta.Value, 10) + [']
				
	If ThisForm.poDataMgr.SqlExecute(lStrQuery,'LocalCli') < 1
		MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexão (LocalCli)')
		Return 0
	EndIf
	Select LocalCli
	Go Top In LocalCli
	If Not Eof([LocalCli])
		lcSub = lcSub + Chr(13) + Chr(13) + ;
				[NOME: ] + Alltrim(LocalCli.RClis) + [ / CPF: ] + Alltrim(LocalCli.Cpfs)
	EndIf
EndIf

Create Cursor dbCabecalho (Titulo c(100), SubTitulo c(100), NomeEmpresa c(100), ImpFuncs l)
Insert Into dbCabecalho (Titulo, SubTitulo, NomeEmpresa, ImpFuncs) Values (lcCab, lcSub, lcEmp, ThisForm.chkFuncs.Value)

Select TmpRelat
Go Top
ENDPROC
