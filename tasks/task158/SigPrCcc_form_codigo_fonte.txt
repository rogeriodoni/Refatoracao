------------------------------------------------------------
* PROCEDURE Click
------------------------------------------------------------
PROCEDURE Click
Local lnTotRegs, lnRegs, llOk

Private _Grupo, _Conta, _Moeda, _Data, _CPros, llConc

With Thisform
	.Conta.Enabled = .F.
	.Estoque.Enabled = .F.
	.btnCusto.Enabled = .F.
	.Processa.Enabled = .F.
	.Cancela.Enabled = .F.
	.LblEnd.Visible = .F.
EndWith

llSaida = .t.

DO While llSaida

	If ThisForm.Conta.Value 

		_Emps  = Padr(ThisForm.OpConta.Get_Empresa.Value,3)
		_Grupo = Padr(ThisForm.OpConta.txtGrupos.Value,10)
		_Conta = Padr(ThisForm.OpConta.txtContas.Value,10)
		_Moeda = Padr(ThisForm.OpConta.txtMoedas.Value,3)
		_Data  = fDtoSQL(ThisForm.OpConta.txtData.Value)
		_Gcm   = Iif(Not Empty(_Grupo) And Not Empty(_Conta) And Not Empty(_moeda),_Grupo + _Conta + _moeda,Space(23))
		llConc = .t.
		
		Wait Window 'Aguarde! Selecionando Registros...' NoWait

		ProgSel = CreateObject( 'fwprogressbar', 'Selecionando Registros...', 1 )
		ProgSel.Titulo.FontBold = .t.
		ProgSel.Show
		ProgSel.Update(.T.)

		Set Talk On
		Set Talk Window
		Set Escape Off
		
		lcWhere = [Where 0 = 0 ]+;
				  Iif(Not Empty(_Grupo),[And Grupos = ']+_Grupo+[' ],[ ])+;
				  Iif(Not Empty(_Conta),[And Contas = ']+_Conta+[' ],[ ])+;
				  Iif(Not Empty(_moeda),[And Moedas = ']+_moeda+[' ],[ ])+;
				  Iif(Not Empty(_Emps), [And Emps = ']+_Emps+[' ],[ ])+;
				  Iif(Not Empty(_Data), [And Datas >= ?_Data ],[ ])
		_pData = Iif(Empty(_Data),Ctod('01/01/1900'),_Data)

		lcWherC = [Where 0 = 0 ]+;
		Iif(Not Empty(_Grupo),[And Grupos = ']+_Grupo+[' ],[ ])+;
				  Iif(Not Empty(_Conta),[And Contas = ']+_Conta+[' ],[ ])+;
				  Iif(Not Empty(_moeda),[And Moedas = ']+_moeda+[' ],[ ])+;
				  Iif(Not Empty(_Emps), [And Emps = ']+_Emps+[' ],[ ])+;
				  Iif(Not Empty(_Data), [And DataConcs >= ?_Data ],[ ])+;
				  [And Concs = ?llConc ]

		If !Empty( _gcm )
		
			Create Cursor TmpConta (Emps c(3), Grupos c(10), Contas c(10), Moedas c(3), CidChaves c(20))
			Index On CidChaves Tag CidChaves
			
			If Empty(_Emps)
				
				lcQuery = [ Select Cemps From SigCdEmp ]
				If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'TmpEmps') < 1)
					=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpEmps)')
					llSaida = .f.
					Loop
				EndIf
				
				Select TmpEmps
				Go Top
				
				Scan 
					Insert into TmpConta (Emps, Grupos,Contas,Moedas,CidChaves) Values ;
							(TmpEmps.Cemps,_Grupo,_Conta,_moeda,Sys(2015)+Sys(2015))
					Select TmpEmps
				EndScan
					
			Else
				Insert into TmpConta (Emps,Grupos,Contas,Moedas,CidChaves) Values ;
				(_Emps,_Grupo,_Conta,_moeda,Sys(2015)+Sys(2015))
			EndIf 
		
		Else

			lcQuery = [ Select Distinct Emps, Grupos, Contas, Moedas, Space(20) as CidChaves From SigMvCcr ]+lcWhere+;
					  [ Union all]+;
					  [ Select Distinct Emps, Grupos, Contas, Moedas, Space(20) as CidChaves From SigMvCcr ]+lcWherc

			If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'TmpConta') < 1)
				=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpConta)')
				llSaida = .f.
				Loop
			EndIf
			Select TmpConta
			Replace all CidChaves with Sys(2015)+Sys(2015)
			Index On CidChaves Tag Cidchaves
			
			ProgSel.Complete( .T.,.t. )
			
		EndIf
		
		Do While .T.

			Select * From TmpConta Into Cursor Selecao Order by Emps,Grupos,Contas,Moedas
			
			Select Selecao

			If Reccount() = 0
				Exit
			EndIf

			lnReg = Reccount('Selecao')
			ProgConta = CreateObject( 'fwprogressbar', 'Recalculando Saldo de Conta Corrente', Recc() )
			ProgConta.Titulo.FontBold = .t.
			ProgConta.Show
			Scan
				lnReg = lnReg - 1
				ThisForm.Get_Registro.Value = lnReg
				ThisForm.Get_Registro.Refresh
				ProgConta.Update( .t. )
				ProgConta.SubTitulo.Caption = Allt( Grupos ) + " : " + Allt( Contas ) + "-" + Allt( Moedas )

				=fRecalculaS( Selecao.Grupos, Selecao.Contas, _pData, Selecao.Moedas )
				
				llOk = fRecalculaS( .t., ThisForm.Podatamgr, .t. )
				If Not llOk
					ThisForm.Podatamgr.Rollback()
					Loop
				EndIf

				llOk = ( ThisForm.PodataMgr.Commit() > 0 )
				If Not llOk
					ThisForm.Podatamgr.Rollback()
					Loop
				EndIf
				
				Select TmpConta
				If Seek(Selecao.CidChaves)
					Delete
				Endif
			EndScan
			
			ProgConta.Complete( .T.)
			Use In Selecao

		Enddo

	EndIf

	If ThisForm.Estoque.Value 

		_Estoque = Padr(ThisForm.OpEstoque.Get_Estoque.Value,10)
		_Empresa = Padr(ThisForm.OpEstoque.Get_Empresa.Value,3)
		_Grupo   = Padr(ThisForm.OpEstoque.txtGrupos.Value,10)
		_CPros   = Padr(ThisForm.OpEstoque.Get_Produto.Value,14)
		_Data    = fDtoSQL(ThisForm.OpEstoque.txtData.Value)

		Wait Window 'Aguarde! Selecionando Registros...' NoWait

		Do Case
			Case !Empty(_Empresa) And !Empty(_Grupo) And !Empty(_Estoque)
				lcWhere = [EmpGruEsts = "] + _Empresa + _Grupo + _Estoque + [" ]
			Case Empty(_Empresa) And Empty(_Grupo) And Empty(_Estoque)
				lcWhere = ''
			Otherwise
				lcEmpIni = Iif(Empty(_Empresa),Space(03),_Empresa)
				lcGruIni = Iif(Empty(_Grupo)  ,Space(10),_Grupo)
				lcEstIni = Iif(Empty(_Estoque),Space(10),_Estoque)
				lcEmpFin = Iif(Empty(_Empresa),Replicate(Chr(254),03),_Empresa)
				lcGruFin = Iif(Empty(_Grupo)  ,Replicate(Chr(254),10),_Grupo)
				lcEstFin = Iif(Empty(_Estoque),Replicate(Chr(254),10),_Estoque)
				lcWhere = [EmpGruEsts Between ']+lcEmpIni+lcGruIni+lcEstIni+[' And ']+lcEmpFin+lcGruFin+lcEstFin+[' ]
		EndCase

		lcWhere = Iif(!Empty(_CPros),Iif(!Empty(lcWhere),lcWhere+[ And ],[])+[ CPros = ?_CPros ],lcWhere)
		lcWhere = Iif(!Empty(_Data),Iif(!Empty(lcwhere),lcWhere+[ And ],[])+[ Datas >= ?_Data ],lcWhere)
		lcWhere = Iif(!Empty(lcWhere),[Where ]+lcWhere,[])
		_pData  = Iif(Empty(_Data),Ctod('01/01/1900'),_Data)

		lcQuery = [ Select Distinct Emps, Grupos, Estos, Cpros, CodCors, CodTams, Space(20) as CidChaves From SigMvHst ]+lcWhere

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'TmpEst') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpEst)')
			llSaida = .f.
			Loop
		EndIf
		
		Select TmpEst
		Replace all CidChaves With Sys(2015)+Sys(2015)
		Index On CidChaves Tag CidChaves
		
		DO While .t.
		
			Select * From TmpEst Into Cursor Selecao
			If Reccount('Selecao') = 0
				Exit
			Endif
			
			Select Selecao
			lnReg = Reccount('Selecao')

			ProgConta = CreateObject( 'fwprogressbar', 'Recalculando Saldo do Estoque', Recc('Selecao') )
			ProgConta.Titulo.FontBold = .t.
			ProgConta.Show

			Select Selecao
			Scan

				lnReg = lnReg - 1
				ThisForm.Get_Registro.Value = lnReg
				ThisForm.Get_Registro.Refresh

				ProgConta.Update( .t. )

				=fRecalculaP( Selecao.Emps, Selecao.Grupos, Selecao.Estos, Selecao.CPros, _pData, Selecao.CodCors, Selecao.CodTams, ThisForm.Podatamgr )

				llOk = fRecalculaP( .t., ThisForm.Podatamgr, .t. )
				If Not llOk
					ThisForm.Podatamgr.Rollback()
					Loop
				EndIf
				
				llOk = ( ThisForm.PodataMgr.Commit() > 0 )
				If Not llOk
					ThisForm.Podatamgr.Rollback()
					Loop
				EndIf
				
				Select TmpEst
				If Seek(Selecao.CidChaves)
					Delete
				EndIf

			Endscan
			ProgConta.Complete( .T., .T. )
			Use In Selecao

		EndDo
		
	EndIf

	If ThisForm.btnCusto.Value 
		lcEmp = Padr(ThisForm.OpCusto.Get_Empresa.Value,3)
		lcPro = Padr(ThisForm.OpCusto.Get_Produto.Value,14)
		ldDat = fDtoSQL(ThisForm.OpCusto.txtData.Value)

		Wait Window 'Aguarde! Selecionando Registros...' NoWait

		lcQuery = [Select Cemps ] + ;
				    [From SigCdEmp ] + ;
				   [Where Not Cemps = Space(3)] + Iif(Empty(lcEmp), [], [ And Cemps = '] + lcEmp + ['])

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'LocalEmp') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalEmp)')
			llSaida = .f.
			Loop
		EndIf
		
		lcQuery = [Select Calccustos From SigCdPac ]
		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'LocalParac') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalParac)')
			llSaida = .f.
			Loop
		EndIf

		llEmp = (LocalParac.CalcCustos <> 1)

		lcQuery = [Select Cpros ] + ;
				    [From SigCdPro ] + ;
				   [Where Not Cpros = Space(14)] + Iif(Empty(lcPro), [], [ And Cpros = '] + lcPro + ['])

		If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'LocalPro2') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalPro2)')
			llSaida = .f.
			Loop
		EndIf

		Select LocalEmp
		Scan
		
			If llEmp
				lcQuery = [Select Distinct Cpros ] + ;
				    [From SigMvEst ] + ;
				   [Where Emps = ']+Localemp.Cemps+[' And Not Cpros = Space(14)] + Iif(Empty(lcPro), [], [ And Cpros = '] + lcPro + ['])

				If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'LocalPro') < 1)
					=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalPro)')
					llSaida = .f.
					Exit
				EndIf
			Else
				Select * From LocalPro2 Into Cursor LocalPro ReadWrite
			Endif

			lnReg = Reccount('LocalPro')

			loCusto = CreateObject('fwprogressbar', 'Preparando Arquivo de Recálculo do Custo de Produtos', lnReg)
			loCusto.Titulo.FontBold = .t.
			loCusto.Show

			Select LocalPro
			Scan

				lnReg = lnReg - 1
				ThisForm.Get_Registro.Value = lnReg
				ThisForm.Get_Registro.Refresh

				loCusto.SubTitulo.Caption = 'Empresa : ' + LocalEmp.Cemps + ' - Produto : ' + LocalPro.CPros
				loCusto.Update(.t.)

				=fRecalculaC(LocalEmp.Cemps, LocalPro.CPros, ldDat, ThisForm.poDataMgr)

				llOk = fRecalculaC(.t., .t., .f., ThisForm.poDataMgr,.t.)
				If Not llOk
					ThisForm.Podatamgr.Rollback()
					Loop
				EndIf
				llOk = ( ThisForm.PodataMgr.Commit() > 0 )
				If Not llOk
					ThisForm.Podatamgr.Rollback()
					Loop
				EndIf

			EndScan
			If Not llEmp
				Exit
			Endif
		EndScan

		loCusto.Complete(.t., .t.)
	EndIf

	If ThisForm.BtnCompra.Value 
	
		Select CrSigOpClU
		Zap
		
		lcEmp = Padr(ThisForm.OpCompra.Get_Empresa.Value,3)
		lcPro = Padr(ThisForm.OpCompra.Get_Produto.Value,14)
		ldDat = fDtoSQL(ThisForm.OpCompra.txtData.Value)

		Wait Window 'Aguarde! Selecionando Registros...' NoWait
		
		lcSql = [Select a.datas,a.Emps,a.Dopes,a.Numes,a.EmpDopNums,a.Valos,a.Contads,a.ContaOs,]+;
				[c.GerGdmis,c.atuCompras,c.TpGdmis,b.cOpers,b.cmoes ]+;
				[From SigMvCab a, SigCdOpe b, SigCdTom c ]+;
				[Where a.Dopes = b.Dopes and b.TipoOps = c.Codigos And ]+;
				[((c.GerGdmis = 1 And c.TpGdmis = 1) Or ( c.GerGdmis = 2 And c.AtuCompras = 1)) And a.Datas >= ?ldDat ]+;

			llSaida = .f.
			Loop
		EndIf
		
		lcSql = [Select EmpDopNums From SigOpClU ]
		If (ThisForm.poDataMgr.SqlExecute(lcSql, 'LocCalcU') < 1)

			llSaida = .f.
			Loop

		Index on EmpDopNums Tag EmpDopNums
		
		Select TprMvCab
		Index on EmpDopNums Tag EmpDopNums

		Select crTpmMvItn
		Index on EmpDopNums Tag EmpDopNums

		Scan

			lnReg = lnReg - 1

					Endif
					
					llData  = TprMvCab.Datas

					lcMoeda = crTpmMvItn.Moedas

								(TprMvCab.Emps,TprMvCab.Dopes,TprMvCab.Numes,TprMvCab.EmpdopNums,;

			Loop
		EndIf

			Endif

			If ThisForm.podatamgr.SqlExecute(lcUpDate,'') < 1
				MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo (Update SigCdCli)')
				Return .f.
			Endif

					[Order by Valors Desc ]

				MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo (LocalCalcU)')
				Return .f.
			Endif

			llData = Iif(Reccount('LocalCalcU') = 0,Null,localCalcU.Datas)

			If ThisForm.podatamgr.SqlExecute(lcUpDate,'') < 1
				MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo (Update SigCdCli 2)')
				Return .f.
			EndIf
			ThisForm.Podatamgr.commit()
		EndScan

		Select Distinct Cpros From CsSelecao Into Cursor Selecao
		
		lnReg = Reccount('Selecao')

		loCusto.Show

		Select Selecao
		Scan
			lnReg = lnReg - 1
			ThisForm.Get_Registro.Value = lnReg
			ThisForm.Get_Registro.Refresh

			loCusto.SubTitulo.Caption = 'Produto '+Selecao.Cpros
			loCusto.Update(.t.)

			lcSql = [Select Top 1 Cpros,Datas,Valors,Moedas From SigOpClU Where Cpros = ']+Selecao.Cpros+[' ]+;
					[Order by Datas Desc ]

			Endif
