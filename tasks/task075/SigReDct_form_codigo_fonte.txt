------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Create Cursor CsRelatorio (cBars n(14), Cpros c(14), Dpros c(40), Nops n(10), cmats c(14), dMats c(40), qtd1s n(12,3),;
			cUni1s c(3), Peso1s n(12,3), cUnip1s c(3), qtd2s n(12,3), cUni2s c(3), Qtds n(6), Quebras n(10), ;
			peso2s n(12,3), cunip2s c(3), PesoFs n(12,3), Ordem n(10), cestos c(1), GrupoOs c(10), ContaOs c(10), ;
			Rclis c(40), Qtdb n(6), ImpCodBar c(20))
Index On Str(Nops,10) + cMats Tag NopMat
Index On Str(Quebras,10) + Str(Nops,10) + Str(Ordem,10) + Str(Cbars,14) Tag Nops

lcEdn  = ThisForm.pcemps + ThisForm.pcDopes + Str(ThisForm.pcNumes,10)
lcEdn2 = ThisForm.pcemps + ThisForm.pcDopes + Str(ThisForm.pcNumes,6)

lcSql = [Select Nops, cmats, sum(qtds) as Qtds, sum(Pesos) as Pesos, sum(Peso2s) as Peso2s From SigCdNei ]+;
		[Where EmpDnPs = ']+lcEdn+[' group by nops, cmats ]
ThisForm.Podatamgr.Sqlexecute(lcSql,'TmpNensi')
Select TmpNensi
Index On Str(Nops,10) + cmats Tag Nops

lcSql = [Select Nops, cbars, Cpros, Qtds From SigOpEtq Where EmpDopNums = ']+lcEdn2+[']
ThisForm.Podatamgr.Sqlexecute(lcSql,'TmpEti')
Select TmpEti
Index On Str(Nops,10) + Str(Cbars,14) Tag Nops

Select TmpEti
Scan
	lcSql = [Select Dpros,cUnis, cUniPs From SigCdPro Where Cpros = ']+TmpEti.Cpros+[']
	=ThisForm.Podatamgr.Sqlexecute(lcsql,'CrTmpPro')
	lnOp    = TmpEti.Nops
	lnOrdem = 0
	
	lcSql = [Select a.Nops,b.GrupoOs,b.ContaOs,c.Rclis From SigOpPic a, SigMvCab b, SigCdCli c ]+;
			[Where a.Nops = ]+Str(lnOp)+[ And a.EmpDopNums = b.EmpDopNums and b.ContaOs = c.Iclis ]

	ThisForm.Podatamgr.Sqlexecute(lcSql,'CrTmpCli')
	
	_ImpCodBar = fAltBar2de5(Transform(TmpEti.Cbars, '@L 99999999999999'))
	
	Select TmpNensi
	Set Order To Nops
	=Seek(Str(lnOp,10))
	
	lcSql = [Select * From SigPrCpo Where Cpros = ']+TmpEti.Cpros+[' Order by Mats ]
	=ThisForm.Podatamgr.Sqlexecute(lcSql,'TmpCompo')
	
	lnQtdB = 0
	Select TmpEti
	Scan While Nops = lnOp
		lnQtdB = lnQtdB + TmpEti.Qtds
		lnOrdem = lnOrdem + 1
		Insert Into CsRelatorio (Nops, cbars, Cpros, Dpros, Ordem, Qtds, GrupoOs, ContaOs, Rclis, Quebras, QtdB, ImpCodBar) Values ;
				(TmpEti.Nops, TmpEti.Cbars, TmpEti.Cpros, crTmpPro.Dpros, lnOrdem, 1,;
					CrTmpCli.GrupoOs, CrTmpCli.ContaOs, CrTmpCli.Rclis, Iif(ThisForm.ImpTotal.Value<>1,TmpEti.Nops,0), TmpEti.Qtds, _ImpCodBar)

		Select TmpNensi
		If Nops = LnOp
			lcSql = [Select a.Dpros,a.cUnis,a.cUnips,b.cestos From SigCdPro a, SigCdUni b where ]+;
					[a.Cpros = ']+TmpNensi.Cmats+[' And a.Cunis = b.Cunis ]
			=ThisForm.Podatamgr.Sqlexecute(lcsql,'TmpPro')
			Replace cmats  With TmpNensi.Cmats,;
					Dmats  With TmpPro.Dpros,;
					qtd1s   With TmpNensi.Qtds,;
					cuni1s  With TmpPro.Cunis,;
					cUnip1s With TmpPro.Cunips,;
					Peso1s  With TmpNensi.Peso2s,;
					PesoFs  With TmpNensi.Pesos, ;
					Cestos  With TmpPro.Cestos in CsRelatorio
			Select TmpNensi
			Skip
		EndIf
	EndScan
	Skip -1

	Select TmpNensi
	Scan While Nops = lnOp
		lnOrdem = lnOrdem + 1
		Insert Into CsRelatorio (Nops, cbars, Cpros, Dpros, Ordem, GrupoOs, ContaOs, Rclis, Quebras, ImpCodBar ) Values ;
				(TmpEti.Nops, 0, TmpEti.Cpros, crTmpPro.Dpros, lnOrdem, CrTmpCli.GrupoOs, CrTmpCli.ContaOs, ;
				CrTmpCli.Rclis, Iif(ThisForm.ImpTotal.Value<>1,TmpEti.Nops,0), _ImpCodBar)

		lcSql = [Select a.Dpros,a.cUnis,a.cUnips,b.Cestos From SigCdPro a, SigCdUni b where ]+;
				[a.Cpros = ']+TmpNensi.Cmats+[' And a.Cunis = b.Cunis ]
		=ThisForm.Podatamgr.Sqlexecute(lcsql,'TmpPro')
		Replace cmats  With TmpNensi.Cmats,;
				Dmats  With TmpPro.Dpros,;
				qtd1s   With TmpNensi.Qtds,;
				cuni1s  With TmpPro.Cunis,;
				cUnip1s With TmpPro.Cunips,;
				Peso1s  With TmpNensi.Peso2s,;
				cestos  With TmpPro.Cestos,;
				PesoFs  With TmpNensi.Pesos in CsRelatorio
	EndScan
	
	Select TmpCompo
	Scan
		Select CsRelatorio
		Set Order To NopMat
		If Not Seek(Str(lnOp,10) + TmpCompo.mats)
			lcSql = [Select Dpros,cUnis,cUnips From SigCdPro where Cpros = ']+TmpCompo.mats+[']
			=ThisForm.Podatamgr.Sqlexecute(lcsql,'TmpPro')
			lnOrdem = lnOrdem + 1
			Insert Into CsRelatorio (Nops, cbars, Cpros, Dpros, Ordem, CMats, dMats,Grupoos,ContaOs,Rclis, Quebras) Values ;
					(TmpEti.Nops, 0, TmpEti.Cpros, crTmpPro.Dpros, lnOrdem, TmpCompo.Mats, TmpPro.Dpros,;
					 CrTmpCli.GrupoOs,CrTmpCli.ContaOs,CrTmpCli.Rclis,Iif(ThisForm.ImpTotal.Value<>1,TmpEti.Nops,0))
		
		EndIf
		Replace qtd2s   With TmpCompo.Qtds * lnQtdB,;
				cuni2s  With Tmpcompo.UniCompos,;
				cUnip2s With TmpCompo.Cunips,;
				Peso2s  With TmpCompo.Pesos * lnQtdB In CsRelatorio
	EndScan
EndScan

If ThisForm.ImpTotal.Value = 1
	Select 9999999999 as Nops, Sum(Qtds) as Qtds, Sum(QtdB) as Qtdb, Sum(Pesofs) as PesoFs From CsRelatorio ;
	Into Cursor Selecao Group by 1

	Select CsRelatorio
	Append Blank
	Replace Nops   With  Selecao.Nops,;
			Qtds   With  Selecao.Qtds,;
			QtdB   With  Selecao.Qtdb,;
			PesoFs With  Selecao.Pesofs In CsRelatorio
Endif

lcSql = [Select Ndopps From SigCdOpd Where Dopps = ']+ThisForm.pcDopes+[']
=ThisForm.Podatamgr.Sqlexecute(lcsql,'TmpOpp')

_ImpCodBar = fAltBar2de5(Transform(TmpOpp.NDopps, '@L 99') + Transform(ThisForm.pcNumes, '@L 9999999999'))

ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'crSigCdEmp', 'Cemps', _Empr, 'Razas')
lcEmp = _Empr + ' - ' + Iif(Eof('crSigCdEmp'), '', crSigCdEmp.Razas)
lcCab = Alltrim(ThisForm.pcDopes) + ' - ' + Str(ThisForm.pcNumes,10)
lcSub = '' 

Create Cursor dbCabecalho (Titulo c(100), SubTitulo m, NomeEmpresa c(100), ImpCodBar c(20))
Insert Into dbCabecalho (Titulo, SubTitulo, NomeEmpresa, ImpCodBar) ;
					Values (lcCab, lcSub, lcEmp, _ImpCodBar)

Select CsRelatorio
Set Order To Nops
Go top

ENDPROC
