PROCEDURE visualizacao
Do Case
	Case This.TipoRelatório.Value = 1
		If This.OptFoto.Value = 1
			Report Form SigReBrf Preview NoConsole ;
				For Iif(_TipoDif = 1 Or ps == '2',.T.,;
				Iif(Not ThisForm.tratapeso,(Saldos # Iif(Empty(ReLidos),Lidos,ReLidos)),;
				((Saldos # Iif(Empty(ReLidos),Lidos,ReLidos)) or (Saldops # Iif(Empty(ReLidops),Lidops,ReLidops)))))
		Else
			If llHorizontal
				Report Form SigReBlh Preview NoConsole ;
				For Iif(_TipoDif = 1 Or ps == '2',.T.,;
				Iif(Not ThisForm.tratapeso,(Saldos # Iif(Empty(ReLidos),Lidos,ReLidos)),;
				((Saldos # Iif(Empty(ReLidos),Lidos,ReLidos)) or (Saldops # Iif(Empty(ReLidops),Lidops,ReLidops)))))
			Else
				Report Form SigReBlr Preview NoConsole ;
				For Iif(_TipoDif = 1 Or ps == '2',.T.,;
				Iif(Not ThisForm.tratapeso,(Saldos # Iif(Empty(ReLidos),Lidos,ReLidos)),;
				((Saldos # Iif(Empty(ReLidos),Lidos,ReLidos)) or (Saldops # Iif(Empty(ReLidops),Lidops,ReLidops)))))
			EndIf 	
		Endif
		
	Case This.TipoRelatório.Value = 2
		Report Form SigReBla Preview NoConsole

	Case This.TipoRelatório.Value = 3
		If CsSigCdPam.BalcTrats
			Report Form SigReBe2 Preview NoConsole
		Else
			Report Form SigReBle Preview NoConsole
		EndIf

	Case This.TipoRelatório.Value = 4
		Report Form SigReBlo Preview NoConsole

	Case This.TipoRelatório.Value = 5
		Do Form SigOpEtq With ThisForm

	Case This.TipoRelatório.Value = 6
		Report Form SigReBlb Preview NoConsole

	Case This.TipoRelatório.Value = 7
		Report Form SigReBel Preview NoConsole
		
	Case This.TipoRelatório.Value = 8
		Report Form SigReBl3 Preview NoConsole
EndCase

If Used('CsRelatorio')
	Use In CsRelatorio
EndIf

lcApagar = Sys(2023) + '\F*.Jpg'
Delete File &lcApagar.

ENDPROC

------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Public lnNumBal, _Empresa, _DProduto, _Leitura, _TipoDif, _Titulo, _TipoLeitura, _TipoAnal, _Produto, _Local,;
		_TotalnLido, _TotalnEsto, _Saldo, _SaldoPs,llHorizontal, impCustos

Local lcEscape,lcTalk,lnDecimals,lcFixed,lnTpCusto,lnTpVenda,TcLinhas,TcTraco1s,TcCabecs,Moedas,Custofs,Pvens

_TipoDif     = This.TipoDiferença.Value
_TipoRel     = This.TipoRelatório.Value 
_TipoLeitura = This.TipoLeitura.Value 
_TipoAnal    = This.TipoAnalítico.Value 
_Produto     = This.Get_Cd_Codigo.Value
_Local       = This.Get_Local.Value
_Etiqueta    = This.Get_Barra.Value
_DProduto    = ''
_Empresa     = Alltrim(CrSigCdEmp.CEmps) + Iif(!Empty(CrSigCdEmp.Razas),' - ' + Alltrim(CrSigCdEmp.Razas),'')
_Leitura     = ''
llHorizontal = .F.
impCustos    = this.optgTpCusto.value
llImpObs	 = (ThisForm.OptObs.Value = 1)
llMostraCtg  = (ThisForm.OptCtg.Value = 1)
llMostraCol  = (ThisForm.OptCol.Value = 1)

Select CsSigCdBal
Go Top
Scatter MemVar Memo

lnNumBal = m.Codigos
ThisForm.TrataPeso = (m.trtporpeso=1) 

lcDsGrupo = ''
lcDsConta = ''

lcDsGrupo = Iif(Seek(m.Grupos,'CrSigCdGcr','Codigos'),Alltrim(CrSigCdGcr.descrs),'')

If Used('TmpCli')
	Use In TmpCli
EndIf
If ThisForm.poDataMgr.SqlExecute([Select iclis,rclis From SigCdCli Where iclis = ']+m.Contas+[' ],'TmpCli') < 1
	MessageBox('Favor reinicializar o processo !!!',16,'Falha na Conexăo (TmpCli)')
	Return(.F.)
EndIf
Select TmpCli
Go Top
If !Eof()
	lcDsConta = Alltrim(TmpCli.RClis)
EndIf

*-[ Monta o Cabecalho do Filtro ]-*
lcDataInis = Iif(IsEmpty(m.DataInis),'',Dtoc(m.DataInis))
lcDataFims = Iif(IsEmpty(m.DataFims),'',Dtoc(m.DataFims))
m.FGrupo   = Iif(!Empty(m.Grupos),				 '     Grupo: ' + Alltrim(m.Grupos) + Iif(!Empty(lcDsGrupo),' - ' + lcDsGrupo,''),'')
m.FEstoque = Iif(!Empty(m.Contas),				 '   Estoque: ' + Alltrim(m.Contas) + Iif(!Empty(lcDsConta),' - ' + lcDsConta,''),'')
m.FPeriodo = Iif(!Empty(lcDataInis + lcDataFims),'   Periodo: ' + lcDataInis + ' ŕ ' + lcDataFims,'')
m.FLocal   = Iif(!Empty(m.LocalIs  + m.LocalFs) ,'     Local: ' + m.LocalIs  + ' ŕ ' + m.LocalFs ,'')
m.FIFor    = Iif(!Empty(m.IForIs   + m.IForFs)  ,'Fornecedor: ' + m.IForIs   + ' ŕ ' + m.IForFs  ,'')
m.FCGru    = Iif(!Empty(m.CGruIs   + m.CGruFs)  ,'     Grupo: ' + m.CGruIs   + ' ŕ ' + m.CGruFs  ,'')
m.FCGGru   = Iif(!Empty(m.CGGruIs  + m.CGGruFs) ,'Gde. Grupo: ' + m.CGGruIs  + ' ŕ ' + m.CGGruFs ,'')
m.FPreco   = Iif(m.PrecoIs + m.PrecoFs > 0      ,'     Preço: ' + Str(m.PrecoIs,9,2) + ' ŕ ' + Str(m.PrecoFs,9,2),'')
m.CObs	   = []
m.llVisCtg = llMostraCtg
m.llVisCol = llMostraCol

If llImpObs 
	m.CObs = Iif(IsEmpty(m.Obs),'','Observaçăo: ' + Alltrim(m.Obs))
EndIf 

Select CsFiltro
Insert Into CsFiltro From MemVar

If Used('CrSMoeda')
	Use In CrSMoeda
EndIf
If Used('CsRelatorio')
	Use In CsRelatorio
EndIf

Do Case 

	Case _TipoRel = 1

        _Titulo    = 'Relatório Resumido do Inventário nş ' + Alltrim(Str(lnNumBal)) + ' - ' + Iif(_TipoDif = 1,'Todos','Com Diferença')
        lnTpCusto  = ThisForm.OptGTpCusto.Value
        lnTpVenda  = ThisForm.OptGTpVenda.Value
        lnDecimals = Set('Decimals')
        lcFixed    = Set('Fixed')
        lcEscape   = Set('Escape')
        lcTalk     = Set('Talk')

        Set Escape Off
        Set Talk On
        Set Decimals To 2
        Set Fixed On

		Wait Window 'Aguarde! Selecionando Produtos...' NoWait
		lStrQuery = [Select a.*,b.CustoFs,b.MoeCusFs,b.PVens As PPVens,b.Moevs,b.FVendas,b.MoePVs,b.Linhas,b.Reffs, ] + ;
					[ltrim(rtrim(isnull(b.Colecoes,'')))+'-'+ltrim(rtrim(isnull(d.descs,''))) as Colecoes,]+;
					[ltrim(rtrim(isnull(b.Categoria,'')))+'-' +ltrim(rtrim(isnull(e.descs,''))) as Categoria,]+;
					[b.PesoMs,Space(200) as Obs, c.TpCustos As TpCus ]+;
					Iif(ThisForm.OptFoto.Value = 1,[, b.dPros, b.FigJpgs ],[])+;
					[From SigIvTrB a, SigCdPro b ]+;
					[Left Join SigCdLin c On c.Linhas = b.Linhas ]+;
					[left Join sigcdctg e on b.categoria = e.cods ]+;  
					[left Join SigCdCol d on b.Colecoes = d.colecoes ]+;
					[Where a.Emps = ?_Empr And a.Codigos = ?lnNumBal And b.CPros = a.CPros ]+;
					[Order By a.CIdChaves ]
		If ThisForm.poDataMgr.SqlExecute(lStrQuery,'Base') < 1
			MessageBox('Favor reinicializar o processo !!!',16,'Falha na Conexăo (Base)')
			Return(.F.)
		EndIf
		Select Base
		Go Top

		*If Not csSigCdBal.Encerras && Se o Balanço Năo Está Encerrado as Quantidades Estăo Apenas No SigIvTrH
			Select Base
			Index On CPros + CodCors + CodTams + Contas Tag CPros
			Replace All Lidos  With 0, Relidos  With 0, ;
						Lidops With 0, Relidops With 0

			lcQuery = [Select CPros, CodCors, CodTams, Contas, CBars, Tipos, Qtds, Pesos, PesReals, IsNull(Obs, Space(200)) as Obs ] + ;
						[From SigIvTrH ] + ;
					   [Where Codigos = ?lnNumBal And ] + ;
					   		 [Emps = ?_Empr Order By CIdChaves ]

			If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crTmpHist') < 1)
				=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpHist)')
				Return 0
			EndIf

			Select crTmpHist
			oProg = CreateObject('fwprogressbar', 'Aguarde!!! Processando Produtos Lidos/Relidos...', Reccount())
			oProg.Show
			Scan
				oProg.Update(.t.)
				If Not Seek(crTmpHist.CPros + crTmpHist.CodCors + crTmpHist.Codtams + crTmpHist.Contas, 'Base', 'CPros')
					Insert Into Base (CPros, CodCors, CodTams, Contas, Emps, Codigos, Obs) ;
							  Values (crTmpHist.CPros, crTmpHist.CodCors, crTmpHist.CodTams, crTmpHist.Contas, _Empr, lnNumBal, crTmpHist.Obs)
				Else
					Replace Obs With crTmpHist.Obs In Base
				EndIf

				llOk = .t.
				If Not Empty(CrTmpHist.CBars)
					lcSql = [Select * from SigIvTrE Where Cbars = ]+Str(CrTmpHist.Cbars,14)+[ And Codigos = ?lnNumBal ]
					If ThisForm.Podatamgr.Sqlexecute(lcsql,'CrTmpEti') > 0
						If Eof('CrTmpEti')
							ThisForm.poDataMgr.CursorQuery('SigOpEtq', 'CsTmpEti', 'CBars', CrTmpHist.CBars, 'CBars')
							If Eof('CsTmpEti')
								ThisForm.poDataMgr.CursorQuery('SigCdPro', 'CrTmpEti', 'CBars', CrTmpHist.CBars, 'CBars')
								If Eof('CrTmpEti')
									llOk = .F.
								EndIf
							EndIf
						EndIf
					EndIf
				EndIf

				If llOk
					If InList(crTmpHist.Tipos, 1, 3)
						Replace Lidos   With Lidos   + crTmpHist.Qtds  In Base
						Replace Lidops  With Lidops  + crTmpHist.Pesos In Base
					Else
						Replace ReLidos  With ReLidos  + crTmpHist.Qtds  In Base
						Replace ReLidops With ReLidops + crTmpHist.Pesos In Base
					EndIf
				EndIf
			EndScan
			oProg.Complete(.t.)
		*EndIf

		Select Base
		Go Top

        Select a.*,Val(Str(a.PPVens * (a.Saldos - Iif(Empty(a.ReLidos),a.Lidos,a.ReLidos)),16,2)) As PVens,;
        		Val(Str(a.PPVens * (a.Saldops - Iif(Empty(a.ReLidops),a.Lidops,a.ReLidops)),16,2)) As PVenPesos,;
			   lnTpCusto As TpCustos,lnTpVenda As TpVendas,Space(200) As TcCabecs,Space(200) As TcTraco1s,;
			   Space(200) As TcLinhas,Space(200) As TcLinhas1,Space(200) As TcLinhas2,Space(200) As TcLinhas3,;
			   Space(200) As TcLinhas4, '1' As Ps, Space(11) AS OrdImg ;
		  From Base a ;
         Order By a.Contas,a.CPros,a.CodCors,a.CodTams ;
          Into Cursor CsBase ReadWrite

		Wait Window 'Aguarde! Calculando Custos...' NoWait

		lnTcLinhas=1

		Select * From CsBase Into Cursor CsRelatorio ReadWrite
		Select CsRelatorio
		Go Top
		Scan
			lnCusto = CsRelatorio.CustoFs
			If Nvl(CsRelatorio.TpCus,'') = '2'
				lnCusto = CsRelatorio.PesoMs * CsRelatorio.CustoFs
			EndIf
			
			Select CsRelatorio
			Replace CsRelatorio.CustoFs With lnCusto * (CsRelatorio.Saldos  - Iif(Empty(CsRelatorio.ReLidos) ,CsRelatorio.Lidos ,CsRelatorio.ReLidos))
			If ThisForm.OptFoto.Value = 1
				If Empty(csRelatorio.FigJpgs) Or IsNull(csRelatorio.FigJpgs)
					Replace OrdImg With 'FBRANCOS' In csRelatorio
				Else
					lcCodigo = Alltrim(csRelatorio.cPros)
					lcCodigo = Strtran(lcCodigo,'/','')
					lcCodigo = Strtran(lccodigo,' ','')
					lcDirectory = Sys(2023)+'\F'+Alltrim(lcCodigo)+'.Jpg'
					Replace OrdImg With 'F' + Alltrim(lcCodigo) In csRelatorio
					lcFoto=Strtran(Strtran(Strtran(csRelatorio.FigJpgs,"data:image/png;base64,",""),"data:image/jpeg;base64,",""),"data:image/jpg;base64,","")
					=StrToFile(Strconv(lcFoto,14),lcDirectory)
				EndIf
			Endif

			If ThisForm.OptEtiq.Value = 1
				If (Iif(Empty(CsRelatorio.ReLidos),CsRelatorio.Lidos,CsRelatorio.ReLidos) > CsRelatorio.Saldos) or ;
					(Iif(Empty(CsRelatorio.ReLidops),CsRelatorio.Lidops,CsRelatorio.ReLidops) > CsRelatorio.Saldops)
					lcSql = [Select cBars from SigIvTrH Where cPros = ']+CsRelatorio.cPros+[' And Codigos = ]+Str(lnNumBal)+;
