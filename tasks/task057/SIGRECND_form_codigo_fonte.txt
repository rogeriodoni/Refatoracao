* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Private pDtR

ldDat = ThisForm.cntLocal.getDtInicial.Value
lcGru = Padr(ThisForm.cntLocal.getGrupo.Value, 10)
lcCon = Padr(ThisForm.cntLocal.getConta.Value, 10)
lcMoe = Padr(ThisForm.cntLocal.getMoeda.Value, 03)

lcTit = ThisForm.Caption
lcSub = 'Data Referęncia : ' + Dtoc(ldDat) + ' - ' + ;
		'Grupo : ' + Alltrim(lcGru) + ' - ' + ;
		'Conta : ' + Iif(Empty(lcCon), 'Todas', Alltrim(lcCon))
ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'crSigCdEmp', 'Cemps', _Empr, 'Razas')
lcCep = Alltrim(_Empr) + Iif(Eof('crSigCdEmp'), '', ' - ' + Alltrim(crSigCdEmp.Razas))

Create Cursor dbCabecalho (NomeEmpresa c(80), Titulo c(80), SubTitulo c(80), Moeda1 c(80))
Insert Into dbCabecalho (NomeEmpresa, Titulo, SubTitulo, Moeda1) Values (lcCep, lcTit, lcSub, lcMoe)

Create Cursor TmpHist (Audits l(1), Datas d(8), Grupos c(10), Contas c(10), RClis c(40), ;
					   Saldo1 n(12, 2), Opers c(1), Hists c(40), Hist2s c(40), Emps c(3), ;
					   Debitos n(12, 2), Creditos n(12, 2), Docus c(10), cIdChaves c(20))
Index On Grupos + Contas + cIdChaves Tag ContData

pDtR = fDtoSQL(ldDat, '23:59:59')

lcQuery = [Select Grupos, Contas, cIdChaves, Datas, Opers, Valors, Moedas, Docus, Emps, DtAudits, Hists, Hist2s ] + ;
			[From SigMvCcr ] + ;
		   [Where Grupos = '] + lcGru + [' And ] + ;
		   		  Iif(Empty(lcCon), [], [Contas = '] + lcCon + [' And ]) + ;
		   		  [Datas <= ?pDtR And ] + ;
		   		  [(DataConcs Is Null Or DataConcs > ?pDtR) ] + ;
		   	[Order By Grupos, Contas, cIdChaves, Datas, Opers, Valors, Docus, Emps, DtAudits]

If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crDados') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crDados)')
	Return 0
EndIf

lcCab = ''
lnSal = 0
Select crDados
oProg = CreateObject('fwprogressbar', 'Processando Operaçőes...', Reccount())
oProg.Show
Scan
	oProg.Update(.t.)
	If (lcCab <> crDados.Grupos + crDados.Contas)
		lcCab = crDados.Grupos + crDados.Contas
		lnSal = 0
	EndIf

	ThisForm.poDataMgr.CursorQuery('SigCdCli', 'crSigCdCli', 'IClis', crDados.Contas, 'RClis')
	lcDes = crSigCdCli.RClis

	lnVal = crDados.Valors
	If (crDados.Moedas <> lcMoe)
		lnCt1 = fBuscarCotacao(crDados.Moedas, crDados.Datas, ThisForm.poDataMgr)
		lnCt2 = fBuscarCotacao(lcMoe, crDados.Datas, ThisForm.poDataMgr)
		lnVal = (crDados.Valors * lnCt1) / lnCt2
	EndIf

	lnSal = lnSal + (lnVal * Iif(crDados.Opers = 'C', 1, -1))

	Insert Into TmpHist (Audits, Datas, Grupos, Contas, RClis, Saldo1, Opers, ;
						 Hists, Hist2s, Emps, Debitos, Creditos, Docus, cIdChaves) ;
				 Values (Not IsEmpty(crDados.DtAudits), crDados.Datas, crDados.Grupos, ;
				 		 crDados.Contas, lcDes, lnSal, crDados.Opers, crDados.Hists, ;
				 		 crDados.Hist2s, crDados.Emps, Iif(crDados.Opers = 'D', lnVal, 0), ;
				 		 Iif(crDados.Opers = 'C', lnVal, 0), crDados.Docus, crDados.cIdChaves)
EndScan
oProg.Complete

Select TmpHist
Set Order To ContData
Go Top
* Fim **************************************************************************************

ENDPROC
