LParameters _Acesso

=DoDefault()

ThisForm.DataSessionId = _Acesso
ThisForm.lacpdf = .t.

&&Cursor com od mesmos Dados do Cursor Direfenca.
SELECT *, 99999999.99 as Deb1s, 99,999999.99 as Cred1s from movaux WHERE transacaos in ( select transacaos FROM dif2 ) INTO CURSOR crGrid readwrite

Replace All crGrid.Debs  With Transform(Val(Debs)/100 ,'@z 9,999,999.99') in crGrid
Replace All crGrid.Creds With Transform(Val(Creds)/100,'@z 9,999,999.99') in crGrid

Replace All crGrid.Valor With Transform(Val(Valor),'@z 9,999,999.99') in crGrid
Replace All crGrid.Docto With Transform(Val(Docto),'@z 9999999999')       in crGrid
