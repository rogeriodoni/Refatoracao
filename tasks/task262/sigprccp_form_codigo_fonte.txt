PROCEDURE processar
Private lnMrki, lnMrkf, pPro, lnEncI, lnEncF

Local lcGrui, lcGruf, lcSGruI, lcSGruF, lcMercI, lcMercF, lcCunii, lcCUnif, lcLini, lcLinf, lcColi, lcColf, lnConta, ;
	lcWhere, lcInicio, lcFinal, lcMoedai, lcMoedaf, lnArred, Custo, Moec, Venda, Moev, Fator, fPeso, Ideal

lcMercI	   = Thisform.getMercI.Value
lcMercF	   = Thisform.getMercF.Value
lcGrui     = Thisform.GetCGrui.Value
lcGruf     = Thisform.GetCGruf.Value
lcSGruI    = Thisform.getSgruI.Value
lcSGruF    = Thisform.getSgruF.Value
lcCunii    = Thisform.GetCUnii.Value
lcCUnif    = Thisform.GetCUnif.Value
lcLini     = Thisform.GetLini.Value
lcLinf     = Thisform.GetLinf.Value
lcColi     = Thisform.GetColi.Value
lcColf     = Thisform.GetColf.Value
lcMoedai   = Thisform.GetMoedai.Value
lcMoedaf   = Thisform.GetMoedaf.Value
lnMrki     = Thisform.Getmrki.Value
lnMrkf     = Thisform.Getmrkf.Value
lnEncI     = Thisform.Get_Enci.Value
lnEncF     = Thisform.Get_Encf.Value
lnArred    = crSigCdPam.Arreds
lnArred    = Iif(lnArred = 0, 1, lnArred)
lnPvenda   = Thisform.Opc_pven.Value
lnReajuste = 1 + (Thisform.Get_Reajuste.Value / 100)
lnMarkup   = Thisform.GetnMrk.Value
lnEncarg   = Thisform.Get_Encargo.Value
lnTpRecal  = Thisform.Opc_Recalc.Value
lcFeitio   = Thisform.Get_Feitio.Value
lnOpcaoM   = Thisform.fwoption1.Value
lnOpcComp  = ThisForm.Opc_Compra.Value
lcIfors    = ThisForm.getCFornecs.Value

lcFtiNew   = Thisform.getnewMkp.Value

If InList(lnTpRecal,7,8) And Empty(lcFtiNew)
	If Not ThisForm.Automatico 
		MessageBox([Favor Informar o Novo Código do MKP!!!],0+64,[Atençăo])
		ThisForm.getNewMkp.SetFocus()
	EndIf 
	Return .F.
EndIf 

Declare laCampo[7]
laCampo[1] = 'CGrus'
laCampo[2] = 'Cunis'
laCampo[3] = 'Linhas'
laCampo[4] = 'Colecoes'
laCampo[5] = 'Moedas'
If lnOpcaoM = 2
	laCampo[5] = 'Moevs'
Endif
laCampo[6] = 'SGrus'
laCampo[7] = 'Mercs'

Declare laVarias[7]
laVarias[1] = 'lcGru'
laVarias[2] = 'lcCuni'
laVarias[3] = 'lcLin'
laVarias[4] = 'lcCol'
laVarias[5] = 'lcMoeda'
laVarias[6] = 'lcSGru'
laVarias[7] = 'lcMerc'

lcWhere = [ ]
For lnConta = 1 To 7 && 5 // Richard em 26/06/2015 - Năo estava incluindo os demais filtros
	lcInicio = Eval(laVarias[lnConta] + 'i')
	lcFinal  = Eval(laVarias[lnConta] + 'f')

	If (lnConta > 1) And (Not Empty(lcInicio) Or Not Empty(lcFinal)) And Not Empty(lcWhere)
		lcWhere = lcWhere + [ And ]
	Endif

	If Empty(lcInicio)
		If Empty(lcFinal)
			lcWhere = lcWhere + []
		Else
			lcWhere = lcWhere + laCampo[lnConta] + [ <= '] + lcFinal + [']
		Endif
	Else
		If Empty(lcFinal)
			lcWhere = lcWhere + laCampo[lnConta] + [ >= '] + lcInicio + [']
		Else
			lcWhere = lcWhere + laCampo[lnConta] + [ Between '] + lcInicio + [' And '] + lcFinal + [']
		Endif
	Endif
Next

lcWhere = Alltrim(lcWhere)
If Empty(lcWhere)
	lcWhere = [ 1=1 ]
Endif

If Right(lcWhere, 3) = [And]
	lcWhere = Subs(lcWhere, 1, Len(lcWhere) - 3)
Endif

With Thisform.Opc_Situacao
	If Inlist(.Value, 1, 2)
		lcWhere = lcWhere + [ And Situas = ] + Str(.Value, 1)
	Endif
EndWith

If Not Empty(lcIfors)
	lcWhere = lcWhere + [ And Ifors = '] + lcIfors + [' ]
EndIf 

If InList(lnOpcComp,1,2)
	lcWhere = lcWhere + [ And ForaLinha = ] + Str(Iif(lnOpcComp=1,0,1), 1)	
EndIf 

lcQuery = [Select * From SigCdPro ] + ;
	[Where ] + lcWhere + Iif(lnMrki > 0, [ And Margems Between ?lnMrki And ?lnMrkf], []) + ;
	Iif(lnEncI>0,[ And Encargos Between ?lnEncI And ?lnEncF ], [ ] )+ ;
	Iif(Not Empty(lcFeitio),[ And ( cFtios = ']+lcFeitio+[' Or cFtioCs = ']+lcFeitio+[') ],[ ])

If (Thisform.poDataMgr.SqlExecute(lcQuery, 'CrSigCdPro') < 1)
	If Not ThisForm.Automatico
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpPro)')
	EndIf 	
	Return .F.
Endif
Select CrSigCdPro
Index On Cpros Tag Cpros

lcQuery = [Select * From SigPrCpo Where 1=0 ]
If (Thisform.poDataMgr.SqlExecute(lcQuery, [TmpPrCpo]) < 1)
	If Not ThisForm.Automatico
		=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (TmpPrCpo)])
	EndIf 
	Return .F.
Endif

Select TmpPrCpo
Index On Cpros Tag Cpros

lcQuery = [Select * From SigPrFti ]
If (Thisform.poDataMgr.SqlExecute(lcQuery, 'CrSigPrFti') < 1)
	If Not ThisForm.Automatico
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (CrSigPrFti)')
	EndIf 
	Return .F.
Endif
Select CrSigPrFti
Index On Cods Tag Cods

lcQuery = [Select a.*, b.Dgrus From SigPrFto a, SigCdGrp b Where ]+;
	[a.Cgrus = b.Cgrus And a.Cgrus <> ' ' ]
If (Thisform.poDataMgr.SqlExecute(lcQuery, 'CrSigPrFtiG') < 1)
	If Not ThisForm.Automatico
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (CrSigPrFtiG)')
	EndIf 
	Return .F.
Endif
Select CrSigPrFtiG
Index On Cods Tag Cods

lcQuery = [Select a.*, b.Dpros From SigPrFto a, SigCdPro b Where ]+;
	[a.Cpros = b.Cpros And a.CGrus = ' ' ]
If (Thisform.poDataMgr.SqlExecute(lcQuery, 'CrSigPrFtiP') < 1)
	If Not ThisForm.Automatico
		=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (CrSigPrFtiP)')
	EndIf 
	Return .F.
Endif
Select CrSigPrFtiP
Index On Cods Tag Cods

Create Cursor TotGrupo (Grupo c(3), Cpros c(14), ValGrupo N(12,3), Moeda c(3))
Index On Grupo + Cpros + Moeda Tag GruMoe

Select CrSigPrCpo
Zap
Index On Cpros Tag Cpros


&& Se Recalcula os Pesos dos Componentes - Richard em 31/10/2013
If (lnTpRecal = 4)
	Select CrSigCdPro
	loBarra = Createobject([fwProgressBar], [Recalculando Pesos de Componentes...], Reccount([crSigCdPro]))
	loBarra.Show
	Scan
		loBarra.Update(.T.)
		pPro = CrSigCdPro.Cpros

		lcQuery = [Select * From SigPrCpo Where CPros = '] + pPro + [']
		If (Thisform.poDataMgr.SqlExecute(lcQuery, [TmpCompo]) < 1)
			If Not ThisForm.Automatico	
				=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (TmpCompo)])
			EndIf 
			Return .F.
		Endif

		Local llAtu

		llAtu = .F.

		Select TmpCompo
		Scan
			lnMtl = 1

			If Not Empty(TmpCompo.Mats)
				lcQuery = [Select a.PesoMs, b.CfgGerGprs ] + ;
					[From SigCdPro a, SigCdGrp b ] + ;
					[Where a.CPros = '] + TmpCompo.Mats + [' And ] + ;
					[a.CGrus = b.CGrus]

				If (Thisform.poDataMgr.SqlExecute(lcQuery, [LocalProCp]) < 1)
					If Not ThisForm.Automatico
						=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (LocalProCp 1)])
						=Messagebox([Os Pesos Podem Năo Ter Sido Calculados! Refaça o Processo!], 16, [Atençăo!!!])
					EndIf 
					Return .F.
				Endif
				Go Top In LocalProCp
				If Not Eof([LocalProCp]) And (LocalProCp.PesoMs > 0) And (Int(Val(Substr(LocalProCp.CfgGerGprs, 9, 1))) > 1)
					If (Int(Val(Substr(LocalProCp.CfgGerGprs, 9, 1))) = 2) && Lança na Unidade 1
						If Not Empty(TmpCompo.UniCompos)
							If Not Empty(TmpCompo.CUniPs) And (TmpCompo.Pesos > 0)
								lnMtl = TmpCompo.Pesos
							Endif
							lnVal = LocalProCp.PesoMs * lnMtl

							lcQuery = [Update SigPrCpo Set Qtds = ] + Str(lnVal, 9, 3) + [ ] + ;
								[Where cIdChaves = '] + TmpCompo.cIdChaves + [']

							If (Thisform.poDataMgr.SqlExecute(lcQuery, []) < 1)
								If Not ThisForm.Automatico
									=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (LocalProCp 1)])
									=Messagebox([Os Pesos Podem Năo Ter Sido Calculados! Refaça o Processo!], 16, [Atençăo!!!])
								EndIf 
								Return .F.
							Endif
							llAtu = .T.
						Endif
					Else && Lança na Unidade 2
						If Not Empty(TmpCompo.CUniPs)
							If Not Empty(TmpCompo.UniCompos) And (TmpCompo.Qtds > 0)
								lnMtl = TmpCompo.Qtds
							Endif
							lnVal = LocalProCp.PesoMs * lnMtl

							lcQuery = [Update SigPrCpo Set Pesos = ] + Str(lnVal, 9, 3) + [ ] + ;
								[Where cIdChaves = '] + TmpCompo.cIdChaves + [']

							If (Thisform.poDataMgr.SqlExecute(lcQuery, []) < 1)
								If Not ThisForm.Automatico
									=Messagebox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (LocalProCp 1)])
									=Messagebox([Os Pesos Podem Năo Ter Sido Calculados! Refaça o Processo!], 16, [Atençăo!!!])
								EndIf 
								Return .F.
							Endif
							llAtu = .T.
						Endif
					Endif
					Use In LocalProCp
				Endif
			Endif
		Endscan
		If llAtu
			Insert Into crProdutos (Cpros, DPros) Values (CrSigCdPro.Cpros, CrSigCdPro.DPros)
		Endif
		Use In TmpCompo
	Endscan
	loBarra.Complete(.T.)
Else


	Select CrSigCdPro
	loBarra = Createobject('fwprogressbar', 'Recalculando Produtos...', Reccount('CrSigCdPro'))
	loBarra.Show
	Scan
		pPro = CrSigCdPro.Cpros

		If Empty(CrSigCdPro.Moedas)
			Replace Moedas With Moecs In CrSigCdPro
		Endif

		lcSql = [Select * From SigPrCpo Where Cpros = ']+pPro+[']
		If Thisform.poDataMgr.SqlExecute(lcSql,'CrSigPrCpo') < 1
			If Not ThisForm.Automatico
				=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (CrSigPrCpo)')
			EndIf 
			Return .F.
		Endif
		*** Tiago - 19/01/2016 - Comentei o if e voltei o código logo abaixo com append para dentro do IF, pois năo estava mais recalculando produtos que năo tinha composiçăo
		*!*			If Reccount('TmpCompo') = 0
		*!*				Loop
		*!*			EndIf
		*!*			Select crSigPrCpo
		*!*			Append From Dbf('TmpCompo')

*!*			If Reccount('TmpCompo') > 0
*!*				Select CrSigPrCpo
*!*				Append From Dbf('TmpCompo')
*!*			Endif

		Select CrSigPrCpo
		Index On Cpros Tag Cpros
		Go Top
		
		Insert Into crProdutos (Cpros, DPros, ValAnt, CustoAfs ) ;
			Values (CrSigCdPro.Cpros, CrSigCdPro.DPros, CrSigCdPro.Pvens, CrSigCdPro.CustoFs )

		Select CrSigCdPro

		=Seek(CrSigCdPro.Moecs, 'crSigCdMoe', 'CMoes' )
		If Not Empty(crSigCdMoe.Moeqs)
			m.Moedas = crSigCdMoe.Moeqs
			_Qtdeqs = Iif(crSigCdMoe.Qtdeqs = 0, 1, crSigCdMoe.Qtdeqs)
		Else
			m.Moedas = CrSigCdPro.Moecs
			_Qtdeqs = 1
		Endif
		Moec = fCarregarCambio(m.Moedas, Date())  * _Qtdeqs

		=Seek(CrSigCdPro.Moepcs, 'crSigCdMoe', 'CMoes' )
		If Not Empty(crSigCdMoe.Moeqs)
			m.Moedas = crSigCdMoe.Moeqs
			_Qtdeqs = Iif(crSigCdMoe.Qtdeqs = 0, 1, crSigCdMoe.Qtdeqs)
		Else
			m.Moedas = CrSigCdPro.Moepcs
			_Qtdeqs = 1
		Endif
		Moepc = fCarregarCambio(m.Moedas, Date())  * _Qtdeqs

		=Seek(CrSigCdPro.Moedas, 'crSigCdMoe', 'CMoes' )
		If Not Empty(crSigCdMoe.Moeqs)
			m.Moedas = crSigCdMoe.Moeqs
			_Qtdeqs = Iif(crSigCdMoe.Qtdeqs = 0, 1, crSigCdMoe.Qtdeqs)
		Else
			m.Moedas = CrSigCdPro.Moedas
			_Qtdeqs = 1
		Endif
		Moedac = fCarregarCambio(m.Moedas, Datetime())  * _Qtdeqs

		=Seek(CrSigCdPro.Cgrus,'CrSigCdGrp','Cgrus')

		Store 0 To lnTQtde, lnTFator, lnTotal, lnTFtio, lnTotCv, lnTotpCv, lnTotEstM

		Select TotGrupo
		Zap

		Select CrSigPrCpo
		=Seek(pPro)
		Scan While Cpros = pPro
			If Empty(CrSigPrCpo.Mats)
				Loop
			Endif

			llOk = .T.
			*-- Rafael - Solicitaçăo do Cleber para por enquanto năo usar os componentes que săo materiais principais
			lcQuery = [Select Distinct Matprincs From SigCdPro Where MatPrincs <> Space(14) and MatPrincs = ']+CrSigPrCpo.Mats+[' ]
			If (Thisform.poDataMgr.SqlExecute(lcQuery, 'crMatPrinc') < 1)
				If Not ThisForm.Automatico
					=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crMatPrinc)')
				EndIf 
				Return 0
			Endif

			Select crMatPrinc
			If Reccount([crMatPrinc])>=1
				llOk = .F.
			Endif

			*!*				If !llOk
			*!*					Select crSigPrCpo
			*!*					Loop
			*!*				EndIf
			*--
			If llOk
				lcSql = [Select Custofs, MoeCusfs, Cunis, Cgrus, Moevs, cUniPs, pVens From SigCdPro ]+;
					[Where Cpros = ']+CrSigPrCpo.Mats+[']
				If Thisform.poDataMgr.SqlExecute(lcSql,'CrCompoPro') < 1
					If Not ThisForm.Automatico
						=Messagebox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (CrCompoPro)')
					EndIf 
					Return .F.
				Endif

				=Seek(CrSigPrCpo.Cgrus,'CrSigCdGrp','Cgrus')

				If Not Eof('crCompoPro') And (lnTpRecal <> 2)
					If CrSigCdGrp.pvCompos = 2
						Replace PCompos   With crCompoPro.Pvens, ;
							Moeds     With crCompoPro.Moevs In CrSigPrCpo
					Else
						Replace PCompos   With crCompoPro.CustoFs, ;
							Moeds     With crCompoPro.MoeCusfs In CrSigPrCpo
					Endif
					Replace UniCompos With crCompoPro.Cunis, ;
						CUniPs    With crCompoPro.CUniPs, ;
						Cgrus 	  With crCompoPro.Cgrus, ;
						DtMovs    With Datetime() In CrSigPrCpo
				Endif

				*!*			Valgr = (CrSigPrCpo.pCompos * CrSigPrCpo.Qtds)
				Valgr = (CrSigPrCpo.PCompos * Iif(CrSigCdGrp.chkInstalas=2, CrSigPrCpo.Pesos, CrSigPrCpo.Qtds))
				Select TotGrupo
				Set Order To GruMoe
				Seek CrSigPrCpo.Cgrus + CrSigPrCpo.Cpros + CrSigPrCpo.Moeds
				If Eof()
					Insert Into TotGrupo (Grupo, Cpros, Moeda) Values (CrSigPrCpo.Cgrus, CrSigPrCpo.Mats, CrSigPrCpo.Moeds)
				Endif
				Replace ValGrupo With ValGrupo + Valgr In TotGrupo

				=Seek(CrSigPrCpo.Moeds, 'crSigCdMoe', 'CMoes' )
				If Not Empty(crSigCdMoe.Moeqs)
					m.Moedas = crSigCdMoe.Moeqs
					_Qtdeqs = Iif(crSigCdMoe.Qtdeqs = 0, 1, crSigCdMoe.Qtdeqs)
				Else
					m.Moedas = CrSigPrCpo.Moeds
					_Qtdeqs = 1
				Endif
				Moem = fCarregarCambio(m.Moedas, Datetime())  * _Qtdeqs

				=Seek(crCompoPro.Moevs, 'crSigCdMoe', 'CMoes' )
				If Not Empty(crSigCdMoe.Moeqs)
					m.Moedas = crSigCdMoe.Moeqs
					_Qtdeqs = Iif(crSigCdMoe.Qtdeqs = 0, 1, crSigCdMoe.Qtdeqs)
				Else
					m.Moedas = crCompoPro.Moevs
					_Qtdeqs = 1
				Endif
				Moev = fCarregarCambio(m.Moedas, Datetime())  * _Qtdeqs

				If Empty(CrSigPrCpo.Moeds)
					Loop
				Endif

				If (lnTpRecal <> 1)
					=Seek(CrSigPrCpo.UniCompos,'CrSigCdUni','Cunis')
					=Seek(CrSigPrCpo.Cgrus,'CrSigCdGrp','Cgrus')

					If (CrSigCdGrp.PCustVens <> 0)
						Replace VlrCvs With Round(CrSigPrCpo.PCompos * (1 + (CrSigCdGrp.PCustVens / 100)), 3) In CrSigPrCpo
					Else
						Replace VlrCvs With CrSigPrCpo.PCompos In CrSigPrCpo
					Endif

					If (CrSigCdUni.PCustVens <> 0)
						Replace QtdCvs With Round(CrSigPrCpo.Qtds * (1 + (CrSigCdUni.PCustVens / 100)), 3) In CrSigPrCpo
					Else
						Replace QtdCvs With CrSigPrCpo.Qtds In CrSigPrCpo
					Endif

					If CrSigPrCpo.Moeds <> crCompoPro.Moevs
						=Seek(CrSigPrCpo.Moeds, 'crSigCdMoe', 'CMoes' )
						If Not Empty(crSigCdMoe.Moeqs)
							m.Moedas = crSigCdMoe.Moeqs
							_Qtdeqs = Iif(crSigCdMoe.Qtdeqs=0, 1, crSigCdMoe.Qtdeqs)
						Else
							m.Moedas = CrSigCdPro.MoeCusfs
							_Qtdeqs = 1
						Endif
						CotMcf = fCarregarCambio(m.Moedas, Datetime()) * _Qtdeqs

						=Seek(crCompoPro.Moevs, 'crSigCdMoe', 'CMoes' )
						If Not Empty(crSigCdMoe.Moeqs)
							m.Moedas = crSigCdMoe.Moeqs
							_Qtdeqs = Iif(crSigCdMoe.Qtdeqs=0, 1, crSigCdMoe.Qtdeqs)
						Else
							m.Moedas = CrSigCdPro.Moedas

		_MarkUpa = Iif(CrSigCdPro.CustoFs=0,0,Round((CrSigCdPro.Pvens * Moev) / (CrSigCdPro.CustoFs * Moecf), 3))
		Replace MarkupA With _MarkUpa In CrSigCdPro

		*-- Preco atual atraves do cambio
		If Inlist(lnTpRecal, 5, 6)
			lnCotId = fBuscarCotacao(CrSigCdPro.Moedas, Datetime(), Thisform.poDataMgr)
			lnCotVd = fBuscarCotacao(CrSigCdPro.Moevs, Datetime(), Thisform.poDataMgr)
			lnPven  = CrSigCdPro.Pvideals*lnCotId/lnCotVd
			lnVlVen = (lnPven / Iif(CrSigCdPro.Encargos<>0,CrSigCdPro.Encargos,1))

			If (lnTpRecal = 6) && Richard em 14/08/2016 - Vianna
				lnPven  = Int(CrSigCdPro.Pvideals*lnCotId/lnCotVd)
				lnVlVen = Int(lnPven / Iif(CrSigCdPro.Encargos<>0,CrSigCdPro.Encargos,1))
			Endif

			Select CrSigCdPro
			Replace Pvens  With lnVlVen
		Endif

		Select crProdutos
		Replace ValAtu  With CrSigCdPro.Pvens,;
			CustoFs With CrSigCdPro.CustoFs In crProdutos

		&&CrSigCdPro.pvideals

		Replace pVarias With Iif(ValAnt = 0, 0, (((ValAtu/ValAnt)-1)*100 ) ) In crProdutos
		Replace cVarias With Iif(CustoAfs = 0, 0, (((CustoFs/CustoAfs)-1)*100 ) ) In crProdutos

		loBarra.Update(.T.)

	Endscan
	loBarra.Complete(.T.)
Endif

Return .T.

ENDPROC

* PROCEDURE atualizar
------------------------------------------------------------
PROCEDURE atualizar
llReturn = .T.

If Not ThisForm.Automatico
	lnOk = MessageBox('Atualiza ???',4+32+256,'Alteraçăo de Preços')
Else
	lnOk = 6
EndIf 

If lnOk = 6
	If Not ThisForm.Automatico
		llImpEtiq = Iif(MessageBox('Confirma a Impressăo das Etiquetas?',4+32+256,'') = 6,.T.,.F.)
	Else
		llImpEtiq = .F.
	EndIf 	

	Select * From CrProdutos Where lMarca = 1 Order By CPros Into Cursor CsProdutos ReadWrite
	Select CsProdutos
	Go Top
	If Eof()
		If Not ThisForm.Automatico
			MessageBox('Nenhum Produto Selecionado !!!',48,'Seleçăo Obrigatória')
			ThisForm.Grd_Produto.Column1.SetFocus
		EndIf 			
		Return(.F.)
	EndIf
	
	loBarra = CreateObject('fwprogressbar','Atualizando os Preços...',Reccount('CsProdutos'))
	loBarra.Show

	Select CsProdutos
	Go Top
	Scan
		loBarra.Update(.T.,'Produto: ' + CsProdutos.CPros)
		If Seek(CsProdutos.CPros,'CrSigCdPro','CPros')
			Select CrSigCdPro
			Scatter MemVar Memo
			m.ImpEtiqs = llImpEtiq
			Insert Into GrSigCdPro From Memvar
			
			lcSql = [Select * From SigCdPro Where Cpros = ']+m.cpros+[']
			If ThisForm.Podatamgr.Sqlexecute(lcSql,'TmpPro2') < 1
				If Not ThisForm.Automatico
					=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpPro2)')
				EndIf 
				Return .F.
			EndIf
			Select TmpPro2
			Scatter Memvar Memo
			m.DataAlts  = Datetime()
			m.HoraAlts  = Substr(Ttoc(m.DataAlts),12,8)
			m.UsuaAlts  = Usuar
			m.cIdChaves = fUniqueIds()
			m.Origem    = Ttoc(Datetime()) + [ SigPrCcp]
			Insert Into CrSigCdPrc From MemVar

			If ThisForm.poDataMgr.SqlExecute([Select * From SigPrCpo Where CPros = ']+m.CPros+[' ],'TmpCompo') < 1
				If Not ThisForm.Automatico
					MessageBox('Favor reinicializar o processo !!!',16,'Falha na Conexăo (TmpCompo)')
				EndIf 
				Return(.F.)
			EndIf
			Select TmpCompo
			Go Top
			Scan
				Scatter MemVar Memo
				m.DataAlts = Datetime()
				m.HoraAlts = Substr(Ttoc(m.DataAlts),12,8)
				m.UsuaAlts = Usuar
				m.cIdChaves = fUniqueIds()
				Insert Into CrSigPrCp2 From MemVar
			EndScan

			If ThisForm.poDataMgr.SqlExecute([Delete From SigPrPrt Where CPros = '] + m.CPros + [' ], []) < 1
				If Not ThisForm.Automatico
					MessageBox('Favor reinicializar o processo !!!',16,'Falha na Conexăo (TmpProt)')
				EndIf 
				Return(.F.)
			EndIf
			
			Select TmpPrCpo
			Set Order To Cpros
			=Seek(m.cpros)
			Scan While Cpros = m.Cpros
				Scatter Memvar Memo
				Insert Into GrSigPrCpo From Memvar
			EndScan
		EndIf
		Select CsProdutos
	EndScan
	loBarra.Complete(.T.)
	
	Wait Clear

	llOk  = .F.
	lnArq = 1

	loBarraFim = CreateObject('fwprogressbar','Atualizando Fisicamente os Arquivos...',5)
	loBarraFim.Show
	
	If  crSigCdPac.nChkSubGrs = 1
		Select grSigCdPro
		Scan
			lcQrySGru  = [Select * From SigCdPsg Where CGrus = ']+grSigCdPro.cGrus+[' Order By nFaixaFins]
			If (ThisForm.poDatamgr.SqlExecute(lcQrySGru,'csSigCdPsg') < 1)
				If Not ThisForm.Automatico
					MessageBox('Favor reinicializar o processo !!!',16,'Falha na Conexăo (csSigCdPsg)')
				EndIf 
				Return(.F.)
			EndIf
			Select csSigCdPsg
			Go Top
			If crSigCdPro.pVens = 0
				lnPVens = grSigCdPro.pvideals
				Locate for nFaixaFins >= lnPVens
				If ! Eof()
					Select grSigCdPro
					Replace sGrus With csSigCdPsg.Codigos
				EndIf
			Else
				lnPVens = grSigCdPro.pVens 
				Locate For nFaixaFins >= lnPVens
				If ! Eof()
					Select grSigCdPro
					Replace sGrus With csSigCdPsg.Codigos
				EndIf
			EndIf
		EndScan
	EndIf 

	loBarraFim.Update(.T.,'SigCdPro (1/4)...')
	llOk = ThisForm.poDataMgr.Update('grSigCdPro')
	If llOk
		lnArq = 2
		loBarraFim.Update(.T.,'SigCdPrc (2/4)...')
		llOk = ThisForm.poDataMgr.Update('CrSigCdPrc')
	EndIf
	If llOk
		lnArq = 3
		loBarraFim.Update(.T.,'SigPrCp2 (3/4)...')
		llOk = ThisForm.poDataMgr.Update('CrSigPrCp2')
	EndIf
	If llOk
		lnArq = 4
		loBarraFim.Update(.T.,'SigPrCpo (4/4)...')
		llOk = ThisForm.poDataMgr.Update('grSigPrCpo')
	EndIf
	loBarraFim.Complete(.T.)
	
	If llOk
		ThisForm.poDataMgr.Commit()
		If Not ThisForm.Automatico
			MessageBox('Processamento Finalizado com Sucesso !!!',64,'')
		EndIf 
		llReturn = .T.
	Else
		ThisForm.poDataMgr.RollBack()
		If Not ThisForm.Automatico
			MessageBox('Falha na Atualizaçăo. Reinicie o Processo !!!',48,'Arquivo: ' + Transform(lnArq,'@L 99'))
		EndIf 
		llReturn = .F.
	EndIf
	Zap In CrSigCdPro
	Zap In CrSigCdPrc
	Zap In CrSigPrCp2
	Zap In CrProdutos
	Zap In grSigCdPro
	Zap In grSigPrCpo
EndIf

This.Enabled = .F.
This.Refresh

Return(llReturn)
ENDPROC

PROCEDURE Processa.Click
If Thisform.Automatico
	=ThisForm.ProcessaAutomatico()
Else

	Select CrProdutos
	If Reccount() > 0
		If Messagebox('Existem Dados Gerados. Deseja Reprocessar? ',4+32,'') = 7
			Return .F.
		Endif
	Endif
	Zap In CrProdutos

	If Thisform.Processar()

		Select CrProdutos
		lnVaria = Thisform.Get_Variacao.Value
		If lnVaria > 0
			Delete For PVarias < lnVaria
		Endif
		If lnVaria < 0
			Delete For PVarias > lnVaria
		Endif
		Set Order To CPros
		Go Top

		This.Parent.Atualiza.Enabled = .T.
		This.Parent.Atualiza.Refresh
		Thisform.Impressăo.Enabled = .T.
		Thisform.Impressăo.Refresh

		Thisform.Grd_Produto.Column1.SetFocus
		Thisform.Grd_Produto.Refresh

	Endif

Endif



Return(.T.)

ENDPROC
