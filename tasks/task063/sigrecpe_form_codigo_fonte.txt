------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
local cNomeEmpresa
local cRelTitulo
local cGrest,cEstoque,cGrupo,cFornecedor,cDescricao,lDescricao
local lcTalk,lcEscape,lnDecimals,lcFixed,lnMouse,lcMacro1,lcMacro2,lnCestmin

cGrupoA    = Padr(thisform.get_cd_grestoque.value,10)
cEstoqA    = Padr(thisform.get_cd_estoque.value,10)
cGrupoB    = Padr(ThisForm.Get_Grupob.Value,10)
cEstoqB    = Padr(ThisForm.Get_ContaB.Value,10)
lnTipo     = ThisForm.Opt_Tipo.Value
lnOpt      = ThisForm.Opc_Situacao.Value
lnOpt      = ThisForm.Opt_CorTam.Value

cRelTitulo   = 'Divergęncias entre Estoque'
cRelTitulo2  = 'Estoque A :'+cGrupoA + '/'+cEstoqA + ' - ' + 'Estoque B :'+cGrupoB + '/'+cEstoqB

lcSql = [Select Razas From SigCdEmp Where Cemps = ']+_Empr+[']
thisform.podatamgr.sqlexecute(lcsql,'TmpEmp')

cNomeEmpresa = Tmpemp.razas

create cursor cscabecalho (NomeEmpresa c(80), RelTitulo c(80), RelTitulo2 c(80))
insert into cscabecalho (NomeEmpresa,RelTitulo, RelTitulo2) values (cNomeEmpresa,cRelTitulo,cRelTitulo2)

IF ThisForm.Processar
	lcSql = [Select * From SigMvEst Where Emps = ']+_Empr+[' And Sqtds > 0 ]
	ThisForm.Podatamgr.Sqlexecute(lcSql,'LocalEst')
	lcSql = [Select Cpros,cproEqs,Reffs,cgrus,dpros,CGrus,Cunis,Linhas,Colecoes,situas From SigCdPro ]
	ThisForm.Podatamgr.Sqlexecute(lcSql,'LocalPro')
	ThisForm.Processar = .f.
ENDIF

If lnTipo = 1
	If lnOpt = 1
		Select a.*,b.Dpros From LocalEst a, LocalPro b Where a.Grupos = cGrupoA And a.Estos = cEstoqA And ;
		a.Emps = _Empr And a.Sqtds > 0 And a.Cpros+a.CodCors+a.CodTams Not In ;
		(Select Cpros+CodCors+CodTams From LocalEst Where Grupos = cGrupoB ;
		And Estos = cEstoqB And Emps=_Empr And sqtds > 0 ) And a.Cpros = b.Cpros ;
		Into Cursor CsRelatorio Order by a.Cpros,a.CodCors,a.CodTams
	Else
		Select a.*,b.Dpros From LocalEst a, LocalPro b Where a.Grupos = cGrupoA And a.Estos = cEstoqA And ;
		a.Emps = _Empr And a.Sqtds > 0 And a.Cpros Not In ;
		(Select Cpros From LocalEst Where Grupos = cGrupoB ;
		And Estos = cEstoqB And Emps=_Empr And sqtds > 0 ) And a.Cpros = b.Cpros ;
		Into Cursor CsRelatorio Order by a.Cpros,a.CodCors,a.CodTams
	EndIf
Else

	lcGrui   = ThisForm.GetCGrui.Value
	lcGruf   = ThisForm.GetCGruf.Value
	lcCUnii  = ThisForm.GetCUnii.Value
	lcCUnif  = ThisForm.GetCUnif.Value
	lcLini   = ThisForm.GetLini.Value
	lcLinf   = ThisForm.GetLinf.Value
	lcColi   = ThisForm.GetColi.Value
	lcColf   = ThisForm.GetColf.Value

	Declare laCampo[4]
	laCampo[1] = 'CGrus'
	laCampo[2] = 'Cunis'
	laCampo[3] = 'Linhas'
	laCampo[4] = 'Colecoes'

	Declare laVarias[4]
	laVarias[1] = 'lcGru'
	laVarias[2] = 'lcCuni'
	laVarias[3] = 'lcLin'
	laVarias[4] = 'lcCol'

	lcWhere = '.T.'
	For lnConta = 1 to 4
		lcInicio = Eval( laVarias[ lnConta ] + 'i' ) 
		lcFinal  = Eval( laVarias[ lnConta ] + 'f' )
		lcWhere  = lcWhere + ' And '
		If Empty( lcInicio )
			If Empty( lcFinal )
				lcWhere = lcWhere + ' .T. '
			Else
				lcWhere = lcWhere + laCampo[ lnConta ] + ' <= ' + ["] + lcFinal + ["]
			Endif
		Else
			If Empty( lcFinal )
				lcWhere = lcWhere + laCampo[ lnConta ] + ' >= ' + ["] + lcInicio + ["]
			Else
				lcWhere = lcWhere + laCampo[ lnConta ] + ' Between ' +;
						 ["] + lcInicio + ["] + ' And ' + ["] + lcFinal + ["]
			Endif
		Endif
	Next

	With ThisForm.Opc_Situacao
		If Inlist( .Value , 1, 2 )
			lcWhere = lcWhere + ' And Situas = ' + Str( .Value, 1 )
		Endif
	Endwith

	Select Cpros,cproEqs,Reffs,cgrus,dpros,'    ' as CodCors, '  ' as CodTams, 0 as Sqtds From LocalPro Where &lcWhere. ;
	Into Cursor TmpPro2

	Create Cursor TmpPro (Cpros c(10), cproeqs c(10), dpros c(40), CodCors c(4), CodTams c(2), Sqtds n(12,2), Reffs c(40), Reff2s c(40) )
	Index On Cpros Tag Cpros
	Append From Dbf('Tmppro2')
	
	Select Cpros,cproEqs,reffs,cgrus,dpros,' ' as CodCors, ' ' as CodTams, 0 as Sqtds From LocalPro Where ;
	cProEqs <> ' ' And CproEqs in (Select Cpros From TmpPro) Into Cursor TmpPro2
	
	Select TmpPro2
	Scan
		If Seek(cproEqs,'TmpPro','Cpros')
			Replace TmpPro.CproEqs With TmpPro2.Cpros,;
					TmpPro.Reff2s  With TmpPro2.Reffs in TmpPro
		EndIf 
	EndScan

	If ThisForm.Opt_reff.Value = 1
		Select Reffs, Max(Reff2s) as reff2s From TmpPro Group by 1 Into Cursor TmpProx
Suspend
		
		Select distinct Padr(reffs,10) as cpros, Space(40) as dpros, ' ' as codcors, ' 'as codtams, 0 as sqtds ;
		  From TmpProx Where Reffs Not In ;
		(Select b.Reffs From LocalEst a, LocalPro b Where a.Grupos = cGrupoB And a.Estos = cEstoqB And a.Emps = _Empr And a.Sqtds > 0 and a.cpros = b.cpros ) ;
		And (reff2s = ' ' Or Reff2s Not in ;
		(Select b.reffs From LocalEst a, LocalPro b Where a.Grupos = cGrupoB And a.Estos = cEstoqB And a.Emps = _Empr And a.Sqtds > 0 and a.cpros = b.cpros ) ) ;
		Into cursor CsRelatorio Order by 1
	Else
		Select * From TmpPro Where Cpros Not In ;
		(Select cpros From LocalEst Where Grupos = cGrupoB And Estos = cEstoqB And Emps = _Empr And Sqtds > 0 ) ;
		And (CproEqs = ' ' Or CproEqs Not in ;
		(Select cpros From LocalEst Where Grupos = cGrupoB And Estos = cEstoqB And Emps = _Empr And Sqtds > 0 ) ) ;
		Into cursor CsRelatorio Order by Cpros
	EndIf
EndIf

m.lcEscape   = Set('Escape')
m.lcTalk     = Set('Talk')
m.lnDecimals = Set('Decimals')
m.lcFixed    = Set('Fixed')
m.lnMouse    = ThisForm.MousePointer

ThisForm.MousePointer = 11

Set Escape Off
Set Talk On
Set Decimals To 2
Set Fixed On

ThisForm.MousePointer = m.lnMouse

Set Talk Off

Set Escape &lcEscape.
Set Talk &lcTalk.
Set Decimals To m.lnDecimals
Set Fixed &lcFixed.

Select CsRelatorio
GoTo Top

ENDPROC
