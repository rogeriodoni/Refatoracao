PROCEDURE procetiqf
LParameters _Saida

Private pTru, pIds

_Pedido = (ThisForm.Relaçăo.Get_Pedido.Value = 1)

_SeqF   = Iif(ThisForm.Get_SeqFim.Value = 0, 9999, ThisForm.Get_SeqFim.Value)
_Divid  = (ThisForm.Etiquetas.Get_Divididas.Value = 1)
_Geral  = (ThisForm.Etiquetas.Get_Imprimir.Value = 1)
_ImpCmp = (ThisForm.Etiquetas.Get_Compo.Value = 1)
_llCera = (ThisForm.Etiquetas.Get_Cera.Value = 1 )


Create Cursor TmpImp (Ordem n(6), Seqs n(4), Nops n(10), CPros c(14), DPros c(40), CUnis c(3), Qtde n(10,3), ;
					  Peso n(9,2), Obs m, Pedido c(13), Datas d, CodBar l, ImpCodBar c(20), CMats c(14), DMats c(50), ;
					  QPedras n(12,3), QConsumo n(12,2), OrdImp n(1), Detalhe1 L, Detalhe2 L, Detalhe3 L, ;
					  Detalhe4 L, Dividida L, Ultimo L, Primeiro L, Quebra L, Ordimg c(17), CodCli c(10), ;
					  DtEnt d, Linhas c(40), dGrupos c(20), CodCors c(4), CodTams c(4), DesCors c(30), ;
					  Numps n(10), Usuars c(10), Reffs c(40), TpInstals c(1), CorPads c(25), Ceras l, ;
					  nFunds n(6),DtFunds d, DataEs d,Idnfs c(40))
Index On Str(Nops, 10) + Str(OrdImp, 1)Tag NopOrdem
Index On Nops Tag Nops
Set Order to Nops

*********   Criaçăo da Imagem em Branco Para o Relatório   ***********************************
lcBrancos = Addbs(JustPath(ThisForm.poDataMgr.GetDatabase())) + 'BrJpg.JPG'
Create Cursor crBranco (Branco M)
Select crBranco
Append Blank
Append Memo Branco From (lcBrancos) Overwrite

=StrToFile(lcFoto, Sys(2023)+'\FBRANCOS.JPG')


If (ThisForm.TpOrdem.Value = 1)
	lcOperacao = crSigCdPam.DoppPads
Else
	lcOperacao = crSigCdPam.DoppServs
EndIf

ThisForm.poDataMgr.CursorQuery('SigCdOpd', 'crTmpCdOpd', 'Dopps', lcOperacao,'Numeras')

							   ['] + _Empr + lcOperacao + [9999999999']
EndIf
_Nops = 0

lcQuery = [Select * ] + ;
		    [From SigOpPic ] + ;
		   [Where ] + _Compara + [ And ] + ;
		   		 Iif(llNumOpe,[0=0 And ],[Nops BetWeen ?pSqi And ?pSqf And ]) + ;

	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TempOpi)')
	Return 0

	_Seq = Val(Right(Str(TempOpi.Nops, 10), 4))
	_Emp = Iif(Empty(Empds), Emps, Empds)

	If (Not _Geral And Imprs ) Or (_Divid And Not Divs)
		Loop
	EndIf

	lcEdp = Emps + Dopps + Str(Nops, 10)
	lcEdn = _Emp + TempOpi.Dopes + Str(TempOpi.Numes, 6)
	ThisForm.poDataMgr.CursorQuery('SigPdMvf', 'TempMfas',  'EmpDNps', lcEdp)
	ThisForm.poDataMgr.CursorQuery('SigCdPro',  'crSigCdPro',  'CPros', TempMfas.CodPds)
	ThisForm.poDataMgr.CursorQuery('SigCdOpd',  'crSigCdOpd',  'Dopps', TempOpi.Dopps)
	ThisForm.poDataMgr.CursorQuery('SigCdOpe',  'crSigCdOpe',  'Dopes', TempOpi.Dopes)
	ThisForm.poDataMgr.CursorQuery('SigCdCor',  'crSigCdCor',  'Cods',  TempOpi.CodCors)
	ThisForm.poDataMgr.CursorQuery('SigCdCor',  'crTmpCor',  'Cods',  CrSigCdPro.CodCors)
	ThisForm.poDataMgr.CursorQuery('SigMvCab', 'crSigMvCab', 'EmpDopNums', lcEdn)
	ThisForm.poDataMgr.CursorQuery('SigCdLin', 'crSigCdLin', 'Linhas', CrSigCdPro.Linhas)
	
	If _llCera And CrSigCdPro.CravCers <> 1
		Loop
	Endif

	_cPedido = TempOpi.Emps + Alltrim(crSigCdOpe.Abrevs ) + ' ' + Alltrim(fGerMascara(TempOpi.Numes))
	_Peso = Round(TempOpi.Qtds * TempOpi.Pesos, 2) && 23/09/2010 - Tava crSigCdPro.Pesoms

	If (_Nops <> Nops)
		_ImpCodBar = fAltBar2de5(Transform(crSigCdOpd.NDopps, '@L 99') + Transform(TempOpi.Nops, '@L 9999999999'))
		_Nops = Nops
		Select TmpImp 
		If Not Eof()
			Replace Ultimo with .t.
		EndIf

		Insert Into TmpImp (Nops, Detalhe2, OrdImp, TpInstals ) Values (_Nops, .t., 2, [])
		_Primeiro = .t.
	Else
		_ImpCodBar = ' '
		_Primeiro = .f.
	EndIf
	
	Insert Into TmpImp (Ordem, Seqs, Cpros, Dpros, CUnis, Qtde, Obs, Pedido, ImpCodBar, Detalhe1, Nops, Datas, Peso, ;
						Dividida, OrdImp, Primeiro, CodCli, DtEnt, Linhas, CodCors, CodTams, DesCors, Numps, Usuars, ;
						Reffs, TpInstals, CorPads, CodBar ) ;
			    Values (_Nop, _Seq, TempMFas.CodPds, crSigCdPro.DPros, crSigCdPro.CUnis, TempOpi.Qtds, Nvl(TempOpi.Obss, ''), ;
			    		_cPedido, _ImpCodBar, .t., TempOpi.Nops, Nvl(TempOpi.Dataes, Ctod('')), _Peso, TempOpi.Divs, ;
			    		1, _Primeiro, crSigMvCab.ContaDs, Nvl(crSigMvCab.PrazoEnts, Ctod('')), Alltrim(crSigCdPro.Linhas)+'-'+ALLTRIM(CrSigCdLin.descs), ;
			    		TempOpi.CodCors, TempOpi.CodTams, crSigCdCor.Descs, TempOpi.Numps, TempOpi.Usuars, ;
			    		CrSigCdPro.Reffs, [], Alltrim(CrSigCdPro.CodCors)+'-'+Alltrim(CrTmpCor.Descs), .t.)

	If Empty(crSigCdPro.FigJpgs) Or IsNull(crSigCdPro.FigJpgs)
		Replace OrdImg With 'FBRANCOS' In TmpImp
	Else
		lcCodigo = Strtran(Strtran(Strtran(TempMfas.CodPds,'/',''),'*',''),' ','')
		lcDirectory = Sys(2023)+'\F'+Alltrim(lcCodigo)+'.Jpg'
		Replace OrdImg With 'F' + Alltrim(lcCodigo) In TmpImp
		lcFoto=Strconv(Strtran(Strtran(Strtran(crSigCdPro.FigJpgs,[data:image/png;base64,],[]),[data:image/jpeg;base64,],[]),[data:image/jpg;base64,],[]),14)
		=StrToFile(lcFoto, lcDirectory)
	EndIf

	If (_Saida = 'I')
		pTru = .t.
		pIds = TempOpi.cIdChaves
		 
		lcQuery = [Update SigOpPic Set Imprs = ?pTru Where cIdChaves = ?pIds]
		If (ThisForm.poDataMgr.SqlExecute(lcQuery, '') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (Update - Proctiraf 1)')
			Return 0
		EndIf
	EndIf
EndScan

Select TmpImp
Replace Ultimo with .t.
Set Order to Nops
Scan
	Nop  = Nops
	nPeso = 0
	Sum Peso to nPeso While Nops = Nop
	Seek Nop
	Replace Peso With nPeso While Nops = Nop
	Skip -1
EndScan
	
Select Nops, CPros, Sum(Qtde) as Qtde, Numps ;
  From TmpImp ;
 Group By 1, 2, 4 ;
  Into Cursor TmpCompo

Select TmpCompo
Scan
	_Primeiro = .t.

	ThisForm.poDataMgr.CursorQuery('SigPrCpo', 'crSigPrCpo', 'CPros', TmpCompo.CPros)
	Select crSigPrCpo
	Scan
		ThisForm.poDataMgr.CursorQuery('SigCdPro', 'crSigCdPro', 'CPros', crSigPrCpo.Mats)
		ThisForm.poDataMgr.CursorQuery('SigCdGrp', 'crSigCdGrp', 'CGrus', crSigCdPro.CGrus)

		If (crSigCdGrp.CEstoqs # 1 And CrSigCdGrp.ImpOps <> 3) Or Not InList(CrSigCdGrp.ImpOps,1,3) Or Not _ImpCmp
			Loop
		Endif

		Insert Into TmpImp (Nops, CMats, DMats, QPedras, Detalhe3, OrdImp, Primeiro, QConsumo, Numps, TpInstals, cunis ) ;
					Values (TmpCompo.Nops, crSigPrCpo.Mats, crSigCdPro.DPros, crSigPrCpo.Qtds * TmpCompo.Qtde, ;
							.t., 3, _Primeiro, crSigPrCpo.QtsCons * TmpCompo.Qtde, TmpCompo.Numps, [], crSigCdPro.Cunis )
		_Primeiro = .f.
	EndScan
EndScan

*** Tiago - 11/01/2012 - Vianna - Alterei para sair apenas uma única vez os processos, e năo a cada pedido que contém na tira  
Select TmpImp 	
Go top
Select Nops, CPros, Sum(Qtde) as Qtde ;
  From TmpImp where !Empty(cpros) ;
 Group By 1, 2 order by Nops ;
  Into Cursor TmpCompo

Select TmpCompo
Scan
	ThisForm.poDataMgr.CursorQuery('SigCdPrf', 'crSigCdPrf', 'Produtos', TmpCompo.CPros,,'Ordems')

	Select crSigCdPrf
	Scan
		If IsEmpty(Obs)
			Loop
		EndIf

		ThisForm.poDataMgr.CursorQuery('SigCdGcr', 'crSigCdGcr', 'Codigos', crSigCdPrf.Grupos)

		Insert Into TmpImp (Nops, DGrupos, Detalhe4, OrdImp, Obs, TpInstals ) ;
				    Values (TmpCompo.Nops, crSigCdGcr.Descrs, .t., 4, Nvl(crSigCdPrf.Obs, ''), [])
	EndScan
EndScan

Select TmpImp
Set Order to NopOrdem
Go Top
_Nops = Nops
Scan
	If (Nops <> _Nops)
		Skip -1
		Replace Quebra with .t.
		Skip +1
		_Nops = Nops
	EndIf
EndScan

Go Bottom
Replace Quebra with .t.
Go Top
If Eof()
	=Messagebox('Nenhuma Seqüęncia Foi Selecionada Para Impressăo!!!', 32, '')
Else
	If (_Saida = 'V')
		Report Form SigReGlL Preview Noconsole
	Else
		Report Form SigReGlL to Print Prompt Noconsole
	EndIf
EndIf

lcApagar = Sys(2023) + '\F*.Jpg'
Delete File &lcApagar.

Use In TmpImp

If (_Saida = 'I')
	ThisForm.Podatamgr.commit()
Endif
ENDPROC

					If lnQt > 7 Or lnQt2 = 2

		Endif
	EndIf
	_ObsPed = _ObsPed + _cPedido + '/'
	_CntLnhO = _CntLnhO + 1

	&& 23/09/2010 - Tava crSigCdPro.Pesoms

			_Obs = _Obs + MLine( TempOpi.Obss, _xLine ) + ' '
		EndIf
