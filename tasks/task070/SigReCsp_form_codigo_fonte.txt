* PROCEDURE processamento

PROCEDURE processamento
lcempresa = Thisform.get_Empresa.Value
lcNmOperacao = Thisform.get_nm_operacao.Value
lccodigo = Thisform.get_codigo.Value
lcdoc = Thisform.get_documento.Value
ldDtInicial = Thisform.get_dt_inicial.Value
ldDtFinal = Thisform.get_dt_final.Value

lcObses = ThisForm.get_Obses.Value
lcConta = ThisForm.getConta.Value
lcGrupo = ThisForm.getGrupo.Value
lcOpCta = Iif(ThisForm.OptConta.Value = 1,[ContaOs],[ContaDs])
lcOpGru = Iif(ThisForm.OptConta.Value = 1,[GrupoOs],[GrupoDs])

lcFiltro = []
If Not Empty(lcGrupo)
	lcFiltro = lcFiltro + [ And A.] + Alltrim(lcOpGru) + [ = '] + lcGrupo + [' ]
EndIf 
If Not Empty(lcConta)
	lcFiltro = lcFiltro + [ And A.] + Alltrim(lcOpCta) + [ = '] + lcConta + [' ]
EndIf 

lcquery = [Select emps,dopes,numes,datas,notas,datars,empdopnums,operadors,usulibs,IsNull(convert(varchar(250),Obses),'') as Obses From SigMvCab a ] +;
	[Where datas BetWeen ?ldDtInicial And ?ldDtFinal ]+;
