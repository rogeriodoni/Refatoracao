------------------------------------------------------------
PROCEDURE baixatotal
Lparameters vCancela

Select TmpTot
ThisForm.CalcTotal()

Select TmpTot
Delete All For ValorPend = 0 And ValorAce = 0

Select TmpConta
_RegisConta = 0
If !Eof()
	_RegisConta = Recno()
Endif
Set Order to 
Replace All Baixa With 0

Select xPar
_RecnoxPar = 0
If !Eof()
	_RecnoxPar = Recno()
Endif
Replace all vpags with 0

Select TmpBaixa
Dele all for !autos
_Autos = .f.
Scan for Autos
	_Autos = .t.
	Exit
Endscan

*** Tiago - 17/12/2012 - Limpa a observaçăo da antecipaçăo será gerado automaticamente com os dias de antecipaçăo
ldDiasAntec = {}
replace Obs with [] in crSigCdPgr		

Select TmpConta
Set Order to CrVencs
Scan For Flag
	_RecTmpConta = Recno()
	_Dopecs = Dopes
	_Numecs = Numes
	_Grupocs = Grupos
	_Contacs = Contas
	_Nopercs = Nopers
	_Empcs = Emps
	_Tipocs = Tipos
	mpend	= Moedas
	cotforma = Cotacaos
	vlpend	= Acerto - BaixaAut
	If vlpend <= 0
		loop
	Endif
	Set Order to DbVencs
	Scan for Moedas = mpend And Flag && Primeiro baixa na moeda igual
		Do Case
			Case Baixa + BaixaAut = Acerto
				VlUsado = 0

			Case Baixa + BaixaAut + VlPend > Acerto
				VlUsado = Acerto - Baixa - BaixaAut

			Case Baixa + BaixaAut + VlPend <= Acerto
				VlUsado = VlPend

		EndCase
		Replace Baixa With Baixa + VlUsado 

		If VlUsado > 0
			Insert Into TmpBaixa ( Empos, Nopers, Tipos, Valos, Moedas, Cotacaos, Grupos, Contas,;
									Dopecs, Numecs, Grupocs, Contacs, Moecs, Nopercs, Tipocs, Valocs, Cotacaocs, Empcs );
						Values ( TmpConta.Emps, TmpConta.Nopers, TmpConta.Tipos,;
								VlUsado, mPend, cotforma, TmpConta.Grupos, TmpConta.Contas,;
								_Dopecs, _Numecs, _Grupocs, _Contacs, mPend, _Nopercs, _Tipocs, VlUsado, cotforma, _Empcs )

			VlPend = VlPend - VlUsado

			Select TmpTot
			If !Seek( mPend )
				Insert Into TmpTot ( Moeda ) Values ( mPend )
			Endif
			Replace ValorAce With ValorAce + VlUsado
		Endif
	Endscan

	If VlPend > 0  && Sobrou dinheiro, procura em outras moedas
		Select TmpConta 
		Scan for Moedas # mPend And Flag
			cottotal = TmpConta.Cotacaos
			VlConv = VlPend / cottotal * cotforma
			Do Case
				Case Baixa + BaixaAut = Acerto
					VlUsado = 0

				Case Baixa + BaixaAut + Round( VlConv, 4 ) > Acerto
					VlUsado = Acerto - Baixa - BaixaAut

				Case Baixa + BaixaAut + Round( VlConv, 4 ) <= Acerto
					VlUsado = Round( VlConv, 4 )

			EndCase
			Replace Baixa With Baixa + Round( VlUsado, 4 )

			If VlUsado > 0
				_VlCred = VlUsado / cotforma * cottotal
				Insert Into TmpBaixa ( Empos, Nopers, Tipos, Valos, Moedas, Cotacaos, Grupos, Contas,;
										Dopecs, Numecs, Grupocs, Contacs, Moecs, Nopercs, Tipocs,;
										Valocs, Cotacaocs, Empcs ) ;
							Values ( TmpConta.Emps, TmpConta.Nopers, TmpConta.Tipos,;
									VlUsado, TmpConta.Moedas, cottotal, TmpConta.Grupos,;
									TmpConta.Contas, _Dopecs, _Numecs, _Grupocs,;
									_Contacs, mPend, _Nopercs, _Tipocs, _VlCred, cotforma, _Empcs )
				VlConv = VlConv - Round( VlUsado, 4 )

				Select TmpTot
				If !Seek( TmpConta.Moedas )
					Insert Into TmpTot ( Moeda ) Values ( TmpConta.Moedas )
				Endif
				Replace ValorAce With ValorAce + Round( VlUsado, 4 )

				VlPend = VlConv / cotforma * cottotal
			Endif
		Endscan
	Endif

	Select TmpConta
	Set Order to CrVencs
	Go _RecTmpConta

	If VlPend > 0
		Select TmpTot
		If !Seek( TmpConta.Moedas )
			Insert Into TmpTot ( Moeda ) Values ( TmpConta.Moedas )
		Endif
		Replace ValorAce With ValorAce + Round( VlPend, 4 )
	Endif

	*** Tiago - 17/12/2012 - Dias de antecipaçăo do título com maior vencimento
	ldDiasAntec = Max(TmpConta.Vencs,ldDiasAntec)
	ndias = ldDiasAntec - Date()
	replace Obs with Iif(!Empty(ndias),[Dias antecipaçăo: ] + Str(ndias),[]) in crSigCdPgr		
Endscan

Select xPar
Set Order to 
Scan
	=ThisForm.PoDatamgr.CursorQuery('SigOpFp','CrFpag','Fpags',xPar.FPags)
	If crFPag.Trocos = 1
		Loop
	Endif
	mpend	= crFPag.Moefpgs
	cotpend = xPar.CotFpgs
	vlpend	= xPar.Valos
	Select TmpConta 
	Set Order to DbVencs
	Scan for Moedas = crFPag.Moefpgs And Flag && Primeiro baixa na moeda igual
		Do Case
			Case Baixa + BaixaAut = Acerto
				VlUsado = 0

			Case Baixa + BaixaAut + VlPend > Acerto
				VlUsado = Acerto - Baixa - BaixaAut

			Case Baixa + BaixaAut + VlPend <= Acerto
				VlUsado = VlPend

		EndCase
		Replace Baixa With Baixa + VlUsado 

		If VlUsado > 0
			Insert Into TmpBaixa ( Empos, Nopers, Tipos, Valos, Moedas, Grupos, Contas,;
									FPags, Bancos, Agencias, NContas, NCheques, Cotacaos ) ;
						Values ( TmpConta.Emps, TmpConta.Nopers, TmpConta.Tipos,;
								VlUsado, mPend, TmpConta.Grupos,;
								TmpConta.Contas, xPar.FPags, xPar.Bancos,;
								xPar.Agencias, xPar.Contas, xPar.Numeros, TmpConta.Cotacaos )
			VlPend = VlPend - VlUsado

			Select TmpTot
			If !Seek( mPend )
				Insert Into TmpTot ( Moeda ) Values ( mPend )
			Endif
			Replace ValorAce With ValorAce + VlUsado
			Replace xPar.Vpags with xPar.VPags + VlUsado
		Endif
	Endscan

	If VlPend > 0 && Sobrou dinheiro, procura em outras moedas
		Select TmpConta 
		Scan for Moedas # crFPag.Moefpgs And Flag
			cotforma = cotpend
			cottotal = TmpConta.Cotacaos
			VlConv = VlPend / cottotal * cotforma
			Do Case
				Case Baixa + BaixaAut = Acerto
					VlUsado = 0

				Case Baixa + BaixaAut + Round( VlConv, 4 ) > Acerto
					VlUsado = Acerto - Baixa - BaixaAut

				Case Baixa + BaixaAut + Round( VlConv, 4 ) <= Acerto
					VlUsado = Round( VlConv, 4 )

			EndCase
			Replace Baixa With Baixa + Round( VlUsado, 4 )

			If VlUsado > 0
				Insert Into TmpBaixa ( Empos, Nopers, Tipos, Valos, Moedas, Grupos, Contas,;
										FPags, Bancos, Agencias, NContas, NCheques, Cotacaos ) ;
							Values ( TmpConta.Emps, TmpConta.Nopers, TmpConta.Tipos,;
									VlUsado, TmpConta.Moedas, TmpConta.Grupos,;
									TmpConta.Contas, xPar.FPags, xPar.Bancos,;
									xPar.Agencias, xPar.Contas, xPar.Numeros, TmpConta.Cotacaos )
				VlConv = VlConv - Round( VlUsado, 4 )

				Select TmpTot
				If !Seek( TmpConta.Moedas )
					Insert Into TmpTot ( Moeda ) Values ( TmpConta.Moedas )
				Endif
				Replace ValorAce With ValorAce + Round( VlUsado, 4 )

				VlPend = VlConv / cotforma * cottotal
			Endif
		Endscan

		Select xPar
		Replace xPar.Vpags with xPar.Valos - Round( VlPend, 2 )
	Endif
Endscan
*****************************************
***  TRATAMENTO DO TROCO  ***
*****************************************
Select xPar
Scan
	=ThisForm.PoDatamgr.CursorQuery('SigOpFp','CrFpag','Fpags',xPar.FPags)
	If crFPag.Trocos # 1
		Loop
	EndIf
	Select xPar
	mTroco = xPar.MoeFpgs
	vTroco = xPar.Valos - xPar.VPags
	cTroco = xPar.CotFpgs
	_RecPar = Recno()
	Select xPar
	Scan
		=ThisForm.PoDatamgr.CursorQuery('SigOpFp','CrFpag','Fpags',xPar.FPags)
		Select xpar
		If crFPag.Trocos = 1
			Loop
		Endif
		If xPar.VPags = xPar.Valos
			Loop
		Endif
		If MoeFpgs # mTroco
			VlPend = vTroco * cTroco / xPar.CotFpgs
		Else
			VlPend = vTroco
		Endif

		Do Case
			Case xPar.VPags + Round( VlPend, 2 ) > xPar.Valos
				VlUsado = xPar.Valos - xPar.VPags
			Case xPar.VPags + Round( VlPend, 2 ) <= xPar.Valos
				VlUsado = Round( VlPend, 2 )
		EndCase

		Replace xPar.VPags With xPar.VPags + VlUsado
		VlPend = VlPend - VlUsado
		Select TmpTot
		If !Seek( xPar.MoeFpgs )
			Insert Into TmpTot ( Moeda ) Values ( xPar.MoeFpgs )
		Endif
		Insert Into TmpBaixa ( Empos, Nopers, Tipos, Valos, Moedas, Grupos, Contas, FPags, ;
							   Bancos, Agencias, NContas, NCheques, Cotacaos ) ;
				Values ( crSigCdPgr.Emps, 0, 'Y', VlUsado, xPar.MoeFpgs, crSigCdPam.GruAcertos,;
						 crSigCdPam.ConAcertos, xPar.FPags, xPar.Bancos, xPar.Agencias,;
						 xPar.Contas, xPar.Numeros, xPar.CotFpgs )
		vTroco = VlPend * xPar.CotFpgs / cTroco
		If vTroco = 0
			Exit
		Endif
	Endscan
	Select xPar
	Go _RecPar
	Replace vPags with Valos - vTroco
Endscan
**********************************************************
***  TRATAMENTO DO CRÉDITO PENDENTE  ***
**********************************************************
Select xPar
Scan
	=ThisForm.PoDatamgr.CursorQuery('SigOpFp','CrFpag','Fpags',xPar.FPags)
	Select xpar
	If crFPag.Trocos = 1
		Loop
	Endif
	If xPar.VPags # xPar.Valos
		mPend = xPar.MoeFpgs
		VlPend = xPar.Valos - xPar.VPags
		Select TmpTot
		If !Seek( mPend )
			Insert Into TmpTot ( Moeda ) Values ( mPend )
		Endif
		Replace ValorAce With ValorAce + VlPend
		If Abs( VlPend * xPar.CotFpgs ) <= crSigCdPam.CancCreds
			lcTipoCred = 'Y'
		Else
			lcTipoCred = 'X'
		Endif
		Insert Into TmpBaixa ( Empos, Nopers, Tipos, Valos, Moedas, Grupos, Contas,;
								FPags, Bancos, Agencias, NContas, NCheques, Cotacaos ) ;
					Values ( crSigCdPgr.Emps, 0, lcTipoCred, VlPend, mPend, Spac( 10 ) ,;
							Spac( 10 ), xPar.FPags, xPar.Bancos, xPar.Agencias,;
							xPar.Contas, xPar.Numeros, xPar.CotFpgs )
		Replace xPar.Vpags with xPar.Valos
	Endif
Endscan
**********************************************************
***  TRATAMENTO DO DÉBITO PENDENTE  ***
**********************************************************
Select TmpConta
Set Order to DbVencs
Scan for Flag 
	VlDifere = Acerto - Baixa - BaixaAut
	If VlDifere > 0
		If !Seek( TmpConta.Grupos + TmpConta.Contas + TmpConta.Moedas + ' ', 'TmpBaixa', 'GruConMoe' )
			Insert Into TmpBaixa ( Empos, Nopers, Valos, Moedas, Cotacaos, Grupos, Contas,;
									Grupocs, Contacs, Moecs, Valocs, Cotacaocs );
						Values ( TmpConta.Emps, TmpConta.Nopers,;
								VlDifere, TmpConta.Moedas, TmpConta.Cotacaos,;
								TmpConta.Grupos, TmpConta.Contas,;
								crSigCdPam.GruAcertos, crSigCdPam.ConAcertos,;
								TmpConta.Moedas, VlDifere, TmpConta.Cotacaos )
		Else
			Select TmpBaixa
			Replace Valos with Valos + VlDifere,;
					Valocs with Valocs + VlDifere
		Endif
		
		*** Tiago - 17/12/2012 - Dias de antecipaçăo do título com maior vencimento
		ldDiasAntec = Max(TmpConta.Vencs,ldDiasAntec)
		ndias = ldDiasAntec - Date()
		replace Obs with Iif(!Empty(ndias),[Dias antecipaçăo: ] + Str(ndias),[]) in crSigCdPgr		
	Endif
Endscan

lnTotal = 0
Select TmpTot
Scan
	If !Seek( Moeda, 'TmpCot' )
		m.Moedas = Moeda
		m.Cotacaos = fCarregarCambio( m.Moedas, crSigCdPgr.Datarcs )
		Insert Into TmpCot From Memvar
	Endif
	lnTotal = lnTotal + Round( ( ValorPend - ValorAce ) * TmpCot.Cotacaos, 2 )
Endscan
If !Seek( crSigCdPam.Moedaps, 'TmpCot' )
	m.Moedas = crSigCdPam.Moedaps
	m.Cotacaos = fCarregarCambio( m.Moedas, crSigCdPgr.Datarcs )
	Insert Into TmpCot From Memvar
Endif
lnTotal = Round( lnTotal / TmpCot.Cotacaos, 2 )
With ThisForm.Pagina.Dados.Pagina.Pendencias
	.Get_ValConv.Value = lnTotal
	.Get_ValConv.Refresh
Endwith 

Select xPar
Go Top
If _RecnoxPar # 0
	Go _RecnoxPar
Endif

Thisform.mOrdemConta()
If _RegisConta # 0
	Go _RegisConta
Endif

With This.Pagina.Dados.Pagina
	.Pendencias.Total.Refresh()
	.Pagamento.fwmemo1.Refresh()

PROCEDURE montaemps
Local lcEmp, lcVir

Select crSigCdEmp 
Locate For .F.
Go Top

Select Cemps ;
  From crSigCdEmp ;
 Where Imps ;
  Into Cursor LocalEmps ReadWrite 
Go Top In LocalEmps

lcEmp = []

Do Case
	Case Reccount([LocalEmps]) = 1
		lcEmp = [ And Emps = '] + LocalEmps.Cemps + [']

	Case Reccount([LocalEmps]) > 1
		lcEmp = [ And Emps In (]
		lcVir = []
		Select LocalEmps
		Scan
			lcEmp = lcEmp + lcVir + ['] + LocalEmps.Cemps + [']
			lcVir = [, ]
		EndScan
		lcEmp = lcEmp + [)]
EndCase

Return lcEmp
ENDPROC

* PROCEDURE Pagina.Dados.Grupo_Salva.Salva.Click
------------------------------------------------------------
PROCEDURE Pagina.Dados.Grupo_Salva.Salva.Click
Local llLancInvalido, lnRecNo

ThisForm.retorna = .t.
If ThisForm.pcEscolha $ "INSERIR/ALTERAR"
	If fVerificaBloqueio( crSigCdPgr.Datarcs, ThisForm.poDatamgr ) # 0

	Endif

	Set Order to Vencs
	_Sel = .f.
	llLancInvalido = .F. 
	Scan
		If Flag And Acerto # 0
			_Sel = .t.
		Endif
		If Flag And Tipos='M' And (Acerto # Valos) And (Acerto # 0)
			llLancInvalido = .T.
		EndIf
	Endscan
	If !_Sel
		Messagebox( 'Nenhuma conta foi selecionada para baixa...', 48, '' )
		Return
	Endif
	
	If llLancInvalido
		Messagebox( 'Lançamento Manual tem que ser totalmente utilizado...', 48, '' )
		Return
	EndIf
	
	Select TmpTot
	lnTotPend = 0
	Scan
		lnTotPend = lnTotPend + ValorPend
	Endscan

		Return
	Endif

		Messagebox('Obrigatório informar Título Banco.',48,'Aviso')

		Return .f.
	Endif
	
	If MessageBox( "Dados corretos ? ", 4 + 32 + 256, "ATENÇĂO!!!" ) = 7
		Select TmpConta
		Return
	Endif

	coper = Iif( ThisForm.Tipo = "C", "D", "C" )

	Into Cursor TmpCli
	Index On Grupos + Contas Tag GruCon

	*** Gravando Contas
	Select TmpCliente
	_ContaCli = 0
	Store '' to _Grupo, _Conta
	Scan
		If !IsEmpty( TmpCliente.Grupo ) Or !IsEmpty( TmpCliente.Conta ) 
			If Seek(TmpCliente.Grupo + TmpCliente.Conta,'TmpCli','GruCon')
				If IsEmpty( _Grupo ) Or IsEmpty( _Conta )
					_Grupo = TmpCliente.Grupo
					_Conta = TmpCliente.Conta
				Endif
				_ContaCli = _ContaCli + 1
			Endif
		Endif
	Endscan
	Select TmpConta
	Set Order to Vencs
	Scan for Flag
		If IsEmpty( _Grupo ) Or IsEmpty( _Conta )
			_Grupo = TmpConta.Grupos
			_Conta = TmpConta.Contas
		Endif
		If !Seek( Grupos + Contas, 'TmpCliente', 'GruConta' )
			Insert Into TmpCliente ( Grupo, Conta ) Value ( Tmpconta.Grupos, TmpConta.Contas )
		Endif
	Endscan

	Select TmpCliente
	Go Top

	Replace crSigCdPgr.Grupos With _Grupo,;
			crSigCdPgr.Contas With _Conta

	*** Acumulando Lançamentos
	Create Cursor TmpLanca ( Grupos c(10), Contas c(10), Moeda c(3), Cotacaos n(14,4),;
							   GrupoC c(10), ContaC c(10), MoedaC c(3), CotacaoC n(14,4),;
							    Hists c(40), Valor n(12,2), ValPag n(12,2), Tipos c(1),;
							    ValorC n(12,2), ValPagC n(12,2), Hist2s c(40) )
	Index on Grupos + Contas + Moeda + Tipos + GrupoC + ContaC + MoedaC + Hists + Hist2s Tag GruConMoe

	Replace all Grupos With crSigCdPgr.Grupos, Contas With crSigCdPgr.Contas for Tipos = "X"

		ThisForm.Podatamgr.CursorQuery('SigOpFp','CrFpag','Fpags',TmpBaixa.Fpags, [ValPends, Infos, Fpags ])

			Loop
		Endif
		_Grupo = TmpBaixa.Grupos
		_Conta = TmpBaixa.Contas
		_Moeda = TmpBaixa.Moedas
		_Cot   = TmpBaixa.Cotacaos
		_Valor = TmpBaixa.Valos
		If IsEmpty( TmpBaixa.Valocs )
			_ValorC = TmpBaixa.Valos
			_CotC   = _Cot
			_MoedaC = _Moeda
		Else
			_ValorC = TmpBaixa.Valocs
			_CotC   = TmpBaixa.Cotacaocs
			_MoedaC = TmpBaixa.Moecs
		Endif
		If TmpBaixa.Tipos = 'X'
			_Cot = TmpBaixa.Cotacaos
			If ThisForm.MoedaMaior # TmpBaixa.Moedas And !IsEmpty( ThisForm.MoedaMaior )
				If !Seek( ThisForm.MoedaMaior, 'TmpCot' )
					m.Moedas = ThisForm.MoedaMaior
					m.Cotacaos = fCarregarCambio( ThisForm.MoedaMaior, crSigCdPgr.Datarcs )
					Insert Into TmpCot From Memvar
				Endif
				_Valor  = Round( TmpBaixa.Valos * _Cot / TmpCot.Cotacaos, 4 )
				_Moeda  = ThisForm.MoedaMaior
				_MoedaC = TmpBaixa.Moedas
				_Cot    = TmpCot.Cotacaos
				_CotC   = TmpBaixa.Cotacaos
			Else
				_Valor = TmpBaixa.Valos
			Endif
			_ValPag = 0
		Else
			If TmpBaixa.Tipos = 'Y'
				_Grupo = crSigCdPam.GruAcertos
				_Conta = crSigCdPam.ConAcertos
			Endif
			_ValPag = _Valor
		Endif

		If crFPag.ValPends = 1 And !( crFpag.Infos = 'C' And COper = 'C' ) And ThisForm.Tipo = 'D'
			Loop
		Endif
		_NOper = fGerUniqueKey('SigMvCcr')

		_Hist2 = ''

		If lnErro < 1
			MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
			Return 0
		EndIf
		
		=ThisForm.Podatamgr.CursorQuery('SigCdCli','CrSigCdCli','Iclis',_Conta,[Hists,TpHists])

		If IsEmpty( crSigCdCli.Hists )
		
			If ! IsEmpty(crSigCdOpe.Hists)

				_Hist = Alltrim( crSigCdOpe.Hists ) + ' '
				
				Do Case 
					Case crSigCdOpe.TpHists = 1
						_Hist = _Hist + crMccr.Nfs
					Case crSigCdOpe.TpHists = 2
						_Hist = _Hist + crMccr.Titulos
					Case cSigCdOpe.TpHists = 3
						_Hist = _Hist + " " + Alltrim( fGerMascara ( crSigCdPgr.Numes ) )

				Endcase
			Else
				_hist = crSigCdPgr.Emps + " " + Alltrim( crSigCdPgr.Dopes ) + " " + Alltrim( fGerMascara( crSigCdPgr.Numes ) ) +;
							' ( ' + Alltrim( crFPag.FPags ) + ' )'
			Endif
		Else
			_Hist = Alltrim(crSigCdPgr.Dopes) + " " + Alltrim( crSigCdCli.Hists ) + ' '
			_histX = crSigCdPgr.Emps + " " + Alltrim( crSigCdPgr.Dopes ) + " " + Alltrim( fGerMascara( crSigCdPgr.Numes ) ) +;

					Else
						_Hist = _Histx
					Endif

		Endif

		If IsEmpty( TmpBaixa.FPags )  
			If _Grupo = TmpBaixa.Grupocs And _Conta = TmpBaixa.Contacs And TmpBaixa.Moedas = TmpBaixa.Moecs
				Loop
			Endif
			_GrupoC = TmpBaixa.Grupocs

			Endif
			Select TmpLanca
			Replace Valor With Valor + _Valor,;
					ValPag With ValPag + _ValPag,;
					ValorC With ValorC + _ValorC
		Else
			_GrupoC = Iif( coper="C", crSigCdPam.gruporecs, crSigCdPam.grupopags )

			Set Order to GruConMoe

			Endif
			Select TmpLanca
			Replace Valor With Valor + _Valor,;
					ValPag With ValPag + _ValPag,;
					ValorC With ValorC + _ValorC
		Endif
	Endscan
	Select TmpLanca
	Scan
		If Valor = 0
			Delete
		Endif
	Endscan
	Select TmpLanca
	Go Top
	If Eof()

			If Tipos = 'X' Or Tipos = 'Y'

	Select TmpCot
	=Seek( lcMoeda )
	Replace Cotacaos with This.Value

	lnCotac = This.Value
	Select TmpConta
	Set Order to
	Scan for Moedas = lcMoeda
		Replace Cotacaos with lnCotac
	Endscan

	Scan for MoedaCs = lcMoeda
		Replace CotacaoCs With lnCotac, ;
				ValoCs    With Round((TmpConta.Cotacaos * TmpConta.Valos) / lnCotac, 2)
	Endscan

	Select xPar
	Scan for MoeFpgs = lcMoeda
		Replace CotFpgs with lnCotac
	Endscan

	Do Form SigOpAlC with crSigOpFp.Moefpgs, .Get_Cotacao.Value, crSigCdPgr.Datarcs to lcRetorno
	If MLine( lcRetorno, 1 ) = 'OK'
		.Get_Cotacao.Value = Val( MLine( lcRetorno, 2 ) )

		If .Get_Cotacao.peAntvalue # .Get_Cotacao.Value 
		
			lcMoeda = crSigOpFp.Moefpgs
			
			Select TmpCot

				Replace Cotacaos with lnCotac
			Endscan
			Select xPar
			Scan for MoeFpgs = lcMoeda
				Replace CotFpgs with lnCotac
			Endscan
		Endif

		.Get_Cotacao.Value = .Get_Cotacao.peAntvalue 
	Endif
