PROCEDURE Init
Lparameters pSNT,lcNomeFrm1

With ThisForm
	.poDataMgr = CreateObject('fSqlConector', ThisForm.Name)
	
	lcQryPar = [Select GrPadVens From SigCdPam]	

	If (.poDataMgr.SqlExecute(lcQryPar, 'crSigCdPam') < 1)
		=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalMccr)')
		Return 0
	EndIf

	pGru = Alltrim(crSigCdPam.GrPadVens)
	thisform.antgru = pGru

	** Verifica se existe conta cadastrada para esse usuario **************************	
	If !fChecaAcesso('SIGCDVIS','ALTCONTA')
		lcQryCta = [Select iclis,rclis From SigCdCli ] + ;
						[Where rclis = '] + usuar + [' AND Grupos = '] + pGru + [']
						
		If (.poDataMgr.SqlExecute(lcQryCta, 'SigCdCli') < 1)
			=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (LocalMccr)')
			Return 0
		EndIf 

		If Empty(SigCdCli.rclis)&& = []
			=MessageBox('Conta năo cadastrada no Cadastro de Contas de Representantes', 0+256, 'Cadastro de Visitas')
			Return .f.				
		Else
			thisform.getCdRepr.Value = Alltrim(SigCdCli.iclis)
			thisform.getDsRepr.Value = Alltrim(SigCdCli.rclis)
			thisform.getDsRepr.Enabled = .F.
			thisform.getCdRepr.Enabled = .F. 
		EndIf
	Else
		ThisForm.getCdRepr.Value   = Space(10)
		ThisForm.getDsRepr.Value   = Space(40) 	
	EndIf 
	** Fim *****************************************************************************

	If (.poDataMgr.pnIdconn > 0)
		.LockScreen = .t.
		.getCdConta.Value = Space(10)
		.getDsConta.Value = Space(20)
		.getDatai.Value       = Date()
		.getDataf.Value       = Date()

		.getDatai.SetFocus
		.Refresh
		.LockScreen = .f.

		=DoDefault()
	Else
		=Messagebox('Impossivel Efetuar Conexăo com Servidor de Banco de Dados...', 0+48, 'CreateObject')
		Return .f.
	EndIf
EndWith
* Fim **************************************************************************************

ENDPROC

------------------------------------------------------------
* PROCEDURE processamento
------------------------------------------------------------
PROCEDURE processamento
Private pDtI, pDtF

pGru = thisform.antgru 

ThisForm.poDataMgr.CursorQuery('SigCdEmp', 'crSigCdEmp', 'Cemps', _Empr, 'Razas')
m.lcEmpresa = _Empr + Iif(Eof('crSigCdEmp'), '', crSigCdEmp.Razas)
m.ldDatai   = ThisForm.getDatai.Value
m.ldDataf   = ThisForm.getDataf.Value
m.lcCconta  = ThisForm.getCdConta.Value
m.lcDconta  = ThisForm.getDsConta.Value
m.lcCrepr  = ThisForm.getCdRepr.Value
m.lcDrepr  = ThisForm.getDsRepr.Value
m.lcVenda = ThisForm.opt_Venda.Value
m.lcPedido = ThisForm.opt_Pedido.Value

ldDtI = Iif(Empty(ThisForm.getDataI.Value), Ctod('01/01/0001'), ThisForm.getDataI.Value)
ldDtF = Iif(Empty(ThisForm.getDataF.Value), Ctod('31/12/9999'), ThisForm.getDataF.Value)
lcCta = Iif(Empty(ThisForm.getCdConta.Value), Space(10), ThisForm.getCdConta.Value)
lcRep = Iif(Empty(ThisForm.getCdRepr.Value),   Space(10), ThisForm.getCdRepr.Value)
lnVen = Iif(ThisForm.opt_Venda.Value=3, 0, ThisForm.opt_Venda.Value)
lnPed = Iif(ThisForm.opt_Pedido.Value=3, 0, ThisForm.opt_Pedido.Value)

pDtI = fDtoSQL(ldDtI)
pDtF = fDtoSQL(ldDtF, '23:59:59')    
		    
lcQuery = [Select a.*, b.rclis as descri, c.rclis as repr, ] + ;
		[case when a.vendas=1 then 'Sim' else 'Năo' end as descvendas, ] + ;
		[case when a.pedidos=1 then 'Sim' else 'Năo' end as descpedidos ] + ;
	      [From SigCdVis a ] + ;
	      [Inner Join SigCdCli b on a.iclis=b.iclis ] + ;
	      [Inner Join SigCdCli c on a.Contas=c.iclis ] + ;	      
		 [Where a.Grupos = ?pGru And ] + ;
		 	[(a.Pedidos = ?lnPed Or ?lnPed='') And ] +;
		 	[(a.Vendas = ?lnVen Or ?lnVen='') And ] +;
		 	[(a.iclis = ?lcCta Or ?lcCta='') And ] +;
		 	[(a.Contas = ?lcRep Or ?lcRep='') And ] +;
		 	   [a.Datas Between ?pDtI And ?pDtF ] + ;
		 [Order By a.contas] 
    				    
If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'crSigCdVis') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpConta)')
	Return 0
EndIf

* inicializa as variáveis de cabeçalho
ThisForm.podatamgr.CursorQuery('SigCdEmp','crSigCdEmp', 'cEmps', _Empr)
lcCbEmpresa = crSigCdEmp.Razas
lcCbTitulo  = 'Relatório de Visitas'
lcCbPeriodo = 'Período: '+dtoc(ldDtI)+' ŕ '+dtoc(ldDtF)

create cursor CsCabecalho (cb_empresa c(80),cb_titulo c(80),cb_periodo c(80))
insert into CsCabecalho (cb_empresa,cb_titulo,cb_periodo);
				 values (lcCbEmpresa,lcCbTitulo,lcCbPeriodo)

Select crSigCdVis
ENDPROC
