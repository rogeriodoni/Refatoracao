* PROCEDURE carregaconta
------------------------------------------------------------
PROCEDURE carregaconta
Create Cursor TmpConta (grupos c(10),contas c(10),flag l,valos n(12,2),acerto n(12,2),moedas c(3),datas d,vencs d,;
						hists c(50),nomes c(50),emps c(3),nopers n(9),dopes c(20),numes n(6),tipos c(1),fpags c(12),;
						opers c(1),baixa n(12,2),baixaaut n(12,2),cotacaos n(12,4),grupocs c(10),contacs c(10),;
						cotorigs n(12,4),moedacs c(3),cotacaocs n(12,4),valocs n(12,2),empos c(3), OrdCopChs n(3))

_Chave = CsSigCqChi.emps + CsSigCqChi.dopes + Str(CsSigCqChi.numes,6)

If Used('TmpPrIt')
	Use In TmpPrIt
EndIf

lStrQuery = [Select emps,dopes,numes,empos,grupos,contas,tipos,nopers,opers,acertos,cotacaos,valos,moedas,hists,vencs,datas ]+;
			[From SigCdPit ]+;
			[Where empdopnums = "]+_Chave+[" ]+;
			[Order By emps,dopes,numes,grupos,contas,nopers ]

lnQueryOk = ThisForm.poDataMgr.SqlExecute(lStrQuery,'TmpPrIt')

If lnQueryOk < 1
	MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
	Return(.F.)
EndIf

Select TmpPrIt
Go Top
Scan
	Scatter MemVar
	m.flag   = .F.
	m.nomes  = ''
	If Seek(m.contas,'CrContas','iclis')
		m.nomes = CrContas.rclis
	EndIf

	Do Case
		Case TmpPrIt.tipos = 'M'
			Scatter Fields moedas,hists,vencs,datas MemVar
			If m.acertos <> 0
				m.Flag  = .T.
			EndIf
	
		Case TmpPrIt.tipos = 'C'
			m.Flag  = .T.
			If Used('TmpMccr')
				Use In TmpMccr
			EndIf
			lStrQuery = [Select emps,nopers,grupos,contas,valors,valpags From SigMvCcr ]+;
						[Where emps = ']+TmpPrIt.empos+[' And nopers = ]+Str(TmpPrIt.Nopers,9)+[ ]
			lnQueryOk = ThisForm.poDataMgr.SqlExecute(lStrQuery,'TmpMccr')
			If lnQueryOk < 1
				MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
				Return(.F.)
			EndIf
			
			Select TmpMccr
			Go Top
			If m.grupos <> TmpMccr.grupos Or m.contas <> TmpMccr.contas
				Select TmpMccr
				Skip
			EndIf

			m.valos  = TmpMccr.valors - TmpMccr.valpags + m.acertos
			
		Case TmpPrIt.tipos = 'P'
			m.Flag  = .T.
			If Used('TmpPar')
				Use In TmpPar
			EndIf
			lStrQuery = [Select emps,nopers,moefpgs,fpags,vpags From SigMvPar ]+;
						[Where emps = ']+TmpPrIt.empos+[' And nopers = ]+Str(TmpPrIt.Nopers,9)+[ ]
			lnQueryOk = ThisForm.poDataMgr.SqlExecute(lStrQuery,'TmpPar')
			If lnQueryOk < 1
				MessageBox('Favor reinicializar o processo.',16,'Falha na Conexăo')
				Return(.F.)
			EndIf
			Select TmpPar
			Go Top
			m.valos  = m.valos - TmpPar.vpags + m.acertos
			m.moedas = TmpPar.moefpgs
			m.hists  = Alltrim(TmpPar.fpags) + ' ' + m.emps + Alltrim(m.dopes) + ' ' + Alltrim(fGerMascara(m.numes))
	EndCase
	m.acerto = m.acertos
	Insert Into TmpConta From MemVar
EndScan
Return
ENDPROC

* PROCEDURE mexibecheques
------------------------------------------------------------
PROCEDURE mexibecheques
LParameters llSeek

ThisForm.LockScreen = .T.
llSeek = Iif(Type('llSeek') = 'L',llSeek,.F.)
If Used('CsSigCqChi')
	UpDate CsSigCqChi Set nMarca1s = 0 Where nMarca1s = 1

PROCEDURE montachq
Lparameters llPosiciona

Private lcDtInicial, lcDtFinal

ThisForm.LockScreen = .T.

If llPosiciona
	lcBusca = csSigCqChi.Bancos + CsSigCqChi.Agencias + CsSigCqChi.ncontas + CsSigCqChi.Ncheques
EndIf

lcDtInicial = fDtoSQL(ThisForm.Dt_Inicial.Value)
lcDtFinal   = fDtoSQL(ThisForm.Dt_Final.Value,'23:59:59')
lcCdGrupo   = ThisForm.getCdGrupos.Value
lcCdConta   = ThisForm.getCdContas.Value

Wait Window 'Aguarde! Selecionando Cheques...' NoWait

If Used('CsSigCqChi')
	ThisForm.GrdCCheques.RecordSource = ''
	Use In CsSigCqChi
EndIf

If Empty(lcCdConta)
	lcQuery = [Select a.emps, a.dopes, a.numes, a.datas, a.bancos, a.agencias, a.ncontas, a.ncheques, a.contas, ] + ;
					 [a.valors, a.favos, a.ncopias, a.emitidos, a.nemissoes, a.cancelas, a.cidchaves, a.JustCanc ] + ;
				[From SigCqChi a ] + ;
			   [Where a.datas Between ?lcDtInicial And ?lcDtFinal And ] + ;
				 	  Iif(Empty(lcCdGrupo), [], [a.Grupos = '] + lcCdGrupo + [' And ]) + ;
				 	 [a.Contas in (Select Distinct ContaDs From SigOpFp Where EmiChqs = 1) ] + ;
			   [Order By a.emps, a.dopes, a.numes, a.datas, a.bancos, a.agencias, a.ncontas, a.ncheques, a.contas, ] + ;
				 	 [a.valors, a.favos, a.ncopias, a.emitidos, a.nemissoes, a.cancelas, a.cidchaves]
Else
	lcQuery = [Select emps, dopes, numes, datas, bancos, agencias, ncontas, ncheques, contas, ] + ;
					 [valors, favos, ncopias, emitidos, nemissoes, cancelas, cidchaves, JustCanc ] + ;
				[From SigCqChi ] + ;
			   [Where datas Between ?lcDtInicial And ?lcDtFinal And ] + ;
					  Iif(Empty(lcCdGrupo), [], [Grupos = '] + lcCdGrupo + [' And ]) + ;
					  [Contas = '] + lcCdConta + [' ] + ;
			   [Order By emps, dopes, numes, datas, bancos, agencias, ncontas, ncheques, contas, ] + ;
					 [valors, favos, ncopias, emitidos, nemissoes, cancelas, cidchaves]
EndIf

If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'TmpChi') < 1)
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (TmpChi - ' + Iif(Empty(lcCdConta), '1', '2') + ')')
	Return .f.
EndIf

Select Emps, Dopes, Numes, Datas, Bancos, Agencias, NContas, NCheques, Contas, Valors, Favos, NCopias, ;
	   NEmissoes, cIdChaves, Iif(Emitidos,1,0) as NEmitidos, Iif(Cancelas,1,0) as NCancelas, 0 as NMarca1s, JustCanc ;
  From TmpChi ;
 Order By Bancos, Agencias, NContas, NCheques ;
  Into Cursor csSigCqChi ReadWrite

Select CsSigCqChi
Index On ncopias               Tag NCopias
Index On nEmitidos             Tag NEmitidos
Index On nCancelas             Tag NCancelas
Index On nMarca1s              Tag NMarca1s
Index On ncheques              Tag NCheques
Index On datas                 Tag Datas
Index On ncontas + ncheques    Tag Conta
Index On contas + Str(ncopias) Tag Contas
Index On Dtos(datas) + bancos + agencias + ncontas + ncheques      Tag Emissao
Index On Str(valors,12,2) + bancos + agencias + ncontas + ncheques Tag Valor
Index On bancos + agencias + ncontas + ncheques                    Tag Cheque
Index On agencias + ncontas + ncheques                             Tag Agencia
Set Order To NCopias
If llPosiciona
	If Not Seek(lcBusca)
		Go Top
	Endif
Else
	Go Top
EndIf

ThisForm.LockScreen = .F.

With ThisForm
	With .GrdCCheques
		.RecordSource = 'CsSigCqChi'
		.clnImprime.ControlSource   = 'CsSigCqChi.nMarca1s'
		.clnDatas.ControlSource     = 'CsSigCqChi.datas'
		.clnContas.ControlSource    = 'CsSigCqChi.contas'
		.clnNCopias.ControlSource   = 'CsSigCqChi.ncopias'
		.clnBancos.ControlSource    = 'CsSigCqChi.bancos'
		.clnAgencias.ControlSource  = 'CsSigCqChi.agencias'
		.clnNContas.ControlSource   = 'CsSigCqChi.ncontas'
		.clnNCheques.ControlSource  = 'CsSigCqChi.ncheques'
		.clnValors.ControlSource    = 'CsSigCqChi.valors'
		.clnSituacaos.ControlSource = [IIf(CsSigCqChi.ncancelas = 1,'Cancelado',IIf(CsSigCqChi.nemissoes > 1,'Reemitido',;
										IIf(CsSigCqChi.nemitidos = 1,'Emitido','Năo Emitido')))]
		.SetAll("DynamicForeColor","Iif(CsSigCqChi.ncancelas = 1,Rgb(255,0,0),Iif(CsSigCqChi.nemitidos = 0,Rgb(0,0,255),Rgb(0,0,0)))","Column")
		.Refresh
	EndWith 
	.cntjustificativa.get_justificativa.ControlSource='CsSigCqChi.JustCanc'
EndWith


Wait Clear

If llPosiciona
	ThisForm.mExibeCheques(.F.)
Else
	ThisForm.mExibeCheques(.T.)
EndIf

ThisForm.Inicial = .f.

Return(.T.)
ENDPROC

		lcNomeImp1 = Lower(AllTrim(CrImp1.cNomeImp1s))

	  From crImp1 ;
	 Where nMarca1s = 1 ;
	 Order By cNomeImp1s ;
	  Into Cursor crImp2 ReadWrite
	
	Select CrImp2
	Index On cNomeImp2s Tag cNomeImp2s
	Go Top
	
	Set Point To ','
	Set Separator To '.'
	  
	Select CrImp2
	Set Order To ('') In CrImp2
	Locate

	If !Eof('CrImp2')
		KeyBoard Chr(32) Plain
		= Inkey()

			m.loLista = CreateObject('fwBuscaInt','CrImp2','cNomeImp2s','','Seleçăo')

			m.loLista.Show()

			m.lcNomeImp1 = CrImp2.cNomeImp1s
			m.lcNomeImp2 = AllTrim(CrImp2.cNomeImp2s)
			
			Set Order To ('') In CsSigCqChi   && Retirar a Ordem e Set Key...

PROCEDURE cmdprocurar.Click
Set Near On

This.Parent.Parent.getBanco.Value = PadR(AllTrim(This.Parent.Parent.getBanco.Value),3)

Select CsSigCqChi
Do Case
	Case !Empty(This.Parent.Parent.getEmissao.Value)
		Set Order To Emissao		
		Seek Dtos(This.Parent.Parent.getEmissao.Value) + This.Parent.Parent.getBanco.Value +;
			This.Parent.Parent.getAgencia.Value + This.Parent.Parent.getConta.Value +;
			This.Parent.Parent.getCheque.Value
		         
	Case !Empty(This.Parent.Parent.getValor.Value)
		Set Order To Valor
		Seek Str(This.Parent.Parent.getValor.Value,12,2) + This.Parent.Parent.getBanco.Value +;
			This.Parent.Parent.getAgencia.Value + This.Parent.Parent.getConta.Value +;
			This.Parent.Parent.getCheque.Value

	Case !Empty(This.Parent.Parent.getBanco.Value)
		Set Order To Cheque
		Seek This.Parent.Parent.getBanco.Value + This.Parent.Parent.getAgencia.Value +;
			This.Parent.Parent.getConta.Value + This.Parent.Parent.getCheque.Value
		         
	Case !Empty(This.Parent.Parent.getAgencia.Value)
		Set Order To Agencia
		Seek This.Parent.Parent.getAgencia.Value + This.Parent.Parent.getConta.Value + ;
			This.Parent.Parent.getCheque.Value
		         
	Case !Empty(This.Parent.Parent.getConta.Value)
		Set Order To Conta
		Seek This.Parent.Parent.getConta.Value + This.Parent.Parent.getCheque.Value
		         
	Case !Empty(This.Parent.Parent.getCheque.Value)
		Set Order To NCheques
		Seek This.Parent.Parent.getCheque.Value
EndCase

Set Near Off

ThisForm.LockScreen = .T.

ThisForm.getCdContas.Enabled   = .T.
ThisForm.getDsContas.Enabled   = .T.
ThisForm.GrdCCheques.Enabled   = .T.
ThisForm.TxtFavorecido.Enabled = .T.
ThisForm.CmdGOk.Enabled        = .T.
ThisForm.CntProcurar.Enabled   = .F.
ThisForm.CntProcurar.Visible   = .F.

ThisForm.mExibeCheques(.F.)
ENDPROC
