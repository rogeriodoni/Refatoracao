PROCEDURE desmembrac
Select xNensi
lnReg = Recno()

llOk = .t.
lcMat = xNensi.Cmats
lnOp = xNensi.Nenvs
lnTpOp = xNensi.TpOPs
lnLote = xNensi.nLotes
lnQtdAnt = xNensi.Qtds
lnPesoAnt = xNensi.Pesos
lnAmarraAnt = xNensi.Amarracao

If lnAmarraAnt <> 0
	Set Order To
	Go Top
	Delete All For FlagIncs And Amarracao = lnAmarraAnt And Cmats <> lcMat
	Set Order To Editar
	Go lnReg
Endif

Select xMFas
Go top
Locate For Nops = lnOp
If Eof()
	Return .f.
Endif
lnQtde = xMFas.Qtds && Quantidade de peças na Op

ThisForm.CtrlAgrega = ThisForm.CtrlAgrega + 1
_xAmarra = ThisForm.CtrlAGrega

lcQuery = [Select a.Mats, a.DCompos, a.Qtds, a.UniCompos, a.Pesos, a.CuniPs, b.Cgrus, b.Varias, c.TpMovs, a.Pesos as Peso2s From SigPrCpo a ] +;
			[Inner Join SigCdPro b On a.Mats = b.Cpros ] +;
			[Inner Join SigCdGrp c On a.Cgrus = c.Cgrus ] +;
			[Where a.Cpros = '] + xNensi.CMats + [']
If ThisForm.poDataMgr.SqlExecute(lcQuery,[crTmpPrCpo]) < 1
	=MessageBox('Favor Reinicializar o Processo!!!', 16, 'Falha na Conexăo (crTmpPrCpo)')
	Return .f.
Endif

*** Calcula o peso fábrica total dos componentes que năo tem peso variável
lnPesoTot = 0
Select crTmpPrCpo
Scan
	ThisForm.poDataMgr.CursorQuery('SigCdUni', 'LocalUni', 'CUnis', crTmpPrCpo.UniCompoS, 'Fators,QtEqualPs,cEstos')
	ThisForm.poDataMgr.CursorQuery('SigCdUni', 'LocalUni2', 'CUnis', crTmpPrCpo.CUniPs, 'Fators,QtEqualPs,cEstos')

	If crTmpPrCpo.Varias # 1
		lnQtdCmp = lnQtde * crTmpPrCpo.Qtds

		loLista.mAddColuna('CUnis', '', 'Uni')
		loLista.Show()
	Endif
	
	*** Tiago - 10/02/2016 - Costantini - Checa se o material padrăo = material principal para năo deixar movimentar componentes diferentes
	If crSigCdOpd.ChkMtPrs = 1 And !Empty(CrSigCdOpd.MatPads) And CrSigCdOpd.MatPads # crListaRemota.MatPrincs
		Messagebox([Componente năo permitido nessa movimentaçăo!],48,[Aviso])
		this.Value = []
		Return 0
	Endif

	lcQuery = [Select Distinct Cgrus From SigCdRpo Where Dopps = ']+CrSigCdOpd.Dopps+[' And not Cgrus = ' ' Order By Cgrus]
	If (ThisForm.poDataMgr.SqlExecute(lcQuery, 'LocalRopt') < 1)

		Return 0
	EndIf
	
	Select LocalRopt
	If Reccount() > 0
		Locate For Cgrus = CrListaRemota.Cgrus
		If Eof()
			=MessageBox('Produto Bloqueado para esta movimentaçăo!', 16, 'Restriçăo')
			this.Value = []			
			Return 0
		Endif
	Endif
	
	This.Value = crListaRemota.CPros
	
	Replace Cmats  With crListaRemota.CPros, ;
			CDescs With crListaRemota.DPros, ;
			CUnis  With crListaRemota.CUnis, ;

	lcMerc = CrListaRemota.Mercs

	If Not CrTmpGru.Tams = 1
		Replace CodTams with ' ' in xNensi
	EndIf
	If CrTmpGru.ServPrds = 1 And This.Value <> ThisForm.AntValue
		lcCat = Padr(This.Value,6)
		Select crSigCdSvc
		Set Order to CaGruCo
		Do Case
			Case Seek(lcCat + TmpNens.Grupoos + TmpNens.Contaos )
			Case Seek(lcCat + TmpNens.Grupoos + Space(10))
			Case Seek(lcCat + Space(20))
		Other
			MessageBox('Este Serviço năo esta definido no cadastro de Serviços!!!',0+16,'')
			Return 0
		Endcase

		*** Tiago - 04/09/11 - Valor do serviço a ser lançado varia de acordo com o percentual do cargo do funcionário. Se estiver sem cargo năo lança pagamento	
		_Grupo = TmpNens.GrupoOs

			_Grupo = TmpNens.GrupoDs

Create Cursor xTmpx (CMats c(14), Qtds n(12,3), Pesos n(12,3))

	If Empty(CMats) Or (CMats = crSigCdPam.Ouros)

	If (LocalPro.Varias <> 1)
		Loop
	EndIf

		Insert Into xTmpx (CMats) Values (xNensi.CMats)
	EndIf
	Replace Qtds  With Qtds  + xNensi.QtFaseAnt, ;
			Pesos With Pesos + xNensi.PsFaseAnt
EndScan

=Seek(TmpNens.Dopps, 'crSigCdOpd', 'Dopps')

Select xTmpx
Scan For Qtds <> 0 And Pesos <> 0
	ThisForm.poDataMgr.CursorQuery('SigCdPro', 'LocalPro', 'CPros', xTmpx.Cmats, 'DPros, CUnis')
	Insert Into xNensi (Emps, Dopps, Numps, CMats, CDescs, CUnis, ;
						Pesos, Qtds, TpOps, FlagIncs, Rateio, Visivel) ;
				Values (TmpNens.Emps, TmpNens.Dopps, TmpNens.Numps, xTmpx.Cmats, LocalPro.Dpros, ;
						LocalPro.CUnis, xTmpx.Pesos, xTmpx.Qtds, crSigCdOpd.Retalhos, .t., 5, .t.)
EndScan

Select xNensi 
Set Order To Editar

ThisForm.Grade.Refresh
* Fim **************************************************************************************
