PROCEDURE processamento
Local llOks, lcCid

llOks = .f.

lcCid = ThisForm.cmbCidades.Value && 1-RJ - Rio de Janeiro
								  && 2-SP - Săo Paulo

Do Case
	Case (Alltrim(lcCid) = [RJ - Rio de Janeiro])
		llOks = ThisForm.PrcRJRioDeJaneiro()
		
	Case (Alltrim(lcCid) = [SP - Săo Paulo])
		llOks = ThisForm.PrcSPSaoPaulo()
EndCase

If llOks
	If (ThisForm.poDataMgr.Commit() < 1)
		=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Commit)])
		llOks = .f.
	EndIf

	ThisForm.AjustaControles(.t.)
	Zap In csOperacoes 
	ThisForm.btnSelecionar.Click(.t.)
EndIf

If Not llOks
	ThisForm.poDataMgr.RollBack()
EndIf
	
Return llOks
* Fim ********************************************************************************************
ENDPROC
