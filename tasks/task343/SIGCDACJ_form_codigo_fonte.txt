PROCEDURE msv_inserir
Local llOks

llOks = .t.

If Not ThisForm.poDataMgr.Update([crSigCdAcJ])
	=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Update - crSigCdAcJ)])
	llOks = .f.
EndIf

If llOKs
	If (ThisForm.poDataMgr.Commit() < 1)
		=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (Commit)])
		llOks = .f.
	EndIf
EndIf

If Not llOks
	ThisForm.poDataMgr.Rollback()
EndIf

NoDefault

Return llOks
* Fim **************************************************************************************
ENDPROC

PROCEDURE Init
Local lcQryCdCli, lcQryCdAcJ, lcKey

lcKey = Replicate(Chr(254), 20)

lcQryCdCli = [Select a.Grupos, a.IClis, a.RClis, a.Razaos, a.Inativas, a.Usuars, a.DataIncs ] + ;
			   [From SigCdGcr b, SigCdCli a ] + ;
			  [Where b.Coletors = 10 And ] + ;
			  		[b.Codigos = a.Grupos]

lcQryCdAcJ = [Select a.*, Space(30) as NComps From SigCdAcJ a Where a.pkChaves = '] + lcKey + [']

With ThisForm
	.poDataMgr = CreateObject([fSqlConector], .Name)

	If (.poDataMgr.pnIdconn > 0)
		With .poDataMgr
			.AddCursor([SigCdCli], [IClis], [crGradeJOB], [], ThisForm.Pagina.Lista.Grade, lcQryCdCli)
			.pfSqlTabela(1).pColuna([RCLis],    [], [], [Nome],   	   345, .t.)
			.pfSqlTabela(1).pColuna([Razaos],   [], [], [Descritivo], 332, .t.)
			.pfSqlTabela(1).pColuna([IClis],    [], [], [Código], 	   080, .t.)
			.pfSqlTabela(1).pColuna([Usuars],   [], [], [Usuário],    080, .t.)
			.pfSqlTabela(1).pColuna([DataIncs], [], [], [Inclusăo],   074, .t.)

			.AddCursor([SigCdAcJ], [pkChaves], [crSigCdAcJ], [], [], lcQryCdAcJ)
		EndWith

		=DoDefault()

		.poDataMgr.Requery([crSigCdAcJ])
		
		With .Pagina.Dados.grdUsuarios
			.AllowHeaderSizing      = .f.
			.AllowRowSizing         = .f.
			.ColumnCount            =  2
			.DeleteMark             = .f.
			.RecordSource           = [crSigCdAcJ]

			.Column1.ControlSource  = [crSigCdAcJ.Usuars]
			.Column2.ControlSource  = [crSigCdAcJ.NComps]
			
			For I=1 To .ColumnCount
				lcCol = Alltrim(Str(I))
				.Column&lcCol..Movable   = .f.
				.Column&lcCol..Resizable = .f.
			Next
		EndWith
		
		.Pagina.Lista.Grupo_Op.Inserir.Enabled  = .f.
		.Pagina.Lista.Grupo_Op.Excluir.Enabled  = .f.
		.Pagina.Lista.Grupo_Op.Procurar.Enabled = .f.
		
		.Pagina.Lista.Grade.SetAll([DynamicForeColor], [Iif(crGradeJOB.Inativas=1, RGB(255,0,0), RGB(0,0,0))], [Column])
		
		.Pagina.Lista.Grade.Column1.SetFocus
	Else
		=Messagebox([Impossível Efetuar Conexăo Com o Servidor de Banco de Dados...], 0+48, [CreateObject])
		Return .f.
	EndIf
EndWith
* Fim **************************************************************************************
ENDPROC

* PROCEDURE Pagina.Lista.Grupo_op.Click

PROCEDURE Pagina.Lista.Grupo_op.Click
Lparameters Opcao

Select crGradeJOB
If Not DoDefault(Opcao)
	ThisForm.mAtivapagina1
	Return .f.
EndIf

If Not Inlist(ThisForm.pcEscolha, [CONSULTAR], [ALTERAR])
	ThisForm.mAtivapagina1
	Return .f.
EndIf

lcJob = crGradeJOB.IClis
 
Select crSigCdAcJ
Zap In crSigCdAcJ

lcQuery = [Select a.*, b.NComps ] + ;
		    [From SigCdAcJ a Left Join SigCdUsu b ] + ;
		   	  [On a.Usuars = b.Usuarios ] + ;
		   [Where a.JOBs = '] + lcJob + [']

If (ThisForm.poDataMgr.SqlExecute(lcQuery, [LocalAcJ]) < 1)
	=MessageBox([Favor Reinicializar o Processo!!!], 16, [Falha na Conexăo (LocalAcJ)])
	Return .f.
EndIf

Select LocalAcJ
Scan
	Scatter Memvar Memo
	Insert Into crSigCdAcJ From Memvar
EndScan

If (ThisForm.pcEscolha = [ALTERAR])
	Append Blank In crSigCdAcJ
	Go Top In crSigCdAcJ
	Go Bottom In crSigCdAcJ
	ThisForm.Pagina.Dados.grdUsuarios.Column1.SetFocus 
	ThisForm.Pagina.Dados.btnInserir.Visible = .t.
	ThisForm.Pagina.Dados.btnExcluir.Visible = .t.
Else
	ThisForm.Pagina.Dados.Grupo_Salva.Salva.SetFocus
	ThisForm.Pagina.Dados.btnInserir.Visible = .f.
	ThisForm.Pagina.Dados.btnExcluir.Visible = .f.
EndIf

ThisForm.Pagina.Dados.grdUsuarios.Refresh
* Fim **************************************************************************************
